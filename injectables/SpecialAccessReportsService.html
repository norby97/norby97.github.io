import { HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class SpecialAccessReportsService extends RootHttpService {
  patientCountVisitsByStudyReport(data) {
    const params = new HttpParams().set('dateFrom', data.dateFrom).set('dateTo',data.dateTo).set('sortType',data.sortType).set('siteIds',data.siteIds);
    return this.http.get<any>(environment.serverUrl + `/reports/patient-status-visit-count-by-study`, {params})
  }
  
  generateWeeklyPerformanceReport(data) {
    const params = new HttpParams().set('siteIds', data.siteIds).set('year',data.year);
    return this.http.get<any>(environment.serverUrl + `/reports/weekly-performance`,{params});
  }
  generateWeeklyPerformance(data) {
    return this.http.post<any[]>(environment.serverUrl + `/reports/weekly-performance`,{});
  }

  getWeekDates(data) {
    const params = new HttpParams().set('year', data.year).set('month', data.month);
    return this.http.get(environment.serverUrl +`/fin-calendar`, {params})
  }

  getAllYearsData(data) {
    const params = new HttpParams().set('siteIds', data.siteIds)
    return this.http.get<any>(environment.serverUrl + `/reports/weekly-performance/all-years`,{params});
  }

  getDailyOpsReport(data) {
    const params = new HttpParams().set('year', data.year).set('month', data.month).set('week', data.week).set('siteIds',data.siteIds);
    return this.http.get<any>(environment.serverUrl + `/reports/daily-ops`,{params});
  }

  getClinicProductivityReport(data) {
    const params = new HttpParams().set('showCurrentQuarter', data.showCurrentQuarter).set('siteIds', data.siteIds)
    return this.http.get<any>(environment.serverUrl + `/reports/clinic-productivity-by-indication/consolidated`,{params});
  }

  getProjectManagementScorecardSummaryReport(data) {
    const params = new HttpParams().set('type', data.type).set('siteIds',data.siteIds)
    return this.http.get<any>(environment.serverUrl + `/reports/site-project-management/`, {params})
  }
  getProjectManagementScorecardSummaryReportcsv(data) {
    const params = new HttpParams().set('type', data.type).set('siteIds',data.siteIds)
    return this.http.get(environment.serverUrl + `/reports/site-project-management/csv`, {params,responseType:'arraybuffer'})
  }

  getCoordinatorEffetivenessReport(data) {
    const params = new HttpParams().set('dateFrom', data.dateFrom).set('dateTo',data.dateTo).set('trialType',data.trialType).set('studyId',data.studyId).set('showProvidersWithActivity',data.showProvidersWithActivity);;
    return this.http.get<any>(environment.serverUrl + `/reports/coordinator-effectiveness/${data.siteId}`, {params})
  }

  getVisitProjectionSummaryForAllVisitsReport(data) {    
    const params = new HttpParams().set('siteIds',data.siteIds)
    return this.http.get<any>(environment.serverUrl + `/reports/visit-projections-summary/all-visits/without-revenue`,{params})
  }

  getVisitProjectionSummaryForCountedVisitsReport(data) {
    const params = new HttpParams().set('siteIds',data.siteIds)
    return this.http.get<any>(environment.serverUrl + `/reports/visit-projections-summary/counted-visits/without-revenue`,{params})
  }
  getVisitProjectionSummaryForAllVisitsWithRevenueReport(data) {
    const params = new HttpParams().set('siteIds',data.siteIds)

    return this.http.get<any>(environment.serverUrl + `/reports/visit-projections-summary/all-visits/with-revenue`,{params})
  }

  getVisitProjectionSummaryForCountedVisitsWithRevenueReport(data) {
    const params = new HttpParams().set('siteIds',data.siteIds)
    return this.http.get<any>(environment.serverUrl + `/reports/visit-projections-summary/counted-visits/with-revenue`,{params})
  }

  getPatientMetricsReport(data){
    const params = new HttpParams().set('showCurrentQuarter', data.showCurrentQuarter).set('siteIds', data.siteIds)
    return this.http.get<any>(environment.serverUrl + `/reports/patient-visit-metrics`,{params});
  }
}
