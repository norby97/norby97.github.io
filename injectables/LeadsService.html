import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { environment } from 'src/environments/environment';
import { Lead } from '../../models/crmbd/lead.model';

@Injectable({
  providedIn: 'root'
})
export class LeadsService {

  constructor(private http: HttpClient) { }
  
  addNewLead(data) {
    return this.http.post(environment.serverUrl + '/lead', data);
  }

  addContactToLead(data) {
    return this.http.post(environment.serverUrl + '/lead/contact/' + data.leadID + '/' + data.contactID, data.request);
  }

  deleteLead(leadID) {
    return this.http.delete(environment.serverUrl + '/lead/' + leadID);
  }

  getLeadbyID(leadID):Observable<Lead> {
    return this.http.get<Lead>(environment.serverUrl + '/lead/' + leadID);
  }

  getLeadNote(leadID):Observable<any[]> {
    return this.http.get<any[]>(environment.serverUrl + '/lead/note/' + leadID,);
  }


  mergeLead(data) {
    return this.http.put(environment.serverUrl + '/lead/merge/' +data.from + '/' + data.to, {});
  }

  editLead(data) {
    return this.http.put(environment.serverUrl + '/lead/' + data.leadID, data.request);
  }

  sendLeadNote(data) {
    return this.http.post(environment.serverUrl + '/lead/note/' + data.leadID, data.request);
  }

  sendLeadToStudyStart(data) {
    return this.http.post(environment.serverUrl + '/study-start', data);
  }

  addArticleToLead(data):Observable<Lead> {
    return this.http.post<Lead>(environment.serverUrl + '/lead/in-the-news/' + data.leadID, {url: data.url  });
  }

  getAll(data): Observable<any[]> {
    const params = new HttpParams().set('sponsorIds', data?.sponsorIds ? data.sponsorIds : [] )
    .set('croIds', data?.croIds ? data.croIds : [])
    .set('ownerIds',data?.ownerIds ? data.ownerIds : [])
    .set('indicationIds',data?.indicationIds ? data.indicationIds : [])
    .set('phaseIds',data?.phaseIds ? data.phaseIds : [])
    .set('siteIds',data?.siteIds ? data.siteIds : [])
    .set('compoundIds',data?.compoundIds ? data.compoundIds : [])
    .set('statuses',data?.statuses ? data.statuses : [])
    .set('hasFollowUp', data?.hasFollowUp ?? false)
    .set('leadIdNumber', data?.leadIdNumber ? data?.leadIdNumber : '')
    .set('skip', data?.skip).set('limit', data?.limit)
    return this.http.get<any[]>(environment.serverUrl + '/lead', {params});
  }

  getAllWithoutFilters(): Observable<Lead[]> {
    return this.http.get<Lead[]>(environment.serverUrl + '/lead/all');
  }

  searchLead(data): Observable<Lead[]> {
    const params = new HttpParams().set('search', data.search)
    return this.http.get<Lead[]>(environment.serverUrl + '/lead/search/' ,{params});
  }

  getLeadFiles(data){
    return this.http.get(environment.serverUrl + `/crm/lead/files/${data}`);
  }

  getLeadFile(data){
    return this.http.get(environment.serverUrl + `/crm/lead/files/${data?.leadId}/${data?.fileId}`,  { responseType: 'arraybuffer' });

  }
deleteLeadFile(data){
    return this.http.delete(environment.serverUrl + `/crm/lead/files/${data?.leadId}/${data?.fileId}`);

  }

  addFileToLead(data){
    const formData: FormData = new FormData();
    formData.append('file', data.file);
    formData.append('type', data.type);
    return this.http.post(environment.serverUrl + `/crm/lead/files/${data?.leadId}` ,formData);

  }
}
