import { HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class TrainingService extends RootHttpService {

  getUsersForSite(data) {
    return this.http.get<any[]>(environment.serverUrl + `/user/from-site/${data}`)
  }

  addTrainee(data) {
    return this.http.post(environment.serverUrl + `/regulatory/user-training`,data)
  }
  
  addTraineeDetails(data) {
    const formData: FormData = new FormData();
    let request = new Blob([JSON.stringify(data.req)], {
      type: 'application/json'
    });
    formData.append('file',data.file);
    formData.append('request',request);
    return this.http.post(environment.serverUrl + `/regulatory/trainings`,formData)

  }

  getTraineesForStudy(data) {
    return this.http.get<any[]>(environment.serverUrl + `/regulatory/user-training/studies/${data}`)
  }

  getTrainings(data){
    return this.http.get<any[]>(environment.serverUrl + `/regulatory/trainings/studies/${data} `)
  }

  downloadTraineeFile(data) {
    return this.http.get(environment.serverUrl + `/regulatory/trainings/${data}/file/download`, {responseType:'arraybuffer'})
  }

  editTrainee(data){
    const formData: FormData = new FormData();
    let request = new Blob([JSON.stringify(data.req)], {
      type: 'application/json'
    });
    formData.append('file',data.file );
    formData.append('request',request);
    return this.http.put<any[]>(environment.serverUrl + `/regulatory/trainings/${data.trainingId}`, formData)
  }

  getNotes(data) {
    const params = new HttpParams().set('includeProjectNotes', data.includepm);
    return this.http.get<any[]>(environment.serverUrl + '/regulatory/trainings/note/'+ data.id, {params});
  }

  sendNote(data) {
    return this.http.post(environment.serverUrl + '/regulatory/trainings/note/' + data.studyStartID,  data.request);
  }

  deleteTrainee(data){
    return this.http.delete(environment.serverUrl + `/regulatory/user-training/${data}`,{})
  }

  deleteScale(data){
    return this.http.delete(environment.serverUrl + `/regulatory/study-training/${data}`,{})
  }
}
