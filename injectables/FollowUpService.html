import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { environment } from 'src/environments/environment';

@Injectable({
  providedIn: 'root'
})
export class FollowUpService {

  constructor(private http: HttpClient) { }

  createFollowUpTask(data) {
    return this.http.post(environment.serverUrl + '/follow-ups/contact/' + data.contactID, data.followTaskDto);
  }
  
  getFollowUpTasksForContact(data) {
    return this.http.get(environment.serverUrl + '/follow-ups/contact/' + data);
  }

  addNoteToFollowUpTask(data) {
    return this.http.post(environment.serverUrl + '/follow-ups/note/' + data.followTaskID, data.request);
  }

  getAllFollowUpsByUserID(data) {
    return this.http.get(environment.serverUrl + '/follow-ups/user/' + data);
  }

  getAllNotesByFollowTaskID(data) {
    return this.http.get(environment.serverUrl + '/follow-ups/' + data);
  }

  closeTask(data) {
    return this.http.put(environment.serverUrl + '/follow-ups/close/' + data.followTaskID, {});

  }

  showIntake() {
    return this.http.put(environment.serverUrl + `/user/ctms/daily-follow-ups/show-intake`, {});
  }

  showScreen() {
    return this.http.put(environment.serverUrl + `/user/ctms/daily-follow-ups/show-screen`, {});
  }

  showOther() {
    return this.http.put(environment.serverUrl + `/user/ctms/daily-follow-ups/show-other`, {});
  }

  showMessages() {
    return this.http.put(environment.serverUrl + `/user/ctms/daily-follow-ups/show-messages`, {});
  }
  
  getPatientMessages(data) {
    return this.http.get<any[]>(environment.serverUrl + `/ctms/answering-services/sites/${data?.siteId}`);
  }

  deletePatientMessage(data) {
    return this.http.delete(environment.serverUrl + `/ctms/answering-services/${data?.answeringServiceId}`, {});
  }

  reassignFollowUps(data){
    return this.http.put(environment.serverUrl + `/ctms/daily-tasks/reassign`, data);
  }

}
