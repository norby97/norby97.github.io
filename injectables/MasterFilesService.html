import { Injectable } from '@angular/core';
import { HttpClient, HttpParams } from '@angular/common/http';
import { environment } from 'src/environments/environment';
import { Observable } from 'rxjs';
import { env } from 'process';
import { FilePath } from '../../models/regulatory/FilePath';
import { FolderStructure } from 'src/app/models/regulatory/folder-structure.model';

@Injectable({
  providedIn: 'root'
})
export class MasterFilesService {

  constructor(private http: HttpClient) { }

  retrieveSiteFolders(data): Observable<FolderStructure> {
    return this.http.get<FolderStructure>(environment.serverUrl + '/master-file/site/' + data.siteID + '/retrieve');
  }

  createNewFolder(data) {
    return this.http.post(environment.serverUrl + '/master-file/site/' + data.id + '/new-folder', data);
  }

  externalPreviewRetrieve(siteID, fileID) {
    return this.http.get(environment.serverUrl + '/file-management/preview/' + siteID + '/download/file/' + fileID, { responseType: 'arraybuffer' });
  }
  uploadFile(data) {
    const formData: FormData = new FormData();
    formData.append('file', data.file);
    return this.http.post(environment.serverUrl + '/master-file/site/' + data.filePath.id + '/new-file', formData, { params: { filePath: JSON.stringify(data.filePath) } });
  }

  moveFile(data) {
    const params = new HttpParams().set('fileId', data.fileId).set('parentId', data.parentId);
    return this.http.put(environment.serverUrl + '/master-file/site/' + data.siteId + '/move-file', params);
  }

  retrieveFile(data) {
    return this.http.get(environment.serverUrl + '/master-file/site/' + data.siteID + '/download/file/' + data.fileID, { responseType: 'arraybuffer' });
  }

  retrieveFileDetails(data): Observable<FilePath> {
    return this.http.get<FilePath>(environment.serverUrl + '/master-file/site/' + data.siteID + '/file/' + data.fileID);
  }

  editFile(data, siteID) {
    return this.http.put(environment.serverUrl + '/master-file/site/' + siteID + '/edit', data);
  }

  linkStudies(data) {
    return this.http.post(environment.serverUrl + '/master-file/site/' + data.siteID + '/' + data.fileID + '/link', data.linkRequests);
  }

  unlink(data) {
    return this.http.post(environment.serverUrl + '/master-file/site/' + data.siteID + '/study/' + data.studyID + '/' + data.fileID + '/unlink', null);
  }

  linkedStudies(data) {
    return this.http.get(environment.serverUrl + '/master-file/site/' + data.siteID + '/' + data.fileID + '/linked');
  }

  updateFile(data) {
    const formData: FormData = new FormData();
    formData.append('physicalFile', data.file);
    return this.http.post(environment.serverUrl + '/master-file/site/' + data.id + '/update/new-version', formData, { params: { newVersionFilePath: JSON.stringify(data.filePath) } });
  }
}
