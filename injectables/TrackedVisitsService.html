import { HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class TrackedVisitsService extends RootHttpService {

  getStatuses(data) {
    const params = new HttpParams().set('studyId', data?.studyId ).set('procedureTypeFilter', data?.procedureTypeFilter).set('dateFrom',data?.dateFrom).set('dateTo', data?.dateTo).set('patientStudyLinkIds', data?.patientStudyLinkIds)
    return this.http.get<any[]>(environment.serverUrl + `/finance/tracked-visits/statuses`,{params} )
  }

  getTrackedProcedures(data) {
    const params = new HttpParams().set('studyId', data?.studyId ).set('procedureTypeFilter', data?.procedureTypeFilter).set('dateFrom',data?.dateFrom).set('dateTo', data?.dateTo).set('patientStudyLinkIds', data?.patientStudyLinkIds)
    return this.http.get<any>(environment.serverUrl + `/finance/tracked-procedures`,{params} )
   
  }
  

  getVisitsForProjectionsReport(data){
    const params = new HttpParams().set('dateFrom',data?.dateFrom).set('dateTo', data?.dateTo)
    return this.http.get<any>(environment.serverUrl + `/ctms/projected-visits/sites/${data?.siteId}/studies`,{params} )
  }

  getTrackedProceduresForHeader(data) {
    return this.http.get<any[]>(environment.serverUrl + `/finance/tracked-procedures/${data}`)
  }

  getTrackedProceduresCSV(data) {
    const params = new HttpParams().set('studyId', data?.studyId ).set('procedureTypeFilter', data?.procedureTypeFilter).set('dateFrom',data?.dateFrom).set('dateTo', data?.dateTo).set('patientStudyLinkIds', data?.patientStudyLinkIds)
    return this.http.get(environment.serverUrl + `/finance/tracked-procedures/csv`,{params, responseType: 'arraybuffer'})
  }

  getTrackedVisits(data) {
    const params = new HttpParams().set('studyId', data?.studyId ).set('procedureTypeFilter', data?.procedureTypeFilter).set('dateFrom',data?.dateFrom).set('dateTo', data?.dateTo).set('patientStudyLinkIds', data?.patientStudyLinkIds)
    return this.http.get<any[]>(environment.serverUrl + `/finance/tracked-visits`,{params})
  }

  getTrackedVisitsStudies(data) {
    const params = new HttpParams().set('studyName', data?.studyName ).set('skip', data?.skip).set('limit', data?.limit).set('dateFrom',data?.dateFrom).set('dateTo', data?.dateTo)
    return this.http.get<any[]>(environment.serverUrl + `/last-studies-activity/${data.siteId}/paginated`,{params})
  }

  getVisitsByStudyId(data) {
    return this.http.get<any[]>(environment.serverUrl + `/finance/studies/visits/studies/${data}`  )
  }


  getPatients(data) {
    return this.http.get<any[]>(environment.serverUrl + `/ctms/patient/tracking/studies/${data}/patients`  )
  }

  exportcsv(data) {
    const params = new HttpParams().set('studyId', data?.studyId ).set('procedureTypeFilter', data?.procedureTypeFilter).set('dateFrom',data?.dateFrom).set('dateTo', data?.dateTo).set('patientStudyLinkIds', data?.patientStudyLinkIds)
    return this.http.get(environment.serverUrl + `/finance/tracked-visits/csv`, {params, responseType: 'arraybuffer'} )
  }

  getNotes(data) {
    const params = new HttpParams().set('includeProjectNotes', data.includepm)
    return this.http.get(environment.serverUrl + `/finance/tracked-visits/study/notes/${data.id}`,{params})
  }

  sendNote(data) {
    return this.http.post(environment.serverUrl + `/finance/tracked-visits/study/notes/${data.studyID}`,data.req)
  }

  getVisitProcedures(data) {
    const params = new HttpParams().set('procedureTypeFilter', data?.procedureTypeFilter )
    return this.http.get(environment.serverUrl + `/finance/tracked-visits/${data.visitId}`, {params})

  }

  deleteVisitProcedure(data) {
    return this.http.delete(environment.serverUrl + `/finance/tracked-visits/${data.trackedVisitId}/${data.trackedProcedureId}`)

  }

  updateVisitProcedure(data) {
    return this.http.put(environment.serverUrl + `/finance/tracked-visits/${data.visitId}`,data?.req )

  }

  deleteVisit(data) {
    return this.http.delete(environment.serverUrl + `/finance/tracked-visits/schedule-visits/${data?.trackedVisitId}`)

    
  }
}
