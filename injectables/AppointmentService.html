import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class AppointmentService extends RootHttpService{

  getDiscrepenciesNumber() {
    return this.http.get<any>(environment.serverUrl + '/ad/budget/discrepancies/count')
  }
  //prescreen 

  getAppointmentDetails(data) {
    return this.http.get<any>(environment.serverUrl + `/ctms/ps-appointments/${data}/schedule`);
  }

  getAppointmentStudies(data){   
    return this.http.get(environment.serverUrl + `/study-start-site/cmts/daily-schedule/${data}`);
  }

  completeScheduled(data) {
    return this.http.post(environment.serverUrl + `/ctms/screen/appointments/prescreen-appointments/${data.appointmentId}`, data.req);
  }
  
  completeRecommend(data) {
    return this.http.put(environment.serverUrl + `/ctms/ps-appointments/${data.appointmentId}/recommend-provider`, data.req);
  }

  completeNotScheduled(data) {
    return this.http.put(environment.serverUrl + `/ctms/ps-appointments/${data.appointmentId}/screen-no-scheduled`, data.req);
  }
  //screen 

  getScreenAppointmentDetails(data) {
    return this.http.get<any>(environment.serverUrl + `/ctms/screen/appointments/${data}/schedule`);
  }

  //visits
  getVisitAppointmentDetails(data) {
    return this.http.get<any>(environment.serverUrl + `/ctms/visits/schedule/${data}/schedule`);
  }

  markArrived(data) {
    return this.http.put(environment.serverUrl + `/ctms/visits/schedule/${data}/arrived`, {});

  }
  markNoShow(data) {
    return this.http.put(environment.serverUrl + `/ctms/visits/schedule/${data}/no-show`, {});
  }

  markNotArrived(data) {
    return this.http.put(environment.serverUrl + `/ctms/visits/schedule/${data}/no-arrived`, {});
  }
  
  markCancel(data) {
    return this.http.put(environment.serverUrl + `/ctms/visits/schedule/${data}/cancel`, {});
  }
  
  markAsShow(data) {
    return this.http.put(environment.serverUrl + `/ctms/visits/schedule/${data}/show`, {});
  }
  
  changeStatus(data) {
    const params = new HttpParams().set('status', data.status);
    return this.http.put(environment.serverUrl + `/ctms/visits/schedule/${data.appointmentId}/status`,params);
  }

  checkOverlap(data): Observable<any[]> {
    return this.http.post<any[]>(environment.serverUrl + `/ctms/appointments/overlapping`, data);
  }

  checkBusinessHours(data){
    const params = new HttpParams().set('date', data.date).set('duration', data.duration)
    return this.http.get(environment.serverUrl + `/commons/business-hours/${data?.siteId}/check`, {params});

  }

  rescheduleVisit(data) {
    return this.http.put(environment.serverUrl + `/ctms/visits/schedule/${data.appointmentId}/reschedule`, data.req);
  }

  getReceptionDeskLists(data) {
    const params = new HttpParams().set('showVisits', data?.showVisits == undefined ? true: data?.showVisits).set('showPrescreens',data?.showPrescreens == undefined ? true: data?.showPrescreens).set('showScreens',data?.showScreens== undefined ? true: data?.showScreens);
    return this.http.get<any>(environment.serverUrl + `/ctms/reception_desk/${data.id}`,{params});

  }
  checkPrescreenOverlap(data) {
    const params = new HttpParams().set('prescreenType', data.type).set('dateTime',data.time);
    return this.http.get(environment.serverUrl + `/ctms/ps-appointments/overlapping/${data.siteId}`,{params,responseType:'text'});
  }

  getCompletedPrescreenFormDetails(data) {
    return this.http.get(environment.serverUrl + `/ctms/ps-appointments/${data}/completed-details`,);
  }

  updateCompletedPrescreenFormDetails(data) {
    return this.http.put(environment.serverUrl + `/ctms/ps-appointments/${data.appoinidtmentId}/completed-details`,data.req);
  }

  getDenovoStudies(data){
    const params = new HttpParams().set('statuses',data.statuses).set('limit',data.pageSize).set('skip',(data.pageNumber*data.pageSize)).set('studyName',data.search);
    return this.http.get<any>(environment.serverUrl + `/commons/studies/include-de-novo/${data.siteId}`,{params});
  }

  getVisitsForTrackedScreens(data){
    return this.http.get<any[]>(environment.serverUrl + `/finance/studies/visits/entry-points/${data}`);
  }
}
