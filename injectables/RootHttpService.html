import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { MatDialog } from '@angular/material/dialog';
import { MatSnackBar } from '@angular/material/snack-bar';
import { SuccessMessageComponent } from '../messages/success-message.component';
import { CopiedMessageComponent } from '../messages/copied-message.component';
import { FileResponseMessageComponent } from '../messages/file-response-message.component';
import { PlaceholderDragMessageComponent } from '../messages/placeholder-drag-message.component';
import { FolderDragMessageComponent } from '../messages/folder-drag-message.component';
import { NotAllowedMessageComponent } from '../messages/not-allowed-message.component';
import { InfoPopupComponent } from '../../components/common/info-popup.component';
import { DebugDialogComponent } from 'src/app/components/common/dialogs/utils/debug-dialog/debug-dialog.component';
import { PermanentErrorDialogComponent } from 'src/app/components/common/dialogs/utils/permanent-error-dialog/permanent-error-dialog.component';
import { environment } from 'src/environments/environment';
 
@Injectable({
    providedIn: 'root'
})
export class RootHttpService {

    constructor(public http: HttpClient,
        private dialog: MatDialog,
        private snackBar: MatSnackBar) {
        }

    public showError(errMessage, disableClose=0) {
        this.dialog.open(InfoPopupComponent, {
            data:{ errMessage,
                   disableClose
            }
        });
    }

    public showPermanentError(errMessage, disableClose=0) {
        this.dialog.open(PermanentErrorDialogComponent, {
            data:{ errMessage,
                   disableClose
            },
            disableClose:true,
        });
    }
    
    public showDebug(err, req) {
        this.dialog.open(DebugDialogComponent , {
            data: {
                err,
                req
            }
            
        });
    }
    showSuccess(){
        this.snackBar.openFromComponent(SuccessMessageComponent, {
            duration: 3000,
        });
    }

    showCopied() {
        this.snackBar.openFromComponent(CopiedMessageComponent, {
            duration: 3000,
        });
    }

    showFileResponse(filename) {
        this.snackBar.openFromComponent(FileResponseMessageComponent, {
            duration: 5000,
            data: {
                file: filename,
            }

        });
    }

    showNotAllowed() {
        this.snackBar.openFromComponent(NotAllowedMessageComponent, {
            duration: 2000,
            panelClass: ['danger-snackbar']

        });
    }

    showFolderDragError() {
        this.snackBar.openFromComponent(FolderDragMessageComponent, {
            duration: 2000,
            panelClass: ['danger-snackbar']

        });
    }

    showPlaceholderDragError() {
        this.snackBar.openFromComponent(PlaceholderDragMessageComponent, {
            duration: 2000,
            panelClass: ['danger-snackbar']

        });
    }

    runMigrations() {
        this.http.get(environment.serverUrl+'/ctms/miscellaneous/appointments/migration').subscribe(res => {})
    }
    runMigrationsClinicProductivityReport() {
        this.http.get(environment.serverUrl+'/reports/clinic-productivity-by-indication/update').subscribe(res => {})
    }
    runMigrationsAdCategories() {
        this.http.put(environment.serverUrl+'/ad/ad-categories/sites/v1/migration',{}).subscribe(res => {})
    }
    runMigrationsUserSites() {
        this.http.post(environment.serverUrl+'/commons/migration',{}).subscribe(res => {})
    }


}
