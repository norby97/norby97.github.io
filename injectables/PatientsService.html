import { formatDate } from '@angular/common';
import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { User } from 'src/app/models/administration/user/User';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class PatientsService extends RootHttpService {

  mailinglistPatient(data) {
    return this.http.put(environment.serverUrl + '/ctms/patients/' + data.patientId +'/mailing-list',  data.req);
  }

  getSMSLogReport(data) {
    const params = new HttpParams().set('studyId', data.studyId == undefined ? '': data.studyId);
    return this.http.get(environment.serverUrl + `/ctms/report/sms-log/${data.smsLogId}`, {params,responseType:'arraybuffer'});
  }

  mergePatients(data) {
    return this.http.post(environment.serverUrl + `/ctms/commons/patients/merge`, data);

  }

  allowSMS1Patient(data) {
    return this.http.put(environment.serverUrl + '/ctms/patients/' + data.patientId +'/allow-sms-one', data.req);
  }

  allowSMS2Patient(data) {
      return this.http.put(environment.serverUrl + '/ctms/patients/' + data.patientId +'/allow-sms-two', data.req);
  }

  coutesyClinicPatient(data) {
    return this.http.put(environment.serverUrl + '/ctms/patients/' + data.patientId +'/courtesy-clinic', data.req);
  }

  doNotContactPatient(data) {
    const params = new HttpParams().set('reasonName', data.reasonName);

    return this.http.put(environment.serverUrl + '/ctms/patients/' + data.patientId +'/do-not-contact', data.req,{params});
  }
  searchPatient(data){ 
    const params = new HttpParams().set('search', data.text);
    return this.http.get(environment.serverUrl + '/ctms/patients/search', { params });
  }

  getPatientEventDetails(data) {
    const params = new HttpParams().set('appointmentType', data.appointmentType);
    return this.http.get<any>(environment.serverUrl + `/ctms/appointment-details/${data.appointmentId}`, {params});
  }
  
  getIntakeDetails(data) {
    const params = new HttpParams().set('intakeDetailsType', data.appointmentType);
    return this.http.get<any>(environment.serverUrl + `/ctms/intake-details/${data.appointmentId}`, {params});
  }
  

  updatePatient(data) {
    return this.http.put(environment.serverUrl + '/ctms/patients/' +data.patientId, data.req);
  }

  findPatients(data) {
    const params = new HttpParams().set('firstName', data.firstName).set('lastName', data.lastName);
    return this.http.get(environment.serverUrl + '/ctms/patients', { params });
  }

  findPatientByID(data) {
    return this.http.get<any>(environment.serverUrl + '/ctms/patients/' + data);
  }

  addPatient(data) {
    return this.http.post(environment.serverUrl + '/ctms/patients', data);
  }

  addAnonymousCall(data){
    return this.http.post(environment.serverUrl + '/ctms/patients/anonymous', data);
  }


  
  getPatientTracks(data) {
    return this.http.get(environment.serverUrl + '/ctms/tracks/patients/'+data);

  }

  createTrackForPatient(data) {
    return this.http.post(environment.serverUrl + '/ctms/tracks/patients/'+data.patientId, data.req);
  }

  updateTrackForPatient(data) {
    return this.http.put(environment.serverUrl + '/ctms/tracks/'+data.trackId, data.req);
  }
  schedulePatient(data) {
    return this.http.post(environment.serverUrl + '/ctms/patients/' +data.patientId, data.req);
  }

  getResources(data):Observable<any> {
    const params = new HttpParams().set('view', data.view).set('showPrescreens',data?.req?.showPrescreens).set('showResources',data?.req?.showResources).set('showVisits',data?.req?.showVisits).set('showScreens',data?.req?.showScreens).set('showPhoneCalls',data?.req?.showPhoneCalls).set('showOffsiteVisits',data?.req?.showOffsiteVisits).set('showMiscellaneous',data?.req?.showMiscellaneous);
    return this.http.get(environment.serverUrl + '/ctms/daily_schedule/resources/'+data.req.id, {params});
  }
  getEvents(data):Observable<any> {
    const params = new HttpParams().set('date', formatDate(data.req.today, 'yyyy-MM-dd', 'en-Us')).set('view', data.view).set('showPrescreens',data?.req?.showPrescreens).set('showResources',data?.req?.showResources).set('showVisits',data?.req?.showVisits).set('showScreens',data?.req?.showScreens).set('showPhoneCalls',data?.req?.showPhoneCalls).set('showOffsiteVisits',data?.req?.showOffsiteVisits).set('showMiscellaneous',data?.req?.showMiscellaneous)
    return this.http.get(environment.serverUrl + '/ctms/daily_schedule/events/' + data.req.id, {params});
  }

  markArrived(data) {
    return this.http.put(environment.serverUrl + '/ctms/ps-appointments/' + data + '/arrived', {});

  }
  markNoShow(data) {
    return this.http.put(environment.serverUrl + '/ctms/ps-appointments/' + data + '/no-show' ,{});
  }

  
  markNotArrived(data) {
    return this.http.put(environment.serverUrl + '/ctms/ps-appointments/' + data + '/no-arrived' ,{});
  }
  
  markCancel(data) {
    return this.http.put(environment.serverUrl + '/ctms/ps-appointments/' + data.appointmentId + '/cancel' ,data.req);
  }
  
  markAsShow(data) {
    return this.http.put(environment.serverUrl + '/ctms/ps-appointments/' + data + '/show' ,{});
  }

  markScreenNoScheduled(data) {
    return this.http.put(environment.serverUrl + '/ctms/ps-appointments/' + data + '/screen-no-scheduled',{});
  
  }

  markUndoCompleted(data) {
    return this.http.put(environment.serverUrl + '/ctms/ps-appointments/' + data + '/undo-completed',{});
  }

  markReschedule(data) {
    return this.http.put(environment.serverUrl + '/ctms/ps-appointments/' + data.appointmentId + '/reschedule' ,data.req);
  
  }

  markRecommendProvider(data) {
    return this.http.put(environment.serverUrl + '/ctms/ps-appointments/' + data + '/recommend-provider' ,{});
  }

  changeStatus(data) {
    const params = new HttpParams().set('status', data.status);
    return this.http.put(environment.serverUrl + `/ctms/ps-appointments/${data.appointmentId}/status`,params);


  }

  //patient reason categories
  getReasonCategories() {
    return this.http.get<any[]>(environment.serverUrl + '/ctms/reason-categories');
  }

  getActiveReasonCategoriesByApptType(data){
    const params = new HttpParams().set('type', data);

    return this.http.get<any[]>(environment.serverUrl + '/ctms/reason-categories/types/active',{params});

    

  }

  disableOrEnableReasonCategory(data) {
    return this.http.put(environment.serverUrl + '/ctms/reason-categories/active-or-inactive/' + data, {});
  }

  updateReasonCategories(data) {
    return this.http.get(environment.serverUrl + '/ctms/reason-categories/' + data.id, data);
  }
  createReasonCategories(data) {
    return this.http.post(environment.serverUrl + '/ctms/reason-categories' ,data);
  }

  deletePatientTrack(data) {
    return this.http.delete(environment.serverUrl + '/ctms/tracks/' + data);
  }

  //patient interest
  deleteInterest(data) {
    return this.http.delete(environment.serverUrl + '/ctms/interests/' + data);
  }


  findInterests(data): Observable<any[]> {
    return this.http.get<any[]>(environment.serverUrl + '/ctms/patients/' +data + '/interests');
  }

  createInterest(data) {
    return this.http.post(environment.serverUrl + '/ctms/patients/'+ data.patientId + '/interests/' + data.interestId, {});
  }
  

  getAdsReferrals(data) {
    return this.http.get(environment.serverUrl + '/ad/calendars/' + data+ '/two-weeks-before' , {});
  }
  
  getAdsReferralsToday(data) {
    const params = new HttpParams().set('date',data.date)
    return this.http.get(environment.serverUrl + '/ad/calendars/' + data.id+ '/two-weeks-before' , {params});
  }
