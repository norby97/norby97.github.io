import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { InTheNews } from 'src/app/models/crmbd/in-the-news.model';
import { StudyStartSiteView } from 'src/app/models/crmbd/study-start-site-view.model';
import { environment } from 'src/environments/environment';
import { DashboardChart } from '../../models/crmbd/dashboard-chart.model';
import { Lead } from '../../models/crmbd/lead.model';
import { StudyStart } from '../../models/crmbd/study-start';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class StudyStartsService  extends RootHttpService   {

  getNotes(data): Observable<StudyStart>{
    const params = new HttpParams().set('includeProjectNotes', data.includepm);
    return this.http.get<StudyStart>(environment.serverUrl + '/study-start-site/notes/' + data.siteID + '/' + data.studyID, {params});
  }
  getNotes2(data): Observable<any[]>{
    const params = new HttpParams().set('includeProjectNotes', data.includepm);
    return this.http.get<any[]>(environment.serverUrl + '/study-start-site/notes/'  + data.studyID, {params});
  }

  getArticles(data) :Observable<InTheNews[]>{
    return this.http.get<InTheNews[]>(environment.serverUrl + '/study-start-site/in-the-news/' + data.studyStartID);
  }

  getSiteStudies(data): Observable<StudyStart[]> {
    return this.http.get<StudyStart[]>(environment.serverUrl + '/study-start-site/' + data.siteID );
  }

  getStudiesNotInSite(data) {
    return this.http.get(environment.serverUrl + '/study-start-site/sites-are-not-in-study/per-site/' + data.studyStartID);
  }
getStudyDetails(data){
  return this.http.get<any>(environment.serverUrl + `/study-start-site/by-site/${data.siteId}/by-study/${data.studyStartId}` );
}

getStudyDetails2(data){
  return this.http.get<any>(environment.serverUrl + `/study-start-site/by-study-start-site-id/${data.studyStartId}` );

}

getSitesInStudy(data){
  return this.http.get<any[]>(environment.serverUrl + `/study-start-site/sites-are-in-study/${data}` );
}

  addNewTrial() {
    return this.http.post(environment.serverUrl + '', {});
  }

  addArticle(data):Observable<StudyStart> {
    return this.http.post<StudyStart>(environment.serverUrl + '/study-start-site/in-the-news/' + data.studyStartID, {url: data.url  });
  }

  getContacts(data):Observable<StudyStart[]> {
    return this.http.get<StudyStart[]>(environment.serverUrl + '/study-start-site/contact/' + data.studyStartID);
  }

  sendNote(data) {
    return this.http.post(environment.serverUrl + '/study-start-site/note/' + data.studyStartID,  data.request);
  }
  
  addContact(data) {
    return this.http.post(environment.serverUrl + '/study-start-site/contact/'+ data.studyStartID + '/' + data.contactID, data.request);
  }

  addScoreCardContact(data) {
    return this.http.post(environment.serverUrl + '/study-start-site/contact/' + data.studyStartID + '/' + data.contactID, data.request);

  }
  
  filter(data): Observable<any>{
    const params = new HttpParams().set('search',data.search).set('skip', data?.skip).set('limit', data?.limit).set('active',data?.active === undefined ? 'true' : data?.active);
    return this.http.post<any>(environment.serverUrl + '/study-start-site/filter/', data?.request);
  }
 

  getNCT(data) {
    return this.http.get(environment.serverUrl + '/study-start-site/ntc/'+ data.studyStartID);
  }

  editStudy(data) {
    return this.http.put(environment.serverUrl + '/study-start-site/'+ data.studyStartID, data.request);
  }

  deleteStudy(data) {
    return this.http.delete(environment.serverUrl + '/study-start-site/' + data.studyStartID);
  }

  changeNTC(data) {
    return this.http.put(environment.serverUrl + '/study-start-site/ntc/' +  data.studyStartID, data.request);
  }

  getDashboard(data): Observable<DashboardChart> {
    return this.http.post<DashboardChart>(environment.serverUrl + '/study-start/dashboard', data);
  }

  changeCommitment(data) {
    return this.http.put(environment.serverUrl + '/study-start-site/commitment/'+ data.studyStartID, data.request);
  }
  getCommitment(data) {
    return this.http.get(environment.serverUrl + '/study-start-site/commitment/'+ data.studyStartID);
  }
  changeStatus(data) {
    const params = new HttpParams().set('status', data.status);
    return this.http.put(environment.serverUrl + '/study-start-site/action-status/' + data.studyStartID, params);
  }

  copyToAnotherSite(data) {
    return this.http.put(environment.serverUrl + '/study-start-site/copy-to-another-site/'+ data.studyStartID, data.request);
  }

  getAll() {
    return this.http.get<StudyStartSiteView[]>(environment.serverUrl + '/study-start-site/all');    
  }

  getAllForLinking(data) {
    return this.http.get<StudyStartSiteView[]>(environment.serverUrl + `/study-start-site/link-study/studies/${data}`);    
  }
  linkStudy(data) {
    return this.http.put(environment.serverUrl + '/study-start-site/link-study/' + data.studyStartID + '/'+ data.studyStartSiteID, {});
  }
  
  getFilter(data) {
    return this.http.post(environment.serverUrl + '/study-start-site/filter/report', data);
  }

  removeContactFromStudy(data) {
    return this.http.delete(environment.serverUrl + '/study-start-site/contact/'  + data.studyStartID + '/' + data.contactID);
  }

  removeNoteFromStudy(data) {
    return this.http.delete(environment.serverUrl + '/study-start-site/note/' +   data.studyStartID + '/' + data.noteID);

  }

  removeArticleFromStudy(data) {
    return this.http.delete(environment.serverUrl + '/study-start-site/in-the-news/' + data.studyStartID + '/' + data.articleID);
  }

  getStudyStarts() {
    return this.http.get<any[]>(environment.serverUrl + '/study-start');    
  }

  getPaymentTypesForStudyStarts(data) {
    return this.http.get<any[]>(environment.serverUrl + `/study-start/${data}/sites`);    
  }

  getPatientsByStudyStartsAndPaymentType(data) {
    return this.http.get<any[]>(environment.serverUrl + `/ctms/patient/tracking/${data.studyStartId}/${data.siteId}/patients`);    
  }
    
  getInvoicesForPatient(data){
    const params = new HttpParams().set('isEdit', 'false');
    return this.http.get<any[]>(environment.serverUrl + `/ctms/tracked-visit/patients/${data}`, {params});    
  }

  getInvoicesForPatientEdit(data){
    const params = new HttpParams().set('isEdit', 'true');
    return this.http.get(environment.serverUrl + `/ctms/tracked-visit/patients/${data}`, {params});    
  }

  getMiscInvoicesForPatient(data) {
    return this.http.get<any[]>(environment.serverUrl + `/finance/miscellaneous/invoices/record-payment/${data.studyStartId}/${data.siteId}`);    

  }
  getAdvInvoicesForPatient(data) {
    return this.http.get<any[]>(environment.serverUrl + `/finance/out/record-payment/${data.studyStartId}/${data.siteId}`);    

  }

  //new era

  getStudyNotes(data) {
    const params = new HttpParams().set('includeProjectNotes', 'true');
    return this.http.get<any[]>(environment.serverUrl + `/study-start-site/notes/${data.siteId}/${data.studyStartId}`, {params});    

  }

  getStudyNotes2(data) {
    const params = new HttpParams().set('includeProjectNotes', 'true');
    return this.http.get<any[]>(environment.serverUrl + `/study-start-site/notes/${data.studyStartId}`, {params});    
  }


  getQueryStudiesCSV(data){
    return this.http.post(environment.serverUrl + `/reports/study-start-sites/filter/csv`,data, {responseType:'arraybuffer'});    
  }


  getStudyStartFiles(data){
    return this.http.get(environment.serverUrl + `/crm/study-start-site/files/${data}`);
  }

  getStudyStartFile(data){
    return this.http.get(environment.serverUrl + `/crm/study-start-site/files/${data?.studyId}/${data?.fileId}`,  { responseType: 'arraybuffer' });

  }
deleteStudyStartFile(data){
    return this.http.delete(environment.serverUrl + `/crm/study-start-site/files/${data?.studyId}/${data?.fileId}`);

  }

  addFileToStudyStart(data){
    const formData: FormData = new FormData();
    formData.append('file', data.file);
    formData.append('type', data.type);
    return this.http.post(environment.serverUrl + `/crm/study-start-site/files/${data?.studyId}` ,formData);

  }


}
