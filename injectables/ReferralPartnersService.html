import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { ContactStudyRole } from 'src/app/models/crmbd/contact-study-role.model';
import { ContactModel } from 'src/app/models/crmbd/contact.model';
import { environment } from 'src/environments/environment';

@Injectable({
  providedIn: 'root'
})
export class ReferralPartnersService {

  constructor(private http: HttpClient) { }
  
  
  findPartner(data) {
    return this.http.get<any>(environment.serverUrl + '/contact/referral-partners/search', {params: data});
  }

  addPartner(data): Observable<ContactModel>  {
    return this.http.post<ContactModel>(environment.serverUrl + '/contact/referral-partners', data);
  }

  getAllByFirstLetter(data) {
    return this.http.get<any>(environment.serverUrl + '/contact/referral-partners', {params: data});
  }

  getPartnerDetails(data) {
    return this.http.get<ContactModel>(environment.serverUrl + '/contact/referral-partners/' + data.contactID);
  }

  createContactStudy(data) {
    return this.http.post(environment.serverUrl + '/study-start-site/contact/referral-partners/' + data.siteId + '/' + data.studyStartId + '/' + data.contactId, {name:data.role});
  }

  deleteContactStudy(data) {
    return this.http.delete(environment.serverUrl + '/contact/study/' + data.contactStudyID);
  }
  
  updateContactDetails(data) {
    return this.http.put(environment.serverUrl + '/contact/referral-partners/' + data.contactID, data.request);
  }

  updateContactOtherDetails(data) {
    return this.http.put(environment.serverUrl + '/contact/referral-partners/other-details/' + data.contactID, data.request);
  }

  updateLinkedInURL(data) {
    return this.http.put(environment.serverUrl + '/contact/referral-partners/linkedin/' + data.contactID , data.request);
  }

  getAllContactNotes(data){
    return this.http.get(environment.serverUrl + '/contact/note/' + data.contactID );
  }

  createContactNote(data) {
    return this.http.post(environment.serverUrl + '/contact/note/', data);
  }
  
  deleteContactNote(data) {
    return this.http.delete(environment.serverUrl + '/contact/note/' + data.contactID + '/' + data.contactNoteID );
  }

  retrieveContactFiles(data) {
    return this.http.get(environment.serverUrl + '/contact/file/' + data);
  }

  uploadContactFile(data) {
    const params = new HttpParams().set('type', data.type);
    const formData: FormData = new FormData();
    formData.append('file', data.file);
    
    return this.http.post(environment.serverUrl + '/contact/file/' + data.contactID, formData, {params});
  }
  
  deleteContactFile(data) {
    return this.http.delete(environment.serverUrl + '/contact/file/' + data.contactID + '/' + data.fileID);
  }

  downloadContactFile(data) {
    return this.http.get(environment.serverUrl + '/contact/file/download/' + data.fileID, {responseType: 'arraybuffer'});
  }

  getLinkedStudyRoles(data): Observable<ContactStudyRole> {
    return this.http.get<ContactStudyRole>(environment.serverUrl + '/contact/study-role/' + data.contactID)
   }

  linkContactToStudy(data) {
  return this.http.get<ContactStudyRole>(environment.serverUrl + '/contact/study-role/' + data.contactID)

  }
  
}
