import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class PermissionsService extends RootHttpService {

  getDefault() {
    return this.http.get(environment.serverUrl + '/permissions/modules' );
  }

  getAllTemplates() {
    return this.http.get<any[]>(environment.serverUrl + '/permissions/template' );
  }

  createTemplate(data) {
    return this.http.post(environment.serverUrl + '/permissions/template', data );
  }

  updateTemplate(data) {
    return this.http.put(environment.serverUrl + '/permissions/template/' + data.id, data.request );
  }

  getTemplateByID(data:any) :any {
    return this.http.get<any>(environment.serverUrl + '/permissions/template/' + data );
  }

  getUserTemplate(data:any) :any {
    return this.http.get(environment.serverUrl + '/permissions/user/'  + data );
  }

  tooglePermissionStatus(data) {
    return this.http.put(environment.serverUrl + '/permissions/template/disable-or-enable/' + data, {});
  }

  editUserTemplate(data: any) {
    return this.http.put(environment.serverUrl + '/permissions/user/'  + data.id, data.request );
  }



}
