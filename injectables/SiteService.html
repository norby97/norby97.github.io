import {Injectable} from '@angular/core';
import {environment} from 'src/environments/environment';
import {Site} from '../../models/common/Site';
import {Observable} from 'rxjs';
import {RootHttpService} from '../utils/root.http.service';
import { SiteUserView } from 'src/app/models/administration/user/site-user-view.model';
import { HttpParams } from '@angular/common/http';

@Injectable({
    providedIn: 'root'
})
export class SiteService extends RootHttpService {

    getAll(companyId): Observable<Array<Site>> {
        return this.http.get<Array<Site>>(environment.serverUrl + '/site/all', {params: {companyId}});
    }

    getAll2(companyId?): Observable<Array<Site>> {
        return this.http.get<Array<Site>>(environment.serverUrl + '/v2/sites');
    }

    getPatientTrackingStudies(data) {
        return this.http.get<any>(environment.serverUrl + `/study-start-site/${data}/finance`);
    }

    getStudies(data) {
        const params = new HttpParams().set('studyName',data?.search).set('page', data?.pageNumber).set('size', data?.pageSize).set('statuses', data.statuses);
        return this.http.get<any>(environment.serverUrl + `/commons/studies/${data.siteId}`, {params});
    }
   
    getAllWithoutPermissions(companyId): Observable<Array<Site>> {
        return this.http.get<Array<Site>>(environment.serverUrl + '/site/crm-db/all', {params: {companyId}});
    }

    save(site: Site): Observable<Site> {
        return this.http.put<Site>(environment.serverUrl + '/site', site);
    }

    delete(siteId: string): Observable<Site> {
        return this.http.delete<Site>(environment.serverUrl + '/site', {params: {siteId}});
    }

    assignRegulatory(data) {
        return this.http.put(environment.serverUrl + '/site/' + data.siteId + '/assign-regulatory/' + data.userId, {} );
    }

    getUserSites(data): Observable<SiteUserView[]> {
        return this.http.get<SiteUserView[]>(environment.serverUrl + '/site/regulatories/' + data);
    }

    //new-era
    getUserSites2(data): Observable<any[]> {
        return this.http.get<any[]>(environment.serverUrl + `/commons/users/${data}/sites`);
    }

    getCompanySites(data){
        return this.http.get<any[]>(environment.serverUrl + `/v2/sites/companies/${data}`);
    }

    getSiteDetails(data){
        return this.http.get(environment.serverUrl + `/v2/sites/${data}`);
    }

    createCompanySite(data){
        return this.http.post(environment.serverUrl + `/v2/sites/companies/${data.siteId}`,data.req);

    }

    updateSiteDetails(data){
        return this.http.put(environment.serverUrl + `/v2/sites/${data.siteId}`,data.req);
    }

    changeSiteStatus(data) {
        return this.http.put(environment.serverUrl + `/v2/sites/${data}/active`,{});

    }
    getSiteBusinessHours(data) {
        return this.http.get(environment.serverUrl + `/commons/business-hours/${data}`);
    } 
    
    editSiteBusinessHours(data) {
        return this.http.post(environment.serverUrl + `/commons/business-hours/${data?.siteId}`,data.req);
    }

    getSiteHolidays(data) {
        return this.http.get<any[]>(environment.serverUrl + `/commons/holidays/${data}`);
    }

    addSiteHoliday(data) {
        return this.http.post(environment.serverUrl + `/commons/holidays/${data?.siteId}`,data.req);

    }

    editSiteHoliday(data) {
        return this.http.put(environment.serverUrl + `/commons/holidays/${data?.siteId}`,data.req);

    }
}
