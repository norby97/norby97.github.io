import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { StudyStart } from 'src/app/models/crmbd/study-start';
import { RegManagerResponse } from 'src/app/models/regulatory/reg-manager-response.model';
import { RegManagerStudy } from 'src/app/models/regulatory/reg-manager-study.model';
import { Study } from 'src/app/models/common/Study';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class RegManagerService extends RootHttpService {

  getTrials(data): Observable<any[]> {
    const params = new HttpParams().set('skip', data?.skip).set('limit', data?.limit).set('siteId', data.siteID).set('statuses',data.statuses)
    return this.http.get<any[]>(environment.serverUrl + '/reg-manager', {params});
  }

  addTrialToRM(data) {
    return this.http.post(environment.serverUrl + '/reg-manager/' + data.studyID, {});
  }

  getTrialDetails(data): Observable<RegManagerStudy> {
    return this.http.get<RegManagerStudy>(environment.serverUrl + '/reg-manager/' + data);
  }

  editTrial(data) {
    return this.http.put(environment.serverUrl + '/reg-manager/information/' + data.regID, data.request);
  }

  addTrialNote(data) {
    return this.http.post(environment.serverUrl + '/reg-manager/note/' + data.regID, data.request);
  }
  
  getRMStudySites(data): Observable<StudyStart[]> {
    return this.http.get<StudyStart[]>(environment.serverUrl + '/reg-manager/sites/' + data);
  }

  getTrialNotes(data) {
    const params = new HttpParams().set('includeProjectNotes', data?.includepm)
    return this.http.get<any[]>(environment.serverUrl + '/reg-manager/note/' + data.request, {params});
  }

  filterTrials(data) {
    return this.http.post(environment.serverUrl + '/lead', data);
  }

  getTrialsNotInRM(data) {
    const params = new HttpParams().set('siteId', data)
    return this.http.get(environment.serverUrl + '/reg-manager/study-start/sites', {params});
  }

  getStartupReport(data): Observable<{milestone: any[], notes: any[]}> {
    return this.http.get<{milestone: any[], notes: any[]}>(environment.serverUrl + '/reg-manager/startup-report/' + data);
  }

  removeTrialFromRM(data) {
    return this.http.delete(environment.serverUrl + '/reg-manager/' + data);
  }

  //archives

  getArchivesBySite(data) {
    return this.http.get(environment.serverUrl + `/regulatory/archive-manager/${data}`);

  } 
  
  getBarcodeData(data) {
    const params = new HttpParams().set('barcode', data)
    return this.http.get(environment.serverUrl + `/regulatory/archive-records/barcode/extract-values`, {params});

  }

  createBox(data) {
    return this.http.post(environment.serverUrl + '/regulatory/archive-boxes',data);

  }

  getBoxDetails(data){
    const params = new HttpParams().set('archiveBoxId', data.boxId).set('studyId', data?.studyId).set('types', data?.types)
    return this.http.get(environment.serverUrl + '/regulatory/archive-records',{params});

    
  }

  getBoxContents(data){
    const params = new HttpParams().set('archiveBoxName', data)
    return this.http.get<any>(environment.serverUrl + '/regulatory/archive-boxes',{params});
  }

  getStudyNotes(data){
    return this.http.get(environment.serverUrl + `/regulatory/archive-manager/notes/studies/${data}`);
  } 
  
  createStudyNote(data){
    return this.http.post(environment.serverUrl + `/regulatory/archive-manager/notes/studies/${data.studyId}`,data.req);
  }

  updateStudyArchive(data){
    return this.http.put(environment.serverUrl + `/regulatory/archive-manager/${data.studyId}`,data.req);

  }
  updateStorageLocation(data){
    return this.http.put(environment.serverUrl + `/regulatory/archive-boxes`,data);

    
  }

  getBoxesByStudy(data){
    return this.http.get(environment.serverUrl + `/regulatory/archive-boxes/${data}`);

  }

  checkIfBoxExists(data){
    const params = new HttpParams().set('archiveBoxName', data)
    return this.http.get(environment.serverUrl + `/regulatory/archive-boxes/check-if-exists`,{params});

    
  }
}
