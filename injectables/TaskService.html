import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { environment } from 'src/environments/environment';
import { GeneralResponse } from '../../models/common/GeneralResponse';
import { OverViewDetailsModel } from '../../models/regulatory/overview-details.model';
import { TaskModel } from '../../models/regulatory/task.model';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class TaskService {

  constructor(private http: HttpClient) { }

  retrieveTasks(data): Observable<TaskModel> {
    const params = new HttpParams().set('pageNr', '0').set('pageSize', '10').set('fileId', data.fileID).set('types', data.type).set('statuses', data.status).set('userId', data.userID);
    return this.http.get<TaskModel>(environment.serverUrl + '/task', { params });
  }

  createTask(data) {
    return this.http.post(environment.serverUrl + '/task', data);
  }

  updateTask(data) {
    return this.http.put(environment.serverUrl + '/task/' + data.taskID, {});
  }

  getTaskNumber() {
    return this.http.get<any>(environment.serverUrl + '/task/count');
  }

  retrieveTaskMessages(data): Observable<TaskModel> {
    const params = new HttpParams().set('pageNr', '0').set('pageSize', '10');
    return this.http.get<TaskModel>(environment.serverUrl + '/task/message/' + data.taskID, { params });
  }

  sendTaskMessages(data): Observable<TaskModel> {
    const params = new HttpParams().set('message', data.message);
    return this.http.post<TaskModel>(environment.serverUrl + '/task/message/' + data.taskID, params);
  }

  assignUsers(data) {
    return this.http.post(environment.serverUrl + '/task/' + data.taskID, data.users);
  }

  getOverviewDetails(): Observable<OverViewDetailsModel> {
    return this.http.get<OverViewDetailsModel>(environment.serverUrl + '/task/overview');
  }

  getFiles(data) {
    const params = new HttpParams().set('esign', data.esign).set('overdue', data.overdue);
    return this.http.get(environment.serverUrl + '/task/all-files', { params });
  }

  esign(data): Observable<GeneralResponse> {
    return this.http.post(environment.serverUrl + '/task/sign', data);
  }
}
