import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class StudyBuildService  extends RootHttpService{

  //get
  getAvailableProcedureGroups(data) {
      return this.http.get<any[]>(environment.serverUrl + `/finance/procedure/conditions/procedures-who-are-available/${data}`);
  }

  getAvailableVisitRules(data) {
    return this.http.get<any[]>(environment.serverUrl + `/finance/visit/conditions/visits-who-are-available/${data}`);
  }

  getAvailableProcedure(data) {
    return this.http.get<any[]>(environment.serverUrl + `/finance/study-detail/${data}/visit-procedures-cost/procedures`);
  }

  addContractedProcedure(data) {
    return this.http.post(environment.serverUrl + `/finance/study-detail/${data?.studyDetailId}/visit-procedures-cost/${data?.procedureId}`,{});
  }

  getContractedProcedure(data) {
    return this.http.get<any>(environment.serverUrl + `/ctms/daily_schedule/providers/${data}`, data);
  }

  removeProvider(data){
    return this.http.delete(environment.serverUrl + `/finance/visit-procedures-cost-providers/${data.visitProcedureCostId}/${data.providerId}`, data);
  }

  copyStudyBuild(data) {
    return this.http.post(environment.serverUrl + `/finance/study-detail/copy/${data.studyDetailId}/studies/${data.studyId}`, {});
  }

  saveProviderProcedures(data) {
    return this.http.post(environment.serverUrl + `/finance/visit-procedures-cost-providers/${data.id}`,data.req);
  }

  getContractedProviderList(data){
    return this.http.get<[{
      id:any,
    procedureName:any,
    procedureId:any,
    visits:any[],
    visitProcedureCostProvider:[{visitProceduresCost: [{cost:any, visitProcedureId:any}]}],
    providers:[{ resource: any;id:any, name:any}],
    }]>(environment.serverUrl + `/finance/study-detail/${data}/visit-procedures-cost`);
  }

  getNotBuildedStudies() {
    return this.http.get(environment.serverUrl + '/finance/no-build-studies');
  }

  buildStudy(data: any) {
    return this.http.post(environment.serverUrl + '/finance/build-study', data);
  }

  getStudyNotes(data) {
    const params = new HttpParams().set('includeProjectNotes', data.includepm)
    return this.http.get(environment.serverUrl + `/finance/study-detail/notes/${data.id}`, {params});
  }

  createStudyNote(data) {
    return this.http.post(environment.serverUrl + `/finance/study-detail/notes/${data?.studyDetailId}`, data.req);
  }

  getAll(data?):Observable<any>{
    const params = new HttpParams().set('active', data.active).set('siteId', data.siteId).set('studyName', data.studyName).set('isBuilt', data.isBuilt).set('page', data?.pageNumber).set('size', data?.pageSize);;
    return this.http.get<any>(environment.serverUrl + '/finance/studies', {params});
  }

  getStudyPath(data) {
    return this.http.get<any>(environment.serverUrl + `/finance/studies/paths/${data.studyDetailId}/${data.budgetVersionId}`);
  }

  getStudyVisits(data) {
    return this.http.get<any>(environment.serverUrl + `/finance/studies/visits/${data.studyDetailId}/${data.budgetVersionId}`);
  }

  getStudyProcedures(data) {
    return this.http.get<any>(environment.serverUrl + `/finance/studies/procedures/${data.studyDetailId}/${data.budgetVersionId}`);
  }

  getStudyAdditionalProcedure(data) {
    return this.http.get<any>(environment.serverUrl + `/finance/studies/procedures/additional/${data.studyDetailId}/${data.budgetVersionId}`);
  }

  getStudyVisitPayment(data) {
    return this.http.get<any>(environment.serverUrl + `/finance/studies/procedures/default/${data.studyDetailId}/${data.budgetVersionId}`);
  }

  getStudyDetails(data) {
    return this.http.get<any>(environment.serverUrl + '/finance/study-detail/'+ data);
  }

  getStudyPathForEnroll(data) {
    return this.http.get<any>(environment.serverUrl + `/finance/studies/paths/studies/`+ data);

  }

  getStudyBillingSetup(data) {
    return this.http.get(environment.serverUrl + `/finance/study-detail/billing-setup/${data}`);
  }


  studyBillingSetup(data) {
    return this.http.post(environment.serverUrl + `/finance/study-detail/billing-setup/${data?.studyDetailId}`, data?.req);
  }

  getProcedureGroups(data) {
    return this.http.get(environment.serverUrl + `/finance/procedure/conditions/${data.studyDetailId}/${data.budgetVersionId}`);

  }

  getVisitConditions(data) {
    return this.http.get(environment.serverUrl + `/finance/visit/conditions/${data.studyDetailId}/${data.budgetVersionId}`);
  }

  getStudyBudgetDetails(data) {
    return this.http.get(environment.serverUrl + `/study-detail-budget-versions/study-detail/${data}`);
  }

  publishStudyVersion(data) {
    return this.http.put(environment.serverUrl + `/study-detail-budget-versions/${data.versionId}/publish`,data.req);
  }

  //edit

  deleteStudyVersion(data){
    return this.http.delete(environment.serverUrl + `/study-detail-budget-versions/${data}`);
  }

  addStudyVersion(data) {
    return this.http.post(environment.serverUrl + `/study-detail-budget-versions/study-detail/${data.studyDetailId}`,data.req);

  }
  makeReviewed(data) {
    return this.http.put<any>(environment.serverUrl + '/finance/study-detail/'+ data +'/reviewed', {});
    
  }
  makeUnReviewed(data) {
    return this.http.put<any>(environment.serverUrl + '/finance/study-detail/'+ data +'/unreviewed',{});
  }

  updateStudyDetails(data) {
    return this.http.put<any>(environment.serverUrl + '/finance/study-detail/' + data.studyID, data.request);
  }

  editStudyPath(data) {
    return this.http.put(environment.serverUrl + `/finance/studies/paths/${data.budgetVersionId}`,data.req);
  }

  editStudyVisits(data) {
    return this.http.put(environment.serverUrl + `/finance/studies/visits/${data.budgetVersionId}`,data.req);
  }

  editStudyProcedures(data) {
    return this.http.put(environment.serverUrl + `/finance/studies/procedures/${data.budgetVersionId}`,data.req);
  }

  editStudyAdditionalProcedures(data) {
    return this.http.post(environment.serverUrl + `/finance/studies/procedures/additional/${data.budgetVersionId}`,data.req);
  }

  editStudyProcedureGroups(data) {
    return this.http.post(environment.serverUrl + `/finance/procedure/conditions/${data.budgetVersionId}`,data.req);
  }

  editStudyVisitRule(data) {
    return this.http.post(environment.serverUrl + `/finance/visit/conditions/${data.budgetVersionId}`,data.req);
  }

  //delete
  deleteProcGroup(data) {
    return this.http.delete(environment.serverUrl + `/finance/procedure/conditions/${data}`);
  }

  deleteVisitRule(data) {
    return this.http.delete(environment.serverUrl + `/finance/visit/conditions/${data}`);
  }

  deletStudyVisit(data) {
    return this.http.delete(environment.serverUrl + `/finance/studies/visits/${data.budgetVersionId}/${data.visitId}`);
  }

  deletStudyProcedure(data) {
    return this.http.delete(environment.serverUrl + `/finance/studies/procedures/${data.budgetVersionId}/${data.procedureId}`);
  }

  deletStudyAdditionalProcedure(data) {
    return this.http.delete(environment.serverUrl + `/finance/studies/procedures/additional/${data.budgetVersionId}/${data.additionalProcedureId}`);
  }
 
  
  //remit locations
  findAllRemitLocations() {
    return this.http.get(environment.serverUrl + `/finance/remit/locations`);
  }

  updateRemitLocation(data) {
    return this.http.put(environment.serverUrl + `/finance/remit/locations/${data.locationId}`, data.req);
  }

  createRemitLocation(data) {
    return this.http.post(environment.serverUrl + `/finance/remit/locations`, data);
  }

  deleteRemitLocation(data) {
    return this.http.delete(environment.serverUrl + `/finance/remit/locations/${data}`);
  }

  getStudyDetail(data) {
    return this.http.get(environment.serverUrl +`/finance/study-detail/studies/${data}`)
    
  }
  getStudyDetailByStudyId(data) {
    return this.http.get(environment.serverUrl +`/finance/study-detail/studies/${data.studyStartId}/${data.siteId}`)
  }

  deactivateVisit(data){
    return this.http.put(environment.serverUrl +`/finance/studies/visits/deactivate/${data.budgetVersionId}/${data.visitId}`,{})
  }

  activateVisit(data){
    return this.http.put(environment.serverUrl +`/finance/studies/visits/activate/${data.budgetVersionId}/${data.visitId}`,{})
  }

  updateBudgetDates(data) {
    return this.http.put(environment.serverUrl +`/study-detail-budget-versions/${data.studyDetailId}`,data.req)

  }


  getKnowledgeableArticles(data){
    return this.http.get<any>(environment.serverUrl +`/finance/study-detail/knowledgeable-articles/${data}`,)

  } 

  getKnowledgeableArticlesbyStudyId(data){
    return this.http.get<any>(environment.serverUrl +`/finance/studies/knowledgeable-articles/${data}`)

  } 

  createKnowledgeableArticles(data){
    return this.http.put(environment.serverUrl +`/finance/study-detail/knowledgeable-articles/${data.studyId}`,data.req)

  } 


  

}
