import { HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { FinancialCalendar } from 'src/app/models/advertising/financial-calendar.model';
import { environment } from 'src/environments/environment';
import { CalendarEvent } from '../../models/advertising/calendar-event.model';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class CalendarService extends RootHttpService {


  getKeywords() {
    return this.http.get<any[]>(environment.serverUrl + '/ad/keywords ' )

  }
  getCalendarCategoryDetails(data) {
    return this.http.get<any[]>(environment.serverUrl + `/study-start-site/find-by-id/${data.studyStartSiteId}` )
    
  }
  downloadDailyScheduleReport(data){
    const params = new HttpParams().set('date', data.date)
    return this.http.get(environment.serverUrl + `/ctms/daily-schedule-report/${data.siteId}`,{params, responseType: 'arraybuffer'}  )
  }

  hideAccrualOnCalendar(data) {
    return this.http.put(environment.serverUrl + `/ad/calendars/${data}/hide-on-accruals`, {} )

  }

  hideDiscrepepencyOnCalendar(data) {
    return this.http.put(environment.serverUrl + `/ad/calendars/${data}/hide-on-discrepancies` ,{})
    
  }

  addKeyword(data) {
    return this.http.post(environment.serverUrl + '/ad/keywords ', data)

  }


  changeAdCalendarCategoryStatus(data) {
    return this.http.put(environment.serverUrl + '/ad/categories/active/' + data,{})
  }

  getCalendarCategories() {
    return this.http.get<any[]>(environment.serverUrl + `/ad/categories` )

  }

  createCalendarCategories(data) {
    return this.http.post(environment.serverUrl + '/ad/categories ', data)
  }

  updateCalendarCategories(data) {
    return this.http.put(environment.serverUrl + '/ad/categories/' + data,{})
  }


  addMediaEvent(data) {
    return this.http.post(environment.serverUrl + '/ad-calendar/add-new', {})
  }

  getByCalendarByMonthAndWeek(data) {
    const params = new HttpParams().set('year', data.year).set('month', data.month);
    return this.http.get<any[]>(environment.serverUrl + '/fin-calendar/', { params })
  }

  getFinWeeks(data): Observable<FinancialCalendar[]> {
    return this.http.post<FinancialCalendar[]>(environment.serverUrl + '/filter/fin-calendar', data);
  }

  getVendors(data) {
    const params = new HttpParams().set('active', data?.active).set('name', data?.search).set('siteId', data?.siteId).set('skip', data?.skip).set('limit', data?.limit)
    return this.http.get<any[]>(environment.serverUrl + '/ad/vendors', { params })
  }

  getCategories() {
    return this.http.get(environment.serverUrl + '/ad/categories', {})
  }

  getStudies(data, showOncalendar) {
    const params = new HttpParams().set('false', showOncalendar);
    return this.http.get(environment.serverUrl + '/ad/calendars/studies/' + data, { params })
  }

  addMediaEventCalendar(data) {
    const formData: FormData = new FormData();
    let request = new Blob([JSON.stringify(data)], {
      type: 'application/json'
    });
    formData.append('scheduleFile', data.scheduleFile);
    formData.append('adFile', data.adFile)
    formData.append('request', request);
    return this.http.post(environment.serverUrl + '/ad/calendars/', formData)
  }

  addStudyToCalendar(data) {
    return this.http.put(environment.serverUrl + '/ad/calendars/add/studies/' + data.studyID, data.request)
  }

  getDefaultCalendar(data): Observable<[FinancialCalendar]> {
    const params = new HttpParams().set('year', data)
    return this.http.get<[FinancialCalendar]>(environment.serverUrl + '/fin-calendar/default', { params })
  }

  getCompanyCalendars() {
    return this.http.get(environment.serverUrl + '/fin-calendar/company');
  }

  createFinancialCalendar(data) {
    const params = new HttpParams().set('year', data.year)
    return this.http.post(environment.serverUrl + '/fin-calendar/company', data, { params });
  }

  getFinancialCalendarWeeks(data): Observable<FinancialCalendar[]> {
    const params = new HttpParams().set('year', data)
    return this.http.get<FinancialCalendar[]>(environment.serverUrl + '/fin-calendar/company/weeks', { params });
  }

  getLastYearLastDate(data) {
    const params = new HttpParams().set('year', data)
    return this.http.get<FinancialCalendar[]>(environment.serverUrl + '/fin-calendar/company/last-date-of-previous-year', { params });
  }

  getCalendarEvents(data): Observable<{ resources: any[], events: any[] }> {

    const params = new HttpParams().set('year', data.year).set('month', data.month).set('showHidden', data.showHidden).set('mediums', data.mediums).set('viewAll',data?.advertising)
    return this.http.get<{ resources: any[], events: any[] }>(environment.serverUrl + '/ad/budget/' + data.siteId, { params });
  }

  mediaEvents(data) {
    return this.http.get(environment.serverUrl + '/ad/ads/studies/' + data.siteId);
  }

  getCurrentDate(): Observable<FinancialCalendar> {
    return this.http.get<FinancialCalendar>(environment.serverUrl + '/current-date/fin-calendar', {});
  }

  getCalendarEventByID(data) {
    return this.http.get<CalendarEvent>(environment.serverUrl + '/ad/calendars/' + data.eventID, {});
  }

  removeStudyFromCalendar(data) {
    return this.http.put<CalendarEvent>(environment.serverUrl + '/ad/calendars/remove/studies/' + data.id, {});
  }

  removeMediaEvent(data){
    return this.http.delete(environment.serverUrl + `/ad/calendars/${data}`);
  }

  toggleInCalendar(data) {
    return this.http.put<CalendarEvent>(environment.serverUrl + '/ad/calendars/' + data.resourceID + '/show-calendar', {});
  }

  toggleZeroWeeks(data) {
    return this.http.put<CalendarEvent>(environment.serverUrl + '/ad/calendars/' + data.resourceID + '/show-zero-weeks', {});
  }

  toggleTriage(data) {
    return this.http.put<CalendarEvent>(environment.serverUrl + '/ad/calendars/' + data.resourceID + '/show-triage', {});
  }

  getCalendarCategoryNotesByAdCalendarID(data) {
    return this.http.get(environment.serverUrl + '/ad/calendars/' + data.id + '/notes');
  }

  getMediaEventEditDetails(data) {
    return this.http.get(environment.serverUrl + '/ad/calendars/' + data.id + '/details');
  }

  getMediaEventEditDetails2(data) {
    return this.http.post<any>(environment.serverUrl + '/ad/calendars/' + data.id + '/week-details', data.request);
  }

  editMediaEventCalendar(data) {
    const formData: FormData = new FormData();
    let request = new Blob([JSON.stringify(data.request)], {
      type: 'application/json'
    });
    formData.append('scheduleFile', data.schedFile);
    formData.append('adFile', data.adFile)
    formData.append('request', request);
    return this.http.put(environment.serverUrl + '/ad/calendars/' + data.calendarID, formData);
  }

  updateCategoryDialog(data) {
    return this.http.put(environment.serverUrl + '/ad/calendars/update/studies/' + data.studyID, data.request);
  }

  sendCategoryNote(data) {
    return this.http.post(environment.serverUrl + '/ad/budgets/notes/' + data.studyID, data.request);
  }


  getMediaEventFile(data) {
    return this.http.get(environment.serverUrl + `/ad/calendars/${data.adCalendarId}/files/${data.fileId}`, {responseType:'arraybuffer'});

  }

  getNotesInCategoryDialog(data) {
    const params = new HttpParams().set('includeProjectNotes', data.includepm)
    return this.http.get(environment.serverUrl + '/ad/budgets/notes/' + data.id, {params});
  }

  findAllByFilter(data) {
    const params = new HttpParams().set('siteId', data.siteId).set('dateFrom', data.dateFrom).set('dateTo',data.dateTo).set('studyId', data.studyId == undefined ? '' : data?.studyId).set('type', data.type).set('indicationIds', data.indicationIds).set('vendorIds', data.vendorIds).set('studyIds', data.studyIds).set('fromAdQuery', data.fromAdQuery)
    return this.http.get<any>(environment.serverUrl + `/reports/overall-media-event-effectiveness`, {params})
  }

  getAdCategories(data) {
    const params = new HttpParams().set('active', data.active)
    return this.http.get<any[]>(environment.serverUrl + `/ad/ad-categories/sites/${data.siteId}`,{params} )
  }

  createAdCategory(data){
    return this.http.post<any>(environment.serverUrl + `/ad/ad-categories/sites/${data.siteId}`, {name:data.name});
  }

  getAdCalendarEventDetails(data) {
    return this.http.get<any>(environment.serverUrl + `/ad/calendars/${data}/event-details`);

  }

  changeAdCalendarMediaEventStatus(data) {
    return this.http.put<any>(environment.serverUrl + `/ad/calendars/${data.calendarId}/status`,{status:data.status});
  }
}
