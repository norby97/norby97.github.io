import { Injectable } from '@angular/core';
import { RootHttpService } from '../utils/root.http.service';
import { environment } from 'src/environments/environment';
import { HttpParams } from '@angular/common/http';

@Injectable({
  providedIn: 'root'
})
export class FindingsService extends RootHttpService {

  getFindings(data) {
    const params  = new HttpParams().set('siteId',data.siteId).set('studyId',data.studyId).set('patientStudyLinkId',data.patientStudyLinkId).set('assignToId',data.assignToId).set('trackedVisitId',data.trackedVisitId).set('findingCategoryId',data.findingCategoryId).set('statuses',data.statuses).set('dateFrom',data.dateFrom).set('dateTo',data.dateTo);
    return this.http.get<any[]>(environment.serverUrl + `/regulatory/qa/findings/filter`,{params});

  }
  
  addFinding(data) {
    return this.http.post(environment.serverUrl + `/regulatory/qa/findings`,data);
  }

  editFinding(data) {
    return this.http.put(environment.serverUrl + `/regulatory/qa/findings/${data?.qaFindingId}`,data.req);
  }
  editMultipleFindings(data) {
    return this.http.put(environment.serverUrl + `/regulatory/qa/findings`,data);
  }
  getAssignToStatistics(data){
    const params  = new HttpParams().set('siteId',data.siteId).set('studyId',data.studyId).set('patientStudyLinkId',data.patientStudyLinkId).set('assignToId',data.assignToId).set('trackedVisitId',data.trackedVisitId).set('findingCategoryId',data.findingCategoryId).set('statuses',data.statuses).set('dateFrom',data.dateFrom).set('dateTo',data.dateTo);
    return this.http.get(environment.serverUrl + `/regulatory/qa/findings/statistics/assigned-users`,{params});      
  }

  getFindingCategoriesStatistics(data){
    const params  = new HttpParams().set('siteId',data.siteId).set('studyId',data.studyId).set('patientStudyLinkId',data.patientStudyLinkId).set('assignToId',data.assignToId).set('trackedVisitId',data.trackedVisitId).set('findingCategoryId',data.findingCategoryId).set('statuses',data.statuses).set('dateFrom',data.dateFrom).set('dateTo',data.dateTo);
    return this.http.get(environment.serverUrl + `/regulatory/qa/findings/statistics/finding-categories`,{params});      
  }

  getStudyStatistics(data){
    const params  = new HttpParams().set('siteId',data.siteId).set('studyId',data.studyId).set('patientStudyLinkId',data.patientStudyLinkId).set('assignToId',data.assignToId).set('trackedVisitId',data.trackedVisitId).set('findingCategoryId',data.findingCategoryId).set('statuses',data.statuses).set('dateFrom',data.dateFrom).set('dateTo',data.dateTo);
    return this.http.get(environment.serverUrl + `/regulatory/qa/findings/statistics/studies`,{params});      
  }

  getVisitStatistics(data){
    const params  = new HttpParams().set('siteId',data.siteId).set('studyId',data.studyId).set('patientStudyLinkId',data.patientStudyLinkId).set('assignToId',data.assignToId).set('trackedVisitId',data.trackedVisitId).set('findingCategoryId',data.findingCategoryId).set('statuses',data.statuses).set('dateFrom',data.dateFrom).set('dateTo',data.dateTo);
    return this.http.get(environment.serverUrl + `/regulatory/qa/findings/statistics/visits`,{params});      
  }

  getSiteAverage(data) {
    return this.http.get(environment.serverUrl + `/regulatory/qa/findings/statistics/${data.siteId}/average`,);      

  }

  getQAStudies(data) {
    return this.http.get<any[]>(environment.serverUrl + `/regulatory/qa/findings/${data}/studies`,);
  }

  getQAPatients(data) {
    return this.http.get<any[]>(environment.serverUrl + `/regulatory/qa/findings/${data}/patient-studylinks`,);
  }
  getQAVisits(data) {
    return this.http.get<any[]>(environment.serverUrl + `/regulatory/qa/findings/${data}/visits`,);
  }
  submitMyResponse(data) { 
    return this.http.post(environment.serverUrl + `/regulatory/qa/finding/notes/${data.qaFindingId}`,data.req,);
  }

  moveToOpen(data){
    return this.http.put(environment.serverUrl + `/regulatory/qa/findings/${data.qaFindingId}/status/open `,{});
  }

  getFindingResponses(data) {
    return this.http.get<any[]>(environment.serverUrl + `/regulatory/qa/finding/notes/${data}`,);
  }

}
