import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { ContactStudyRole } from 'src/app/models/crmbd/contact-study-role.model';
import { environment } from 'src/environments/environment';
import { ContactModel } from '../../models/crmbd/contact.model';

@Injectable({
  providedIn: 'root'
})
export class ContactService {

  constructor(private http: HttpClient) { }

  findContact(data) {
    return this.http.get(environment.serverUrl + '/contact/search', {params: data});
  }


  getContactsRecentView() {
    return this.http.get<any[]>(environment.serverUrl + '/contact/all/recent-changes');

  }

  addContact(data): Observable<ContactModel> {
    return this.http.post<ContactModel>(environment.serverUrl + '/contact', data);
  }

  getAllByFirstLetter(data) {
    return this.http.get<any>(environment.serverUrl + '/contact', {params: data});
  }

  getContactsPaginated(data): Observable<any> {
    const params = new HttpParams().set('search',data.search).set('page', data?.pageNumber).set('size', data?.pageSize);
    return this.http.get<any[]>(environment.serverUrl + '/contact/paginated', {params})
  }

  getContactDetails(data):Observable<ContactModel> {
    return this.http.get<ContactModel>(environment.serverUrl + '/contact/' + data.contactID);
  }

  createContactStudy(data) {
    return this.http.post(environment.serverUrl + '/study-start-site/contact/' + data.siteId + '/' + data.studyStartId + '/' + data.contactId, {name:data.role});
  }

  deleteContactStudy(data) {
    return this.http.delete(environment.serverUrl + '/contact/study/' + data.contactStudyID);
  }
  
  updateContactDetails(data) {
    return this.http.put(environment.serverUrl + '/contact/' + data.contactID, data.request);
  }

  updateContactOtherDetails(data) {
    return this.http.put(environment.serverUrl + '/contact/other-details/' + data.contactID, data.request);
  }

  updateLinkedInURL(data) {
    return this.http.put(environment.serverUrl + '/contact/linkedin/' + data.contactID , data.request);
  }

  getAllContactNotes(data){
    return this.http.get(environment.serverUrl + '/contact/note/' + data.contactID );
  }

  createContactNote(data) {
    return this.http.post(environment.serverUrl + '/contact/note/', data);
  }
  
  deleteContactNote(data) {
    return this.http.delete(environment.serverUrl + '/contact/note/' + data.contactID + '/' + data.contactNoteID );
  }

  retrieveContactFiles(data) {
    return this.http.get(environment.serverUrl + '/contact/file/' + data);
  }

  uploadContactFile(data) {
    const params = new HttpParams().set('type', data.type);
    const formData: FormData = new FormData();
    formData.append('file', data.file);
    
    return this.http.post(environment.serverUrl + '/contact/file/' + data.contactID, formData, {params});
  }
  
  deleteContactFile(data) {
    return this.http.delete(environment.serverUrl + '/contact/file/' + data.contactID + '/' + data.fileID);
  }

  downloadContactFile(data) {
    return this.http.get(environment.serverUrl + '/contact/file/download/' + data.fileID, {responseType: 'arraybuffer'});
  }

  getContactAudits(data) {
    return this.http.get(environment.serverUrl + '/audit/contact/' + data.contactID);

  }

  removeContactFromLead(data) {
    return this.http.delete(environment.serverUrl + '/lead/contact/' + data.leadID + '/' + data.contactID);
  }

  removeArticleFromLead(data) {
    return this.http.delete(environment.serverUrl + '/lead/in-the-news/' + data.leadID + '/' + data.articleID);
  }

  removeNoteFromLead(data) {
    return this.http.delete(environment.serverUrl + '/lead/note/' + data.leadID + '/' + data.noteID);
  }

  getLinkedStudyRoles(data): Observable<ContactStudyRole> {
    return this.http.get<ContactStudyRole>(environment.serverUrl + '/contact/study-role/' + data.contactID)
   }

  linkContactToStudy(data) {
  return this.http.get<ContactStudyRole>(environment.serverUrl + '/contact/study-role/' + data.contactID)

  }
}
