  rescheduledAppointmentsReport(data) {
    const params = new HttpParams().set('dateFrom', data.dateFrom).set('dateTo',data.dateTo).set('siteIds', data.siteIds).set('reasonIds', data.reasonIds).set('userIds', data.userIds).set('appointmentTypes', data.appointmentTypes).set('type', data.type);
    return this.http.get<any>(environment.serverUrl + `/reports/rescheduled-appointments`, {params})
  }
