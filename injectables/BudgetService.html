import { HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { environment } from 'src/environments/environment';
import { Budgets } from '../../models/advertising/budgets.model';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class BudgetService extends RootHttpService {

  getStudyBudgets(data): Observable<Budgets>{
    return this.http.get<Budgets>(environment.serverUrl + '/ad/budgets/' + data);
  }

  createBudget(data) {
    const formData: FormData = new FormData();
    let request = new Blob([JSON.stringify(data.request)], {
      type: 'application/json'
    });
    formData.append('documentFile', data.documentFile == undefined ? '' : data.documentFile);
    formData.append('request', request);
    return this.http.post(environment.serverUrl + '/ad/budgets/' +data.studyStartSiteId, formData);
  }

  getBudgetNotes(data) {
    const params = new HttpParams().set('includeProjectNotes', data.includepm);
    
    return this.http.get(environment.serverUrl + '/ad/budgets/notes/' + data.studyID, {params});
  }

  addBudgetNotes(data) {
    return this.http.post(environment.serverUrl + '/ad/budgets/notes/' + data.studyID, data.request);
  }

  getBudgetFile(data){
    return this.http.get(environment.serverUrl + `/ad/budgets/download/${data}`);
  }

  getReimbursable(data: any) {
    return this.http.get<any>(environment.serverUrl + '/ad/budgets/available/' + data )
  }

  downloadDocumentation(data) {
    return this.http.get(environment.serverUrl + `/ad/budgets/download/${data}`, {responseType:'arraybuffer'})

  }
 
}
