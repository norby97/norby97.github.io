import { HttpClient, HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class MiscellaneousService extends RootHttpService {

  findAll(data) {
    return this.http.get(environment.serverUrl + '/ctms/miscellaneous/appointments');
  }

  getMiscDetails(data):Observable<any> {
    return this.http.get<any>(environment.serverUrl + `/ctms/miscellaneous/appointments/${data}/schedule`);
  }


  getNotScheduledVisits(data) {
   const params = new HttpParams().set('date', data.date);
    return this.http.get(environment.serverUrl + `/ctms/visits/schedule/study/${data.studyId}/patient-study-link/${data.patientStudyLinkId}/visits-who-is-not-scheduled`, {params});
  }

  scheduleVisitTrack(data) {
    return this.http.post(environment.serverUrl + `/ctms/visits/schedule/patient-study-link/${data.patientStudyLinkId}`, data.req);
  }

  
  createMiscApp(data) {
    return this.http.post(environment.serverUrl + '/ctms/miscellaneous/appointments', data);
  }

  createMiscAppForPatient(data) {
    return this.http.post(environment.serverUrl + '/ctms/miscellaneous/appointments/patients/' + data.patientId,data.req );
  }

  getMiscResources(data): Observable<any> {
   const params = new HttpParams().set('siteId', data);
    return this.http.get<any>(environment.serverUrl + '/ctms/daily-schedule/resources' ,{params});
  }

  getMiscUsers(data): Observable<any> {
    const params = new HttpParams().set('siteId', data);
    return this.http.get<any>(environment.serverUrl + '/user/ctms/show-on-schedule',{params} );
  }

  cancelMiscAppointement(data) {
    return this.http.put(environment.serverUrl + `/ctms/miscellaneous/appointments/${data}/cancel`,{});

  }


  markArrived(data) {
    return this.http.put(environment.serverUrl + `/ctms/miscellaneous/appointments/${data}/arrived`, {});

  }
  markNoShow(data) {
    return this.http.put(environment.serverUrl + `/ctms/miscellaneous/appointments/${data}/no-show`, {});
  }

  
  markNotArrived(data) {
    return this.http.put(environment.serverUrl + `/ctms/miscellaneous/appointments/${data}/no-arrived`, {});
  }
  
  markCancel(data) {
    return this.http.put(environment.serverUrl + `/ctms/miscellaneous/appointments/${data}/cancel`, {});
  }
  
  markAsShow(data) {
    return this.http.put(environment.serverUrl + `/ctms/miscellaneous/appointments/${data}/show`, {});
  }
  
  changeStatus(data) {
    const params = new HttpParams().set('status', data.status);
    return this.http.put(environment.serverUrl + `/ctms/miscellaneous/appointments/${data.appointmentId}/status`,params);
  }
  
  rescheduleMiscAppointment(data) {
    return this.http.put(environment.serverUrl + `/ctms/miscellaneous/appointments/patients/${data.patientId}/${data.appointmentId}/reschedule`,data.req);
  }

  reschedulePatientlessMiscApp(data) {
    return this.http.put(environment.serverUrl + `/ctms/miscellaneous/appointments/${data.appointmentId}/reschedule`,data.req);
  }

  checkOverlap(data) {
    return this.http.post<any[]>(environment.serverUrl + '/ctms/appointments/overlapping/miscellaneous',data );
  }
}
