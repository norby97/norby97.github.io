import { Injectable } from '@angular/core';
import { HttpClient, HttpParams } from '@angular/common/http';
import { environment } from 'src/environments/environment';
import { Observable } from 'rxjs';
import { RootHttpService } from '../utils/root.http.service';
import { FilePath } from '../../models/regulatory/FilePath';
import { FolderStructure } from 'src/app/models/regulatory/folder-structure.model';

@Injectable({
  providedIn: 'root'
})
export class FileService extends RootHttpService {

  uploadFile(data) {
    const formData: FormData = new FormData();
    let request = new Blob([JSON.stringify(data.filePath)], {type: 'application/json'});
    data.file.forEach(element => {
      formData.append('file', element);
    });
    formData.append('filePath', request);
    return this.http.post(environment.serverUrl + '/file-management/study/' + data.filePath[0].id + '/new-file', formData);
  }

  externalPreviewRetrieve(studyID, fileID) {
    return this.http.get(environment.serverUrl + '/file-management/preview/' + studyID + '/download/file/' + fileID, { responseType: 'arraybuffer' });
  }

  updateFile(data) {
    const formData: FormData = new FormData();
    formData.append('physicalFile', data.file);
    return this.http.post(environment.serverUrl + '/file-management/study/' + data.id + '/update/new-version', formData, { params: { newVersionFilePath: JSON.stringify(data.filePath) } });
  }

  searchFile(data) {
    const params = new HttpParams().set('pageNr', '0').set('pageSize', '5').set('text', data.text);
    return this.http.get(environment.serverUrl + '/file-management/search/', {params});
  }

  markReviewed(data) {
    const params = new HttpParams().set('fileId', data.fileId).set('reviewFlag', data.reviewFlag);
    return this.http.post(environment.serverUrl + '/file-management/study/' + data.studyID + '/mark-review', params);
  }

  getReviewedFiles(data): Observable<[]> {
    return this.http.get<[]>(environment.serverUrl + '/file-management/study/' + data.studyID + '/review');
  }

  createNewFolder(data) {
    return this.http.post(environment.serverUrl + '/file-management/study/' + data.id + '/new-folder', data);
  }

  retrieveFile(data) {
    if (data.addSignaturePage) {
      const params = new HttpParams().set('addSignaturePage', 'true');
      return this.http.get(environment.serverUrl + '/file-management/study/' + data.studyID + '/open/file/' + data.fileID, { responseType: 'arraybuffer', params });
    } else {
      return this.http.get(environment.serverUrl + '/file-management/study/' + data.studyID + '/open/file/' + data.fileID,{ responseType: 'arraybuffer' });
    }
  }
 downloadFile(data) {
    if (data.addSignaturePage) {
      const params = new HttpParams().set('addSignaturePage', 'true');
      return this.http.get(environment.serverUrl + '/file-management/study/' + data.studyID + '/download/file/' + data.fileID, { responseType: 'arraybuffer', params });
    } else {
      return this.http.get(environment.serverUrl + '/file-management/study/' + data.studyID + '/download/file/' + data.fileID,{ responseType: 'arraybuffer' });
    }
  }
  retrieveFileDetails(data): Observable<FilePath> {
    return this.http.get<FilePath>(environment.serverUrl + '/file-management/study/' + data.studyID + '/file/' + data.fileID);
  }

  editFile(data, studyID) {
    return this.http.put(environment.serverUrl + '/file-management/study/' + studyID + '/edit', data);
  }

  retrieveFolderStructure(data): Observable<FolderStructure> {
    return this.http.get<FolderStructure>(environment.serverUrl + '/file-management/study/' + data + '/retrieve');
  }

  moveFile(data) {
    const params = new HttpParams().set('fileId', data.fileId).set('parentId', data.parentId);
    return this.http.put(environment.serverUrl + '/file-management/study/' + data.studyId + '/move-file', params);
  }

  downloadStudy(data) {
    return this.http.post(environment.serverUrl + '/download/study/' + data, {});
  }

  getAll(data) {
    const params = new HttpParams().set('expired', data.expired);
    return this.http.get(environment.serverUrl + '/file-management/all-files', { params });
  }

}
