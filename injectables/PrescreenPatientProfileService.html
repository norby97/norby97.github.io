import { isPlatformBrowser } from '@angular/common';
import { ComponentFactoryResolver, ComponentRef, Inject, Injectable, Injector, PLATFORM_ID } from '@angular/core';
import { PatientProfileComponent } from 'src/app/components/ctms/patient-profile/patient-profile.component';

@Injectable({
  providedIn: 'root'
})
export class PrescreenPatientProfileService {
  private windowRef: ComponentRef<PatientProfileComponent>;

  constructor(private componentFactoryResolver: ComponentFactoryResolver, private injector: Injector, @Inject(PLATFORM_ID) private platformId: Object) {}
  
  open(id,name) {
    name=name?.replace('(','%28').replace(')','%29') ?? 'Patient Profile'

    const factory = this.componentFactoryResolver.resolveComponentFactory(PatientProfileComponent);
    if (isPlatformBrowser(this.platformId)) {
      const win = window.open(`/ctms/patients/${id}/${name}`, '_blank','height=1000,width=1300');
      const div = win.document.createElement('div');
      const style = win.document.createElement('style');
   
      win.document.head.appendChild(style);
      win.document.body.appendChild(div);
      const factory = this.componentFactoryResolver.resolveComponentFactory(PatientProfileComponent);
      
      this.windowRef = factory.create(this.injector, [], div);
    }
  }

  close() {
    if (this.windowRef) {
      this.windowRef.destroy();
    }
    
  }
    
}
