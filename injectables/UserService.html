import { Injectable } from '@angular/core';
import { environment } from 'src/environments/environment';
import { Observable } from 'rxjs';
import { RootHttpService } from '../utils/root.http.service';
import { HttpParams } from '@angular/common/http';
import { GeneralResponse } from 'src/app/models/common/GeneralResponse';
import { Company } from '../../models/administration/company/Company';
import { ChangePasswordRequests } from '../../models/administration/user/ChangePasswordRequests';
import { User } from '../../models/administration/user/User';
import { UserInviteToken } from '../../models/administration/user/UserInviteToken';

@Injectable({
    providedIn: 'root'
})
export class UserService extends RootHttpService {

    getAll(): Observable<User[]> {
        return this.http.get<User[]>(environment.serverUrl + '/user/all');
    }
    getAllAdministration(data) {
        const params = new HttpParams().set('externalUsers',data?.externalUsers).set('activeUsers',data?.activeUsers)
        return this.http.get<User[]>(environment.serverUrl + '/v2/users/administration',{params});
    }

    getAll2(data?) {
        const params = new HttpParams().set('siteId',data?.siteId == undefined ? null: data?.siteId)
        return this.http.get<User[]>(environment.serverUrl + '/v2/users', {params});
    }
    getUserByID(id): Observable<User> {
        return this.http.get<User>(environment.serverUrl + '/user', { params: { id } });
    }

    create(user: User): Observable<User> {
        return this.http.post<User>(environment.serverUrl + '/user/' + user.templateId , user);
    }

    resetLoginCounter(id): Observable<User> {
        return this.http.put<User>(environment.serverUrl + '/user/reset-failed-login-counter/' + id, {});
    }
    updateUserProfileDetails(form: any): Observable<User> {
        const formData: FormData = new FormData();
        
        
        formData.append('profilePicture', form.profilePicture);
        return this.http.post<User>(environment.serverUrl + '/user/' + form.userId + '/updateProfileDetails', formData,
            {
                params: {
                    firstName: form.firstName,
                    lastName: form.lastName,
                    phone: form.phone,
                }
            });
    }

    confirmUserInviteToken(userInviteToken: string): Observable<UserInviteToken> {
        return this.http.get<UserInviteToken>(environment.serverUrl + '/user/confirmUserInviteToken', { params: { userInviteToken } });
    }

    registerUser(userRegister: any): Observable<User> {
        return this.http.post<User>(environment.serverUrl + '/user/register', userRegister);
    }

    forgotPassword(data): Observable<GeneralResponse> {
        const params = new HttpParams().set('email', data.email);
        return this.http.post<GeneralResponse>(environment.serverUrl + '/auth/forgot-password', params);
    }

    updatePassword(changePasswordRequests: ChangePasswordRequests): Observable<User> {
        return this.http.post<User>(environment.serverUrl + '/user/change-password', changePasswordRequests);
    }

    generatePin(fdaDocument: any, userId: string): Observable<Company> {
        const formData: FormData = new FormData();
        if (fdaDocument) {
            formData.append('fdaDocument', fdaDocument);
        }
        return this.http.post<Company>(environment.serverUrl + '/user/' + userId + '/generatePin', formData);
    }

    revokePin(userId: string): Observable<Company> {
        return this.http.get<Company>(environment.serverUrl + '/user/' + userId + '/revokePin');
    }

    updateUser(data) {
        return this.http.put(environment.serverUrl + '/user', data);
    }

    nonExclusiveLower(data?): Observable<[User]> {
        if (data) {
            const params = new HttpParams().set('view', 'FULL').set('studyId', data.studyID);
            return this.http.get<[User]>(environment.serverUrl + '/user/all/non-exclusive-lower', { params });
        } else {
            const params = new HttpParams().set('view', 'FULL');
            return this.http.get<[User]>(environment.serverUrl + '/user/all/non-exclusive-lower', { params });
        }
    }

    removeProfilePicture(data): Observable<{ profilePictureUrlPrefix }> {
        return this.http.delete<{ profilePictureUrlPrefix }>(environment.serverUrl + '/user/' + data.userID + '/profile-picture');

    }

    getUsersBySite(data): Observable<[User]> {
        if (data.accountType) {
            const params = new HttpParams().set('accountType', data.accountType)
            return this.http.get<[User]>(environment.serverUrl + '/site/' + data.siteID + '/get-regulatory', { params });
        } else {
            return this.http.get<[User]>(environment.serverUrl + '/site/' + data.siteID + '/get-regulatory');
        }
    }

    getUsersWithPIN(): Observable<[User]> {
        return this.http.get<[User]>(environment.serverUrl + '/user/allWithPin');
    }

    resendInvite(data) {
        return this.http.post(environment.serverUrl + '/user/' + data.userID + '/resend-invite', {});
    }

    userSetLockScreen() {
        return this.http.put(environment.serverUrl + '/user/lock-screen', {});
    }

    getIfUserLockedScreened() {
        return this.http.get(environment.serverUrl + '/user/lock-screen', {});
    }

    makeActive(data) {
        return this.http.put(environment.serverUrl + '/user/active/' + data, {});

    }

    makeInactive(data) {
        return this.http.put(environment.serverUrl + '/user/inactive/' + data, {});
    }

    updateUserPreferences(data) {
        return this.http.put(environment.serverUrl + '/user/preferences', data);
    }

    //refactoring
    makeUserSiteStatus(data) {
        return this.http.put(environment.serverUrl + `/commons/users/${data.userId}/sites/${data.siteId}/active`, {});
    }

    makeUserSiteManager(data) {
        return this.http.put(environment.serverUrl + `/commons/users/${data.userId}/sites/${data.siteId}/manager`, {});
    }

    makeUserSiteShowOnSchedule(data) {
        return this.http.put(environment.serverUrl + `/commons/users/${data.userId}/sites/${data.siteId}/show-on-schedule`, {});
    }

    makeUserSitePrincipalInvestigator(data) {
        return this.http.put(environment.serverUrl + `/commons/users/${data.userId}/sites/${data.siteId}/principal-investigator`, {});
    }

    getUserNotifications(data) {
        return this.http.get<any[]>(environment.serverUrl + `/commons/notification-setup/users/${data}`);
    }

    getUserNotificationsToShow(data) {
        const params = new HttpParams().set('skip', data.skip).set('limit', data.limit)

        return this.http.get<any[]>(environment.serverUrl + `/commons/notifications`, {params});
    }

    getUserNotificationsCount() {
        return this.http.get(environment.serverUrl + `/commons/notifications/count`);
    }

    updateUserNotification(data) {
        return this.http.put(environment.serverUrl + `/commons/notification-setup/users/${data.userNotificationId}`, data.req);
    }

    dismissUserNotification(data){
        return this.http.delete(environment.serverUrl + `/commons/notifications/${data}`);
    }
    dismissAllUserNotification(){
        return this.http.delete(environment.serverUrl + `/commons/notifications/users`);
    }
}
