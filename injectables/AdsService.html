import { Injectable } from '@angular/core';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class AdsService extends RootHttpService {

  getAds(data) {
    return this.http.get<any>(environment.serverUrl + '/ad/ads/studies/' + data, {});
  }

  getMediaAds(data) {
    return this.http.get(environment.serverUrl + '/ad/ads/all/studies/' + data, {});
  }

  getCategories() {
    return this.http.get(environment.serverUrl + '/ad/categories/',{});
  }

  createAd(data) {
    const formData: FormData = new FormData();
    let request = new Blob([JSON.stringify(data.request)], {
      type: 'application/json'
    });
    formData.append('audioFile',data.audioFile );
    formData.append('scriptFile',data.scriptFile)
    formData.append('request',request);
    return this.http.post(environment.serverUrl + '/ad/ads/studies/' + data.studyStartID, formData  );
  }

  approveAd(data) {
    const formData: FormData = new FormData();
    let request = new Blob([JSON.stringify(data.request)], {
      type: 'application/json'
    });
    formData.append('request',request );
    formData.append('approvedFile',data.request.approvedFile)
    return this.http.put(environment.serverUrl + '/ad/ads/approve/' + data.adID, formData);
  }

  disapproveAd(data) {
    return this.http.put(environment.serverUrl + '/ad/ads/disapprove/' + data.adID, data.request);
  }

  createStatusToAdd(data) {
    return this.http.put(environment.serverUrl + '/ad/ads/statuses/' + data.adID, data.request);
  }
  
}
