import { HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class LedgerService extends RootHttpService{

  createMiscInvoice(data) {
    return this.http.post(environment.serverUrl + `/finance/miscellaneous/invoices/studies/${data.studyId}`,data.req);
  }

  getInvoices(data) {
    const params = new HttpParams().set('studyId', data.studyId).set('dateFrom', data.dateFrom).set('dateTo', data.dateTo);
    return this.http.get(environment.serverUrl + `/finance/ledger`,{params});
  }


  getAdvInvoice(data) {
    return this.http.get(environment.serverUrl + `/finance/out/invoices/download/${data}`, {responseType:'arraybuffer'});
  }

  getMiscInvoice(data) {
    return this.http.get(environment.serverUrl + `/finance/miscellaneous/invoices/download/${data}`, {responseType:'arraybuffer'});
  }

  editAdvertisingPayment(data) {

  }

  getMiscDetailsForEdit(data) {
    const params = new HttpParams().set('studyId', data.studyId).set('dateFrom', data.dateFrom).set('dateTo', data.dateTo).set('isEdit', 'true');
    return this.http.get<any>(environment.serverUrl + `/finance/ledger`,{params});
  }


  editMiscPayment(data){
    return this.http.post<any>(environment.serverUrl + `/finance/miscellaneous/invoices/studies/${data.studyId}`, data.req);
  }

  getTimerAvailablePatients(data) {
    const params = new HttpParams().set('date', data)
    return this.http.get<any>(environment.serverUrl + `/ctms/timers/patient-appointments`, {params});
  }

  createPatientTimer(data) {
    return this.http.post<any>(environment.serverUrl + `/ctms/timers`, data);
  }

  getPatientTimers() {
    return this.http.get<any>(environment.serverUrl + `/ctms/timers`);
  }

  deletePatientTimer(data) {
    return this.http.delete<any>(environment.serverUrl + `/ctms/timers/${data}`);

  }
}
