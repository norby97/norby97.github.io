import { HttpParams } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { environment } from 'src/environments/environment';
import { RootHttpService } from '../utils/root.http.service';

@Injectable({
  providedIn: 'root'
})
export class TrackingService extends RootHttpService  {

  removePatientFromStudy(data){
    return this.http.delete(environment.serverUrl + `/ctms/patient/tracking/${data}  `,);
    
  }

  linkVisit(data) {
    return this.http.put(environment.serverUrl + `/ctms/patient/tracking/link-visit/${data.patientStudyLinkId}/${data.patientId}/${data.visitId}`, {});
  }

  createNewPatientAndAddToStudy(data) {
    return this.http.post(environment.serverUrl + `/ctms/patient/tracking/studies/${data.studyId}/new-patient`, data.req);
  }
  
  existingPatientAndAddToStudy(data) {
    return this.http.post(environment.serverUrl + `/ctms/patient/tracking/studies/${data.studyId}/existing-patient/${data.patientId}`, data.req);
  }

  trackVsit(data) {
    return this.http.post(environment.serverUrl + `/ctms/patient/tracking/track-visit/${data.patientStudyLinkId}`,data.req);
  }

  retrackVisit(data) {
    return this.http.post(environment.serverUrl + `/ctms/patient/tracking/retrack-visit/${data.trackedVisitId}/${data.patientStudyLinkId}`,data.req);
  }

  untrackVisit(data) {
    return this.http.post(environment.serverUrl + `/ctms/patient/tracking/untrack-visit/${data.trackedVisitId}`,data.req);
  }

  findAllVisitsForPatient(data):Observable<any[]> {
    return this.http.get<any[]>(environment.serverUrl + `/ctms/patient/tracking/visits/${data}`, {});
  }

  getAllDetailsByVisit(data) {
    return this.http.get(environment.serverUrl + `/ctms/patient/tracking/${data.patientStudyLinkId}/visits/details/${data.visitId}/${data.id}`, {});
  }

  findAllTrackingPatientByStudy(data):Observable<any[]> {
    const params = new HttpParams().set('active', data.status);
    return this.http.get<any[]>(environment.serverUrl + `/ctms/patient/tracking/${data.id}`, {params});
  }

  getEntryPoints(data) {
    return this.http.get<any[]>(environment.serverUrl + `/finance/studies/visits/entry-points/${data}`,);
  }
  
  deleteVisit(data) {
    return this.http.delete(environment.serverUrl + `/ctms/visits/schedule/${data} `,);
  }
  
}
