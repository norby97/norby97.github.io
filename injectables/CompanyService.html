import {Injectable} from '@angular/core';
import {environment} from 'src/environments/environment';
import {RootHttpService} from '../utils/root.http.service';
import {Observable} from 'rxjs';
import { HttpParams } from '@angular/common/http';
import { AllCompanyDetailsDto } from '../../models/administration/company/AllCompanyDetailsDto';
import { Company } from '../../models/administration/company/Company';
import { CompanyDetailsDto } from '../../models/administration/company/CompanyDetailsDto';

@Injectable({
    providedIn: 'root'
})
export class CompanyService extends RootHttpService {

    getByID(id): Observable<CompanyDetailsDto> {
        const params = new HttpParams().set('id', id)
        return this.http.get<CompanyDetailsDto>(environment.serverUrl + '/company', {params});
    }

    getAll(): Observable<AllCompanyDetailsDto> {
        return this.http.get<AllCompanyDetailsDto>(environment.serverUrl + '/company/all');
    }

    save(company: Company): Observable<Company> {
        return this.http.put<Company>(environment.serverUrl + '/company', company);
    }

    submitBlankFdaForm(fdaDocument: any, companyId: string): Observable<Company> {
        const formData: FormData = new FormData();
        formData.append('fdaDocument', fdaDocument);
        return this.http.post<Company>(environment.serverUrl + '/company/' + companyId + '/submitBlankFdaForm', formData);
    }

    getAdvertisingSetupDetails(data) {
        return this.http.get<any>(environment.serverUrl + `/administration/company/account-type/${data}`,);
    }

    editAdvertisingSetupDetails(data) {
        return this.http.post(environment.serverUrl + `/administration/company/account-type/${data.companyId}`, data.req);
    }

    getCompanyPhoneNumbers(data) {
        return this.http.get(environment.serverUrl + `/administration/companies/${data.companyId}/phoneNumbers`,);
    }

    getPhoneNumbers(data) {
        const params = new HttpParams().set('pattern', data)
        return this.http.get(environment.serverUrl + `/administration/plivo/phone-numbers`,{params});
    }

    editPhoneNumbers(data) {
        return this.http.post(environment.serverUrl + `/administration/companies/${data.companyId}/phoneNumbers`, data.req);
    }

   getListen360(data) {
        return this.http.get<any>(environment.serverUrl + `/commons/companies/${data}/listen360 `);
    }

    editListen360(data) {
        return this.http.post(environment.serverUrl + `/commons/companies/${data.companyId}/listen360`, data.req);
    }

    addNotificationTypeToCompany(data){
        return this.http.post(environment.serverUrl + `/commons/notification-setup/companies`, data);
    }

    getAllNotificationTypes(data){
        return this.http.get<any>(environment.serverUrl + `/commons/notification-setup/companies/${data}`);
    }

    updateCompanyNotificationType(data) {
        return this.http.put(environment.serverUrl + `/commons/notification-setup/companies`, data);

    }
}
