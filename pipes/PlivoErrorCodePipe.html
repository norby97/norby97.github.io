import { Pipe, PipeTransform } from '@angular/core';

@Pipe({
  name: 'plivoErrorCode'
})
export class PlivoErrorCodePipe implements PipeTransform {

  transform(value: number): string {
    if(value === null) {
      return ''; 
    }
    if(value === 10) {
      return 'invalid-message'; 
    }
    if(value === 20) {
      return 'network-error'; 
    }
    if(value === 30) {
      return 'spam-detected'; 
    }
    if(value === 40) {
      return 'invalid-source-numbe'; 
    }
    if(value === 50) {
      return 'invalid-destination-number'; 
    }
    if(value === 60) {
      return 'loop-detected'; 
    }
    if(value === 70) {
      return 'destination-permanantly-unavailable'; 
    }
    if(value === 80) {
      return 'destination-temporarily-unavailable'; 
    }
    if(value === 90) {
      return 'no-route-available'; 
    }
    if(value === 100) {
      return 'prohibited-by-carrier'; 
    }
    if(value === 110) {
      return 'message-too-long'; 
    }
  
    if(value === 160) {
      return 'dlt-registration-issue'; 
    }
    if(value === 200) {
      return 'source-number-blocked-by-stop-from-destination-number'; 
    }
    if(value === 300) {
      return 'failed-to-dispatch-message'; 
    }
    if(value === 420) {
      return 'message-expired'; 
    }
    if(value === 450) {
      return 'destination-country-disabled'; 
    }
    if(value === 900) {
      return 'insufficient-credit'; 
    }
    if(value === 910) {
      return 'account-disabled'; 
    }
    if(value === 950) {
      return 'daily-messaging-limit-reached'; 
    }
    if(value === 1000) {
      return 'unknown-error'; 
    }
    
  } 

}
