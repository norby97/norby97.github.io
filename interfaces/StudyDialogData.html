export class StudyDialogComponent implements OnInit {
  
    formGroup: UntypedFormGroup;
    @Output() studySuccessfullySaved = new EventEmitter();
    defaultPrincipalInvestigator;
    siteSelected = false;
    investigators;
    sites
    loading = false;
    @HostListener('window:keyup.esc') onKeyUp() {
      this.dialogRef.close(false);
    }
    
    @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
        event.returnValue = false;
    }
    constructor(
        
        @Inject(MAT_DIALOG_DATA) public data: StudyDialogData,
        private formBuilder: UntypedFormBuilder,
        public dialog: MatDialog,
        private sccipService: SccipService,
        private siteService: SiteService,
        private authService: AuthenticationService,
        private utilService: UtilService,
        private userService: UserService,
        public dialogRef: MatDialogRef<StudyDialogComponent>,
        private studyService: StudyService,
        private binderService: BindersTemplateService
    ) {
    }
    sponsors;
    filteredSponsors: Observable<Sponsor[]>;
    cros;
    filteredCros: Observable<Sponsor[]>;
    users;
    binderTemplates;
    studyDetails: Study;
    ngOnInit() {
        if(this.data?.study?.id)  {
            this.siteSelected = true;
            this.studyService.getInvestigators(null).subscribe(res => {
               this.investigators = res;
            });
            this.studyService.getStudy(this.data.study.id).subscribe(res => {
                this.studyDetails = res;
                this.formGroup = this.formBuilder.group({
                    id: this.studyDetails?.id,
                    sponsor: this.formBuilder.group({
                        name: [this.studyDetails?.sponsor?.name, Validators.required]
                    }),
                    protocol: [this.studyDetails?.protocol, Validators.required],
                    cro: this.formBuilder.group({
                        name: this.studyDetails?.cro?.name
                    }),
                    nickname: this.studyDetails?.nickname,
                    indNumber: this.studyDetails?.indNumber,
                    principalInvestigator: [this.studyDetails?.principalInvestigator, Validators.required],
                    subInvestigators: [this.studyDetails?.subInvestigators],
                    site: [this.studyDetails?.site, Validators.required],
                    siteNumber: this.studyDetails?.siteNumber,
                    leadCrc: [this.studyDetails?.leadCrc, Validators.required],
                    backupCrc: [this.studyDetails?.backupCrc],
                    structure: [this.data?.study?.structure, this.data?.study?.id ? null : Validators.required ]
        
                });
            });
        } else {
            this.siteSelected = false;
        }
        
        this.sccipService.getCros().subscribe(res => {
            this.cros = res;
        })
        this.sccipService.getSponsors().subscribe(res => {
            this.sponsors = res;
        })
        if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
            this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
          } else {
             this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
              this.sites = res;
            });
          }
        this.userService.getAll2().subscribe(res => {
            this.users = res;
        })
       
        this.binderService.retrieve({ companyID: this.authService.loginUserValue.company.id }).subscribe(res => {
            this.binderTemplates = res.templates;
        });
        this.formGroup = this.formBuilder.group({
            id: this.studyDetails?.id,
            sponsor: this.formBuilder.group({
                name: [this.studyDetails?.sponsor?.name, Validators.required]
            }),
            protocol: [this.studyDetails?.protocol, Validators.required],
            cro: this.formBuilder.group({
                name: this.studyDetails?.cro?.name
            }),
            nickname: this.studyDetails?.nickname,
            indNumber: this.studyDetails?.indNumber,
            principalInvestigator: [this.studyDetails?.principalInvestigator, Validators.required],
            subInvestigators: [this.studyDetails?.subInvestigators],
            site: [this.studyDetails?.site, Validators.required],
            siteNumber: this.studyDetails?.siteNumber,
            leadCrc: [this.studyDetails?.leadCrc, Validators.required],
            backupCrc: [this.studyDetails?.backupCrc],
            structure: [this.data?.study?.structure, this.data?.study?.id ? null : Validators.required ]

        });
        this.defaultPrincipalInvestigator = this.data?.study?.principalInvestigator?.firstName;
        this.filteredSponsors = this.formGroup.get('sponsor.name').valueChanges.pipe(
            startWith(''),
            map(value => this._filter(value, this.sponsors))
        );
        this.filteredCros = this.formGroup.get('cro.name').valueChanges.pipe(
            startWith(''),
            map(value => this._filter(value, this.cros))
        );
        
        
      
    }

    getInvestigator(event) {
        this.studyService.getInvestigators(event.value.id).subscribe(res => {
            this.investigators =res;
            this.siteSelected = true;
        })
    }

    private _filter(value: string, items: Array<any>) {
        const filterValue = value?.toLowerCase();
        return items?.filter(item => {
            return item.name?.toLowerCase().indexOf(filterValue) === 0;
        });
    }

    compareId(o1: any, o2: any) {
        return (o1?.id === o2?.id);
    }

    submit() {
        this.loading = true;
        let duplicate = false;
        if (this.formGroup.valid) {
            // this.data.studies.forEach(study => {
            //     if (study.protocol === this.formGroup.get('protocol').value && study.sponsor.name === this.formGroup.get('sponsor').value.name
            //          &&  study.site.id === this.formGroup.get('site').value.id && this.formGroup.controls.protocol.touched && this.formGroup.controls.protocol.touched) {
            //             duplicate = true;
            //             this.studyService.showError('Sponsor and Protocol pair already exists in Site!');
            //     }
            // });    
            if (duplicate === false) {
                this.studyService.save(this.formGroup.value).subscribe(
                    res => {
                        if (res.responseStatus !== 'SUCCESS') {
                            this.studyService.showError(res.responseMessage);
                        } else {
                            this.utilService.studyAdded = 1;
                            this.studySuccessfullySaved.emit();
                            this.dialog.closeAll();

                        }
                    }, err => {
                        this.loading = false;
                      }
                );
            }
        } else {
            this.formGroup.markAllAsTouched();
            this.loading = false;
        }
    }

}
