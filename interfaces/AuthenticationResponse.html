import { Injectable } from '@angular/core';
import { BehaviorSubject, Observable } from 'rxjs';
import { HttpClient, HttpParams } from '@angular/common/http';
import { environment } from '../../../environments/environment';
import { Router } from '@angular/router';
import { UntypedFormGroup } from '@angular/forms';
import { User } from '../../models/administration/user/User';
import { MatDialog } from '@angular/material/dialog';
import { InfoPopupComponent } from '../../components/common/info-popup.component';
import { GeneralResponse } from '../../models/common/GeneralResponse';
import { AppLoadService } from '../utils/AppLoadService';
import { UtilService } from '../utils/UtilService';
import { SiteService } from '../regulatory/site.service';
import { ToastrService } from 'ngx-toastr';
import { UserPreferencesService } from '../utils/user-preferences.service';
declare var $:any;
export interface AuthenticationResponse {
    showSecurityCodeField: boolean;
    loginUser: User;
    message: string;
}

export interface LogInUserDto extends GeneralResponse {
    modules: any;
    user: User;
    token: string;
    ip: string;
}

@Injectable({ providedIn: 'root' })
export class AuthenticationService {
    private loginUserSubject: BehaviorSubject<User>;
    public apploading = false;
    public loginUser: Observable<User>;
    authenticationResponse: AuthenticationResponse = {
        showSecurityCodeField: false,
        loginUser: null,
        message: ''
    };

    constructor(
        private http: HttpClient,
        public dialog: MatDialog,
        private siteService: SiteService,
        private appService: AppLoadService,
        private userPreferences: UserPreferencesService,
        private toastr:ToastrService,
        private router: Router
    ) {
       
        const subjectValue: User = JSON.parse(localStorage.getItem('siteCentric.loginUser'));
        this.loginUserSubject = new BehaviorSubject<User>(subjectValue);
        this.loginUser = this.loginUserSubject.asObservable();
       
    }

    public  setloginUserValue(user) {
        user.modules = this.loginUserPermissions
        this.loginUserSubject.next(user);
    }

    public get loginUserValue(): User {
        return this.loginUserSubject.value;
    }

    public get loginUserPermissions(): any[] {
        return this.loginUserSubject.value.modules;
    }


    login(loginForm: UntypedFormGroup): AuthenticationResponse {
        localStorage.removeItem('siteCentric.token');
        this.http.put<LogInUserDto>(environment.serverUrl + '/auth/login', {
            user: {
                email: loginForm.value.email.toLowerCase(),
                password: loginForm.value.password,
            },
            password: loginForm.value.password,
            securityCode: loginForm.value.securityCode
        }).subscribe(
            res => {
                if (res.responseStatus !== 'SUCCESS') {
                    this.showError({errMessage:res.responseMessage});
                } else if (loginForm.value.securityCode) {
                    this.appService.onLoginProcess.emit(true);
                    this.apploading = true;
                    localStorage.setItem('siteCentric.token', res.token);
                    res.user.modules = Object.assign({},res.modules);
                    localStorage.setItem('siteCentric.loginUser', JSON.stringify(res.user));
                    this.loginUserSubject.next(res.user);
                    this.siteService.getAll2(this.loginUserValue?.company?.id).subscribe(result => {
                        localStorage.setItem('siteCentric.sites', JSON.stringify(result));
                        if(this.loginUserValue?.accountType != 'EXTERNAL') {
                            if(res.user.preferencesLayout == null ||res.user.preferencesLayout == undefined || res.user.preferencesLayout == '' )  {
                                localStorage.setItem('scUserSettings',JSON.stringify({selectedSite:result?.length ? result[0] : {id:null, alias:'', name:''} }));
                            } else {
                                localStorage.setItem('scUserSettings',res?.user?.preferencesLayout);
                            }
                            this.userPreferences.setuserState('loggedIn',true,true);
                        }
                        this.appService.onLogin.emit(true);
                        this.apploading = false;
                    }, err => {
                        this.apploading = false;
                      });

                } else {
                    this.authenticationResponse.showSecurityCodeField = true;
                    this.authenticationResponse.loginUser = res.user;
                    this.authenticationResponse.message = '';
                }
            }, err => {
                this.apploading = false;
            }        
        );
        return this.authenticationResponse;
    }

    logout() {
        this.userPreferences.setuserState('scoreCardSearchText','');
        this.userPreferences.setuserState('saveScoreCardSearch',false);
        if (this.loginUserValue) {
      
            if(JSON.parse(localStorage.getItem('scUserSettings'))?.clearUponLogout == true){
                localStorage.setItem('scUserSettings',JSON.stringify({selectedSite:JSON.parse(localStorage.getItem('siteCentric.sites'))[0], collapseMenu:'off', fixedNavbar:'off',showSuccessMessages:true,clearUponLogout:true}))
                this.userPreferences.setuserState('loggedIn',false,true);
              }
            this.http.put<LogInUserDto>(environment.serverUrl + '/auth/logout', { user: this.loginUserValue }).subscribe(
                res => {
                    if (res.responseStatus !== 'SUCCESS') {
                        this.showError(res.responseMessage);
                        this.appService.onLogOut.emit(false);
                    } else {
                        this.appService.onLogOut.emit(true);
                        $('body').removeClass('fixed-sidebar');
                        $('#page-wrapper').css('padding','0');
                    }
                },
                err => {
                    this.appService.onLogOut.emit(false);
                }
            );
            this.router.navigate(['/login']);
            localStorage.removeItem('siteCentric.loginUser');
            localStorage.removeItem('siteCentric.token');
            this.toastr.clear();
            this.dialog.closeAll();
            this.loginUserSubject.next(null);
        }
        
    }

    checkLockScreenPassword(data): Observable<User> {
        const params = new HttpParams().set('password', data.password).set('token', data.token);
        return this.http.get<User>(environment.serverUrl + '/auth/password/match', { params });
    }

    private showError(errMessage) {
        this.dialog.open(InfoPopupComponent, {
            data: errMessage
        });
    }

    confirmForgot(data): Observable<GeneralResponse> {
        const params = new HttpParams().set('newPassword', data.newPassword).set('token', data.token);
        return this.http.get<GeneralResponse>(environment.serverUrl + '/auth/forgot-password/confirm', { params });
    }
}
