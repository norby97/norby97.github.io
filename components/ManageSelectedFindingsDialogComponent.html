import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormBuilder, UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { Observable, of } from 'rxjs';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { UserService } from 'src/app/services/administration/user.service';
import { SccipService } from 'src/app/services/crm_bd/sccip.service';
import { FindingsService } from 'src/app/services/regulatory/findings.service';
import { SiteService } from 'src/app/services/regulatory/site.service';

@Component({
  selector: 'app-manage-selected-findings-dialog',
  templateUrl: './manage-selected-findings-dialog.component.html',
  styleUrls: ['./manage-selected-findings-dialog.component.scss']
})
export class ManageSelectedFindingsDialogComponent implements OnInit {
  findingForm:UntypedFormGroup;
  loading = false;
  usersloading = false;
  sites;
  users;
  userscopy;
  findingcategories =[];
  findingcategoriesloading = false;
  findingcategoriescopy =[];
  @HostListener('window:keyup.esc') onKeyUp() {
    this.matdialogref.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
    constructor( public matdialogref: MatDialogRef<ManageSelectedFindingsDialogComponent>,
    private findingService: FindingsService,
    private sccipService: SccipService,
              private userService:UserService,
              private formBuilder: UntypedFormBuilder,
        private siteService:SiteService,
        private authService:AuthenticationService,
      @Inject(MAT_DIALOG_DATA) public data: any) { }
  
  ngOnInit(): void {
  if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
    this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
  } else {
     this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
      this.sites = res;
    });
  }
  this.findingForm = new UntypedFormGroup({
    assignToId: new UntypedFormControl(-1,Validators.required),
    findingCategoryId: new UntypedFormControl(-1,Validators.required),
    status: new UntypedFormControl(-1,Validators.required),
    findingNotes: new UntypedFormControl('',Validators.required),
  });
  this.getUsers({value:this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id)})
  }
 
  searcher7 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    let temp = []
    if(search == '') {
        this.findingcategories = this.findingcategoriescopy;
    } else {
        this.findingcategoriescopy.map(elem => {
            if(elem?.name?.toLowerCase().indexOf(search?.toLowerCase()) > -1) {
                temp.push(elem);
            }
        });
        this.findingcategories = temp;
    }
  
    return of(this.findingcategories)
  }

  getUsers(event) {
    this.usersloading = true;
    this.userService.getAll2({siteId:event?.value.id}).subscribe(res => {
      this.users = res;
      this.userscopy = res;
      this.usersloading = false
    });
    this.findingcategoriesloading = true;
    this.sccipService.getFindingCategories({active:true}).subscribe(res => {
      this.findingcategories = res;
      this.findingcategoriescopy = res;

      this.findingcategoriesloading = false;

    })
  }
  edit() {
    this.loading = true;
    if(this.findingForm.valid){
      const data = {
        qaFindingIds:[],
        status:this.findingForm.controls.status.value == -1 ? null :this.findingForm.controls.status.value == 'RESOLVED' ?'COMPLETED':this.findingForm.controls.status.value,
        assignToId:this.findingForm.controls.assignToId.value == -1 ? null :this.findingForm.controls.assignToId.value.id ,
        qaFindingCategoryId:this.findingForm.controls.findingCategoryId.value == -1 ? null :this.findingForm.controls.findingCategoryId.value.id,
        note:this.findingForm.controls.findingNotes.value
      }
      this.data?.findings.map((elem,i) => {if(this.data?.checks[i]) {data.qaFindingIds.push(elem.id); }})
      this.findingService.editMultipleFindings(data).subscribe(res => {
        this.matdialogref.close(true);
        this.loading = false;
      }, err => {
        this.loading = false;

      })
    } else {
      this.findingForm.markAllAsTouched();
      this.loading = false;

    }
  }

}
