export class AccrualsComponent implements OnInit {

  constructor(private dialog: MatDialog,
              private caledarService: CalendarService,
              private siteService: SiteService,
              private userPreferences:UserPreferencesService,

              }

getYear(data){
  return new Date(data).getFullYear();
}
getMonth(data){
  return new Date(data).getMonth()+1;
}

  ngOnInit(): void {
    this.filterForm = new UntypedFormGroup({
      siteId: new UntypedFormControl('', Validators.required),
      date : new UntypedFormControl( moment(JSON.parse(localStorage.getItem('scUserSettings'))?.selectedMonth) || moment())
    })
    
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.filterForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.filterForm.controls.siteId.updateValueAndValidity();
      this.filter(this.filterForm)

    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.filterForm.controls.siteId.setValue( this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id) );
        this.filterForm.controls.siteId.updateValueAndValidity();
        this.filter(this.filterForm)
      });
    } 
  }

    printPreview(){
      this.accrualsprint = !this.accrualsprint
      }

  get getkeys() {
    return Object.values(this.studiesAccruals)
  }
  toggleTableRows() {
    this.isTableExpanded = !this.isTableExpanded;
    
    this.userPreferences.setuserState('accrualsExpanded',this.isTableExpanded)
    this.dataSource.data.forEach((row: any) => {
      row.isExpanded = this.isTableExpanded;
    })
  }
  groupBy(arr, property) {
    return arr.reduce(function (memo, x) {
        if (!memo[x[property]]) { memo[x[property]] = []; }
        memo[x[property]].push(x);
        return memo;
    }, {});
}
getTotalperCol(colname) {
  let total = 0
  this.studiesAccruals.map(elem => {
    total += parseFloat(elem[`${colname}`])
  })
  return total;
}
  filter(form) {
    this.loading = true;
    this.selectedSiteName = this.filterForm.controls.siteId.value.name
    this.columnsToDisplay[0] = this.selectedSiteName
    if(this.filterForm.valid) {
      this.caledarService.getCurrentDate().subscribe(res => {
        this.current.mt = res?.mt;
        this.current.yr = res?.yr;
        this.actualDate.mt = res?.mt
        this.actualDate.yr = res?.yr
        this.userPreferences.setuserState('selectedSite',this.filterForm.controls.siteId.value)
        this.userPreferences.setuserState('selectedMonth',this.filterForm.controls.date.value,true)
        this.invoiceService.getAccruals({siteId: this.filterForm.controls.siteId.value.id, yr:moment(this.filterForm.controls.date.value).format('YYYY') , mt: moment(this.filterForm.controls.date.value).format('MM')}).subscribe(res => {
          this.accruals = res.monthAccruals;
          this.totalaccruals = res.totalAccruals;
          this.studiesAccruals = res?.studiesAccruals ?? [];
          let total ={
            budgeted:0,
            invoiced:0,
            reimb:0,
            nonreimb:0,
            outstanding:0
          }
          this.studytotalaccruals = total
          Object.values(this.studiesAccruals).map(elem => {
            total.budgeted += elem.cost;
            total.invoiced += elem.invoicedTotal;
            total.reimb += this.minus(elem.reimb, elem.invoicedReimb);
            total.nonreimb += this.minus(elem.nonReimb, elem.invoicedNonReimb);
          })
          total.outstanding = total.budgeted - total.invoiced
          this.studytotalaccruals = total;
          this.currentOutstanding.reimb = res.totalReimbursable;
          this.currentOutstanding.nonReimb = res.totalNonReimbursable;
          this.currentOutstanding.total = res.totalCost;
          this.totalaccrualsSortedWeeks = Object.values(this.groupBy(res.totalAccruals, 'finCalendarWeekId')).map(obj => obj)
            this.accrualsSortedWeeks =Object.values(this.groupBy(res.monthAccruals, 'finCalendarWeekId')).map(obj => obj)
         

          
          this.dataSource2 = new MatTableDataSource(this.totalaccrualsSortedWeeks);
          this.dataSource = new MatTableDataSource(this.accrualsSortedWeeks);
         
          this.loading = false;
  
  
        })
      }, err => {
        this.loading = false;
      })
      
    } else {
      this.filterForm.markAllAsTouched()
      this.loading = false;

    }
   
  }
  

  minus(a,b) {
    if(Number.isNaN(Math.abs(parseFloat(a ? a : 0) - parseFloat(b ? b : 0)))) {
      return 0;
    } else {
      return Math.abs(parseFloat(a ? a : 0) - parseFloat(b ? b : 0));
    }
    
  }
  
  chosenMonthHandler(event, dp, input) {
    dp.close();
    this.actualDate ={
      yr: moment(event).format('YYYY'),
      mt: moment(event).format('MM')
    }
    this.filterForm.controls.date.setValue(moment(event).format('YYYY-MM-DD'));
    this.filterForm.controls.date.updateValueAndValidity();
  }

  removeAccrual(data) {
    const ref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: `Are you sure you want to remove  ${data.eventName} + ${data.indicationName}  ?`
      }
    })
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.caledarService.hideAccrualOnCalendar(data.eventId).subscribe(res => {

          this.filter(this.filterForm);   
        })
      }
    })
  }
  
}
