import { trigger, state, style, transition, animate } from '@angular/animations';
import { Component, OnInit, ViewChild } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort, Sort } from '@angular/material/sort';
import { MatTableDataSource } from '@angular/material/table';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { RegManagerService } from 'src/app/services/regulatory/reg-manager.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { AddTrialToRegManagerDialogComponent } from '../../common/dialogs/regulatory/add-trial-to-reg-manager-dialog/add-trial-to-reg-manager-dialog.component';

@Component({
  selector: 'app-reg-manager',
  animations: [trigger('detailExpand', [
    state('collapsed', style({ height: '0px', minHeight: '0' })),
    state('expanded', style({ height: '*' })),
    transition('expanded <=> collapsed', animate('225ms cubic-bezier(0.4, 0.0, 0.2, 1)')),
  ])],
  templateUrl: './reg-manager.component.html',
  styleUrls: ['./reg-manager.component.scss']
})
export class RegManagerComponent implements OnInit {
  columnsToDisplay: string[] = ['Status', 'Indication', 'PI', 'Sponsor', 'Protocol' ,'Compound', 'CRO', 'Lead CRC', 'Backup CRC', 'Site #', 'IRB', 'Lab' , 'Action'];
  dataSource: MatTableDataSource<any>;
  expandedElement: [] | null;
  filterForm: UntypedFormGroup;
  actualPage = 0;
  selectedSite;
  skip = 0;
  limit=5000;
  @ViewChild(MatSort, { static: true }) sort: MatSort;
  sites;
  trials=[];
  trialscopy=[];
  notes=[];
  loading = false;
  notesloading = false;
  constructor(private siteService: SiteService,
              private regService: RegManagerService,
              private userPreferences: UserPreferencesService,
              private dialog: MatDialog,
              private authService: AuthenticationService) { }

  ngOnInit(): void {
    this.filterForm = new UntypedFormGroup({
      siteId: new UntypedFormControl(''),
    });
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.filterForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.selectedSite = this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id).id;
        this.filterForm.controls.siteId.updateValueAndValidity();
      this.refreshTable();
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.filterForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.selectedSite = this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id).id;
        this.filterForm.controls.siteId.updateValueAndValidity();
        this.refreshTable();
      });
    }
    // this.siteService.getAllWithoutPermissions(this.authService.loginUserValue.company.id).subscribe(res => {
    //   this.sites = res;
    //   this.filterForm.controls.siteId.setValue(this.sites[0].id);
    //   this.selectedSite = this.sites[0].id;
    //   this.filterForm.controls.siteId.updateValueAndValidity();
    //   this.refreshTable();

    // });
   
  }
  openBarcodeLabelDialog(value){
    value.studyName = value?.sponsorName + ' ' +value?.protocolName 
    const ref =  this.dialog.open(CreateBoxLabelDialogComponent, {
      minWidth: '70vw',
      panelClass:'custom-dialog-container',
      maxHeight:'87vh',
      maxWidth:'80vw',
      width:'auto',
      height:'auto',
      disableClose: true,
      data: {
        type:'REG',
        site: this.filterForm.controls.siteId.value,
        study: {id:value?.studyStartSiteId},
        value
      }
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
      }
    })
  }

  getNotes(id) {
this.notesloading = true;  
    this.notes =[]
    this.regService.getTrialNotes({includepm:true, request:id}).subscribe(res => {
      this.notes = res;
      this.notesloading = false;  
    })
  }
  onScroll() {
    this.loading=true;
    
    this.actualPage= this.actualPage+1
    const data = {
      skip: this.skip + (this.actualPage*this.limit),
      limit: this.limit,
      siteID: this.selectedSite,
      statuses: ['MOVING_FORWARD','ACTIVE','COMPLETED']

    }
    this.regService.getTrials(data).subscribe(res =>  {
      this.trials = [...this.trials, ...res];
      this.sortData(this.sort)
      this.dataSource = new MatTableDataSource(this.trials.filter(elem => elem.id));
      this.loading = false;
    }, err => {
      this.loading = false;
    });
  }

  filter(event: any) {
    this.actualPage = 0;
    this.sort.direction ='';
    this.sort.active = null;
    this.selectedSite = event.value === undefined ? null : event.value.id
    const data = {
      skip: this.skip + (this.actualPage*this.limit),
      limit: this.limit,
      siteID: event.value === undefined ? null : event.value.id,
      statuses: ['MOVING_FORWARD','ACTIVE','COMPLETED']
    }
    this.regService.getTrials(data).subscribe(res =>  {
      this.trials = res;
      this.trialscopy = res;
      this.dataSource = new MatTableDataSource(this.trials.filter(elem => elem.id));
    });
  }


  getColor(status) {
    if(status == 'ACTIVE') return 'text-green2'
      else if(status == 'MOVING_FORWARD') return 'text-lila'
        else if(status == 'ADD_ON') return 'text-yellow'
          else if(status == 'PURSUING') return 'text-kek'
          else return ''
  }

  openRegManagerDialog(study) {
    const dialogRef = this.dialog.open(RegManagerDialogComponent, {
      data: { study ,
        site:this.filterForm.controls.siteId.value.id},
      height: 'auto',
      maxHeight:'80vh',
      minWidth:'80vw',
      width: 'auto',
      disableClose: true,
      panelClass:'custom-dialog-container',
    })
    dialogRef.afterClosed().subscribe(res => {
      if(res) {
        this.refreshTable();
      }
  });
  }
  deleteFromRM(studyDetails) {
  
    const dialogRef = this.dialog.open(FeedBackDialogComponent, {
      minWidth: 'auto',
      maxHeight:'80vh',
      maxWidth:'80vw',
      width: 'auto',
      height:'auto',
      disableClose: true,
      data: {
          message: `Are you sure you want to remove ${studyDetails?.sponsorName} ${studyDetails?.protocolName} from Reg Manager?`,
      }
  });
  dialogRef.afterClosed().subscribe((res) => {
      if (res === true) {
        this.regService.removeTrialFromRM(studyDetails.id).subscribe(res => {
            this.refreshTable()
        })
        
      }
  });
   
  }

  sortData(sort: Sort) {
    const data = this.trialscopy.slice();
    if (!sort.active || sort.direction === '') {
      this.trials = data;
      return;
    }
     this.trials = data?.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'status': return this.compare(a.studyStatus?.toLowerCase(), b.studyStatus?.toLowerCase(), isAsc);
        case 'indication': return this.compare(a.indicationName?.toLowerCase(), b.indicationName?.toLowerCase(), isAsc);
        case 'protocol': return this.compare(a.protocolName?.toLowerCase(), b.protocolName?.toLowerCase(), isAsc);
        case 'pi': return this.compare(a.principalInvestigatorFullName?.toLowerCase(), b.principalInvestigatorFullName?.toLowerCase(), isAsc);
        case 'cro': return this.compare(a.croName?.toLowerCase(), b.croName?.toLowerCase(), isAsc);
        case 'leadcrc': return this.compare(a.leadCrcFullName?.toLowerCase(), b.leadCrcFullName?.toLowerCase(), isAsc);
        case 'backupcrc': return this.compare(a.backupCrcFullName?.toLowerCase(), b.backupCrcFullName?.toLowerCase(), isAsc);
        case 'sponsor': return this.compare(a.sponsorName?.toLowerCase(), b.sponsorName?.toLowerCase(), isAsc);
        case 'compound': return this.compare(a.compoundName?.toLowerCase(), b.compoundName?.toLowerCase(), isAsc);
        case 'sitenr': return this.compare(a.siteNr?.toLowerCase(), b.siteNr?.toLowerCase(), isAsc);
        case 'lab': return this.compare(a.labValueName?.toLowerCase(), b.labValueName?.toLowerCase(), isAsc);
        case 'irb': return this.compare(a.irbValueName?.toLowerCase(), b.irbValueName?.toLowerCase(), isAsc);
      }
    });
    this.dataSource = new MatTableDataSource(this.trials.filter(elem => elem.id));

  }
 
 
  compare(a: number | string | Date, b: number | string | Date, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }

  
  applyFilter(event,inputdom) {
    let temp = []  
    if(event.target.value == '') {
        this.dataSource = new MatTableDataSource(this.trialscopy.filter(elem => elem.id));
    } else {
        this.trialscopy.map(elem => {
            if(elem.sponsorName.toLowerCase().indexOf(event.target.value.toLowerCase()) > -1 || 
               elem.protocolName.toLowerCase().indexOf(event.target.value.toLowerCase()) > -1) {
                temp.push(elem);
            }
        });
        this.trials = temp;
        this.dataSource = new MatTableDataSource(this.trials.filter(elem => elem.id));
      }
}

  refreshTable() {
    this.loading = true;
    this.actualPage = 0;
    this.sort.direction ='';
    this.sort.active = null;
    this.userPreferences.setuserState('selectedSite',this.filterForm.controls.siteId.value,true)
    const data = {
      skip: this.skip + (this.actualPage*this.limit),
      limit: this.limit,
      siteID: this.selectedSite,
      statuses: ['MOVING_FORWARD','ACTIVE','COMPLETED']

    }
    this.regService.getTrials(data).subscribe(res =>  {
      this.trials = res;
      this.trialscopy = res;
      this.dataSource = new MatTableDataSource(this.trials.filter(elem => elem.id));
      this.loading = false
    });
  }

  openAddTrialDialog() {
   const dialogRef =  this.dialog.open(AddTrialToRegManagerDialogComponent, {
    data:{
      studyStartId: this.selectedSite
    },
    height: 'auto',
    maxHeight:'80vh',
    maxWidth:'120vh',
    width: 'auto',
    disableClose: true,
    panelClass:'custom-dialog-container',
    });

    dialogRef.afterClosed().subscribe(res => {
        if(res) {
          this.refreshTable();
        }
    });
  }
 

}
