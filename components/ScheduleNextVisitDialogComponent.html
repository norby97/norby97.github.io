import { NgxMatDateFormats, NgxMatDateAdapter, NGX_MAT_DATE_FORMATS } from '@angular-material-components/datetime-picker';
import { formatDate } from '@angular/common';

import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MomentDateAdapter, MAT_MOMENT_DATE_ADAPTER_OPTIONS } from '@angular/material-moment-adapter';
import { DateAdapter, MAT_DATE_LOCALE } from '@angular/material/core';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { forkJoin } from 'rxjs';

import { MY_FORMATS } from 'src/app/app.component';
import { CustomNgxDatetimeAdapter } from 'src/app/directives/ngx-mat-select/ngx-date-format';

@Component({
  selector: 'app-schedule-next-visit-dialog',
  templateUrl: './schedule-next-visit-dialog.component.html',
  styleUrls: ['./schedule-next-visit-dialog.component.scss'],
  providers:[
    { provide: NGX_MAT_DATE_FORMATS , useValue: MY_FORMATS}  
  ]
})
export class ScheduleNextVisitDialogComponent implements OnInit {
  visitForm: UntypedFormGroup;
  visits;
  miscUsers = []
  miscResources = [];
  resChecks = [];
  userChecks = [];
  smsChecks = [];
  reminders;
  reasonCats;
  confirmed = false;
  defaultTime = [new Date().getHours(), 0 , 0] 
  visitLength = 15;
  visitsloading = false;
  loading = false; 
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public utilService: UtilService,
              public dialogRef: MatDialogRef<ScheduleNextVisitDialogComponent>,
              private patientService: PatientsService,
              private miscService: MiscellaneousService,
              private siteService: SiteService,
              private dialog: MatDialog,
              private smsService: SmsService,
              private appointmentService: AppointmentService,
    @Inject(MAT_DIALOG_DATA) public data: any,      
    ) { 
      this.dpheader = DatepickerHeaderComponent;
  }



    ngOnInit(): void {
      
    
      if(!this.data?.reschedule) {
        this.visitForm = new UntypedFormGroup({
          screenDate :new UntypedFormControl('',Validators.required),
          duration :new UntypedFormControl('', Validators.required),
          notes: new UntypedFormControl(),
          reasonCat: new UntypedFormControl( ),
          visit: new UntypedFormControl(''),
    
        });
        this.visitForm.controls.visit.setValidators(Validators.required);
        this.visitForm.controls.visit.updateValueAndValidity()
      
        
      } else {
        this.visitLength = parseInt(this.data?.visit?.duration)
        this.visitForm = new UntypedFormGroup({
          screenDate :new UntypedFormControl(this.data?.patientDetails?.appointedDate,Validators.required),
          duration :new UntypedFormControl(this.data?.visit?.duration, Validators.required),
          notes: new UntypedFormControl(),
          reasonCat: new UntypedFormControl('',Validators.required),
          visit: new UntypedFormControl(this.data?.appointmentId),
    
        });  
        this.visitForm.controls.reasonCat.setValidators(Validators.required);
        this.visitForm.controls.reasonCat.updateValueAndValidity()
      }
      const r2 = this.patientService.getActiveReasonCategoriesByApptType('VISIT')
      const r3 = this.miscService.getMiscResources(this.data?.site.id)
      const r4 = this.miscService.getMiscUsers(this.data?.site.id)
      const r5 = this.smsService.findSMSRemainders()
      forkJoin([r3,r4,r2,r5]).subscribe(res => {
        this.miscResources = res[0];
        this.miscUsers = res[1];
        this.miscResources.map(elem => this.resChecks.push(false))
        this.miscUsers.map(elem => this.userChecks.push(false))
        this.reasonCats = res[2].sort((a,b)=> this.compare(a?.name?.toLowerCase() , b?.name?.toLowerCase(),true));
        this.reminders = res[3];
        if(this.data.reschedule) {
          this.miscUsers.forEach((elem,i) => {
            if(this.data?.patientDetails?.providerIds?.includes(elem.id)) {
              this.userChecks[i] = true
            } else {
              this.userChecks[i] = false
            }
            
          })
          this.miscResources.forEach((elem,i) => {
            if(this.data?.patientDetails?.providerIds?.includes(elem.id)) {
              this.resChecks[i] = true
            } else {
              this.resChecks[i] =false
            }
            
          })
        }
      });
    }

    compare(a: number | string, b: number | string, isAsc: boolean) {
      return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }


    loadVisits(event) {
      this.visitsloading = true;
      this.miscService.getNotScheduledVisits({patientStudyLinkId: this.data?.selectedPatient?.id, studyId: this.data?.study?.id, date:formatDate(event.target.value,  'yyyy-MM-dd', 'en-US')}).subscribe(res => {
        this.visits = res;
        this.visitsloading = false;
  
      }, err => {
        this.visitsloading = false;
      })    
    }

    
    loadDuration(event,type) {
      if(type) {
        this.visitLength = parseInt(event?.value)
        this.visitForm.controls.duration.setValue(parseInt(event?.value));
        this.visitForm.controls.duration.updateValueAndValidity()
      } else {
        this.visitLength = parseInt(event?.value?.length)
        this.visitForm.controls.duration.setValue(parseInt(event?.value?.length));
        this.visitForm.controls.duration.updateValueAndValidity()
      }
      
   }
  
    markAsConfirmed(status) {
      this.confirmed = status;
    }
  
    schedule(visitForm) {
      this.loading = true;
      if(visitForm.valid) {
        const data ={
          appointmentId: this.data?.reschedule == true ? this.data?.appointmentId : '',
          patientStudyLinkId: this.data?.reschedule == true ? this.data?.selectedPatient?.id : '',
          req : {
            visitId:this.visitForm.controls.visit.value.id,
            visitDateTime:formatDate(this.visitForm.controls.screenDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),
            duration: this.visitLength,
            reasonId:this.visitForm.controls.reasonCat.value,
            notes:this.visitForm.controls.notes.value,
            markAsConfirmed: this.confirmed,
            resourcesIds:[],
            userIds:[]
          }
        }
        this.miscUsers.forEach((elem,i) => {
          if(this.userChecks[i]) {
            data.req.userIds.push(elem.id);
          }
        });
        this.miscResources.forEach((elem,i) => {
          if(this.resChecks[i]) {
            data.req.resourcesIds.push(elem.id);
          }
        });

        const data2 = {
          userIds: data?.req?.userIds,
          resourceIds: data?.req?.resourcesIds,
          appointmentDate:data?.req?.visitDateTime,
          duration:data?.req?.duration,
          removeAppointmentId: this.visitForm.controls.visit.value.id,

        }
     
            this.utilService.isWithinBusinessHours(this.data?.site?.id, new Date(data?.req.visitDateTime),formatDate(data?.req?.visitDateTime,'hh:mm a','en-US'),data?.req?.duration).subscribe(result => {
              if(result) {
                this.appointmentService.checkOverlap(data2).subscribe(res => {
                  if(res.length == 0) {
                    if(this.data?.reschedule) {
                      this.appointmentService.rescheduleVisit(data).subscribe(res => {
                        this.loading = false;

                        this.dialogRef.close(true);
                      }, err => {
                        this.loading = false;
                      })
                    } else {
                      this.miscService.scheduleVisitTrack(data).subscribe(res => {
                        this.loading = false;

                        this.dialogRef.close(true);
                      }, err => {
                        this.loading = false;
                      })
                    }
                  } else {
                    const ref = this.dialog.open(OverlapDialogComponent,{
             
                      data: {
                        type:'SCREEN',
                        date:data?.req?.visitDateTime,
                          siteId:this.data?.site?.id,
                          misc: false,
                        res,
                      },
                      width: 'auto',
                      height: 'auto',
                      minWidth:'40vw',
                      panelClass:'custom-dialog-container',
                      disableClose: true,
                      maxHeight:'80vh'
                    })
                    ref.afterClosed().subscribe(res => {
                      if(res) {
                        if(this.data?.reschedule) {
                          this.appointmentService.rescheduleVisit(data).subscribe(res => {
                            this.loading = false;

                            this.dialogRef.close(true);
                          }, err => {
                            this.loading = false;
                          })
                        } else {
                          this.miscService.scheduleVisitTrack(data).subscribe(res => {
                            this.loading = false;

                            this.dialogRef.close(true);
                          }, err => {
                            this.loading = false;
                          })
                        }
                      } else {
                        this.loading = false;
                      }
                    })
                  }
                })
              } else {
                const ref12 = this.dialog.open(BusinessHoursHolidaysDialogComponent,{
                  data: {
                    siteId: this.data?.site?.id,
                    isholiday:false,
                    appointmentDate:data?.req?.visitDateTime,
                  },
                  width: 'auto',
                  height: 'auto',
                  panelClass:'custom-dialog-container',
                  disableClose: true,
                  maxHeight:'80vh',
                  minWidth:'40vw'
                })

                ref12.afterClosed().subscribe(result2 => {
                  if(result2) {
                    this.appointmentService.checkOverlap(data2).subscribe(res => {
                      if(res.length == 0) {
                        if(this.data?.reschedule) {
                          this.appointmentService.rescheduleVisit(data).subscribe(res => {
                            this.loading = false;

                            this.dialogRef.close(true);
                          }, err => {
                            this.loading = false;
                          })
                        } else {
                          this.miscService.scheduleVisitTrack(data).subscribe(res => {
                            this.loading = false;

                            this.dialogRef.close(true);
                          }, err => {
                            this.loading = false;
                          })
                        }
                      } else {
                        const ref = this.dialog.open(OverlapDialogComponent,{
             
                          data: {
                            type:'SCREEN',
                            date:data?.req?.visitDateTime,
                              siteId:this.data?.site?.id,
                              misc: false,
                            res,
                          },
                          width: 'auto',
                          height: 'auto',
                          minWidth:'40vw',
                          panelClass:'custom-dialog-container',
                          disableClose: true,
                          maxHeight:'80vh'
                        })
                        ref.afterClosed().subscribe(res => {
                          if(res) {
                            if(this.data?.reschedule) {
                              this.appointmentService.rescheduleVisit(data).subscribe(res => {
                                this.loading = false;

                                this.dialogRef.close(true);
                              }, err => {
                                this.loading = false;
                              })
                            } else {
                              this.miscService.scheduleVisitTrack(data).subscribe(res => {
                                this.loading = false;

                                this.dialogRef.close(true);
                              }, err => {
                                this.loading = false;
                              })
                            }
                          } else {
                            this.loading = false;
                          }
                        })
                      }
                    })
                  } else {
                    this.loading = false;
                  }
                })
              }
            })
             
             
      
      
      } else {
        this.loading = false;
        this.visitForm.markAllAsTouched();
      }
    }
