 columnSums() {
  return this.visits?.map((visit) => {
  var visitCosts = this.simplifiedProcedures?.map((patient) =>{
    var temp = parseFloat(patient.visits.find((pv) => pv.id == visit.id && pv.applies )?.amount || 0)
    if( temp == undefined || isNaN(temp)  || temp == null) {
      return {amount:0, overhead:0 ,visitId:visit.id, overheadapplies: false,}
    } else{ 
      if(patient.procedure.applyOverhead && this.firstFormGroup.get('overhead').value != 0) {
        return {overheadapplies: true,overhead: (temp *(parseFloat(this.firstFormGroup.get('overhead').value  == '' ? 0 :this.firstFormGroup.get('overhead').value )/100)) , amount:temp , visitId:visit.id} 
      } else {
        return {amount:temp || 0, visitId:visit.id , overhead:0, overheadapplies: false}

      }
    }
  });
  var overheadCostSum = 0;
  var visitPaymentCosts
  var temp = parseFloat(this.visitPayment.find((pv) => pv.visitId === visit.id  )?.amount || 0)
  if( temp == undefined || isNaN(temp)  || temp == null) {
    return 0
  } else{ 
  if(this.visitPaymentApplyOverhead && this.firstFormGroup.get('overhead').value  != 0) {
     visitPaymentCosts =temp 
     overheadCostSum = overheadCostSum+ temp *(parseFloat(this.firstFormGroup.get('overhead').value  == '' ? 0 :this.firstFormGroup.get('overhead').value )/100)
  } else { 
    visitPaymentCosts = temp || 0
  }
}

var visitCostSum = visitCosts.reduce((sum, amount) => (sum + (amount.amount)), 0);

  visitCostSum = visitCostSum + parseFloat(visitPaymentCosts)
  
  overheadCostSum = overheadCostSum + visitCosts.reduce((sum, amount) =>(sum + (amount.overheadapplies ? amount.overhead: 0)), 0)
  const totalCostSum = visitCostSum + overheadCostSum;
  return {
    subtotal:visitCostSum,
    overheadtotal:overheadCostSum,
    adjusted:totalCostSum
  }
});

}

  sortData(sort: Sort) {
    const data = this.visitRows.slice();
    if (!sort.active || sort.direction === '') {
        this.visitRows = data;
        return;
    }
    this.visitRows = data.sort((a, b) => {
        const isAsc = sort.direction === 'asc';
        switch (sort.active) {
            case 'name': return this.compare(a.name?.toLowerCase(), b.name?.toLowerCase(), isAsc);
            default: return 0;
        }
    });
}

compare(a: number | string, b: number | string, isAsc: boolean) {
  return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
}

refreshTotals(status, i, i2=0) {
  if(status != undefined) {
    if(status == true) {
      this.procedureTotals.forEach((procedure,index) => {
        if(index == i) {
          procedure.subtotal = 0;
          procedure.adjusted = 0;
          procedure.overheadtotal = 0;
        }
      });
      this.procedures?.forEach((elem) => {
        elem.visits.forEach((element,index) => {
          if(this.visitPayment[index]?.amount == 0) {
            if(index == i && elem.visits[index].applies== true)  {
              this.procedureTotals[i].subtotal += parseFloat(elem.cost);
              if(elem.details.applyOverhead ) {
                this.procedureTotals[i].overheadtotal += parseFloat(this.percentage(parseFloat(this.firstFormGroup.get('overhead').value), elem.cost));  
              }
              this.procedureTotals[i].adjusted = parseFloat(this.procedureTotals[i].subtotal) + parseFloat(this.procedureTotals[i].overheadtotal);
            } 
          } else {
            this.procedureTotals[index].subtotal = parseFloat(this.visitPayment[index]?.amount);
            this.procedureTotals[index].adjusted = parseFloat(this.procedureTotals[index].subtotal) + parseFloat(this.procedureTotals[index].overheadtotal);
          }
         
        });
      });
    } else {
      this.procedures?.forEach((elem, prodind) => {
        elem.visits.forEach((element,index) => {
          if(index == i && prodind == i2 && elem.visits[index].applies== false && this.visitPayment[index]?.amount == 0) {
            this.procedureTotals[i].subtotal -= parseFloat(elem.cost);
            if(elem.details.applyOverhead ) {
              this.procedureTotals[i].overheadtotal -= parseFloat(this.percentage(parseFloat(this.firstFormGroup.get('overhead').value), elem.cost));  
            }
            this.procedureTotals[i].adjusted = parseFloat(this.procedureTotals[i].subtotal) + parseFloat(this.procedureTotals[i].overheadtotal);

      });
    }
   
   
  } else {
    this.procedureTotals.forEach(procedure => {
      procedure.subtotal = 0;
      procedure.adjusted = 0;
      procedure.overheadtotal = 0;
    });
    
    this.procedures?.forEach((elem, ind) => {

        if(this.visitPayment[index].amount == 0) {
          if(elem.cost != 0 ) {
            if(elem.visits[index].applies== true ) {
              if(elem.details.applyOverhead ) {
                this.procedureTotals[index].overheadtotal += parseFloat(this.percentage(parseFloat(this.firstFormGroup.get('overhead').value), elem.cost));  
              }
              this.procedureTotals[index].subtotal += parseFloat(elem.cost);

            }

            this.procedureTotals[index].adjusted = parseFloat(this.procedureTotals[index].subtotal) + parseFloat(this.procedureTotals[index].overheadtotal);
        }
            });
        });
       
      }
    this.grandtotal = 0;

    })
  }
