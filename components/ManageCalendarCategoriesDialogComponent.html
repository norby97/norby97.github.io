import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { CalendarService } from 'src/app/services/advertising/calendar.service';
import { SccipService } from 'src/app/services/crm_bd/sccip.service';

@Component({
  selector: 'app-manage-calendar-categories-dialog',
  templateUrl: './manage-calendar-categories-dialog.component.html',
  styleUrls: ['./manage-calendar-categories-dialog.component.scss']
})
export class ManageCalendarCategoriesDialogComponent implements OnInit {
  loading = false;
  categories;
  categoriescopy;
  active = true;
  categoryForm: FormGroup;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(@Inject(MAT_DIALOG_DATA) public data: any,
  public dialogRef: MatDialogRef<ManageCalendarCategoriesDialogComponent>,
  private caledarService: CalendarService,
  private sccipService: SccipService,

  ) { }

  ngOnInit(): void {
  this.categoryForm = new FormGroup({
      name: new FormControl('', Validators.required),
      search: new FormControl('')
    })
    this.getCategories()
  }

  getCategories() {
    this.caledarService.getAdCategories({siteId:this.data?.site?.id, active:null}).subscribe(res => {
      this.categories = res;
      this.categoriescopy = res;
    })
  }
 

  addCategory() {
    if(this.categoryForm.valid) {
      this.caledarService.createAdCategory({name: this.categoryForm.controls.name.value,siteId:this.data?.site?.id}).subscribe(res => {
        this.getCategories();
        this.categoryForm.controls.name.reset('')
        this.categoryForm.controls.search.reset('')
      })
    } else {
      this.categoryForm.markAllAsTouched()
    }
  }
  applyFilter(event) {
    let temp = []
    if(event.target.value == '') {
        this.categories = this.categoriescopy;
    } else {
        this.categoriescopy.map(elem => {
            if(elem?.name?.toLowerCase().indexOf(event.target.value.toLowerCase()) > -1 ) {
                temp.push(elem);
            }
        });
        this.categories = temp;
    }
}
  toggleStatus(event, category) {
    if(event) {
      this.sccipService.makeAdCategoryForSiteActive({siteId:this.data?.site.id,categoryId:category.categoryId}).subscribe(res => {
        this.getCategories()
      })
    } else {
      this.sccipService.makeAdCategoryForSiteInactive({siteId:this.data?.site.id,categoryId:category.categoryId}).subscribe(res => {
        this.getCategories()
      })
    }
  }

}
