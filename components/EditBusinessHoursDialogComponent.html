  convert12To24(time: string): string {
    if(time == null) {
      return null
    }
    const [timeStr, period] = time.split(' ');
    const [hourStr, minuteStr, secondStr] = timeStr.split(':');
    let hour = parseInt(hourStr, 10);
    const minute = parseInt(minuteStr, 10);
    const second = parseInt(secondStr || '00', 10);
    if (period === 'PM' && hour < 12) {
      hour += 12;
    } else if (period === 'AM' && hour === 12) {
      hour = 0;
    }
    const hourStr24 = hour < 10 ? '0' + hour : hour;
    return `${hourStr24}:${minute < 10 ? '0' + minute : minute}`;
  }

    swapdates(day,time1: string, time2: string, time3: string, time4: string) {
      const format = 'h:mm A';
      let isAscending = true;
      if(time2 == null && time3 == null && time4 == null && time1 == null) return false;
    
      if(time1 == null && time2 == null && time3 == null) return true;
      if(time1 == null && time2 == null && time4 == null) return true;
      if(time1 == null && time3 == null && time4 == null) return true;
      if(time2 == null && time1 == null && time3 == null) return true;
      if(time2 == null && time1 == null && time4 == null) return true;
      if(time2 == null && time3 == null && time4 == null) return true;
      if(time1 != null && time4 == null) return true;
      if(time2 != null && time3 == null) return true;
      const times = [time1, time2, time3, time4].map(time => moment(time, format));
    
      for (let i = 0; i < times.length - 1; i++) {
        if (times[i].isAfter(times[i + 1])) {
          isAscending = false
          break;
        }
      }
  
    return !isAscending;
    }
