  refreshNotifications(){
    this.loading = true;
    const data ={
        skip:this.skip*this.limit,
        limit:this.limit
    }
    this.userService.getUserNotificationsToShow(data).subscribe(res =>{
        if(res.length==0){
            this.nomore = true;
        }
        if(this.skip == 0) {
            this.notifications = res;
            this.notifications.forEach(elem => {
                    elem.createdDate = elem.createdDate.replace('T',' ') + ' UTC'
                    elem.createdDate = elem.createdDate.toString()
                    
                    
            })
            this.loading = false;
           
        }else {
            this.notifications = [...this.notifications, ...res];
            this.loading = false;
            this.notifications.forEach(elem => {
                elem.createdDate = elem.createdDate.replace('T',' ') + ' UTC'
                elem.createdDate = elem.createdDate.toString()
                                    })

        }
            
        this.skip++
    } ,err => {
        this.loading = false;

    })
        
}
