import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormGroup, UntypedFormBuilder, UntypedFormControl, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { forkJoin, Observable, of } from 'rxjs';
import { startWith, map } from 'rxjs/operators';
import { Costs } from 'src/app/models/advertising/costs.model';
import { AdsService } from 'src/app/services/advertising/ads.service';
import { BudgetService } from 'src/app/services/advertising/budget.service';
import { CalendarService } from 'src/app/services/advertising/calendar.service';
import { SccipService } from 'src/app/services/crm_bd/sccip.service';
import { AddMediaEventDialogComponent } from '../add-media-event-dialog/add-media-event-dialog.component';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';
@Component({
  selector: 'app-edit-event-dialog',
  templateUrl: './edit-event-dialog.component.html',
  styleUrls: ['./edit-event-dialog.component.scss']
})
export class EditEventDialogComponent implements OnInit {
  addMediaEventForm: UntypedFormGroup;
  addMediaEventForm2: UntypedFormGroup;
  firstStep = true;
  vendors=[];
  fileName = 'No file uploaded!';
  file2Name = 'No file uploaded!';
  fileToUpload;
  totalCost;;
  hideCategoryStep2 = false;
  file2ToUpload;

  categoriescopy;

  studyReimbBudget;

  availableBudget = 1000;
  weeks;
  studySelected =false;
  studies;
  costs: Costs[] = []
  secondStep = false;

  detailsloading

  actualAdID;

  filteredIndications;
  actualDetails;
  actualDetails2; 
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(private calendarService: CalendarService,
              private sccip: SccipService,
              private budgeService: BudgetService,
              public dialogRef: MatDialogRef<AddMediaEventDialogComponent>,
              private formBuilder: UntypedFormBuilder,
              private adService: AdsService,
              @Inject(MAT_DIALOG_DATA) public data,) {  
                this.dpheader = DatepickerHeaderComponent;

               }

  ngOnInit(): void {
  
    this.actualDetails = this.data.actualDetails;
    this.actualDetails2 = this.data.actualDetails2
    const categories = this.calendarService.getAdCategories({siteId:this.data?.data?.site?.id ? this.data?.data?.site?.id : this.data?.data?.site?.siteId,active:true});
    const studies = this.calendarService.getStudies(this.data?.data?.site?.id ? this.data?.data?.site?.id : this.data?.data?.site?.siteId, true);

 
    forkJoin([categories, studies]).subscribe(res => {
      this.categories = res[0];
      this.categoriescopy = res[0];
      this.studies = res[1];
    })

    this.addMediaEventForm2 = new UntypedFormGroup({
      reimbPortion: new UntypedFormControl(''),
      adId: new UntypedFormControl('',),
      scheduleFile: new UntypedFormControl(''),
      keywords: new UntypedFormControl(''),
      adFile: new UntypedFormControl(''),
    })
    this.addMediaEventForm = new UntypedFormGroup({
      eventName: new UntypedFormControl('', Validators.required),
      vendorId: new UntypedFormControl(this.data?.actualDetails?.vendor,  Validators.required),
      studyId: new UntypedFormControl(this.data?.actualDetails?.study?.id == null ? -1 :this.data?.actualDetails?.study?.id ),
      indication:new UntypedFormControl(this.data?.actualDetails?.indication?.id),
      category: new UntypedFormControl(this.data?.actualDetails?.category?.id),
      cost: new UntypedFormControl('', Validators.required),
      dateFrom: new UntypedFormControl('', Validators.required ),
      dateTo: new UntypedFormControl('', Validators.required )
    })
    
    this.setStudy({value:this.data?.actualDetails?.study?.id})
    this.data.actualDetails2.medium =this.data?.actualDetails2.mediumType 
    this.setValidators({value:this.data?.actualDetails2})

    }
    
    searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
      let temp = []
      if(search == '') {
          this.categories = this.categoriescopy;
      } else {
          this.categoriescopy.map(elem => {
              if(elem.name.toLowerCase().indexOf(search.toLowerCase()) > -1 ) {
                  temp.push(elem);
              }
          });
          this.categories = temp;
      }
    
      return of(this.categories)
    }
    searcher = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
      const data = {
        search,
        pageNumber: pageNumber-1,
        pageSize,
      }
      if(pageNumber - 1 != 0) {
        this.sccip.getindIcationsPaginated(data).subscribe(res => {
          res.content.forEach(element => {
              this.indications.push(element);
          });
        })
      } else {
        this.sccip.getindIcationsPaginated(data).subscribe(res => {
          this.indications = res.content;
          this.indications.push({id:this.actualDetails?.indication?.id, name:this.actualDetails?.indication?.name, ngxSelected:1,active:true})
        })
      }
      
      return of(this.indications)
    }

    private _filter(value: string, items: Array<any>) {
      const filterValue = value?.toLowerCase();
      return items?.filter(item => {
          return item.name?.toLowerCase().indexOf(filterValue) === 0;
      });
    } 


    searcher3 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
      const data = {
        search,
        active:true,
        limit:pageSize,
        skip: 0 + ((pageNumber-1)*pageSize),
        siteId:  this.data?.data?.site?.id
      }
      if(pageNumber-1 != 0) {
        this.calendarService.getVendors(data).subscribe(res => {
          this.vendors = [...this.vendors, ...res];
  
        })
      } else {
       this.calendarService.getVendors(data).subscribe(res => { 
          this.vendors =  []
          this.vendors = [...res]
          this.vendors.push({id:this.data?.actualDetails?.vendor?.id, name:this.data?.actualDetails?.vendor?.name})
        
        })
      }
      return of(this.vendors)
    }

  addEvent(form){
    if(this.addMediaEventForm.valid) {
      const dt = new Date(this.addMediaEventForm.controls.dateTo.value, );
        const dateto = Date.UTC(dt.getFullYear(), dt.getMonth(), dt.getDate());
        const df = new Date(this.addMediaEventForm.controls.dateFrom.value, );
        const datefrom = Date.UTC(df.getFullYear(), df.getMonth(), df.getDate());

      const data = {
        id:this.data.data.arg.event.extendedProps.adCalendarId,
        request : {
        dateFrom: new Date( datefrom),
        dateTo:  new Date( dateto),
        }
      }
      this.totalCost = this.addMediaEventForm.controls.cost.value
      this.calendarService.getMediaEventEditDetails2(data).subscribe(res => {
        this.weeks = res.weeks;
        this.actualAdID = res?.ad?.id;
        this.costs = [];
        this.weeks.forEach(element => {
          this.costs.push({nonreimb:element.nonReimb, id: element.finCalendarWeekId, adBudgetWeeklyId: element.adBudgetWeeklyId, reimb:element.reimb, total: element.cost});       
        });
        this.reimbursableCost = this.getSum(1);
        this.nonReimbCost = this.getSum(2);
        this.firstStep = false;
        this.secondStep = true;
        this.getReimb();
        
        if(this.addMediaEventForm.get('studyId').value != -1 && this.addMediaEventForm.get('studyId').value != '-1'  && this.addMediaEventForm.get('studyId').value != '') {
          this.hideCategoryStep2 = true;
          this.getByStudy(this.addMediaEventForm)
        } else {
          this.hideCategoryStep2 = false
          this.nonReimbCost = this.totalCost;
        }
      })
    } else {
      this.addMediaEventForm.markAllAsTouched();
    }
    
  }

  getByStudy(form) {
      this.studyID = this.addMediaEventForm.controls.studyId.value;
      this.adService.getMediaAds(this.studyID).subscribe(res => {
        this.ads = res;
      })
    
  }

  updateCosts() {
    this.reimbursableCost = this.getSum(1);
    this.nonReimbCost = this.getSum(2);
  }

  addMediaEvent(form) {
    this.loading = true;
   
    if(this.addMediaEventForm2.valid) {
      
      if(this.getSum(1) == this.reimbursableCost && this.getSum(2) == this.nonReimbCost && this.getSum(3) == this.totalCost) {
        const dt = new Date(this.addMediaEventForm.controls.dateTo.value, );
        const dateto = Date.UTC(dt.getFullYear(), dt.getMonth(), dt.getDate());
        const df = new Date(this.addMediaEventForm.controls.dateFrom.value, );
        const datefrom = Date.UTC(df.getFullYear(), df.getMonth(), df.getDate());

        const data = {
          calendarID: this.data.data.arg.event.extendedProps.adCalendarId,
          request: {
            eventName: this.addMediaEventForm.controls.eventName.value, 
            vendorId: this.addMediaEventForm.controls.vendorId.value.id, 
            studyId: this.addMediaEventForm.controls.studyId.value == -1 ? null :this.addMediaEventForm.controls.studyId.value  , 
            indicationId: this.addMediaEventForm.get('indication').value, 
            totalCost: parseFloat(this.totalCost)?.toFixed(2), 
            nonReimbursable: parseFloat(this.nonReimbCost)?.toFixed(2),
            reimbursable:parseFloat(this.reimbursableCost)?.toFixed(2),
            adId: this.actualDetails.study.id == -1 ? this.actualAdID : this.addMediaEventForm2.controls.adId.value, 
            adCategoryId: this.addMediaEventForm.get('category').value, 
            keywords: [], 
            dateTo: new Date( dateto),
            dateFrom: new Date( datefrom),
            siteId: this.data?.data?.site?.id ?  this.data?.data?.site?.id : this.data?.data?.site?.siteId,
            weeksBudget: []
          },
      
          adFile: this.actualDetails.study.id == -1 ? null : this.fileToUpload,
          schedFile:  this.actualDetails.study.id == -1 ? null :  this.file2ToUpload
        }
        this.costs.map((elem,index) => { data.request.weeksBudget.push({weekId: elem.id, adBudgetWeeklyId: elem.adBudgetWeeklyId, dateFrom: this.weeks[index]?.dateFrom, dateTo: this.weeks[index]?.dateTo, reimbursable: elem.reimb, nonReimbursable: elem.nonreimb, totalCost: (parseFloat(elem.nonreimb)+ parseFloat(elem.reimb))?.toFixed(2) });})
        this.calendarService.editMediaEventCalendar(data).subscribe(res => {

          this.dialogRef.close(true);
        },err => {
          this.loading =false;
        })
      } else {
        this.loading = false;
        this.calendarService.showError("Provided costs don't add up!")
      }
    } else {
      this.loading = false;
      this.addMediaEventForm2.markAllAsTouched();
    }
  }
  setValidators(event) {
      if( this.actualDetails.study.name =='General Ad' ){
        this.addMediaEventForm2.get('adId').setValidators(null)
        this.addMediaEventForm2.get('adId').updateValueAndValidity();
      } else {
        if((event.value.medium == 'CENTRAL_CAMPAIGN'  ||event.value.medium =='COMMUNITY_OUTREACH' || event.value.medium=='DATABASE')){
          this.addMediaEventForm2.get('adId').setValidators(null)
          this.addMediaEventForm2.get('adId').updateValueAndValidity();
        } else {

          this.addMediaEventForm2.get('adId').setValidators(Validators.required)
          this.addMediaEventForm2.get('adId').updateValueAndValidity();
        }
      }
      
    }
  setStudy(event) {
    if(event.value != null && event.value != 'null'&& event.value != -1   ) {
      this.studySelected = true;
      this.studyID = event.value
   
      this.addMediaEventForm.get('category').setValidators(null);
      this.addMediaEventForm.get('category').updateValueAndValidity();
      if(event.value.medium == 'CENTRAL_CAMPAIGN'  ||event.value.medium =='COMMUNITY_OUTREACH' || event.value.medium=='DATABASE'){
        this.addMediaEventForm2.get('adId').setValidators(null)
        this.addMediaEventForm2.get('adId').updateValueAndValidity();
      } else {
        this.addMediaEventForm2.get('adId').setValidators(Validators.required)
        this.addMediaEventForm2.get('adId').updateValueAndValidity();
      }
      this.studyReimbBudget
      this.addMediaEventForm.get('indication').setValidators(null)
      this.addMediaEventForm.get('indication').updateValueAndValidity();
      this.getReimb();
    } else {
      this.studyID = null;
      this.actualDetails.study.id = -1
      this.actualDetails.study.name = 'General Ad'
      this.studySelected = false;
      this.addMediaEventForm2.get('adId').setValidators(null)
      this.addMediaEventForm2.get('adId').updateValueAndValidity();
      this.addMediaEventForm.get('category').setValidators(Validators.required);
      this.addMediaEventForm.get('category').updateValueAndValidity();
      this.addMediaEventForm.get('indication').setValidators(Validators.required)
      this.addMediaEventForm.get('indication').updateValueAndValidity();
    }
    
  }

  getReimb() {
    if(this.studyID != null && this.studyID != 'null' && this.studyID != -1) {
      this.budgeService.getReimbursable(this.studyID).subscribe(res => {
        this.studyReimbBudget = res;
      })
    } else {
      this.studyReimbBudget = 0;
      this.reimbursableCost = 0;
      this.costs.map(elem => elem.reimb = 0)
    }
    
  }

  splitEvenlyPreview() {
    this.costs.forEach((elem, index)=> {
      if(index == this.costs.length+1){
        const allButLastReimb = parseFloat(this.reimbursableCost.toString()) - (this.costs.length-1 * parseFloat(this.costs[0].reimb))
        const allButLastNonReimb = this.nonReimbCost - (this.costs.length-1 * parseFloat(this.costs[0].nonreimb))
        allButLastReimb == 0 ? elem.reimb = (parseFloat(this.reimbursableCost.toString()) / parseFloat(this.costs.length.toString())).toString()  :  elem.reimb = allButLastReimb.toFixed(2).toString()
        allButLastNonReimb == 0 ? elem.reimb = (parseFloat(this.nonReimbCost.toString()) / parseFloat(this.costs.length.toString())).toString() : elem.nonreimb = allButLastNonReimb.toFixed(2).toString();
      } else {
        elem.reimb = (parseFloat(this.reimbursableCost.toFixed(2).toString()) / parseFloat((this.costs.length ).toFixed(2).toString()))
        elem.nonreimb = (parseFloat(this.nonReimbCost.toFixed(2).toString()) / parseFloat((this.costs.length).toFixed(2).toString()))
      }
    });
  }

  removesplitEvenlyPreview() {
    this.costs.forEach(elem => elem = {nonreimb:'0', id: elem.id, reimb:'0', total:'0' })
  }

  getSum(type) {
    let sum = 0;
    if(type == 1) {
      this.costs.forEach(element => {
        sum += parseFloat(element.reimb)
      });
      return parseFloat(sum.toFixed(2));
    }
    if(type == 2) {
      this.costs.forEach(element => {
        sum += parseFloat(element.nonreimb)
      });
      return parseFloat(sum.toFixed(2));
    }
    if(type == 3) {
      this.costs.forEach(element => {
        sum += parseFloat(element.reimb) + parseFloat(element.nonreimb)
      });
      return parseFloat(sum.toFixed(2));
    }
  }

  get2sum(i){    
    this.costs[i].total = (parseFloat(this.costs[i].nonreimb)  + parseFloat(this.costs[i].reimb)).toFixed(2);
    return  parseFloat((parseFloat(this.costs[i].nonreimb)  + parseFloat(this.costs[i].reimb)).toFixed(2));
    
  }

  getMinus(event) {
    this.nonReimbCost = this.totalCost - parseFloat(event.target.value)
    this.reimbursableCost =  this.totalCost - this.nonReimbCost;
  }
