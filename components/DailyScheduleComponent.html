setBusinessHours(newSite=false) {

  if(newSite) {
    this.siteService.getSiteBusinessHours(this.siteForm.controls.siteID.value.id).subscribe(res => {
      this.savedBusinessHours = res;
      this.currentSiteBusinessHours = res;
      this.savedBusinessHours?.map(element => {
        if(element.startTime != null && element.endTime != null){

          if(element.day == 'MONDAY') {
            this.calendarBusinessHours.push({daysOfWeek:[1],startTime:element.startTime, endTime:element.startBreakTime})
            this.calendarBusinessHours.push({daysOfWeek:[1],startTime:element.endBreakTime, endTime:element.endTime})
          }
           if(element.day == 'TUESDAY') {
            this.calendarBusinessHours.push({daysOfWeek:[2],startTime:element.startTime, endTime:element.startBreakTime})
            this.calendarBusinessHours.push({daysOfWeek:[2],startTime:element.endBreakTime, endTime:element.endTime})
          }
           if(element.day == 'WEDNESDAY') {
            this.calendarBusinessHours.push({daysOfWeek:[3],startTime:element.startTime, endTime:element.startBreakTime})
            this.calendarBusinessHours.push({daysOfWeek:[3],startTime:element.endBreakTime, endTime:element.endTime})
          }
           if(element.day == 'THURSDAY') {
            this.calendarBusinessHours.push({daysOfWeek:[4],startTime:element.startTime, endTime:element.startBreakTime})
            this.calendarBusinessHours.push({daysOfWeek:[4],startTime:element.endBreakTime, endTime:element.endTime})
          }
           if(element.day == 'FRIDAY') {
            this.calendarBusinessHours.push({daysOfWeek:[5],startTime:element.startTime, endTime:element.startBreakTime})
            this.calendarBusinessHours.push({daysOfWeek:[5],startTime:element.endBreakTime, endTime:element.endTime})
          }
           if(element.day == 'SATURDAY') {
            this.calendarBusinessHours.push({daysOfWeek:[6],startTime:element.startTime, endTime:element.startBreakTime})
            this.calendarBusinessHours.push({daysOfWeek:[6],startTime:element.endBreakTime, endTime:element.endTime})
          }
           if(element.day == 'SUNDAY') {
            this.calendarBusinessHours.push({daysOfWeek:[7],startTime:element.startTime, endTime:element.startBreakTime})
            this.calendarBusinessHours.push({daysOfWeek:[7],startTime:element.endBreakTime, endTime:element.endTime})
          }
        }
              
      });
        
        
      const timeSlotMap: { [key: string]: any } = {};

      for (const entry of this.calendarBusinessHours) {
        const key = `${entry.startTime}-${entry.endTime}`;
        if (!timeSlotMap[key]) {
          timeSlotMap[key] = {
            daysOfWeek: [...entry.daysOfWeek],
            startTime: entry.startTime,
            endTime: entry.endTime
          };
        } else {
          timeSlotMap[key].daysOfWeek.push(...entry.daysOfWeek);
        }
      }

      const unifiedTimeSlots = Object.values(timeSlotMap).map(value => ({
        daysOfWeek: [...new Set(value.daysOfWeek)],
        startTime: value.startTime,
        endTime: value.endTime
      }));
      this.calendarBusinessHours2 = unifiedTimeSlots

      this.siteService.getSiteHolidays(this.siteForm.controls.siteID.value.id).subscribe(res=> {
        this.currentSiteHolidays = res;
        res?.map(elem => {
          if(moment(elem.date).isSame(moment(this.currentDay),'day')) {
              this.isHolidayToday = true;
              this.holiday = elem;
            
          }
        })

      })
      this.calendarOptions.businessHours = this.calendarBusinessHours2;
      this.ref.detectChanges() 
    })
  } else {
    this.calendarOptions.businessHours = this.calendarBusinessHours2;
    this.ref.detectChanges()  
  }
}

  }
