import { Component, Input, EventEmitter, OnDestroy, OnInit, Output } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { FeedBackDialogComponent } from '../feed-back-dialog/feed-back-dialog.component';

@Component({
  selector: 'app-countdown',
  templateUrl: './countdown.component.html',
  styleUrls: ['./countdown.component.scss']
})
export class CountdownComponent implements OnInit, OnDestroy {
  @Input() seconds: any;
  @Output() deleteClicked: EventEmitter<any> = new EventEmitter();
  @Input() name: any;
  secondsobj;
  minute;
  second;
  hour;
  interval;
  constructor(private dialog:MatDialog) { }
  ngOnDestroy(): void {
    if(this.interval) {
      clearInterval(this.interval)
     }
  }
  showTime : string;
  ngOnInit() {
    this.calculateRemaining(this.seconds?.timerStartAt,this.seconds?.minutes )
    //setInterval(this.convertTime, 10000,this.seconds);
  }

  calculateRemaining(start,duration) {
    var x = new Date(start)
    var y = new Date()
    let seconds = Math.round(Math.abs(x.getTime() - y.getTime())/1000);
    let durationSeconds = duration*60;
    if(durationSeconds < seconds) {
        this.secondsobj= 0;
        this.convertTime();
    } else {
        this.secondsobj =  durationSeconds - seconds;
        this.convertTime();
    }
}

  convertTime(){
    if(this.secondsobj > 0) {
      if(this.interval) {
        clearInterval(this.interval)
       }
      this.interval = setInterval(() => {
        if(this.secondsobj <= 0) {
          this.timeout();
          clearInterval(this.interval)
        }
        var seconds = this.secondsobj;
        let dayTime = (60*60*24)
        var day = Math.floor(seconds/dayTime)
        let rmainingTime = (seconds-(dayTime*day));
        let hoursTime = (60*60);
        this.hour = Math.floor(rmainingTime/(hoursTime))
        rmainingTime = (rmainingTime-(hoursTime*this.hour));
        let minTime = 60;
        this.minute = Math.floor(rmainingTime/minTime)
        rmainingTime = (rmainingTime-(minTime*this.minute));
        this.second = Math.floor(rmainingTime)
        this.secondsobj--;
       
      }, 1000);
    } else {
      this.hour =0;
      this.minute = 0;
      this.second = 0;
      this.name = this.name + ' (EXPIRED)'
      if(this.interval) {
        clearInterval(this.interval)
       }
    }
  
  }

  timeout() {
    this.hour =0;
    this.minute = 0;
    this.second = 0;
    this.name = this.name + ' (EXPIRED)'
    this.secondsobj = null;
  }


  deleteTimer() {
    const ref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to delete ' + this.name + '?'
      }
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
       if(this.interval) {
        clearInterval(this.interval);
       }
       this.deleteClicked.emit(1)

      }
    });
  }

}
