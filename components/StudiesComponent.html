  studies
  studiesCopy

      this.filterForm.controls.siteId.updateValueAndValidity();

       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.filterForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.filterForm.controls.siteId.updateValueAndValidity();
      });
    }
    this.filterStudies()
  }

  sortData(sort: Sort) {
    const data = this.studies?.content;
    if (!sort.active || sort.direction === '') {
      this.studiesCopy = data;
      this.studies.content = this.studiesCopy;
      return;
    }
     this.studiesCopy = data?.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'study': return this.compare(a?.studyName?.toLowerCase(), b?.studyName?.toLowerCase(), isAsc);
        case 'builddate': return this.compare(new Date(a?.studyBuiltDate).getTime(), new Date( b?.studyBuiltDate).getTime(), isAsc);
        case 'buildby': return this.compare(a?.buildBy == null ? '' : a?.buildBy, b?.buildBy == null ? '' :b?.buildBy, isAsc);
        case 'site': return this.compare(a?.siteValue?.toLowerCase(), b?.siteValue?.toLowerCase(), isAsc);
        case 'revnr': return this.compare(a?.revisionNumber?.toLowerCase(), b?.revisionNumber?.toLowerCase(), isAsc);
        default: return 0;
      }
    });
    this.studies.content = this.studiesCopy;
    this.dataSource = new MatTableDataSource(this.studies.content);

}


compare(a: number | string, b: number | string, isAsc: boolean) {
  return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
}

filterStudies() {
  this.actualPage = 0;
  this.loading = true;    
  if(this.filterForm.controls.siteId.value.id != null) {
    this.userPreferences.setuserState('selectedSite',this.filterForm.controls.siteId.value,true)

  }
  const data = {
    pageNumber: 0,
   pageSize: 30,
    active:this.filterForm.controls.active.value, 
    siteId: this.filterForm.controls.siteId.value.id,
    studyName:this.filterForm.controls.findStudySponsor.value,
    isBuilt: this.filterForm.controls.isbuilt.value,
  }
  this.studyBuildService.getAll(data).subscribe(res => {
    this.studies = res;
    this.dataSource = new MatTableDataSource(res.content);
   
    this.loading = false;
  }, err => {
    this.loading = false;
  })
}

getNextPage(index, size) {
  this.actualPage = index;
  const data = {
    pageNumber: index,
    pageSize: size,
    active:this.filterForm.controls.active.value, 
    siteId: this.filterForm.controls.siteId.value.id,
    studyName:this.filterForm.controls.findStudySponsor.value,
    isBuilt: this.filterForm.controls.isbuilt.value,
  }
  this.studyBuildService.getAll(data).subscribe(res => {
    this.studies = res;
    this.dataSource = new MatTableDataSource(res.content);
    

  })
}

}
