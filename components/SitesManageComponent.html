export class SitesManageComponent implements OnInit {
    @Input() companyId: string;
    displayedColumns: string[] = ['name', 'active', 'actions'];
    dataSource: MatTableDataSource<Site>;
    @ViewChild(MatSort, {static: true}) sort: MatSort;


    constructor(
        private siteService: SiteService,
        public utilService: UtilService,
        public dialog: MatDialog
    ) {
    }

    ngOnInit(): void {
        this.siteService.getCompanySites(this.companyId).subscribe(res => {
            this.dataSource = new MatTableDataSource(res);
            this.dataSource.sort = this.sort;
            this.dataSource.sortingDataAccessor = (item, property) => {
                switch (property) {
                    case 'name':
                        return item.name;
                    default:
                        return item[property];
                }
            };
        });
    }
    openBusinessHoursDialog(site:Site){
        const dialogRef = this.dialog.open(EditBusinessHoursDialogComponent, {
            height: 'auto',
            maxHeight:'80vh',
            minWidth:'80vw',
            width: 'auto',
            disableClose: true,
            panelClass:'custom-dialog-container',
            data: {
                site,
            }
        });
        dialogRef.afterClosed().subscribe(() => {
            this.ngOnInit();
        });
    }
    openHolidaysDialog(site:Site){
        const dialogRef = this.dialog.open(AddHolidayDialogComponent, {
            height: 'auto',
            maxHeight:'80vh',
            minWidth:'40vw',
            width: 'auto',
            disableClose: true,
            panelClass:'custom-dialog-container',
            data: {
                site,
            }
        });
        dialogRef.afterClosed().subscribe(() => {
            this.ngOnInit();
        });
    }
    openSiteDialog(site: Site, edit: boolean) {
        site.company = {
            id: this.companyId
        };
        const dialogRef = this.dialog.open(SiteDialogComponent, {
            height: 'auto',
            maxHeight:'80vh',
    maxWidth:'50vw',
            width: 'auto',
            disableClose: true,
            panelClass:'custom-dialog-container',
            data: {
                site,
                edit
            }
        });
        dialogRef.afterClosed().subscribe(() => {
            this.ngOnInit();
        });
    }

    delete(siteId: string) {
        this.siteService.delete(siteId).subscribe(
            res => {
                if (res.responseStatus !== 'SUCCESS') {
                    this.siteService.showError(res.responseMessage);
                } else {

                    this.ngOnInit();
                }
            }
        );
    }
    changeStatus(site){
        this.siteService.changeSiteStatus(site.id).subscribe(res => {
            this.ngOnInit()
        }, err =>{
            this.siteService.showError(err.error);
        })
    }

    applyFilter(event: Event) {
        const filterValue = (event.target as HTMLInputElement).value;
        this.dataSource.filter = filterValue.trim().toLowerCase();
      
    }
}
