  search(form) {
    this.loading= true;
    this.searched =true;
    const data = {
     request: {
      skip:0,
      limit:this.limit,
      compoundIds: this.queryForm.controls.compound.value,
      croIds: this.queryForm.controls.cro.value,
      indicationIds: this.queryForm.controls.diagnosis.value,
      populationIds: this.queryForm.controls.population.value,
      principalInvestigatorIds: this.queryForm.controls.pi.value,
      protocolIds:this.queryForm.controls.protocol.value,

      bdIndication:this.queryForm.controls.bdIndication.value,
      phaseIds: this.queryForm.controls.phase.value,
      sponsorIds:this.queryForm.controls.sponsor.value,
      statuses: this.queryForm.controls.status.value,
      trialTypes: this.queryForm.controls.type.value,
      years:this.queryForm.controls.years.value,
      type: this.actualStatus =='LAST_NOTES' ? 'INCLUDE_NOTES' : 'INCLUDE_PROTOCOL_TITLE'
     }
    }
    this.currentSearchType =  this.actualStatus =='LAST_NOTES'? 'INCLUDE_NOTES' : 'INCLUDE_PROTOCOL_TITLE'
 
    this.studyStartService.filter(data).subscribe(res => {
      this.studies = res;
      this.studiesCopy =res;
      this.dataSource = new MatTableDataSource(res);
      if(this.currentSearchType=='INCLUDE_NOTES'){
        this.isTableExpanded = false;
        this.dataSource.data.forEach((row: any) => {
          row.isExpanded = this.isTableExpanded;
        })
      }
      this.loading= false;

    },err => {
    this.loading= false;

    });
    if(this.authService.loginUserValue.modules[4].permissions[4].permission === 'EDIT') {
      this.studyStartService.getFilter(data).subscribe(res => {
        this.actualReport = res;

      }, err => {
    this.loading= false;;

      })
    }

  }

  sortData(sort: Sort) {
    const data = this.studies;
    if (!sort.active || sort.direction === '') {
      this.studiesCopy = data;
      this.studies = this.studiesCopy;
      return;
    }
     this.studiesCopy = data?.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'study': return this.compare(a.studyName?.toLowerCase(), b?.studyName?.toLowerCase(), isAsc);
        case 'site': return this.compare(a.siteValueAlias?.toLowerCase(), b?.siteValueAlias?.toLowerCase(), isAsc);
        case 'protocol': return this.compare(a.protocol?.toLowerCase(), b?.protocol?.toLowerCase(), isAsc);
        case 'idnumber': return this.compare(a.idNumber, b?.idNumber, isAsc);
        case 'ind': return this.compare(parseFloat(a?.indNumber == null ? 0 : a?.indNumber), parseFloat(b?.indNumber== null ? 0 : b?.indNumber), isAsc);
        case 'phase': return this.compare(a.phase?.toLowerCase(), b?.phase?.toLowerCase(), isAsc);
        case 'type': return this.compare(a.studyType?.toLowerCase(), b?.studyType?.toLowerCase(), isAsc);
        case 'cro': return this.compare(a.croName?.toLowerCase(), b?.croName?.toLowerCase(), isAsc);
        case 'pi': return this.compare(a.principalInvestigatorFullName?.toLowerCase(), b?.principalInvestigatorFullName?.toLowerCase(), isAsc);
        case 'indication': return this.compare(a.indicationName?.toLowerCase(), b?.indicationName?.toLowerCase(), isAsc);
        case 'protocol': return this.compare(a.protocolName?.toLowerCase(), b?.protocolName?.toLowerCase(), isAsc);
        case 'status': return this.compare(this.utilService.getMasterStatus(a.studyStatus),this.utilService.getMasterStatus(b.studyStatus), isAsc);
        case 'fpv': return this.compare(new Date(a.startDate),new Date(b.startDate), isAsc);
        case 'sponsor': return this.compare(a.sponsorName?.toLowerCase(), b?.sponsorName?.toLowerCase(), isAsc);
        case 'compound': return this.compare(a.compoundName?.toLowerCase(), b?.compoundName?.toLowerCase(), isAsc);
        default: return 0;
      }
    });
    this.studies = this.studiesCopy;
    this.dataSource = new MatTableDataSource(this.studies);

  }
  getColor(status) {
    if(status === 'ACTIVE') return 'text-green2'
      else if(status === 'MOVING_FORWARD') return 'text-warning'
        else if(status === 'ADD_ON') return 'text-kek'
          else if(status === 'PURSUING') return 'text-lila'
          else if(status === 'DELETED') return 'text-danger'
          else if(status === 'SPONSOR_CLOSED') return 'text-danger'
          else if(status === 'COMPLETED') return 'text-green'
          else return ''
  }
  compare(a: number | string | Date, b: number | string | Date, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
