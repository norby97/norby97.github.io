@Component({
  selector: 'app-add-patient-to-study-final-step-dialog',

export class AddPatientToStudyFinalStepDialogComponent implements OnInit {
  addPatientForm: UntypedFormGroup;
  prescreensYesForm: UntypedFormGroup;
  prescreensNoForm: UntypedFormGroup;
  users;
  indications;
  allowsms = {
    one: false,
    two: false
  }
  sites;
  guardian = false;
  formsubmitted = false;
  schedulePreScreen = false;
  confirms = {
    sms: false,
    email: false
  }
  communityPartners;
  reasonCats;
  miscUsers = []
  miscResources = [];
  resChecks = [];
  userChecks = [];
  defaultTime = [new Date().getHours(), 0 , 0] 
  loading = false;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;

  constructor(public dialogRef: MatDialogRef<AddPatientDialogComponent>,
              private appointmentService: AppointmentService,
              private patientService: PatientsService,
              private authService: AuthenticationService,
              private dialog:MatDialog,
              private siteService: SiteService,
              public utilService: UtilService,    
              public userService: UserService,
              private trackService: TrackingService,
              private miscService:MiscellaneousService,
              private sccipService: SccipService ,
                 @Inject(MAT_DIALOG_DATA) public data: any,) { }

  ngOnInit(): void {
    this.addPatientForm = new UntypedFormGroup({
      siteId: new UntypedFormControl(this.data?.siteId),
      dob: new UntypedFormControl('',Validators.required),
      gender: new UntypedFormControl('',Validators.required ),
      genderyouid: new UntypedFormControl(  ),
      racialg: new UntypedFormControl( ),
      email: new UntypedFormControl( this.data?.schedule == true ? this.data?.patientDetails?.email : '' ),
      guardian: new UntypedFormControl( ),
      sfirstname: new UntypedFormControl( '',Validators.required ),
      slastname: new UntypedFormControl('',Validators.required ),
      phone1: new UntypedFormControl(  ),
      phoneType1: new UntypedFormControl( ),
      phone1sms: new UntypedFormControl(''),
    });

    this.userService.getAll2({siteId:this.data?.siteID}).subscribe(res => {
      this.users = res;
    });
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.addPatientForm.controls.siteId.setValue(this.data?.siteId);
      this.addPatientForm.controls.siteId.updateValueAndValidity();
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.addPatientForm.controls.siteId.setValue(this.data?.siteId);
      this.addPatientForm.controls.siteId.updateValueAndValidity();
      });
    }
    this.sccipService.getindIcations().subscribe(res => {
      this.indications = res;
    });
   
    const r3 = this.miscService.getMiscResources(this.data?.siteId)
    const r4 = this.miscService.getMiscUsers(this.data?.siteId)
    forkJoin([r3,r4]).subscribe(res => {
      this.miscResources = res[0];
      this.miscUsers = res[1];
    });
    
    
    this.prescreensNoForm = new UntypedFormGroup({
      subjectnr: new UntypedFormControl(''),
      alias: new UntypedFormControl('',Validators.required),
      reasonCat: new UntypedFormControl(''),
      followUpDate: new UntypedFormControl(''),
      screenDate :new UntypedFormControl('',Validators.required),
      study :new UntypedFormControl('',),
      duration :new UntypedFormControl('',Validators.required),
      notes: new UntypedFormControl(),
      refnotes: new UntypedFormControl(),
    });
      this.prescreensNoForm.controls.alias.setValue(this.data?.patientDetails?.firstName == null ? '' :this.data?.patientDetails?.firstName[0]  +( this.data?.patientDetails?.middleName == null || this.data?.patientDetails?.middleName == '' ? '-' : this.data?.patientDetails?.middleName ) + this.data?.patientDetails?.lastName == null ? '' : this.data?.patientDetails?.lastName[0]);
      this.addPatientForm.controls.sfirstname.setValue(this.data?.patientNames?.firstName == null ? null : this.data?.patientNames?.firstName);
      this.addPatientForm.controls.slastname.setValue(this.data?.patientNames?.lastName == null ? null :this.data?.patientNames?.lastName );
      this.addPatientForm.controls.slastname.updateValueAndValidity();
      this.addPatientForm.controls.sfirstname.updateValueAndValidity();
      this.prescreensNoForm.controls.alias.updateValueAndValidity();
      
  }


  toogle(num) {
    if(num ==1) {
      this.allowsms.one = !this.allowsms.one;
    } else if (num ==2) {
      this.allowsms.two = !this.allowsms.two;
    } else {
      this.guardian = !this.guardian
    }

    
  }
  add() {
  this.loading = true;
  if(this.data?.schedule) {
      if(this.prescreensNoForm.valid && this.addPatientForm.valid) {
        const data  = {
          studyId: this.data?.study.id,
          patientId: this.data?.patientDetails.id,
          req: {
            duration: this.prescreensNoForm.controls.duration.value,
            alias: this.prescreensNoForm.controls.alias.value,
            screenNumber:this.prescreensNoForm.controls.subjectnr.value,
            resourcesIds: [],
            screenDateTime: formatDate(this.prescreensNoForm.controls.screenDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),
            userIds: []
          }
        }
        this.miscResources.forEach((elem,i) => {
              if(this.resChecks[i]) {
                data.req.resourcesIds.push(elem.id);
              }
            });
        this.miscUsers.forEach((elem,i) => {
          if(this.userChecks[i]) {
            data.req.userIds.push(elem.id);
          }
        });
        this.trackService.existingPatientAndAddToStudy(data).subscribe(res => {

  this.loading = false;
  this.dialogRef.close(true);
        }, err => {
          this.loading = false;
        })
      } else {
  this.loading = false;
  this.addPatientForm.markAllAsTouched();
        this.prescreensNoForm.markAllAsTouched();
      }
    } else {
      if(this.prescreensNoForm.valid && this.addPatientForm.valid) {
        const data  = {
          studyId: this.data?.study.id,
          req: {
            duration: this.prescreensNoForm.controls.duration.value,
            alias: this.prescreensNoForm.controls.alias.value,
            screenNumber:this.prescreensNoForm.controls.subjectnr.value,
            resourcesIds: [],
            screenDateTime: formatDate(this.prescreensNoForm.controls.screenDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),
            userIds: [],
            birthDate: this.addPatientForm.controls.dob.value,
            email: this.addPatientForm.controls.email.value,
            firstName:  this.addPatientForm.controls.sfirstname.value,
            gender: this.addPatientForm.controls.gender.value,
            genderIdentity: this.addPatientForm.controls.genderyouid.value,
            lastName: this.addPatientForm.controls.slastname.value,
            phoneDescriptionOne: '',
            phoneNumberOne: this.addPatientForm.controls.phone1.value,
            phoneTypeOne: this.addPatientForm.controls.phoneType1.value,
            race: this.addPatientForm.controls.racialg.value,
            siteId: this.data?.siteId
            
          }
        }
        this.miscResources.forEach((elem,i) => {
              if(this.resChecks[i]) {
                data.req.resourcesIds.push(elem.id);
              }
            });
        this.miscUsers.forEach((elem,i) => {
          if(this.userChecks[i]) {
            data.req.userIds.push(elem.id);
          }
        });



            this.utilService.isWithinBusinessHours(this.data?.siteId, new Date(this.prescreensNoForm.controls.screenDate.value),formatDate(this.prescreensNoForm.controls.screenDate.value,'hh:mm a','en-US'),30).subscribe(result => {
              if(result) {

                this.trackService.createNewPatientAndAddToStudy(data).subscribe(res => {

                  this.loading = false;
                  this.dialogRef.close(true);
                  }, err => {
                    this.loading = false;
                  })
              } else {
                const ref12 = this.dialog.open(BusinessHoursHolidaysDialogComponent,{
                  data: {
                    siteId: this.data?.siteId,
                    isholiday:false,
                    appointmentDate:this.prescreensNoForm.controls.screenDate.value
                  },
                  width: 'auto',
                  height: 'auto',
                  panelClass:'custom-dialog-container',
                  disableClose: true,
                  maxHeight:'80vh',
                  minWidth:'40vw'
                })

                ref12.afterClosed().subscribe(result2 => {
                  if(result2) {
                    this.trackService.createNewPatientAndAddToStudy(data).subscribe(res => {

                      this.loading = false;
                      this.dialogRef.close(true);
                      }, err => {
                        this.loading = false;
                      })
                  } else {
                    this.loading = false;
                  }
                })
              }
            })
                
             
     
      } else {  
        this.loading = false;
        this.addPatientForm.markAllAsTouched();
        this.prescreensNoForm.markAllAsTouched();
      }
    }
  }
}
