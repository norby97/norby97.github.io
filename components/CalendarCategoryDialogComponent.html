import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormBuilder, UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { map, startWith } from 'rxjs/operators';
import { StudyBudgetsComponent } from 'src/app/components/advertising/study-budgets/study-budgets.component';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { CalendarService } from 'src/app/services/advertising/calendar.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { FeedBackDialogComponent } from '../../utils/feed-back-dialog/feed-back-dialog.component';
import { CategoryStudyBudgetDialogComponent } from '../category-study-budget-dialog/category-study-budget-dialog.component';
import { Observable, of } from 'rxjs';
@Component({
  selector: 'app-calendar-category-dialog',
  templateUrl: './calendar-category-dialog.component.html',
  styleUrls: ['./calendar-category-dialog.component.scss']
})
export class CalendarCategoryDialogComponent implements OnInit {
  sites; 
  notes;
  loading = false;
  includepm = true;
  noteForm: UntypedFormGroup;
  projectMNote = false;
  filteredCategories 
  editCategoryForm: UntypedFormGroup;
  categories;
  categoriescopy;
  categoryDetails;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<CalendarCategoryDialogComponent>,
              private calendarService: CalendarService,
              private dialog: MatDialog,
              private formBuilder: UntypedFormBuilder,
              private authService: AuthenticationService,
              private siteService: SiteService,
              @Inject(MAT_DIALOG_DATA) public data,) { }

  ngOnInit(): void {
    this.noteForm = new UntypedFormGroup({
      note: new UntypedFormControl('', Validators.required),
      sites: new UntypedFormControl([]),
    });
    this.calendarService.getCalendarCategoryDetails({studyStartSiteId: this.data?.id}).subscribe(res => {
        this.categoryDetails = res;
    })
    this.editCategoryForm = this.formBuilder.group({
      category: new UntypedFormControl('', Validators.required),
    });
    this.calendarService.getAdCategories({siteId:this.data.siteId, active:null}).subscribe(res => {
      this.categories = res;
      this.categoriescopy = res;
      
    })
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
      });
    }
   
    this.getNotes(this.data.id)
   
  }
  private _filter(value: string, items: Array<any>) {
    const filterValue = value?.toLowerCase();
    return items?.filter(item => {
        return item.name?.toLowerCase().indexOf(filterValue) === 0;
    });
  } 
  getNotes(id) {
    const data = {
      id ,
      includepm:this.includepm
    }
    this.calendarService.getNotesInCategoryDialog(data).subscribe(res => {
      this.notes = res
      
    });
  }

  editCategory(editCategoryForm) {
    this.loading = true;
    if(this.editCategoryForm.valid) {
      const data = {
        studyID:this.data.id,
        request: {
          adCategoryId: this.editCategoryForm.get('category').value
        }
    }
    this.calendarService.updateCategoryDialog(data).subscribe(res => {

      this.loading= false;
      this.dialogRef.close(true);
    })
    } else {
      this.loading= false;
      this.editCategoryForm.markAllAsTouched();

    }
    
  }

  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    let temp = []
    if(search == '') {
        this.categories = this.categoriescopy;
    } else {
        this.categoriescopy.map(elem => {
            if(elem.name.toLowerCase().indexOf(search.toLowerCase()) > -1 ) {
                temp.push(elem);
            }
        });
        this.categories = temp;
    }
      
    return of(this.categories)
  }

 removeStudyFromCalendar() {
  const dialogRef = this.dialog.open(FeedBackDialogComponent, {
    minWidth: 'auto',
    maxHeight:'80vh',
    maxWidth:'80vw',
    width: 'auto',
    height:'auto',
    disableClose: true,
    data: {
        message: 'Are you sure you want to remove ' + this.data?.props?.name + ' from calendar?',
    }
});
    dialogRef.afterClosed().subscribe((res) => {
        if (res === true) {
          this.calendarService.removeStudyFromCalendar({id: this.data.id}).subscribe(res => {

            this.dialogRef.close(true);
          })
        } 
    });
 
 }

  toggle() {
    this.projectMNote = !this.projectMNote;
  }

  openCalendarCategoryStudyBudget() {
    const dialog = this.dialog.open(CategoryStudyBudgetDialogComponent, {
      width: '70%',
      height: 'auto',
      panelClass:'custom-dialog-container',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw',
      data: {
        studyID: this.data.id,
        studyName: this.data?.props.name
      }
      
    });
    dialog.afterClosed().subscribe(res => {
      if(res) {
      }
    })  }

  sendNote(form) {
    if(this.noteForm.valid) {
      const data = {
        studyID: this.data.id,
        request: {
          note: this.noteForm.controls.note.value,
          projectManagementNote: this.projectMNote,
          sites: this.noteForm.controls.sites.value,
          startupNote: false
        }

      }
    
      this.calendarService.sendCategoryNote(data).subscribe(res => {
        this.getNotes(this.data.id)
        this.noteForm.reset();
        this.noteForm.controls.sites.setValue([]);
        this.noteForm.controls.sites.updateValueAndValidity();
      })
    } else {
      this.noteForm.markAllAsTouched();
    }
  }
}
