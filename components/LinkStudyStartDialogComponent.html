import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { StudyStartsService } from 'src/app/services/crm_bd/study-starts.service';

@Component({
  selector: 'app-link-study-start-dialog',
  templateUrl: './link-study-start-dialog.component.html',
  styleUrls: ['./link-study-start-dialog.component.scss']
})
export class LinkStudyStartDialogComponent implements OnInit {
  sites: any[] = [];
  linkForm: UntypedFormGroup;
  loading = true;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogref.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(@Inject(MAT_DIALOG_DATA) public data: any,
              public dialogref: MatDialogRef<LinkStudyStartDialogComponent>,
              private studyStartService: StudyStartsService) { }

  ngOnInit(): void {
    this.linkForm = new UntypedFormGroup({
      site: new UntypedFormControl('', Validators.required)
    })
    this.studyStartService.getAllForLinking(this.data.studyStartID).subscribe(res => {
      this.sites = res;
      this.loading = false;
    }, err => {
      this.loading = false;
    });
  }

  linkStudy(form) {
    this.loading =true;
    if(this.linkForm.valid) {
      const data ={
        studyStartID: this.data.studyStartID,
        studyStartSiteID: this.linkForm.controls.site.value,
      }
      this.studyStartService.linkStudy(data).subscribe(res => {

        this.dialogref.close(true);
        this.loading =false;
      });
    } else {
      this.loading =false;
      this.linkForm.markAllAsTouched();
    }
  }

}
