import { AfterViewInit, Component, DoCheck, EventEmitter, Input, OnChanges, OnInit, Output, SimpleChanges } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { PatientsService } from 'src/app/services/ctms/patients.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { ManagePatientInterestDialogComponent } from '../dialogs/ctms/manage-patient-interest-dialog/manage-patient-interest-dialog.component';
import { Sort } from '@angular/material/sort';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { MargePatientDialogComponent } from '../dialogs/ctms/marge-patient-dialog/marge-patient-dialog.component';
import { PatientMiniViewDialogComponent } from '../dialogs/ctms/patient-mini-view-dialog/patient-mini-view-dialog.component';

@Component({
  selector: 'app-patient-search',
  templateUrl: './patient-search.component.html',
  styleUrls: ['./patient-search.component.scss']
})
export class PatientSearchComponent implements OnInit, OnChanges {
  @Input() results;
  @Input() resultscopy;
  @Input() loading = true;
  @Input() searchedText;
  patientDetails;
  patientNotes= []; 
  @Output() redirected: EventEmitter<any> = new EventEmitter();
  sites;
  patientChecks = [];
  filterbySitevalue = ''
  constructor(private patientService: PatientsService,
              private dialog:MatDialog,
              private authService: AuthenticationService,
              private siteService: SiteService,
              public utilService: UtilService) { }
  ngOnChanges(changes: SimpleChanges): void {
    if(changes?.searchedText != undefined || changes?.searchedText != null) {
      this.loading = true;
    }
    if(changes?.searchedText == undefined || changes?.searchedText == null) {
      this.loading = false;
    }
  }
 

  applyFilter(event) {
    let temp = []
    if(event.target.value == '') {
        this.resultscopy = this.results;
    } else {
        this.results?.map(elem => {
            if(elem.fullName.toLowerCase().indexOf(event.target.value.toLowerCase()) > -1 || elem.lastNote.toLowerCase().indexOf(event.target.value.toLowerCase()) > -1 || elem.email.toLowerCase().indexOf(event.target.value.toLowerCase()) > -1  ) {
                temp.push(elem);
            }
        });
        this.resultscopy = temp;
    }
}
  ngOnInit(): void {
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
      });
  } 

}

openPatientMiniViewDialog(patient) {

if(!patient?.doNotContact) {
  patient.patientId = patient.id
  const ref = this.dialog.open(PatientMiniViewDialogComponent, {
    width: 'auto',
    maxHeight:'80vh',
  maxWidth:'80vw',
    height:'auto',
    panelClass:'custom-dialog-container',
    disableClose: true,
    data: {
      dontshowfullprofilebutton:true,
      fromTracking:false,
      patient:patient,
      site: {id:patient?.siteId}
    }
  });
  
} else {
  this.siteService.showError('Patient marked as DO NOT CONTACT.')
}
  
}
filterbySite(id) {
  let temp = []
  if(id == null) {
    this.results = this.resultscopy;
    this.filterbySitevalue = '';

} else {
  this.filterbySitevalue = id;

    this.resultscopy.map(elem => {
        if(elem.siteId.toLowerCase().indexOf(id.toLowerCase()) > -1 ) {
            temp.push(elem);
        }
    });
    this.results = temp;
    this.patientChecks = []
    this.results.map(elem => {
      this.patientChecks.push(false);
    })
}
}

openMergePatienDialog() {
  const ref = this.dialog.open(MargePatientDialogComponent,  {
    width: 'auto',
    height: 'auto',
    disableClose: true,
    maxHeight:'90vh',
    minHeight:'80vh',
    minWidth:'80vw',
  panelClass:'custom-dialog-container',
  maxWidth:'90vw',
    data: {
      checks:this.patientChecks,
      patients:this.results
    },
  });
  ref.afterClosed().subscribe(res => {
    if(res) {
      this.filterbySitevalue = ''
      this.loading = true
      this.patientService.searchPatient({text:this.searchedText}).subscribe(res => {
        this.results = res;
        this.resultscopy = res;
        this.patientChecks = []
        this.results.map(elem => {
          this.patientChecks.push(false);
        })
        this.loading = false;
      }, err => {
        this.loading = false;
      });  
    }
    
  }) 
}

  getPatientDetails(id){
    this.patientService.findPatientByID(id).subscribe(res => {
      this.patientDetails = res;
    })
    this.patientService.getLast5Notes(id).subscribe(res=> {
      this.patientNotes=res;
      this.patientNotes.forEach(elem => {
        if (elem?.noteDate) {
          elem.noteDate = elem?.noteDate?.replace('T',' ') + ' UTC'
          elem.noteDate = elem?.noteDate?.toString()       
         }      
      })
    })
  }
  openManageInterestDialog(patient) {
    const ref = this.dialog.open(ManagePatientInterestDialogComponent,  {
      width: 'auto',
      height: 'auto',
      disableClose: true,
      maxHeight:'80vh',
    panelClass:'custom-dialog-container',
    maxWidth:'80vw',
      data: {
        patientDetails: patient
      },
    });
    ref.afterClosed().subscribe(res => {
      this.patientService.findPatientByID(patient.patientId).subscribe(res => {
        this.patientDetails = res;
      })
    }) 
  }
  compare(a: number | string | Date, b: number | string | Date, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
  
  sortData(sort: Sort) {
     this.resultscopy = [...this.results]
    const data = this.results;
    if (!sort.active || sort.direction === '') {
      this.results = data;
      return;
    }
    this.results = data.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'last': return this.compare(new Date(a?.lastActivity).getTime(), new Date(b?.lastActivity).getTime(), isAsc);
        case 'site': return this.compare(a?.siteAlias?.toLowerCase(), b?.siteAlias?.toLowerCase(), isAsc);
        case 'name': return this.compare(a?.fullName.toLowerCase(), b?.fullName?.toLowerCase(), isAsc);

      }
    });
  }
}
