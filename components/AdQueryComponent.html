  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
  

  sortData(sort: Sort) {
    const data = this.resultsCopy
    if (!sort.active || sort.direction === '') {
        this.results = data;
        return;
    }
    this.resultsCopy = data.sort((a, b) => {
        const isAsc = sort.direction === 'asc';
        switch (sort.active) {
            case 'Total Cost  (Reimb. | NonReimb.)': return this.compare(a?.totalCost,b?.totalCost, isAsc);
            case 'Dates Running/Media Event':  return this.compare(new Date(a.dateFrom).getTime(),new Date(b.dateFrom).getTime(), isAsc);
            case 'Study/Indication': return this.compare(a.studyName?.toLowerCase(), b.studyName?.toLowerCase(), isAsc);
            case '$ Adv. Return': return this.compare(a.advertisingReturn, b.advertisingReturn, isAsc);
            case 'RAND/$ per RAND.': return this.compare(a.randomized, b.randomized, isAsc);
            case 'Calls/$ per call': return this.compare(a.calls, b.calls, isAsc);
            case 'PS/$ per PS': return this.compare(a.ps, b.ps, isAsc);
            case 'S.SCR/$ per S.SCR': return this.compare(a.scheduledScr, b.scheduledScr, isAsc);
            case 'SCR/$ per SCR': return this.compare(a.completedScr, b.completedScr, isAsc);

            default: return 0;
        }
    });
    this.results =this.resultsCopy
}
