  sortData(sort: Sort) {
    const data = this.leads;
    if (!sort.active || sort.direction === '') {
      this.leadsCopy = data;
      this.leads = this.leadsCopy;
      return;
    }
     this.leadsCopy = data?.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'lastaction': return this.compare(new Date(a?.actionDate) , new Date(b?.actionDate), isAsc);
        case 'followup': return this.compare(new Date(a?.followUpDate), new Date(b?.followUpDate), isAsc);
        case 'indication': return this.compare(a?.indication?.toLowerCase(), b?.indication?.toLowerCase(), isAsc);
        case 'status': return this.compare(a?.idNumber, b?.idNumber, isAsc);
        case 'cro': return this.compare(a?.cro?.toLowerCase(), b?.cro?.toLowerCase(), isAsc);
        case 'owner': return this.compare(a?.owner?.toLowerCase(), b?.owner?.toLowerCase(), isAsc);
        case 'sponsor': return this.compare(a?.sponsor?.toLowerCase(), b?.sponsor?.toLowerCase(), isAsc);
        case 'compound': return this.compare(a?.compound?.toLowerCase(), b?.compound?.toLowerCase(), isAsc);
        case 'sites': return this.compare(a?.sites?.length, b?.sites?.length, isAsc);
        case 'phase': return this.compare(a?.phase?.toLowerCase(), b?.phase?.toLowerCase(), isAsc);
        default: return 0;
      }
    });
    this.leads = this.leadsCopy;
    this.dataSource = new MatTableDataSource(this.leads);

  }
  
  compare(a: number | string | Date, b: number | string | Date, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
