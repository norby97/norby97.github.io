import { AfterViewInit, ChangeDetectorRef, Component, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { Observable, of } from 'rxjs';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { TrackedVisitsService } from 'src/app/services/finance/tracked-visits.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { AdvertisingReportService } from 'src/app/services/reports/advertising-report.service';

@Component({
  selector: 'app-label-generator',
  templateUrl: './label-generator.component.html',
  styleUrls: ['./label-generator.component.scss']
})
export class LabelGeneratorComponent implements OnInit,AfterViewInit {
  labelForm:UntypedFormGroup;
  sites;
  studies = []
  studiescopy =[]
  studiesloading = false
  patients = []
  patientscopy = []
  patientsloading = false;
  loading = false;
  actualPage = 0;
  lastsearch = ''
  generated=false;
  data: {
    type,
    siteId,
    studyID,
    patientStudyLinkId
  }={
    type:'PAT',
    siteId:'',
    studyID:'',
    patientStudyLinkId:''
  }
  selectedType = ''
  constructor(private siteService:SiteService,
              private cdr: ChangeDetectorRef,
              private trackedVisitsService:TrackedVisitsService,
              private authService:AuthenticationService,
              private advertisingService:AdvertisingReportService) { }
  ngAfterViewInit(): void {
    this.cdr.detectChanges();
  }

  ngOnInit(): void {
    this.labelForm = new UntypedFormGroup({
      site:new UntypedFormControl('',Validators.required),
      type:new UntypedFormControl('SOURCE',Validators.required),
      study:new UntypedFormControl('',Validators.required),
      patient:new UntypedFormControl('',Validators.required),
    });
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.labelForm.controls.site.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.labelForm.controls.site.updateValueAndValidity();
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.labelForm.controls.site.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.labelForm.controls.site.updateValueAndValidity();
      });
    }
  }


  setValidators(event){
    this.generated = false;
    this.selectedType = event.value
    if(event.value =='SOURCE') {
      this.data.type='PAT';
      this.labelForm.controls.patient.setValue('');
      this.labelForm.controls.patient.setValidators(Validators.required);
      this.labelForm.controls.patient.updateValueAndValidity();
    } else {
      if(event.value =='CONTRACTS'){
        this.data.type = 'CON'
      } else {
        this.data.type = 'REG'
      }
      this.labelForm.controls.patient.setValidators(null);
      this.labelForm.controls.patient.setValue('');
      this.labelForm.controls.patient.updateValueAndValidity();
    }
  }

  searcher3 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    let temp = []
    if(search == '') {
        this.patients = this.patientscopy;
    } else {
        this.patientscopy.map(elem => {
            if(elem?.alias?.toLowerCase().indexOf(search.toLowerCase()) > -1  || elem?.subjectNumber?.toLowerCase().indexOf(search.toLowerCase()) > -1) {
                temp.push(elem);
            }
        });
        this.patients = temp;
    }
    
    return of(this.patients)
  }

  generate(){
    if(this.labelForm.valid) {
      this.data.siteId = this.labelForm.controls.site.value.id
      this.data.studyID = this.labelForm.controls.study.value.id
      this.data.patientStudyLinkId = this.labelForm.controls.patient.value.id
      this.generated=true;
    } else {
      this.labelForm.markAllAsTouched()
    }
  }

  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
  this.studiesloading = true;
  this.generated=false;
    if(this.lastsearch != search || pageNumber == 1) {
      this.actualPage = 0
    }
    const data = {
      search,
      pageNumber: this.actualPage,
      pageSize:30,
      siteId:   this.labelForm.controls.site.value.id,
      trialType: '',
      dateFrom:  '' ,
      dateTo: '', 
    }
    if(pageNumber - 1 != 0) {
      this.advertisingService.getStudiesLastActivityPaginated(data).subscribe(res => {
        this.studies =[...this.studies,...res]
        this.studiesloading = false;
      })
    } else {
      this.generated = false;
      this.advertisingService.getStudiesLastActivityPaginated(data).subscribe(res => {
        this.studies =[...res]
        this.studiesloading = false;
      })
    }
    this.actualPage++;
    return of(this.studies)
  }

  getPatients(){
    this.patientsloading = true;
    this.trackedVisitsService.getPatients(this.labelForm.controls.study.value.id).subscribe(res => {
      this.patients = res;
      this.patientscopy = res;
      this.patientsloading = false;
    } ,err => {
      this.patientsloading = false;

    })
  }

}
