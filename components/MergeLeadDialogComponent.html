import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormGroup, UntypedFormControl, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { UserService } from 'src/app/services/administration/user.service';
import { LeadsService } from 'src/app/services/crm_bd/leads.service';
import { Lead } from 'src/app/models/crmbd/lead.model';

@Component({
  selector: 'app-merge-lead-dialog',
  templateUrl: './merge-lead-dialog.component.html',
  styleUrls: ['./merge-lead-dialog.component.scss']
})
export class MergeLeadDialogComponent implements OnInit {
  leads:Lead[] = [];
  mergeForm;
  actualPage = 0
  skip = 0;
  leadloading =false;
  limit = 30
  sourcenotes= []
  targetNotes = [];
  loading = false;
  currentID = null;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(@Inject(MAT_DIALOG_DATA) public data: any,
              private leadService: LeadsService,
              private userService: UserService,
              public dialogRef: MatDialogRef<MergeLeadDialogComponent>,
              ) { }

  ngOnInit(): void {
    this.mergeForm = new UntypedFormGroup({
      leadToMerge: new UntypedFormControl(this.data.lead.id, Validators.required),
      leadIdNumber: new UntypedFormControl('', Validators.required),
      code:new UntypedFormControl('', Validators.required),
      targetLead: new UntypedFormControl('', Validators.required),
    });
    this.currentID = this.makeid(6);
    this.leadService.getLeadNote( this.data.lead.id).subscribe(res => {
      this.sourcenotes= res;
    })
   
  }


  makeid(length) {
    var result           = '';
    var characters       = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
    var charactersLength = characters.length;
    for ( var i = 0; i < length; i++ ) {
        result += characters.charAt(Math.floor(Math.random() * charactersLength));
    }
    return result;
}
  applyFilter(event) {
    this.leadloading = true;
    this.actualPage = 0;
   const data = {
      skip: this.skip + (this.actualPage*this.limit),
       limit: this.limit,
      sponsorIds: '',
      ownerIds: '',
      croIds:'',
      indicationIds: '',
      phaseIds: '',
      compoundIds: '',
      statuses: '',
      hasFollowUp: false,
      siteIds: '',
      leadIdNumber: this.mergeForm.controls.leadIdNumber.value,
    }
    this.leadService.getAll(data).subscribe(res => {
      this.leads = res;
      if(this.leads.length ==1) {
        this.mergeForm.controls.targetLead.setValue(this.leads[0].id);
        this.leadService.getLeadNote(this.leads[0].id).subscribe(res => {
          this.targetNotes= res;
        }, err => {
          this.leadloading = false;
        })
      }
      this.leadloading = false;
    });
    
  }


  merge(form) {
    this.loading = true;
    if(this.mergeForm.valid ) {
      if(this.currentID == this.mergeForm.controls.code.value) {
        const data = {
          from : this.data.lead.id,
          to: this.mergeForm.get('targetLead').value
        }
        this.leadService.mergeLead(data).subscribe(res => {
          this.loading = false;
          this.dialogRef.close(true);
        }, err => {
          this.loading = false;
        })
      } else {
        this.userService.showError('Wrong confirmation code!');
        this.currentID = this.makeid(6);
        this.mergeForm.controls.code.reset('');
        this.loading = false;

      }
      
    } else {
      this.mergeForm.markAllAsTouched();
    }
  }

  loadNotes(event) {
    this.leadService.getLeadNote( event.target.value).subscribe(res => {
      this.targetNotes= res;
    })
  }
}
