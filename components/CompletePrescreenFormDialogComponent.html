  ngOnInit(): void {


    const data = {
      search:'',
      pageNumber: 0,
     pageSize: 30,
      siteId:   this.data?.site,
      statuses: ['MOVING_FORWARD','ADD_ON','ACTIVE']
    }
    const r1 = this.userService.getAll2({siteId:this.data?.site})
    const r2 = this.siteService.getStudies(data)
    const r3 = this.patientService.getReferralsPaginated({search:'',pageNumber:0,pagSize:30 })
    const r4 = this.sccipService.getindIcationsPaginated({search:'',pageNumber:0,pagSize:30 })
    const r5 = this.patientService.getActiveReasonCategoriesByApptType('PRESCREEN')
    const r6 = this.miscService.getMiscResources(this.data?.site)
    const r7 = this.miscService.getMiscUsers(this.data?.site)
    forkJoin([r1,r2,r3,r4,r5,r6,r7]).subscribe(res => {
      this.users = res[0];
      this.studies = res[1].content;
      this.communityPartners2 = res[2].content
      this.indications = res[3].content;
      this.reasonCats = res[4].sort((a,b)=> this.compare(a?.name?.toLowerCase() , b?.name?.toLowerCase(),true));
      this.miscResources = res[5];
      this.miscUsers = res[6];
    });

      if(this.data?.edit) {
        if(this.data?.patientDetails?.referralSource =='ADS') {
          this.patientService.getAdsReferralsToday({id:this.data?.site,date:formatDate(new Date(), 'yyyy-MM-dd','en-Us')}).subscribe(res => {
            this.communityPartners = res;
            this.showRefs = true;
          })
        }
      this.prescreenForm = new UntypedFormGroup({
        refSource: new UntypedFormControl(this.selectedRefSource = this.data?.patientDetails.referralSource !='UNKNOWN'&& this.data?.patientDetails.referralSource !='WAITLIST' &&  this.data?.patientDetails.referralSource !=null  ? this.data?.patientDetails.referralSource :'', Validators.required),
        comPartners:new UntypedFormControl(this.data?.patientDetails.referralSource =='ADS' ? this.data?.patientDetails.adId : this.data?.patientDetails.referralId ),
        outcome :new UntypedFormControl(this.data?.patientDetails.subStatus =='SCR_SCHEDULED' ? 'SCREEN_SCHEDULED' :this.data?.patientDetails.subStatus =='REFERRED_OUT' ? 'RECOMMEND_PROVIDER': 'SCREEN_NOT_SCHEDULED', Validators.required),
        performedBy:new UntypedFormControl(this.authService.loginUserValue.id,Validators.required),
        screenDate :new UntypedFormControl(new Date(this.data?.patientDetails.appointedDate)),
        study :new UntypedFormControl(''),
        duration :new UntypedFormControl(180),
        notes: new UntypedFormControl(this.data?.patientDetails?.internalNotes),
        comPartners2: new UntypedFormControl(''),
        refnotes: new UntypedFormControl(),
        waitlist1: new UntypedFormControl(),
        waitlist2: new UntypedFormControl(),
        reasonCat: new UntypedFormControl(),
        followUpDate: new UntypedFormControl(),
        address1:new UntypedFormControl(this.data?.patientDetails.addressOne),
        address2:new UntypedFormControl(this.data?.patientDetails.addressTwo),  
        city:new UntypedFormControl(this.data?.patientDetails.city),
        state:new UntypedFormControl(this.data?.patientDetails.state),
        zip:new UntypedFormControl(this.data?.patientDetails.zip),
        upcoming: new UntypedFormControl(),
      });
      this.appointmentService.getCompletedPrescreenFormDetails(this.data.arg.event.extendedProps.appointmentId).subscribe(res => {
        this.completedDetails = res;
        this.prescreenForm.controls.refnotes.setValue(this.completedDetails?.referralNotes);
        this.prescreenForm.controls.refnotes.updateValueAndValidity();        
      })
      this.setOutcome({value:this.data?.patientDetails.subStatus =='SCR_SCHEDULED' ? 'SCREEN_SCHEDULED' :this.data?.patientDetails.subStatus =='REFERRED_OUT' ? 'RECOMMEND_PROVIDER': 'SCREEN_NOT_SCHEDULED'})
      
      
    } else {
      var day = new Date(new Date());
      var next4Day = new Date(day);
      next4Day.setDate(day.getDate() + 2);
      while (next4Day.getDay() === 0 || next4Day.getDay() === 6) {
        next4Day.setDate(next4Day.getDate() + 1);
      }
      next4Day.setMinutes(0);
      next4Day = new Date(next4Day)
      this.patientService.getAdsReferrals(this.data?.site).subscribe(res => {
        this.communityPartners = res;
        this.showRefs = true;
      })
      this.prescreenForm = new UntypedFormGroup({
        refSource: new UntypedFormControl(    this.selectedRefSource = this.data?.patientDetails.referralSource !='UNKNOWN'&& this.data?.patientDetails.referralSource !='WAITLIST' &&  this.data?.patientDetails.referralSource !=null  ? this.data?.patientDetails.referralSource :''
                , Validators.required),
        comPartners:new UntypedFormControl(this.data?.patientDetails.referralSource =='ADS' ? this.data?.patientDetails.adId : this.data?.patientDetails.referralId),
        outcome :new UntypedFormControl(this.data?.patientDetails.subStatus, Validators.required),
        performedBy:new UntypedFormControl(this.authService.loginUserValue.id,Validators.required),
        screenDate :new UntypedFormControl(new Date(next4Day)),
        study :new UntypedFormControl(),
        duration :new UntypedFormControl(180),
        notes: new UntypedFormControl(this.data?.patientDetails?.internalNotes),
        comPartners2: new UntypedFormControl(),
        refnotes: new UntypedFormControl(),
        waitlist1: new UntypedFormControl(),
        waitlist2: new UntypedFormControl(),
        reasonCat: new UntypedFormControl(),
        followUpDate: new UntypedFormControl(''),
        upcoming: new UntypedFormControl(),
        address1:new UntypedFormControl(this.data?.patientDetails?.addressOne),
        address2:new UntypedFormControl(this.data?.patientDetails?.addressTwo),  
        city:new UntypedFormControl(this.data?.patientDetails?.city),
        state:new UntypedFormControl(this.data?.patientDetails?.state),
        zip:new UntypedFormControl(this.data?.patientDetails?.zip),
      });
    
    }
    this.selectedRefSource = this.data?.patientDetails.referralSource !='UNKNOWN'&& this.data?.patientDetails.referralSource !='WAITLIST' &&  this.data?.patientDetails.referralSource !=null  ? this.data?.patientDetails.referralSource :null;
    if(this.selectedRefSource === 'ADS') {
      this.prescreenForm.controls.comPartners.setValue(this.data?.patientDetails.adId);
      this.prescreenForm.controls.comPartners.setValidators(Validators.required);
      this.prescreenForm.controls.comPartners.updateValueAndValidity();
    } else if( this.selectedRefSource == 'PHYSICIAN_COMMUNITY_REFERRAL' ) {
      
      this.prescreenForm.controls.comPartners.setValue(this.data?.patientDetails.referralId);
      this.prescreenForm.controls.comPartners.setValidators(Validators.required);
      this.prescreenForm.controls.comPartners.updateValueAndValidity();
      
        
    } else {
      this.prescreenForm.controls.comPartners.setValidators(null);
      this.prescreenForm.controls.comPartners.updateValueAndValidity();
    }  
  }

  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
}

  openPatientMiniViewDialog(patient, type=0) {
    const ref = this.dialog.open(PatientMiniViewDialogComponent, {
      width: 'auto',
    
      maxHeight:'80vh',
    panelClass:'custom-dialog-container',
    maxWidth:'80vw',
      height:'auto',
      disableClose: true,
      data: {
        fromTracking:false,
        patient: type == 0 ? patient : {patientId: patient.patientId},
        site: this.data?.site,
      }
    });

      if(res) {
        this.dialogRef.close(true);
      }
    })
  }

  openManageInterestDialog(patient) {

      this.appointmentService.getAppointmentDetails(this.data.arg.event.extendedProps.appointmentId).subscribe(res => {
        this.data.patientDetails = res;

  }

  searcher = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        res.content.forEach(element => {
            this.indications.push(element);
        });
      })
    } else {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.indications = res.content;
      })
    }
    return of(this.indications)
  }

  searcher6 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {

  prescreen(form) {
    this.loading = true;
    if(this.data?.edit) {    

      if(this.prescreenForm.valid) {
        const data = {
