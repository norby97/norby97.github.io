import { formatDate } from '@angular/common';
import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { UserService } from 'src/app/services/administration/user.service';
import { StudyStartsService } from 'src/app/services/crm_bd/study-starts.service';
import { ScoreCardService } from 'src/app/services/ctms/score-card.service';
import { StudyService } from 'src/app/services/regulatory/study.service';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';

@Component({
  selector: 'app-edit-scoreboard-dialog',
  templateUrl: './edit-scoreboard-dialog.component.html',
  styleUrls: ['./edit-scoreboard-dialog.component.scss']
})
export class EditScoreboardDialogComponent implements OnInit {
  cardForm:UntypedFormGroup;
  users;
  loading = false;
  commitmentForm 
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor( public dialogRef:MatDialogRef<EditScoreboardDialogComponent>,
            private userService: UserService,
            private studystartService: StudyStartsService,
            private scoreCardService: ScoreCardService,
     @Inject(MAT_DIALOG_DATA) public data: any,) {
      this.dpheader = DatepickerHeaderComponent;
   }
     
     ngOnInit(): void {
     
     
    this.cardForm = new UntypedFormGroup({
      startDate: new UntypedFormControl(this.data?.scoreDetails?.enrollmentStartDate),
      endDate: new UntypedFormControl(this.data?.scoreDetails?.enrollmentEndDate),

      siteNumber: new UntypedFormControl(this.data?.scoreDetails?.siteNumber),
      nickname: new UntypedFormControl(this.data?.scoreDetails?.siteNickname),
      coordinator: new UntypedFormControl(this.data?.scoreDetails?.coordinatorId),
      backup: new UntypedFormControl(this.data?.scoreDetails?.backupId),
    })
    this.commitmentForm = new UntypedFormGroup({
      screened: new UntypedFormControl(this.data.commitment.screened, Validators.required),
      actualScreened :new UntypedFormControl(this.data.commitment.actualScreened, Validators.required),
      screenFailures: new UntypedFormControl(this.data.commitment.screenFailures, Validators.required),
      actualScreenFailures :new UntypedFormControl(this.data.commitment.actualScreenFailures, Validators.required),

    })
    this.userService.getAll2({siteId:this.data?.scoreDetails.siteId}).subscribe(res => {
      this.users = res;
    })
  }

  getPercetage(partialValue, totalValue) {
    if(isNaN((100 * partialValue) / totalValue) || !isFinite((100 * partialValue) / totalValue)) {
       return 0;
    }  else {
      return (100 * partialValue) / totalValue;
    }
  }

  
  edit() {
    this.loading = true;
    if(this.cardForm.valid) {
        const data = {
          studyId: this.data?.scoreId,
          req: {
            enrollmentStartDate: formatDate(this.cardForm.controls.startDate.value, 'yyyy-MM-dd','en-Us'),
             enrollmentEndDate:this.cardForm.controls.endDate.value ? formatDate(this.cardForm.controls.endDate.value, 'yyyy-MM-dd','en-Us') : null,
             siteNumber:this.cardForm.controls.siteNumber.value,
             studyNickName:this.cardForm.controls.nickname.value,
             coordinatorId:this.cardForm.controls.coordinator.value,
             backupId:this.cardForm.controls.backup.value,
             commitmentDto: {
              screened: this.commitmentForm.controls.screened.value,
              actualScreened: this.commitmentForm.controls.actualScreened.value,
              screenFailures: this.commitmentForm.controls.screenFailures.value,
              actualScreenFailures: this.commitmentForm.controls.actualScreenFailures.value,
              completers: this.data.commitment.completers,
              actualCompleters: this.data.commitment.actualCompleters,
              studyPhaseCommitmentRequests:[...this.data.commitment.studyPhaseCommitmentRequests.map(elem => {
                return {id:elem.id ,discontinuations:elem.discontinuations, enrolled: elem.enrolled}
              })]
             }
          }
        }
        if(this.commitmentForm.valid) {
          this.scoreCardService.updateScoreInStudy(data).subscribe(res => {
            this.loading = false;
            this.dialogRef.close(true)
          }, err => {
            this.loading = false;
          })
          
        } else {this.commitmentForm.markAllAsTouched();
          this.loading = false;
        }
        

    } else {
      this.loading = false;

      this.cardForm.markAllAsTouched();
    }
  }

}
