@Component({
  selector: 'app-working-list',
  animations: [trigger('detailExpand', [
    state('collapsed', style({ height: '0px', minHeight: '0' })),
    state('expanded', style({ height: '*' })),
    transition('expanded <=> collapsed', animate('225ms cubic-bezier(0.4, 0.0, 0.2, 1)')),
])],
  templateUrl: './working-list.component.html',
  styleUrls: ['./working-list.component.scss']
})
export class WorkingListComponent implements OnInit, OnDestroy {
  workingLists = [];
  userForm: UntypedFormGroup;
  users=[];
  userscopy=[];
  usersloading = false;
  selectedUser = '';
  listDetails;
  listPatients =[];
  listPatientscopy =[];
  patientChecks = [];
  autorefresh = false;
  intervalFn;
  loading = false;
  loading2 = false;
  isFetching = false;
  actualFilePathId;
  limit =50;
  skip=0;
  actualPage=0;
  selectedListId;
  sites=[];
  allCheck = false;
  notesloading = false;
  selectedPatients = []
  excludedPatients = []
  columnsToDisplay: string[] = ['Name (Gender/Age)', 'Phone/Email', 'Interest', 'Attempts', 'Actions'];
  dataSource: MatTableDataSource<any[]>;
  expandedElement: any | null;
  @ViewChild(MatPaginator, { static: true }) paginator: MatPaginator;
  @ViewChild(MatSort, { static: true }) sort: MatSort;
  constructor(
              private recruitingService: RecruitingService,
              public utilService: UtilService,    
              private dialog:MatDialog,
              private siteService: SiteService,
              private userService: UserService,    
              private userPreferences: UserPreferencesService,
              private authService: AuthenticationService,
              private patientService: PatientsService,
       ) { }

  ngOnDestroy(): void {
    clearInterval(this.intervalFn);
  }

getNotes(i,patient){
  this.notesloading = true;
  this.patientService.getLast5Notes(patient.id).subscribe(res => {
    this.listPatients[i].notes = res
    this.listPatients[i].notes?.map(elem => {
      elem.noteDate = elem.noteDate.replace('T',' ') + ' UTC'
      elem.noteDate = elem.noteDate.toString()
    })
    this.notesloading = false
  });
}

  ngOnInit(): void {
    this.userForm = new UntypedFormGroup({
      user: new UntypedFormControl(''),
      siteID:new UntypedFormControl(''),
    })
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.userForm.controls.siteID.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.userForm.controls.siteID.updateValueAndValidity();
      this.getUsers({value:this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id)})
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.userForm.controls.siteID.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.userForm.controls.siteID.updateValueAndValidity();
        this.getUsers({value:this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id)})
      });
    }
    this.loading2 = true;
    this.recruitingService.getWorkingListForUser('').subscribe(res => {
      this.workingLists = res;
      this.loading2 = false;
    },err => {
      this.loading2 = false;

    });
  }
  onScroll() {
    this.actualPage++;
    this.isFetching = true
    const data = {
      skip: this.skip + (this.limit*this.actualPage),
      limit:this.limit,
      req:this.listDetails.id
    }

    this.recruitingService.getWorkingListPatients(data).subscribe(res => {
      this.listPatients = [...this.listPatients, ...res]
      this.listPatientscopy = [...this.listPatients, ...res]
      res.forEach(element => {
        this.patientChecks.push(false);
      });
      this.dataSource = new MatTableDataSource(this.listPatients);
      this.dataSource.paginator = this.paginator;
      this.dataSource.sort = this.sort;
      this.isFetching = false;
      if(res?.length != 0){

        this.allCheck = false;
      }

    });
  
  }
  getListPatients(item) {
    this.listDetails = item;
    this.loading =true;
    this.patientChecks = []
    this.actualPage = 0;
    this.allCheck = false;
    const data = {
      skip: this.skip + (this.limit*this.actualPage),
      limit:this.limit,
      req:item.id
    }

    this.recruitingService.getWorkingListPatients(data).subscribe(res => {
      this.listPatients = res;
      this.listPatientscopy = res;
      this.listPatients.map(elem => {
        
        
        
        this.patientChecks.push(false)})
    
      this.dataSource = new MatTableDataSource(res);
      this.dataSource.paginator = this.paginator;
      this.dataSource.sort = this.sort;
      this.loading =false;

  });
  }
  getListPatientswithoutLoading(item) {
    this.listDetails = item;
    this.patientChecks = []
    this.actualPage = 0;
    const data = {
      skip: this.skip + (this.limit*this.actualPage),
      limit:this.limit,
      req:item.id
    }

    this.recruitingService.getWorkingListPatients(data).subscribe(res => {
      this.listPatients = res;
      this.listPatientscopy = res;
      this.listPatients.map(elem => {
        this.patientChecks.push(false)})
      
      this.dataSource = new MatTableDataSource(this.listPatients);
      this.loading =false;
      
  });
  }
  sortData(sort: Sort) {
  
    const data = this.listPatientscopy;
    if (!sort.active || sort.direction === '') {
      this.listPatients = data;
      return;
    }
    this.listPatientscopy = data.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
       case 'Name (Gender/Age)':
          return this.compare(a.hasSms, b.hasSms, isAsc);
      }
    });
    this.listPatients = this.listPatientscopy
  }
  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
  refresh(){
    this.loading2= true;
    this.workingLists = []
    this.recruitingService.getWorkingListForUser(this.selectedUser).subscribe(res => {
      this.workingLists = res;
      this.loading2 = false;
    },err => {
      this.loading2 = false;

    });
  }
  change() {
    this.autorefresh = !this.autorefresh;
    if(this.autorefresh) {
      this.intervalFn =setInterval( () => {
        this.renderWorkingLists({value:{id:this.selectedUser}})
        this.getListPatientswithoutLoading(this.listDetails);

       }, 10000)
    } else {
      clearInterval(this.intervalFn);
    }
  }

  renderWorkingLists(event) {
    this.loading = true;
    this.userPreferences.setuserState('selectedSite',this.userForm.controls.siteID.value,true)

    this.selectedUser = event.value.id;
      this.recruitingService.getWorkingListForUser(event.value.id).subscribe(res => {
        this.workingLists = res;
        this.loading =false;
      })
  }


  getUsers(event) {
    this.usersloading = true;
    
    this.userPreferences.setuserState('selectedSite',this.userForm.controls.siteID.value)
    this.userService.getAll2({siteId:event.value.id}).subscribe(res => {
      this.users = res;
      this.userscopy = res;
      this.usersloading = false;
    },err =>{
      this.usersloading = false;

    });
  }

  openImportCSVDialog(){
    const ref = this.dialog.open(ImportCsvRecruitingDialogComponent, {
      data: {
        siteID:this.userForm.controls.siteID.value.id
      },
      width: 'auto',
      height: 'auto',
      disableClose: true,
      maxHeight:'80vh',
      minHeight:'25vh',
    maxWidth:'80vw',
    minWidth:'50vw',
      panelClass:'custom-dialog-container'
    });
    
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.renderWorkingLists({value:{id:this.selectedUser}})
        this.listDetails = null;
      }
    })
  }

  deleteWorkingList(workingList) {
    const dialogref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to remove ' + workingList.name + '?',
      },
      disableClose: true,
    } ) 
    dialogref.afterClosed().subscribe(res => {
      if(res) {
        this.recruitingService.deleteWorkingList(workingList.id).subscribe(res => {

          this.recruitingService.getWorkingListForUser(this.selectedUser).subscribe(res => {
            this.workingLists = res;
          });
        })
      }     
    })
    
  }
  toogleAll(status) {
    this.allCheck = status;
    this.excludedPatients = [];
    this.selectedPatients = []
    for(let i =0; i< this.patientChecks.length; i++) {
    
      if(!this.listPatients[i]?.patientSmsReplied) {

        this.patientChecks[i] =status;
      } else {
        this.excludedPatients.push(this.listPatients[i].id)
        this.patientChecks[i] =false
      }
    }
  }
  
  isAnyPatientChecked() {
    let found =false;
    this.patientChecks.map(elem => {
      if(elem == true) {
        found = true;
      }
    })
    return found;
  }

  toggleStatus(status, index) {
    if(this.allCheck) {
      this.patientChecks[index] = status;
      if(!status) {
        if(this.excludedPatients.indexOf(this.listPatients[index].id) == -1 ) {
          this.excludedPatients.push(this.listPatients[index].id);
        }
      } else {
        this.excludedPatients.splice(this.excludedPatients.indexOf(this.listPatients[index].id),1)
      }
    } else {
      this.patientChecks[index] = status;
      if(status) {
        if(this.selectedPatients.indexOf(this.listPatients[index].id) == -1 ) {
          this.selectedPatients.push(this.listPatients[index].id);
        }
      } else {
       this.selectedPatients.splice(this.selectedPatients.indexOf(this.listPatients[index].id),1)
      }
    }


  }

  deletePatientFromWorkingList(patient){
    const dialogref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to remove ' +patient.firstName +', '+patient.lastName + ' from ' +this.listDetails.name + '?',
      },
      disableClose: true,
    } ) 

    dialogref.afterClosed().subscribe(res => {
      if(res) {
        const data = {
          workingListId: this.listDetails.id,
          patientId: patient.id
        }
          this.recruitingService.deletePatientFromWorkingList(data).subscribe(res => {

            this.getListPatients(this.listDetails);
          })
      }
    })
    
  }

  incrementAttempt(patient){
    const data = {
      workingListId: this.listDetails.id,
      patientId: patient.id
    }
      this.recruitingService.increasePatientAttempt(data).subscribe(res => {

        this.getListPatientswithoutLoading(this.listDetails);
      })
  }

  decrementAttempt(patient){
    const data = {
      workingListId: this.listDetails.id,
      patientId: patient.id
    }
      this.recruitingService.decreasePatientAttempt(data).subscribe(res => {

        this.getListPatientswithoutLoading(this.listDetails);
      })
  }
  openManageInterestDialog(patient) {
    patient.patientId = patient.id
    patient.patientFullName = patient.firstName + ' ' + patient.lastName
    const ref = this.dialog.open(ManagePatientInterestDialogComponent,  {
      width: 'auto',
