import { NgxMatDateFormats, NgxMatDateAdapter, NGX_MAT_DATE_FORMATS } from '@angular-material-components/datetime-picker';
import { formatDate } from '@angular/common';
import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { FormGroup, UntypedFormBuilder, UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MAT_MOMENT_DATE_ADAPTER_OPTIONS } from '@angular/material-moment-adapter';
import { MAT_DATE_LOCALE } from '@angular/material/core';
import { MatDialogRef, MatDialog, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Observable, of } from 'rxjs';

import { MY_FORMATS } from 'src/app/app.component';
import { CustomNgxDatetimeAdapter } from 'src/app/directives/ngx-mat-select/ngx-date-format';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { ContactService } from 'src/app/services/crm_bd/contact.service';
import { SccipService } from 'src/app/services/crm_bd/sccip.service';
import { PatientsService } from 'src/app/services/ctms/patients.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';


@Component({
  selector: 'app-anonymous-call-dialog',
  templateUrl: './anonymous-call-dialog.component.html',
  styleUrls: ['./anonymous-call-dialog.component.scss'],
  providers:[
    { provide: NGX_MAT_DATE_FORMATS , useValue: MY_FORMATS} 
  ]
})
export class AnonymousCallDialogComponent implements OnInit {
  communityPartnersContacts;
  communityPartnersReferrals;
  ads
  reasonCats;
  defaultTime = [new Date().getHours(), 0 , 0] 
  selectedRefSource = '';
  indications;
  addPatientForm: FormGroup;
  showRefs = false;
  loading = false; 
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<AnonymousCallDialogComponent>,
    private formBuilder: UntypedFormBuilder,              
    private contactService:ContactService,
    private patientService: PatientsService,
    public utilService: UtilService,              
    private sccipService: SccipService ,
    private authService: AuthenticationService, 
    @Inject(MAT_DIALOG_DATA) public data: any,) {
      this.dpheader = DatepickerHeaderComponent;
     }

  ngOnInit(): void {
    this.sccipService.getindIcationsPaginated({search:'',pageNumber:0,pageSize:30 }).subscribe(res => {
      this.indications = res.content;
    });
    this.patientService.getReferralsPaginated({search: '',pageSize: 30, pageNumber: 0}).subscribe(res => {
      this.communityPartnersReferrals = res.content;
    })
    this.patientService.getAdsReferrals(this.data?.siteId).subscribe(res => {
      this.ads = res;
      this.showRefs = true;
    })
    this.addPatientForm = new UntypedFormGroup({
      callDate: new UntypedFormControl(new Date(new Date().setMinutes(0)), Validators.required),
      siteId: new UntypedFormControl(this.data?.siteId),
      type: new UntypedFormControl('info_call',Validators.required),
      refSource: new UntypedFormControl('', Validators.required),
      interest: new UntypedFormControl('', Validators.required),
      comPartners: new UntypedFormControl(''),
      notes: new UntypedFormControl(''),
    });
  }

  getCommunityPartners(event) {
    this.selectedRefSource = event.value;
    if(event.value === 'ADS') {
      this.addPatientForm.controls.comPartners.setValidators(Validators.required);
      this.addPatientForm.controls.comPartners.updateValueAndValidity();
      this.addPatientForm.controls.comPartners.enable();
      
    } else if( event.value === 'PHYSICIAN_COMMUNITY_REFERRAL') {
      this.addPatientForm.controls.comPartners.enable();
      this.addPatientForm.controls.comPartners.setValidators(Validators.required);
      this.addPatientForm.controls.comPartners.updateValueAndValidity();
 
    } else if( event.value === 'CONTACT_DATABASE') {
      this.addPatientForm.controls.comPartners.enable();
      this.addPatientForm.controls.comPartners.setValidators(null);
      this.addPatientForm.controls.comPartners.updateValueAndValidity();
    
    } else {
      this.addPatientForm.controls.comPartners.setValidators(null);
      this.addPatientForm.controls.comPartners.updateValueAndValidity();
        this.addPatientForm.controls.comPartners.disable();
    } 

  }

  searcher = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        res.content.forEach(element => {
            this.indications.push(element);
        });
      })
    } else {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.indications = res.content;
      })
    }
    return of(this.indications)
  }

  searcher4 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.patientService.getReferralsPaginated(data).subscribe(res => {
        res.content.forEach(element => {
            this.communityPartnersReferrals.push(element);
        });
      })
    } else {
      this.patientService.getReferralsPaginated(data).subscribe(res => {
        this.communityPartnersReferrals = res.content;
      })
    }
    return of(this.communityPartnersReferrals)
  }

  
  searcher5 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.contactService.getContactsPaginated(data).subscribe(res => {
        res.content.forEach(element => {
            this.communityPartnersContacts.push(element);
        });
      })
    } else {
      this.contactService.getContactsPaginated(data).subscribe(res => {
        this.communityPartnersContacts = res.content;
      })
    }
    return of(this.communityPartnersContacts)
  }

  submit() {
  this.loading = true;
  if(this.addPatientForm.valid) {
      const data = {
        callDateTime: formatDate(this.addPatientForm.controls.callDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),
        interestId:this.addPatientForm.controls.interest.value,
        referralSource:this.selectedRefSource ,
        adId: this.selectedRefSource === 'ADS' ?  this.addPatientForm.controls.comPartners.value : '',
        referralId:this.addPatientForm.controls.interest.value,
        notes:this.addPatientForm.controls.notes.value,
        siteId:this.addPatientForm.controls.siteId.value,
      }
      this.patientService.addAnonymousCall(data).subscribe(res => {

          this.loading = false;
          this.dialogRef.close(true);
      }, err => {
        this.loading = false;
      })
    } else {
          this.loading = false;
          this.addPatientForm.markAllAsTouched();
    }
  }
}
