  columnSums() {
    return this.visitRows?.map((visit) => {
    var visitCosts = this.simplifiedProcedures?.map((patient) =>{
      var temp = parseFloat(patient.visits.find((pv) => pv.id == visit.id && pv.applies )?.amount || 0)
      if( temp == undefined || isNaN(temp)  || temp == null) {
        return {amount:0, overhead:0 ,visitId:visit.id, overheadapplies: false,}
      } else{ 
        if(patient.procedure.applyOverhead && this.data?.overhead != 0) {
          return {overheadapplies: true,overhead: (temp *(parseFloat(this.data?.overhead  == '' ? 0 :this.data?.overhead )/100)) , amount:temp , visitId:visit.id} 
        } else {
          return {amount:temp || 0, visitId:visit.id , overhead:0, overheadapplies: false}

        }
      }
    });
    var overheadCostSum = 0;
    var visitPaymentCosts
    var temp = parseFloat(this.visitPayments.find((pv) => pv.visitId === visit.id  )?.amount || 0)
    if( temp == undefined || isNaN(temp)  || temp == null) {
      return 0
    } else{ 
    if(this.vpapplyOverhead && this.data?.overhead  != 0) {
       visitPaymentCosts =temp 
       overheadCostSum = overheadCostSum+ temp *(parseFloat(this.data?.overhead  == '' ? 0 :this.data?.overhead )/100)
    } else { 
      visitPaymentCosts = temp || 0
    }
  }

  var visitCostSum = visitCosts.reduce((sum, amount) => (sum + (amount.amount)), 0);

    visitCostSum = visitCostSum + parseFloat(visitPaymentCosts)
    
    overheadCostSum = overheadCostSum + visitCosts.reduce((sum, amount) =>(sum + (amount.overheadapplies ? amount.overhead: 0)), 0)
    const totalCostSum = visitCostSum + overheadCostSum;
    return {
      subtotal:visitCostSum,
      overheadtotal:overheadCostSum,
      adjusted:totalCostSum
    }
  });

  }

  calculateTotals() {
    this.grandtotal = 0;
    this.procedureTotals.forEach(procedure => {
      procedure.subtotal = 0;
      procedure.adjusted = 0;
      procedure.overheadtotal = 0;
    });

    this.procedures.forEach(procedure => {
      procedure.visits.forEach((visit,i) => {
        if(this.visitPayments[i].amount > 0) {
          this.procedureTotals[i].subtotal = this.visitPayments[i].amount
          if(procedure.applyOverhead ) {
            this.procedureTotals[i].overheadtotal += parseFloat(this.percentage(parseFloat(this.data.overhead), this.visitPayments[i].amount));  
          }
          this.procedureTotals[i].adjusted = parseFloat(this.procedureTotals[i].subtotal) + parseFloat(this.procedureTotals[i].overheadtotal);

        } else {
          
          let index = -1;
          this.visitRows.forEach((allvisit,j) => {
            if(visit.id == allvisit.id) {
              index = j;

            }
          });
          this.procedureTotals[index].subtotal += parseFloat(procedure.amount);

          if(procedure.applyOverhead ) {
            this.procedureTotals[index].overheadtotal += parseFloat(this.percentage(parseFloat(this.data.overhead),procedure.amount));  
          }
          this.procedureTotals[index].adjusted = parseFloat(this.procedureTotals[index].subtotal) + parseFloat(this.procedureTotals[i].overheadtotal);
        }
      })
    })
    

    this.procedureTotals.forEach(elem => {
      this.grandtotal += parseFloat(elem.adjusted);
    })   
    
  }

    this.procmatrix.push([]);
    this.visitRows?.forEach(elem => {
      this.simplifiedProcedures[this.simplifiedProcedures.length-1]?.visits?.push({name: elem.name, id:elem.id ,applies: false, path:elem.path, amount:0})
    });

  }

  percentage(percent, total) {
    return ((percent/ 100) * total).toFixed(2)
  }

  checkIfApplies(i,i2,visitId) {
    return this.procmatrix[i][i2]
  }

  refreshCount() {
    this.refreshTotals();
    this.simplifiedProcedures.map(elem =>elem.checkCount = 0);
    this.simplifiedProcedures.map(elem => {
      elem.visits.map(visit => {
        if(visit.applies) {
          elem.checkCount++;
        }
      })
    })
  }

  deleteProcedureRow(index) {
    this.procedures.splice(index, 1);
    if(this.procedures.length == 0) {
      this.procedureTotals.forEach((procedure,index) => {
          procedure.subtotal = 0;
          procedure.adjusted = 0;
          procedure.overheadtotal = 0;
      });
    } else {
      this.refreshTotals()
    }
  }
  
  changeType(index, status) {
    if(this.simplifiedProcedures[index].procedure.type == status) {
      this.simplifiedProcedures[index].procedure.type = 'NONE'
    } else{
        this.simplifiedProcedures[index].procedure.type = status;
    }
  }

  edit() {
    const data = {
      budgetVersionId: this.data?.budgetVersionId,
      req: {
        applyOverhead: this.vpapplyOverhead,
        applyWithholding: this.vpapplyWithholding,
        invoiceable: this.vpInvoinceable,
        procedureDefaultRequest: this.visitPayments,
        procedures: []
      }
    }
    this.simplifiedProcedures.map((elem,i) => {
      data.req.procedures.push({
        id:elem.procedure?.id ? elem.procedure?.id : null,
        procedureId:elem.procedure?.procedure.id,
         type:elem.procedure?.type,
         amount:elem.procedure?.amount,
         applyOverhead:elem.procedure?.applyOverhead,
         applyWithholding:elem.procedure?.applyWithholding,
         invoiceable:elem.procedure?.invoiceable,
         proceduresVisits: elem.visits
      });
      this.simplifiedProcedures[i].visits.forEach((elem,j) => {
          data.req.procedures[data.req.procedures.length-1].proceduresVisits.push({
            studyVisitId: elem.id,
            inProcedure: elem.applies,
            amount: elem.amount
          })
      })

    })
      
    this.studyBuildService.editStudyProcedures(data).subscribe(res => {
      this.dialogRef.close(true);
    })
  }
}
