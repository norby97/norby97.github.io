import { ChangeDetectorRef, Component, DoCheck, EventEmitter, Input, OnChanges, OnDestroy, OnInit, Output, SimpleChanges } from '@angular/core';
import { AuthenticationService } from '../../../services/administration/authentication.service';
import { UntypedFormBuilder, UntypedFormGroup } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { InquiryCallDialogComponent } from '../dialogs/ctms/inquiry-call-dialog/inquiry-call-dialog.component';
import { NavigationEnd, Router } from '@angular/router';
import { BehaviorSubject, Subject } from 'rxjs';
import { DatePipe, Location } from '@angular/common';
import { ToastrService } from 'ngx-toastr';
import { ProfileUpdatePasswordDialogComponent } from '../../administration/profile/profile-update-password-dialog.component';
import { UtilService } from 'src/app/services/utils/UtilService';
import { UserService } from 'src/app/services/administration/user.service';
import { UserPreferencesService } from 'src/app/services/utils/user-preferences.service';
import { FeedBackDialogComponent } from '../dialogs/utils/feed-back-dialog/feed-back-dialog.component';

declare var $: any;
@Component({
    selector: 'topnavbar',
    templateUrl: 'topnavbar.template.html',
    styleUrls: ['./topnavbar.component.scss']
})
export class TopNavbarComponent implements OnInit,DoCheck, OnChanges,OnDestroy {
    @Input() isSearched
    @Input() loading
    @Input() patientSearch
    @Output() searched: EventEmitter<any> = new EventEmitter();
    @Output() dialogClosed = new EventEmitter();
    showBack:Subject<boolean> = new BehaviorSubject(false);
    condition = false;
    searchForm: UntypedFormGroup;
    textLength;
    myreg = /%20/g
    myreg2 = /%28/g;
    myreg3 = /%29/g;
    myreg4 = /%2/g;
    myreg5 = /[0-9]/g;
    hidemenu = false;
    passwordExpireDays = null;
    showPasswordExpireWarning = false;
    toogleClass = false;
    localStorageSupport = true;
    actualPageTitle = '';
    notifications = []
    interval
    nomore = false;
    notificationscount;
    skip=0;
    limit=10;
    constructor(
        public authService:AuthenticationService,
        public formBuilder: UntypedFormBuilder,
        private location: Location,
        private datePipe: DatePipe,
        public userService:UserService,
        public router: Router,
        private userpreferencesService: UserPreferencesService,

    ngOnDestroy(): void {
        clearInterval(this.interval);
    }
    ngOnChanges(changes: SimpleChanges): void {

        if(changes?.isSearched != undefined  || changes.isSearched != null) {
            if(changes?.isSearched?.currentValue == false && changes?.isSearched?.previousValue == true) {
                this.searchForm.reset('')
            }
            
        }
        if(changes?.patientSearch != undefined  || changes.patientSearch != null) {
            if(changes?.patientSearch?.currentValue == false && changes?.patientSearch?.previousValue == true) {
                this.searchForm.reset('')
            }
            
        }
    }

    ngDoCheck(): void {
        
        this.showBack.next(this.myreg5.test(this.router?.url))
        this.showBack.subscribe(res => {
            this.condition = res;
        })
      
    }
   
    goBack() {
        this.location.back();
    }
   
    ngOnInit(): void {
        
        if (this.localStorageSupport) {
            if(JSON.parse(localStorage.getItem('scUserSettings'))?.collapseMenu == 'on') {
                this.toogleClass = true;
                $('body').removeClass('fixed-sidebar');
                $('body').addClass('mini-navbar');
                $('#page-wrapper').css('padding','0');

                this.SmoothlyMenu();
                this.userpreferencesService.setuserState('collapseMenu','on')
            };
            if(JSON.parse(localStorage.getItem('scUserSettings'))?.collapseMenu == null || JSON.parse(localStorage.getItem('scUserSettings'))?.collapseMenu == undefined ) { 
                this.SmoothlyMenu();
                if(this.authService.loginUserValue) {

                    $('#page-wrapper').css('padding','0 0 0 220px');
                }

                this.userpreferencesService.setuserState('collapseMenu','off')
            }
            if(JSON.parse(localStorage.getItem('scUserSettings'))?.fixedNavbar == 'on') {
                $('.navbar-static-top').removeClass('navbar-static-top').addClass('navbar-fixed-top');
                $('body').removeClass('boxed-layout');
                $('body').addClass('fixed-nav');
                this.userpreferencesService.setuserState('fixedNavbar','on')
            };
            if(JSON.parse(localStorage.getItem('scUserSettings'))?.fixedNavbar == null || JSON.parse(localStorage.getItem('scUserSettings'))?.fixedNavbar == undefined ) { 
                this.userpreferencesService.setuserState('fixedNavbar','off')
            }
        }
        this.showPasswordExpireWarning = JSON.parse(localStorage.getItem('siteCentric.loginUser'))?.showPasswordExpireMessage;
        this.passwordExpireDays = JSON.parse(localStorage.getItem('siteCentric.loginUser'))?.passwordExpireCountdownDays;
        this.searchForm = this.formBuilder.group({
            text: ''
        });
        if(this.showPasswordExpireWarning && this.passwordExpireDays <= 7) {
            this.toastr.warning(`Click here to change your password.`, 'Your password is about to expire!',{
                toastClass:'bg-warning ngx-toastr',
                timeOut:0,
                positionClass:"toast-bottom-left",
                extendedTimeOut:0,
                closeButton:true,
                tapToDismiss:false,
                disableTimeOut:true,
                titleClass:' font-weight-bold',
                messageClass:'dotsdots'
              }).onTap.subscribe(res => {
                this.dialog.open(ProfileUpdatePasswordDialogComponent, {
                    minWidth: 'auto',
                    data:{
                        hideclose:false,
                    },
                    maxHeight:'80vh',
                    maxWidth:'80vw',
                    width: 'auto',
                    height:'auto',
                    disableClose: true,
                    panelClass:'custom-dialog-container'
                    });              });
              if(this.passwordExpireDays == 1) {
                this.dialog.open(ProfileUpdatePasswordDialogComponent, {
                    minWidth: 'auto',
                    data:{
                        hideclose:true,
                    },
                    maxHeight:'80vh',
                    maxWidth:'80vw',
                    width: 'auto',
                    height:'auto',
                    disableClose: true,
                    panelClass:'custom-dialog-container'
                    });
              }
        }
        $('#collapse-menu').click(() => {
            this.toogleClass = !this.toogleClass;
            if (this.toogleClass) {
                $('body').removeClass('fixed-sidebar');
                $('body').addClass('mini-navbar');
                $('#page-wrapper').css('padding','0');

                this.SmoothlyMenu();
                this.userpreferencesService.setuserState('collapseMenu','on',true)
            } else {
                $('body').removeClass('mini-navbar');
                $('body').addClass('fixed-sidebar');
                if(this.authService.loginUserValue){

                    $('#page-wrapper').css('padding','0 0 0 220px');
                }

                this.SmoothlyMenu();
                if (this.localStorageSupport) {
                    this.userpreferencesService.setuserState('collapseMenu','off',true)
                }
            }
        });
        this.userService.getUserNotificationsCount().subscribe(res =>{
            this.notificationscount = res;
           
        })
       this.interval =  setInterval( () => {
        
        this.userService.getUserNotificationsCount().subscribe(res =>{
            this.notificationscount = res;
        })
        },60000)
    }

    refreshNotifications(){
        this.loading = true;
        const data ={
            skip:this.skip*this.limit,
            limit:this.limit
        }
        this.userService.getUserNotificationsToShow(data).subscribe(res =>{
            if(res.length==0){
                this.nomore = true;
            }
            if(this.skip == 0) {
                this.notifications = res;
                this.notifications.forEach(elem => {
                        elem.createdDate = elem.createdDate.replace('T',' ') + ' UTC'
                        elem.createdDate = elem.createdDate.toString()
                    
                    
                })
                this.loading = false;
           
            }else {
                this.notifications = [...this.notifications, ...res];
                this.loading = false;
                this.notifications.forEach(elem => {
                    elem.createdDate = elem.createdDate.replace('T',' ') + ' UTC'
                    elem.createdDate = elem.createdDate.toString()
                                        })

            }
            
            this.skip++
        } ,err => {
            this.loading = false;

        })
        
    }

    SmoothlyMenu() {
        if (!$('body').hasClass('mini-navbar') || $('body').hasClass('body-small')) {
            $('.side-menu').hide();
            setTimeout(
                function () {
                    $('.side-menu').fadeIn(400);
                }, 200);
        } else if ($('body').hasClass('fixed-sidebar')) {
            $('.side-menu').hide();
            setTimeout(
                function () {
                    $('.side-menu').fadeIn(400);
                }, 100);
        } else {
            $('.side-menu').removeAttr('style');
        }
    }

      }
    search(form) {
        if (this.textLength) {
            this.searched.emit({ form , url:this.router.routerState.snapshot.url});
        }
    }

    length(e: string) {
        this.textLength = e.length < 3 ? false : true;
        this.process(e);
    }

    process(value) {
        value = value.replace(/^a-zA-Z0-9 ]/g, '');
    }

    deleteAllNotifications(){
        const dialogRef = this.dialog.open(FeedBackDialogComponent, {
            minWidth: 'auto',
            maxHeight:'80vh',
            maxWidth:'80vw',
            width: 'auto',
            height:'auto',
            disableClose: true,
            data: {
                message: 'Are you sure you want to dismiss all notifications?',
            }
        });
        dialogRef.afterClosed().subscribe((res) => {
            if (res === true) {
            this.userService.dismissAllUserNotification().subscribe(res => {
                this.userService.getUserNotificationsCount().subscribe(res =>{
                    this.notificationscount = res;
                })
                this.refreshNotifications();
            })
            } else {
                this.dialog.closeAll();
            }
        });
    }

    openInquiryCallDialog() {
      this.dialog.open(InquiryCallDialogComponent, {
            maxHeight:'80vh',
            maxWidth:'80vw',
            minWidth: '40vw',
            width: 'auto',
            height:'auto',
            disableClose: true,
            panelClass:'custom-dialog-container',

          });
    }
    dismissNotification(notification){
        const ref = this.dialog.open(FeedBackDialogComponent, {
            data: {
              message: 'Are you sure you want to dismiss this notification?'
            }
          });
          ref.afterClosed().subscribe(res => {
            if(res) {
            this.userService.dismissUserNotification(notification.id).subscribe(res => {
  
              this.refreshNotifications();   
              })
            }
          });

    }

    refreshElapsedTime(){
        this.skip=0;
        this.notifications = []
      this.refreshNotifications()
    }
}
