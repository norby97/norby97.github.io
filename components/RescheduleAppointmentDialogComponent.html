import { NgxMatDateFormats, NgxMatDateAdapter, NGX_MAT_DATE_FORMATS } from '@angular-material-components/datetime-picker';
import { formatDate } from '@angular/common';
import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MAT_MOMENT_DATE_ADAPTER_OPTIONS } from '@angular/material-moment-adapter';
import { MAT_DATE_LOCALE } from '@angular/material/core';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { forkJoin } from 'rxjs';

import { MY_FORMATS } from 'src/app/app.component';
import { CustomNgxDatetimeAdapter } from 'src/app/directives/ngx-mat-select/ngx-date-format';
import { AppointmentService } from 'src/app/services/ctms/appointment.service';
import { MiscellaneousService } from 'src/app/services/ctms/miscellaneous.service';
import { PatientsService } from 'src/app/services/ctms/patients.service';
import { ScreenAppointmentService } from 'src/app/services/ctms/screen-appointment.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { ManagePatientInterestDialogComponent } from '../manage-patient-interest-dialog/manage-patient-interest-dialog.component';
import { OverlapDialogComponent } from '../overlap-dialog/overlap-dialog.component';
import { ShowCalendarDialogComponent } from '../show-calendar-dialog/show-calendar-dialog.component';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';
import { SiteService } from 'src/app/services/regulatory/site.service';
import moment from 'moment';
import { BusinessHoursHolidaysDialogComponent } from '../../utils/business-hours-holidays-dialog/business-hours-holidays-dialog.component';


@Component({
  selector: 'app-reschedule-appointment-dialog',
  templateUrl: './reschedule-appointment-dialog.component.html',
  styleUrls: ['./reschedule-appointment-dialog.component.scss'],
  providers:[
    { provide: NGX_MAT_DATE_FORMATS , useValue: MY_FORMATS}  
  ]
})
export class RescheduleAppointmentDialogComponent implements OnInit {
  rescheduleForm: UntypedFormGroup;
  confirmed = false;
  reasonCats;
  confirms = {
    sms: false,
    email: false
  }
  miscResources;
  miscUsers;
  resChecks = [];
  userChecks = [];
  studies;
  loading = false;
  screenconfirmed = false;
  defaultTime = [new Date().getHours(), 0 , 0] 
  screenForm: UntypedFormGroup; 
  dpheader;
  visits
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<RescheduleAppointmentDialogComponent>,
    private screenAppointMentService: ScreenAppointmentService,
    private miscService: MiscellaneousService,
    private siteService: SiteService,
    public utilService: UtilService,
    private dialog: MatDialog,
    private appointmentService: AppointmentService,
    @Inject(MAT_DIALOG_DATA) public data: any,      
    private patientService: PatientsService) { 
      this.dpheader = DatepickerHeaderComponent;
    }


  ngOnInit(): void {
    
    
    
    if(this.data?.visitType == 'SCREEN'){
      this.patientService.getActiveReasonCategoriesByApptType('SCREEN').subscribe(res => {
        this.reasonCats = res.sort((a,b)=> this.compare(a?.name?.toLowerCase() , b?.name?.toLowerCase(),true));
      });
    }
    if(this.data?.visitType == 'VISIT'){
      this.patientService.getActiveReasonCategoriesByApptType('VISIT').subscribe(res => {
        this.reasonCats = res.sort((a,b)=> this.compare(a?.name?.toLowerCase() , b?.name?.toLowerCase(),true));
      });
    }
    if(this.data?.visitType == 'PRESCREEN'){
      this.patientService.getActiveReasonCategoriesByApptType('PRESCREEN').subscribe(res => {
        this.reasonCats = res.sort((a,b)=> this.compare(a?.name?.toLowerCase() , b?.name?.toLowerCase(),true));
      });
    }
   
    const r1 = this.appointmentService.getAppointmentStudies(this.data?.site)
    const r3 = this.miscService.getMiscResources(this.data?.site)
    const r4 = this.miscService.getMiscUsers(this.data?.site)
    forkJoin([r1,r3,r4]).subscribe(res => {
      this.studies = res[0];
      this.miscResources = res[1];
      this.miscUsers = res[2]; 
      this.miscResources.map(elem => this.resChecks.push(false))
      this.miscUsers.map(elem => this.userChecks.push(false))
      if(this.data.reschedule) {
        this.miscUsers.forEach((elem,i) => {
          if(this.data?.patientDetails?.providerIds?.includes(elem.id)) {
            this.userChecks[i] = true
          } else {
            this.userChecks[i] = false
          }
          
        })
        this.miscResources.forEach((elem,i) => {
          if(this.data?.patientDetails?.providerIds?.includes(elem.id)) {
            this.resChecks[i] = true
          } else {
            this.resChecks[i] =false
          }
          
        })
      }
      
    });
     this.screenForm = new UntypedFormGroup({
      screenDate :new UntypedFormControl(this.data?.patientDetails?.appointedDate,Validators.required),
      duration :new UntypedFormControl(this.data?.patientDetails?.duration,Validators.required),
      notes: new UntypedFormControl(),
      study: new UntypedFormControl(this.data?.patientDetails?.studyId),
      refnotes: new UntypedFormControl(),
      visit:new UntypedFormControl(this.data?.showvisitdropdown == true ? this.data.visitid:''),
      reason: new UntypedFormControl('',Validators.required),
    });
    if(this.data?.showvisitdropdown){
      this.appointmentService.getVisitsForTrackedScreens(this.data?.study?.id).subscribe(res =>{
        this.visits = res;
        this.visits.forEach(element => {
            if(element.masterStudyVisitId == this.data.visitid){
              this.screenForm.controls.visit.setValue(element);
              this.screenForm.controls.visit.updateValueAndValidity();
            }
        });
      })
    }
    this.rescheduleForm = new UntypedFormGroup({
        confirmed: new UntypedFormControl(),
        notes: new UntypedFormControl(),
        reasonCat: new UntypedFormControl('',Validators.required),
        rescheduleDateTime: new UntypedFormControl(this.data?.patientDetails?.appointedDate,Validators.required),
        type: new UntypedFormControl(this.data.patientDetails.type,Validators.required),
    });
    if(this.data?.reschedule) {
      this.rescheduleForm.controls.rescheduleDateTime.setValue(this.data.patientDetails.appointedDate);
      this.rescheduleForm.controls.rescheduleDateTime.updateValueAndValidity();
      this.screenForm.controls.screenDate.setValue(this.data.patientDetails.appointedDate);
      this.screenForm.controls.screenDate.updateValueAndValidity();
      if(this.data?.visitType == 'SCREEN')
      this.screenForm.controls.screenDate.setValue(this.data.patientDetails.appointedDate);
      this.screenForm.controls.screenDate.updateValueAndValidity();
      this.screenForm.controls.duration.setValue(this.data.patientDetails.duration);
      this.screenForm.controls.duration.updateValueAndValidity();
      this.rescheduleForm.controls.rescheduleDateTime.setValue(this.data.patientDetails.appointedDate);
      this.rescheduleForm.controls.rescheduleDateTime.updateValueAndValidity();
    }
   
  }


  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
}

  rescheduleAppointment(form) {
    this.loading = true;

        if(this.data?.visitType == 'SCREEN') {
          if(this.screenForm.valid) {
            const data ={
              appointmentId: this.data.arg.event.extendedProps.appointmentId,
              req: {
                duration: this.screenForm.controls.duration.value,
                internalNotes: this.screenForm.controls.notes.value,
                resourcesIds: [],
                isConfirmed: this.screenconfirmed,
                screenDateTime: formatDate(this.screenForm.controls.screenDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),
                reasonId: this.screenForm.controls.reason.value,
                studyId: this.screenForm.controls.study.value,
                userIds: []
              }
                
            }
            this.miscResources.forEach((elem,i) => {
              if(this.resChecks[i]) {
                data.req.resourcesIds.push(elem.id);
              }
            });
            this.miscUsers.forEach((elem,i) => {
              if(this.userChecks[i]) {
                data.req.userIds.push(elem.id);
              }
            });
              const data2 = {
              userIds: data?.req?.userIds,
              resourceIds: data?.req?.resourcesIds,
              appointmentDate:this.rescheduleForm.controls.rescheduleDateTime.value,
              duration:data?.req?.duration,
              removeAppointmentId: this.data.arg.event.extendedProps.appointmentId,

            }






              this.utilService.isWithinBusinessHours(this.data?.site, formatDate(this.screenForm.controls.screenDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),formatDate(this.screenForm.controls.screenDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),data?.req?.duration).subscribe(result => {
                if(result) {
                  this.appointmentService.checkOverlap(data2).subscribe(res => {
                    if(res.length == 0) {
                      this.screenAppointMentService.markReschedule(data).subscribe(res => {

                        this.dialogRef.close(true);
                      }, err => {
                        this.loading = false;
                      })
                    } else {
                      const ref = this.dialog.open(OverlapDialogComponent,{
                        data: {
                          misc: false,
                          res,
                          type:'SCREEN',
                          siteId:this.data?.site,
                          date:this.screenForm.controls.screenDate.value
                        },
                        width: 'auto',
                        height: 'auto',
                        panelClass:'custom-dialog-container',
                        disableClose: true,
                        maxHeight:'80vh'
                      })
                      ref.afterClosed().subscribe(res => {
                        if(res) {
                          this.screenAppointMentService.markReschedule(data).subscribe(res => {

                            this.dialogRef.close(true);
                          }, err => {
                            this.loading = false;
                          })
                        } else {
                          this.loading = false;
                        }
                      })
                    }
                  }, err => {
                    this.loading = false;
                  })
                } else {
                  const ref12 = this.dialog.open(BusinessHoursHolidaysDialogComponent,{
                    data: {
                      siteId: this.data?.site,
                      isholiday:false,
                      appointmentDate:this.screenForm.controls.screenDate.value,
                    },
                    width: 'auto',
                    height: 'auto',
                    panelClass:'custom-dialog-container',
                    disableClose: true,
                    maxHeight:'80vh',
                    minWidth:'40vw'
                  })

                  ref12.afterClosed().subscribe(result2 => {
                    if(result2) {
  
                      this.appointmentService.checkOverlap(data2).subscribe(res => {
                        if(res.length == 0) {
                          this.screenAppointMentService.markReschedule(data).subscribe(res => {

                            this.dialogRef.close(true);
                          }, err => {
                            this.loading = false;
                          })
                        } else {
                          const ref = this.dialog.open(OverlapDialogComponent,{
                 
                            data: {
                              misc: false,
                            date:this.screenForm.controls.screenDate.value,
                            siteId:this.data?.site,
                            type:'SCREEN',
                              res,
                            },
                            width: 'auto',
                            height: 'auto',
                            minWidth:'40vw',
                            panelClass:'custom-dialog-container',
                            disableClose: true,
                            maxHeight:'80vh'
                          })
                          ref.afterClosed().subscribe(res => {
                            if(res) {
                    
                              this.screenAppointMentService.markReschedule(data).subscribe(res => {
                                this.loading = false;

                                this.dialogRef.close(true);
                              }, err => {
                                this.loading = false;
                              })
                            }else {
                              this.loading = false;
                            }
                  
                          })
                        }
                      })
                    } else {
                      this.loading = false;
                    }
                  })
                }
              })
                 
             






          
          } else {
            this.loading = false;
            this.screenForm.markAllAsTouched()
          }
          
        } else {
          if(this.rescheduleForm.valid) {
            const data ={
              appointmentId: this.data.arg.event.extendedProps.appointmentId,
              req: {
                confirmed: this.confirmed,
                notes: this.rescheduleForm.controls.notes.value,
                reasonCategoryId: this.rescheduleForm.controls.reasonCat.value,
                rescheduleDateTime: formatDate(this.rescheduleForm.controls.rescheduleDateTime.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),
                type: this.rescheduleForm.controls.type.value,
              }
              
            }


                this.utilService.isWithinBusinessHours(this.data?.site, new Date(this.rescheduleForm.controls.rescheduleDateTime.value),formatDate(this.rescheduleForm.controls.rescheduleDateTime.value,'hh:mm a','en-US'),30).subscribe(result =>{
                  if(result) {
                    this.patientService.markReschedule(data).subscribe(res=> {
                      this.loading = false;

                      this.dialogRef.close(true);
                    }, err => {
                      this.loading = false;
                    })
                  } else {
                    const ref12 = this.dialog.open(BusinessHoursHolidaysDialogComponent,{
                      data: {
                        siteId: this.data?.site,
                        isholiday:false,
                        appointmentDate:this.rescheduleForm.controls.rescheduleDateTime.value,
                      },
                      width: 'auto',
                      height: 'auto',
                      panelClass:'custom-dialog-container',
                      disableClose: true,
                      maxHeight:'80vh',
                      minWidth:'40vw'
                    })

                    ref12.afterClosed().subscribe(result2 => {
                      if(result2) {
                        this.patientService.markReschedule(data).subscribe(res=> {
                          this.loading = false;

                          this.dialogRef.close(true);
                        }, err => {
                          this.loading = false;
                        })
                     
                      } else {
                        this.loading = false;
                      }
                    })
                  }
             
                })
                 
          } else {
            this.loading = false;
            this.rescheduleForm.markAllAsTouched();
          }
          
        }
  }

  markAsConfirmed(status) {
    this.confirmed = status;
  }
  markAsConfirmedscreen(status) {
    this.screenconfirmed = status;
  }
  showCalendarDialog() {
    if(this.data?.visitType == 'SCREEN')  {
      this.dialog.open(ShowCalendarDialogComponent, {
        data: {
          siteId:this.data?.site,
          type: 'RESOURCES_VISITS_SCREENS',
          date:  formatDate(this.screenForm.controls.screenDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us')
  
        },
        width: 'auto',
        height: 'auto',
        panelClass:'custom-dialog-container',
        disableClose: true,
        maxHeight:'80vh',
        maxWidth:'90vw',
        minWidth:'70vw',
      })
    } else {
      this.dialog.open(ShowCalendarDialogComponent, {
        data: {
          siteId:this.data?.site,
          type: 'PRESCREEN',
          date:  formatDate(this.rescheduleForm.controls.rescheduleDateTime.value, 'yyyy-MM-ddTHH:mm', 'en-Us')
  
        },
        width: 'auto',
        height: 'auto',
        panelClass:'custom-dialog-container',
        disableClose: true,
        maxHeight:'80vh',
        minWidth:'60vw',
      })
    }
    
  }
  openManageInterestDialog(patient) {
    const ref = this.dialog.open(ManagePatientInterestDialogComponent,  {
      width: 'auto',
      height: 'auto',
      disableClose: true,
    panelClass:'custom-dialog-container',
      maxHeight:'80vh',
    maxWidth:'80vw',
      data: {
        patientDetails: patient
      },
    });
    ref.afterClosed().subscribe(res => {
      this.appointmentService.getAppointmentDetails(this.data.arg.event.extendedProps.appointmentId).subscribe(res => {
        this.data.patientDetails = res;
      })
    }) 
  }


}
