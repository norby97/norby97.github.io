  sortData(sort: Sort) {
    const data = this.invoices;
    if (!sort.active || sort.direction === '') {
      this.invoicesCopy = data;
      this.invoices = this.invoicesCopy;
      return;
    }
     this.invoicesCopy = data?.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'invoicedate': return this.compare(new Date(a?.invoiceDate) , new Date(b?.invoiceDate), isAsc);
        case 'dateentered': return this.compare(new Date(a?.dateEntered), new Date(b?.dateEntered), isAsc);
        case 'type': return this.compare(a.invoiceType?.toLowerCase(), b?.invoiceType?.toLowerCase(), isAsc);
        case 'createdby': return this.compare(a.idNumber?.toLowerCase(), b.idNumber?.toLowerCase(), isAsc);
        case 'vendor': return this.compare(a.vendorName?.toLowerCase(), b.vendorName?.toLowerCase(), isAsc);
        case 'study': return this.compare(a.studyName?.toLowerCase(), b.studyName?.toLowerCase(), isAsc);
        case 'invoicenr': return this.compare(a.invoiceNumber?.toLowerCase(), b.invoiceNumber?.toLowerCase(), isAsc);
        case 'total': return this.compare(a.invoiceTotal?.toLowerCase(), b.invoiceTotal?.toLowerCase(), isAsc);
        case 'status': return this.compare(a.invoiceStatus?.toLowerCase(), b.invoiceStatus?.toLowerCase(), isAsc);
        default: return 0;
      }
    });
    this.invoices = this.invoicesCopy;

  }
  
  compare(a: number | string | Date, b: number | string | Date, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }

  this.invoices.map(elem => {
    if(this.invoiceForm.controls.find.value == 'SENT_INVOICES') {

  searcher7 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    if(pageNumber == 1) {
      this.studypageNumber = 1;
      this.invoiceForm.controls.vendore.reset([])
      this.invoiceForm.controls.vendore.updateValueAndValidity();
    } 
    const data = {
      search,
      active:true,
      limit:pageSize,
      skip: 0 + ((this.studypageNumber-1)*pageSize),
      siteID: this.invoiceForm.controls.sites.value.id,
    }
    if(pageNumber-1 != 0) {
      this.vendorService.getVendors(data).subscribe(res => {
        this.vendorslist = [...this.vendorslist,...res];
       
      })
    } else {
     this.vendorService.getVendors(data).subscribe(res => { 
        this.vendorslist = res
      
      })
    }
    this.studypageNumber++
    return of(this.vendorslist)
  }
