import { Component, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { VendorService } from 'src/app/services/advertising/vendor.service';
import { AddVendorDialogComponent } from '../../common/dialogs/advertising/add-vendor-dialog/add-vendor-dialog.component';
import { ErrorDialogService } from '../../common/dialogs/utils/generic-warning-dialog/error-dialog.service';
import { Sort } from '@angular/material/sort';
import { FeedBackDialogComponent } from '../../common/dialogs/utils/feed-back-dialog/feed-back-dialog.component';

@Component({
  selector: 'app-ad-vendors',
  templateUrl: './ad-vendors.component.html',
  styleUrls: ['./ad-vendors.component.scss']
})
export class AdVendorsComponent implements OnInit {
  vendors;
  vendorscopy;
  filterForm: UntypedFormGroup;
  sites;
  index =0;
  skip=0
  limit=30;
  researchSites: any;
  loading = false;
  companyId: any;
  constructor(private dialog: MatDialog,
              private siteService: SiteService,
              private authService: AuthenticationService,
              private vendorService: VendorService,
              private dialogService: ErrorDialogService) { }

  ngOnInit(): void {
    this.filterForm = new UntypedFormGroup({
      active: new UntypedFormControl(true),
     search: new UntypedFormControl(''),
      siteId: new UntypedFormControl(null),
    })
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.filterForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.filterForm.controls.siteId.updateValueAndValidity();
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.filterForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.filterForm.controls.siteId.updateValueAndValidity();
      });
    }
    this.refreshVendors();

  }

  refreshVendors() {
    this.loading = true;
    this.index = 0;
    const data = {
      active: this.filterForm.controls.active.value,
     search: this.filterForm.controls.search.value,
      siteID:  this.filterForm.controls.siteId.value.id,
      skip: this.skip + (this.index*this.limit),
      limit: this.limit,
    }

    this.vendorService.getVendors(data).subscribe(res => {
      this.vendors = [...res]
      this.vendorscopy = [...res]
      this.loading = false;
    }, err => {
      this.loading = false;
    })
  }
  onScroll() {
    this.index++;
    const data = {
      active: this.filterForm.controls.active.value,
      search: this.filterForm.controls.search.value,
      siteID: this.filterForm.controls.siteId.value.id,
      skip: this.skip + (this.index*this.limit),
      limit: this.limit,
    }
    this.loading = true;
    this.vendorService.getVendors(data).subscribe(res => {
      this.vendors = [...this.vendors, ...res]
      this.vendorscopy = [...this.vendorscopy, ...res]
      this.loading = false;

    }, err => {
      this.loading = false;
    });
  
  }
  editVendor(vendor) {
    const dialogref =this.dialog.open(AddVendorDialogComponent, {
      data: {
        vendor
      },
      height: 'auto',
      maxHeight:'80vh',
    maxWidth:'80vw',
      width: 'auto',
      disableClose: true,
      panelClass:'custom-dialog-container',
      
  });
  dialogref.afterClosed().subscribe(res => {
    if(res) {
      this.refreshVendors()
    }
  })
  }
  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
  
  sortData(sort: Sort) {
    const data = this.vendors.slice();
    if (!sort.active || sort.direction === '') {
        this.vendors = data;
        return;
    }
    this.vendors = data.sort((a, b) => {
        const isAsc = sort.direction === 'asc';
        switch (sort.active) {
            case 'vendor': return this.compare(a.vendorName?.toLowerCase(), b.vendorName?.toLowerCase(), isAsc);
            case 'status': return this.compare(a.active?.toLowerCase(), b.active?.toLowerCase(), isAsc);
            case 'phone': return this.compare(a.repPhone?.toLowerCase(), b.repPhone?.toLowerCase(), isAsc);
            case 'name': return this.compare(a.repName?.toLowerCase(), b.repName?.toLowerCase(), isAsc);
            case 'site': return this.compare(a.site?.siteAlias?.toLowerCase(), b.site?.siteAlias?.toLowerCase(), isAsc);
            case 'medium': return this.compare(a.medium?.toLowerCase(), b.medium?.toLowerCase(), isAsc);
            case 'email': return this.compare(a.repEmail?.toLowerCase(), b.repEmail?.toLowerCase(), isAsc);
            default: return 0;
        }
    });
}

  filter(form) {
    this.loading = true;
    this.index = 0
    const data = {
      active: this.filterForm.controls.active.value,
      siteID: this.filterForm.controls.siteId.value.id,
      search: this.filterForm.controls.search.value,
      skip: this.skip + (this.index*this.limit),
      limit: this.limit,
    }
    this.vendorService.getVendors(data).subscribe(res => {

      this.vendors = [...res]
      this.vendorscopy = [...res]
      this.loading = false;
    }, err => {
      this.loading = false;
    })
  }

  addNewVendor() {
    const dialogref =this.dialog.open(AddVendorDialogComponent, {
      height: 'auto',
      maxHeight:'80vh',
    maxWidth:'80vw',
      width: 'auto',
      disableClose: true,
      panelClass:'custom-dialog-container',
      data: {
        vendor: null
      }
  });

  this.siteService.getAll2(this.companyId).subscribe(res => {
    this.researchSites = res;
    if (!this.researchSites.length) {
        this.dialogService.openDialog({
            title: 'No Research Site created.',
            message: 'No site has been created on your company, create a site to add a new vendor.',
            confirmText: 'Ok, got it!'
        })
    }
});

  dialogref.afterClosed().subscribe(res => {
    if(res) {
      this.refreshVendors()
    }
  })
  }

  applyFilter(event) {
    let temp = []
    if(event.target.value == '') {
        this.vendors = this.vendorscopy;
    } else {
        this.vendorscopy.map(elem => {
            if(elem.name.toLowerCase().indexOf(event.target.value.toLowerCase()) > -1 || elem.medium.toLowerCase().indexOf(event.target.value.toLowerCase()) > -1 ||  elem.repName.toLowerCase().indexOf(event.target.value.toLowerCase()) > -1 ) {
                temp.push(elem);
            }
        });
        this.vendors = temp;
    }
}

  disableVendor(vendor) {
    const ref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to inactivate ' + vendor.vendorName + '?'
      }
    })
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.vendorService.disableVendor(vendor.id).subscribe(res => {

          this.refreshVendors();   
        })
      }
    })
    
  }

  enableVendor(vendor) {
    const ref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to activate ' + vendor.vendorName + '?'
      }
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
      this.vendorService.enableVendor(vendor.id).subscribe(res => {

        this.refreshVendors();   
        })
      }
    });
  }
}
