  sortData(sort , list) {
    let temp = []
    temp = list.slice();
    const data = temp?.sort((a, b) => {
      const isAsc = sort.direction === 'desc';
      switch (sort.active) {
        case 'name': return this.compare(new Date(a?.dateTime).getTime(), new Date(a?.dateTime).getTime(), isAsc);
        default: return 0;
      }
    });
    return data
   

  }
 
  compare(a: number | string | Date, b: number | string | Date, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
