import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { RecordPaymentService } from 'src/app/services/finance/record-payment.service';

@Component({
  selector: 'app-record-payment-add-invoice-dialog',
  templateUrl: './record-payment-add-invoice-dialog.component.html',
  styleUrls: ['./record-payment-add-invoice-dialog.component.scss']
})
export class RecordPaymentAddInvoiceDialogComponent implements OnInit {
  totalAmount = 0;
  trackedProcedures = [];
  totals= [];
  payedFull = false;
  payedWitholding = false;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(@Inject(MAT_DIALOG_DATA) public data: any,
              private recordPaymentService: RecordPaymentService,
              public dialogRef: MatDialogRef<RecordPaymentAddInvoiceDialogComponent> ) { }

  ngOnInit(): void {
  
    this.resetGrandTotal();
    this.calcGrandTotal()
  }

  percentage(percent, total) {
    return ((percent/ 100) * total).toFixed(2)
  }

  resetFields(){
    if(!this.data.edit) {
      this.trackedProcedures.forEach(elem => {
        elem.paymentAmount = null;
        elem.withHoldingAmount = null;
      });
      if(this.data.type == 'Visits') {
        this.data?.invoice.trackedProcedures.forEach(elem =>{
          elem.paymentAmount = null;
          elem.withHoldingAmount = null;
        });  
      }
      this.totals.fill(0);
      this.totalAmount =0;
    }
    this.dialogRef.close(false);
  }

  resetGrandTotal() {
    if(this.data?.type == 'Visits') {
      this.data.invoice.trackedProcedures.map(elem=>  {
        this.totals.push(isNaN(this.add(elem.paymentAmount, elem.withHoldingAmount)) ? 0 :this.add(elem.paymentAmount, elem.withHoldingAmount) )
        elem.visitId = this.data.invoice.visitId;
       
        elem.patientStudyLinkId = this.data.invoice.patientStudyLinkId;
        elem.visitScheduleId =this.data.invoice.visitScheduleId;
      });
      this.data.invoice.type = this.data.type;
      this.trackedProcedures = this.data.invoice.trackedProcedures;

    } else if (this.data?.type == 'Advertising') {
      this.trackedProcedures = this.data.invoice.invoiceDetails;
      this.data.invoice.invoiceDetails.map(elem=>  {
      this.totals.push(isNaN(this.add(elem.paymentAmount, 0)) ? 0 :this.add(elem.paymentAmount,0) )

      });
      this.data.invoice.type = this.data.type;

    } else {
      this.trackedProcedures = this.data.invoice.miscellaneousDetails;
      this.data.invoice.miscellaneousDetails.map(elem=>  {
      this.totals.push(isNaN(this.add(elem.paymentAmount,0)) ? 0 :this.add(elem.paymentAmount, 0) )
      });
      this.data.invoice.type = this.data.type;    
    }
    

  }

  allFull(){
    this.payedFull = true;
    this.payedWitholding = false;
    this.totals.fill(0);
    if(this.data.type == 'Visits') {
      this.trackedProcedures.forEach((elem,i) => {
        elem.paymentAmount = this.trackedProcedures[i].amount
        elem.withHoldingAmount = 0;
        this.totals[i]= this.add(elem.paymentAmount, elem.withHoldingAmount)
      });
      this.calcGrandTotal();
    } else if(this.data.type == 'Miscellaneous') {
      this.trackedProcedures.forEach((elem,i) => {
        elem.paymentAmount = this.trackedProcedures[i].amount
        this.totals[i]= this.add(elem.paymentAmount, this.totals[i])
      });
      this.calcGrandTotal();
    }
    

  }
  

  allWithHolding() {
    this.payedWitholding = true;
    this.payedFull = false;
    this.totals.fill(0);
    this.trackedProcedures.forEach((elem,i) => {
      elem.paymentAmount = parseFloat(this.trackedProcedures[i].amount) - parseFloat(this.percentage(this.data?.withHoldingAmount,this.trackedProcedures[i].amount))
      elem.withHoldingAmount = this.percentage(this.data?.withHoldingAmount,this.trackedProcedures[i].amount) || 0;
      this.totals[i]= this.add(elem.paymentAmount, elem.withHoldingAmount)
    });
    this.calcGrandTotal();

  }

  resetTotal(index) {
    this.totals[index]= this.add(this.trackedProcedures[index].paymentAmount,  this.trackedProcedures[index].withHoldingAmount)
  }

  calcGrandTotal() {
    this.totalAmount = 0;
    this.totals.forEach(elem => {
      this.totalAmount = this.add(elem,this.totalAmount);
    })
  }

  calcTotal(event,index,type){
    this.payedFull = false;
    this.payedWitholding = false;
    this.totals[index]=0;
    if(this.data?.type =='Visits') {
      if(type == 1) {
        this.trackedProcedures[index].paymentAmount = event.target.value;
        this.trackedProcedures[index].withHoldingAmount = 0;
        this.totals[index]= this.add(this.trackedProcedures[index].paymentAmount,  this.trackedProcedures[index].withHoldingAmount)
        
      } else {
        this.trackedProcedures[index].paymentAmount = this.trackedProcedures[index].amount - parseFloat(event.target.value);
        this.trackedProcedures[index].withHoldingAmount = event.target.value;
        this.totals[index]= this.add(this.trackedProcedures[index].paymentAmount,  this.trackedProcedures[index].withHoldingAmount)
        
      }
      this.calcGrandTotal();

    } else {
        this.trackedProcedures[index].paymentAmount = event.target.value;
        this.totals[index]= this.add(this.trackedProcedures[index].paymentAmount,  this.totals[index])
       
    this.calcGrandTotal();

    }
    
  }

  calcFull(index){
    this.trackedProcedures[index].paymentAmount = this.trackedProcedures[index].amount;
    this.trackedProcedures[index].withHoldingAmount = 0;
    this.totals[index]= this.add(this.trackedProcedures[index].paymentAmount,  this.trackedProcedures[index].withHoldingAmount)
    this.calcGrandTotal()
  }

  calcFullMisc(index) {
    this.trackedProcedures[index].paymentAmount = this.trackedProcedures[index].amount;
    this.totals[index]= this.trackedProcedures[index].amount;
    this.calcGrandTotal()

  }

  calcWitholding(index) {
    this.trackedProcedures[index].paymentAmount = parseFloat(this.trackedProcedures[index].amount) - parseFloat(this.percentage(this.data?.withHoldingAmount,this.trackedProcedures[index].amount))
    this.trackedProcedures[index].withHoldingAmount =  this.percentage(this.data?.withHoldingAmount,this.trackedProcedures[index].amount) || 0;
    this.totals[index]= this.add(this.trackedProcedures[index].paymentAmount,  this.trackedProcedures[index].withHoldingAmount);
    this.calcGrandTotal()
  }


  add(y, z) {
    return  +y + +z;
  }

  clearCache() {
    if(this.data.edit) {
      const data = {
          recordPaymentProcedureCacheIds: [],
          recordPaymentMiscellaneousCacheIds:[],
          recordPaymentAdvertisingCacheIds:[]
        
       }
      this.trackedProcedures.map(elem =>{
        if(this.data.type =='Visits' ){
            data.recordPaymentProcedureCacheIds.push(elem.cacheId);
        } else if(this.data.type  =='Advertising') {
          data.recordPaymentAdvertisingCacheIds.push(elem.cacheId);
        } else {
          data.recordPaymentMiscellaneousCacheIds.push(elem.cacheId);
        }
      })
      this.recordPaymentService.clearPayment(data).subscribe(res => {

        this.dialogRef.close(1);
      })
    } else {
      this.dialogRef.close(1);
    }
    

  
    
  }

  submit() {
    
  this.data.invoice.trackedProcedures = this.trackedProcedures;
  this.data.invoice.paymentAmount = this.totalAmount;
  this.dialogRef.close({
    payedFull: this.payedFull,
    payedWitholdin:this.payedWitholding,
    invoice: this.data.invoice})
  }
}
