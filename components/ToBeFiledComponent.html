import { Component, OnInit, ChangeDetectorRef, ViewChild } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { ActivatedRoute } from '@angular/router';
import { FileService } from 'src/app/services/regulatory/file.service';
import { StudyService } from 'src/app/services/regulatory/study.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { MasterFilesService } from 'src/app/services/regulatory/master-files.service';
import { InitAuditTableService } from '../../administration/audit/audit-view-table.component/init-audit-table-service';
import printJS from 'print-js';
import { FileDialogComponent } from '../../common/dialogs/regulatory/file-dialog/file-dialog.component';
import { FileCorrespondenceDialogComponent } from '../../common/dialogs/regulatory/file-correspondence-dialog/file-correspondence-dialog.component';
import { ToBeFiledService } from 'src/app/services/regulatory/to-be-filed.service';
import { UserService } from 'src/app/services/administration/user.service';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatSort, Sort } from '@angular/material/sort';
import { UtilService } from 'src/app/services/utils/UtilService';
import { FeedBackDialogComponent } from '../../common/dialogs/utils/feed-back-dialog/feed-back-dialog.component';

@Component({
    selector: 'app-to-be-filed',
    templateUrl: './to-be-filed.component.html',
    styleUrls: ['./to-be-filed.component.scss']
})
export class ToBeFiledComponent implements OnInit {
    companyID;
    fileEmailID = null;
    show = [].fill(false);
    emails;
    url = null;
    studyName = '';
    actualFileName = null;
    actualFileID = null;
    actualFile = null;
    actualFileDetails = null;
    siteID;
    siteEmail;
    filed = [];
    emailSelected = false;
    siteSelected = false;
    companySites = [];
    user;
    assignedSite;
    sites: UntypedFormGroup;
    assignedSiteNumber = 0;
    loading = false;
    studiesloading = true;
    @ViewChild(MatSort) sort:MatSort;
    constructor(private dialog: MatDialog,
        private route: ActivatedRoute,
        private fileService: FileService,
        private ref: ChangeDetectorRef,
        private studyService: StudyService,
        public utilService: UtilService,
        private siteService: SiteService,
        private toBeFiledService: ToBeFiledService,
        private masterService: MasterFilesService,
        public initAuditTableService: InitAuditTableService,
        private userService: UserService) { }

    ngOnInit(): void {
        this.sites = new UntypedFormGroup({
            site: new UntypedFormControl('')
        });
        this.companyID = JSON.parse(localStorage.getItem('siteCentric.loginUser')).company.id;
        this.userService.getUserByID(JSON.parse(localStorage.getItem('siteCentric.loginUser')).id).subscribe(res => {
            this.user = res;
        });
        this.studiesloading = true;
        this.siteService.getAll2(this.companyID).subscribe(res => {
            localStorage.setItem('siteCentric.sites', JSON.stringify(res));

            this.assignedSiteNumber = 0;
            this.companySites = res
            this.siteSelected = true;
            this.companySites = res;
            this.assignedSite = res[0];
            this.siteID = this.assignedSite.id;
            this.sites.get('site').setValue(this.assignedSite.id + '_' + this.assignedSite.name);
            this.renderSiteEmails({ value: this.assignedSite.id + '_' + this.assignedSite.name } );
         
            this.studiesloading = false;

        }, err => {
            this.studiesloading = false;
          });
    }


    sortData(sort: Sort) {
        const data = this.emails.slice();
        if (!sort.active || sort.direction === '') {
            this.emails = data;
            return;
        }
        this.emails = data.sort((a, b) => {
            const isAsc = sort.direction === 'asc';
            switch (sort.active) {
                case 'name': return this.compare(a.from?.toLowerCase(), b.from?.toLowerCase(), isAsc);
                case 'date': return this.compare(new Date(a.timestamp).getTime(), new Date(b.timestamp).getTime(), isAsc);
                default: return 0;
            }
        });
    }
    
    
    compare(a: number | string, b: number | string, isAsc: boolean) {
      return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
    }

    renderSiteEmails(event) {
        this.siteSelected = true;
        this.actualFileName = null;
        this.show.fill(false);
        this.emailSelected = false;
        this.siteID = event.value.split('_')[0];
        this.siteEmail = this.siteID + '@portal.sitecentric.us';
        this.toBeFiledService.retrieveEmailsForSite({ siteID: this.siteID }).subscribe(res => {
            this.emails = res.filedEmails;
            this.sortData(this.sort)
        });

    }

    copied() {
        this.studyService.showCopied();
    }

    retrieveFile(id, name) {
        this.loading = true;
        this.toBeFiledService.retrieveFiledDocuments({ siteID: this.siteID, fileEmailID: id }).subscribe(res => {
            this.filed = res.filedDocuments;
        });
        this.emailSelected = false;
        this.actualFileID = id;
        this.actualFileName = name;
        const data = {
            fileID: id,
            fileEmailID: this.fileEmailID,
            siteID: this.siteID,

        };
        this.toBeFiledService.retrieveFile(data).subscribe(res => {
            this.url = res;
            this.actualFile = res;
            this.loading = false;
        }, err => {
            this.loading = false;
          });
    }

    printFile() {
        if (this.url === null) {
            this.studyService.showError('Please select a PDF first.');
        } else {
            const base64 = btoa(
                new Uint8Array(this.url)
                    .reduce((data, byte) => data + String.fromCharCode(byte), '')
            );

            printJS({ printable: base64, type: 'pdf', showModal: false, base64: true });
        }
    }

    openFileDialog() {
        const dialogRef = this.dialog.open(FileDialogComponent, {
            height: 'auto',
            disableClose: true,
            width: 'auto',
            data: {
                siteID: this.siteID,
                emailID: this.fileEmailID,
                fileID: this.actualFileID
            }
        });
        dialogRef.afterClosed().subscribe((res) => {
            this.toBeFiledService.retrieveEmailsForSite({ siteID: this.siteID }).subscribe(result => {
                this.emails = result.filedEmails;
            });
        });
    }

    openFileCorrespondeceDialog() {
        const dialogRef = this.dialog.open(FileCorrespondenceDialogComponent, {
            height: 'auto',
            width: 'auto',
            disableClose: true,
            data: {
                siteID: this.siteID,
                emailID: this.fileEmailID
            }
        });
        dialogRef.afterClosed().subscribe((res) => {
            this.toBeFiledService.retrieveEmailsForSite({ siteID: this.siteID }).subscribe(result => {
                this.emails = result.filedEmails;
            });
        });
    }

    toogleFiles(i, id, from) {
        this.emailSelected = true;
        this.fileEmailID = id;
        this.actualFile = null;
        this.actualFileID = null;
        this.actualFileName = from;
        this.toBeFiledService.retrieveFiledDocuments({ siteID: this.siteID, fileEmailID: this.fileEmailID }).subscribe(res => {
            this.filed = res.filedDocuments;
        });
        const data = {
            siteID: this.siteID,
            emailID: this.fileEmailID,
            fileID: this.fileEmailID,
            fileEmailID: this.fileEmailID,
        }
        this.toBeFiledService.retrieveFile(data).subscribe(res => {
            this.url = res;
            this.actualFile = res;

        });
        this.show.fill(false);
        this.show[i] = !this.show[i];
    }

    deleteEmail(id) {
        const dialogRef = this.dialog.open(FeedBackDialogComponent, {
            minWidth: 'auto',
            maxHeight:'80vh',
            maxWidth:'80vw',
            width: 'auto',
            height:'auto',
            disableClose: true,
            data: {
                message: 'Are you sure you want to delete this email?',
            }
        });
            dialogRef.afterClosed().subscribe((res) => {
                if (res === true) {
                    this.toBeFiledService.deleteFiledEmail({ siteID: this.siteID, fileEmailID: id }).subscribe(res => {
                        this.toBeFiledService.retrieveEmailsForSite({ siteID: this.siteID }).subscribe(result => {
                            this.emails = result.filedEmails;
                        });
                    });
                } 
            });
      
    }
}
