import { AfterViewInit, Component, HostListener, OnInit } from '@angular/core';

import { AppLoadService } from './services/utils/AppLoadService';
import { FileService } from './services/regulatory/file.service';
import { UserService } from './services/administration/user.service';
import { MatDialog } from '@angular/material/dialog';
import { LockScreenComponent } from './components/common/lock-screen/lock-screen.component';
import { PatientsService } from './services/ctms/patients.service';
import { BnNgIdleService } from 'bn-ng-idle';
import { lockScreenConfig } from './app.module';
import { HttpClient } from '@angular/common/http';
import { NgxMatDateFormats } from '@angular-material-components/datetime-picker';
import { ActivatedRoute, NavigationEnd, Router } from '@angular/router';
import { UtilService } from './services/utils/UtilService';
import { DashboardService } from './services/utils/dashboard.service';

declare var $:any;

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss']
})
export class AppComponent implements OnInit,AfterViewInit { 
  title = 'siteCentric CTMS';
  isSearched = false;
  patientSearch = false;
  loginProcess = false;
  searchResults;
  cheatcode=0;
  loading = false;
  searchedText= '';
  devEnviroment = false;
  interval;
  @HostListener('document:mousemove', ['$event'])
  onMouseMove(event: MouseEvent): void {
    if(!this.appLoadService.userIdle){

      localStorage.setItem('idleSeconds',lockScreenConfig.idleTime.toString())
    }

  }
  @HostListener('window:keyup',['$event']) onKeyUp(event) {
    if(event.key =='Control') {
      this.cheatcode++;
      if(this.cheatcode ==10 ){
        this.utilService.setDeveloperView(true);
      }
    } else {
      this.cheatcode = 0;
      this.utilService.setDeveloperView(false);
    }
        
  }
  
  constructor(
    public authService: AuthenticationService,
    public appLoadService: AppLoadService,
    private bnIdle: BnNgIdleService,
    private https:HttpClient,
    private router:Router,
    private activeRoute:ActivatedRoute,
    public utilService:UtilService,
    private toastrService:ToastrService,
    private userPreferences:UserPreferencesService,
    private fileService: FileService,
    private patientService: PatientsService,
    private dialog: MatDialog,
    private userService: UserService) {
     
     }
  ngAfterViewInit(): void {
    $('.collapse-link').on('click', function (e) {
      e.preventDefault();
      var ibox = $(this).closest('div.ibox');
      var button = $(this).find('i');
      var content = ibox.children('.ibox-content');
      content.slideToggle(200);
      button.toggleClass('fa-chevron-up').toggleClass('fa-chevron-down');
      ibox.toggleClass('').toggleClass('border-bottom');
      setTimeout(function () {
          ibox.resize();
          ibox.find('[id^=map-]').resize();
      }, 50);
  });
  }

  ngOnInit() {
    this.appLoadService.onLoginProcess.subscribe(res => {
      this.loginProcess = res;
    })
    if(!this.authService.loginUserValue || !this.getToken()) {
      $('body').removeClass('fixed-sidebar');
      $('#page-wrapper').css('padding','0 !important');

    }
          const isIE1 = /^.*MSIE*./i.test(window.navigator.userAgent);
        if (isIE1) {
          this.appLoadService.userIdle = true;
          this.userService.showPermanentError('Unsupported Browser! Please use Google Chrome or Edge.')
        } else  if(navigator.userAgent.toLowerCase().indexOf('firefox') > -1){
          this.appLoadService.userIdle = true;
          this.userService.showPermanentError('Unsupported Browser! Please use Google Chrome or Edge.')
        } else {
          if(location.origin.indexOf('dev') != -1) {
            this.devEnviroment = true;
          }
          if(this.appLoadService.userIdle) {
            this.dialog.open(LockScreenComponent,{disableClose: true, backdropClass:'blurred-backdrop'});
          } else {
            this.idleTimerStart();
          }
        
          this.appLoadService.onLogin.subscribe(data => {
            this.loginProcess = true;
            if (data === true) {
              if(this.authService.loginUserValue.accountType=='EXTERNAL') {
                this.router.navigate(['/regulatory/dashboard'],{replaceUrl:true});
                this.utilService.forceReload()
              } else {
                if(JSON.parse(localStorage.getItem('scUserSettings'))?.homePage != undefined) {
                  this.router.navigate([decodeURI(JSON.parse(localStorage.getItem('scUserSettings'))?.homePage),{replaceUrl:true}])
                  this.utilService.forceReload()

                } else {
                  this.userPreferences.setuserState('homePage','/my-dashboard/'
                  +encodeURI(this.authService.loginUserValue.firstName).replace('(','').replace(')','')+' '+encodeURI(this.authService.loginUserValue.lastName).replace('(','').replace(')',''),true)
                  this.router.navigate(['/my-dashboard', encodeURI(this.authService?.loginUserValue?.firstName.replace('(','').replace(')','') + ' ' + this.authService?.loginUserValue?.lastName.replace('(','').replace(')',''))],{replaceUrl:true})
                  this.utilService.forceReload()
 
                }
              }
              this.idleTimerStart();
            } else {
              this.dialog.closeAll();
              this.bnIdle?.stopTimer();
              this.authService.logout()
            }
          });
          this.appLoadService.onLogOut.subscribe(data => {
            if(this.bnIdle) {
              this.bnIdle?.stopTimer();
              $('#page-wrapper').css('padding','0');

            }
           
            this.dialog.closeAll();
            this.authService.logout()
            this.router.navigate(['/login'])
          });
        }
      
  }

  idleTimerStart() {
    localStorage.setItem('idleSeconds',lockScreenConfig.idleTime.toString())
    if(this.authService.loginUserValue) {
      this.userService.getIfUserLockedScreened().subscribe(res => {
        if(res == true) {
          this.dialog.closeAll();
          this.appLoadService.userIdle = true; 
         const dialogref = this.dialog.open(LockScreenComponent, {disableClose:true, backdropClass:'blurred-backdrop'});
         dialogref.afterClosed().subscribe(res => {
          if(res) {
            this.idleTimerStart();
          }
        })
        
        } else { 
          const ref = this.bnIdle.startWatching(lockScreenConfig.idleTime).subscribe((isTimedOut: boolean) => {
            this.interval = setInterval( () => {
              localStorage.setItem('idleSeconds',(parseInt(localStorage.getItem('idleSeconds'))-1).toString())
              if(((parseInt(localStorage.getItem('idleSeconds')))<=0 ||parseInt(localStorage.getItem('idleSeconds')) ==null)) {
               clearInterval(this.interval)
              }
               },1000)
          
            if (isTimedOut && ((parseInt(localStorage.getItem('idleSeconds')))<=0 ||parseInt(localStorage.getItem('idleSeconds')) ==null)  ) {
              if(this.bnIdle) {
                this.bnIdle?.stopTimer();
              }                    
                  ref?.unsubscribe();
                   this.appLoadService.userIdle = true; 
                    this.userService.userSetLockScreen().subscribe(res => {
                    
                      this.dialog.closeAll();
                      const dialogref = this.dialog.open(LockScreenComponent,{disableClose: true, backdropClass:'blurred-backdrop'});
                      dialogref.afterClosed().subscribe(res => {
                        if(res) {
                          this.idleTimerStart();
                        }
                      })
                  });
              }
          });
        }
      });
      
    }
  }

  onSearch(event) {
    this.loading =true;
    this.searchedText = event.form.value.text
    if(this.authService.loginUserValue.accountType =='REGULATORY' || this.authService.loginUserValue.accountType =='EXTERNAL' || this.authService.loginUserValue.accountType =='REGULAR') {
      this.fileService.searchFile(event.form.value).subscribe(res => {
        this.searchResults = res;
        this.loading = false;

      }, err => {
        this.loading = false;
      });
      this.isSearched = true;
      this.patientSearch = false;
    } else {
      if(event.url.split('/')[1] =='ctms' || event.url.split('/')[1] =='my-dashboard') {
        this.patientService.searchPatient(event.form.value).subscribe(res => {
          this.searchResults = res;
          this.loading = false;
        }, err => {
          this.loading = false;
        });
        this.isSearched = false;
        this.patientSearch = true;
      }  else {
        this.fileService.searchFile(event.form.value).subscribe(res => {
          this.searchResults = res;
          this.loading = false;
        }, err => {
          this.loading = false;
        });
        this.isSearched = true;
        this.patientSearch = false;
      }
      this.loading = false;
    }
    
  }

  hideSearch() {
    this.isSearched = false;
    this.patientSearch = false;
  }

  getToken() {
    if(localStorage.getItem('siteCentric.token') != null) {
      return localStorage.getItem('siteCentric.token');
    } else return false;
    
  }
}
