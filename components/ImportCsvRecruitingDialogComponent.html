import { DialogRef } from '@angular/cdk/dialog';
import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { UserService } from 'src/app/services/administration/user.service';
import { PatientsService } from 'src/app/services/ctms/patients.service';
import { WorkingListService } from 'src/app/services/ctms/working-list.service';

@Component({
  selector: 'app-import-csv-recruiting-dialog',
  templateUrl: './import-csv-recruiting-dialog.component.html',
  styleUrls: ['./import-csv-recruiting-dialog.component.scss']
})
export class ImportCsvRecruitingDialogComponent implements OnInit {
  fileUploadForm:UntypedFormGroup;
  fileName='';
  fileToUpload;
  workingForm:UntypedFormGroup;
  showError = false;
  headerValues=[];
  firstStep = true;
  includedob = true;
  errorStep = false;
  imported = false;
  ads 
  importerrors;
  users;
  adsForm: UntypedFormGroup
  selectedAdID = '';
  importing = false;
  matchingValues =['','','','','','']
    patientDBHeaderValues=[ 'firstName','lastName', 'dateOfBirth','phoneNumber','email','note' ]
  
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef:MatDialogRef<ImportCsvRecruitingDialogComponent>,
              private patientService: PatientsService,
              private userService: UserService,
              @Inject(MAT_DIALOG_DATA) public data: any,
              private workingListService:WorkingListService) { }

  ngOnInit(): void {
    this.patientService.getAdsReferrals(this.data?.siteID).subscribe(res => {
      this.ads =res;
    })
    this.workingForm = new UntypedFormGroup({
      name : new UntypedFormControl('', Validators.required),
      expDate : new UntypedFormControl('', Validators.required),
      userId : new UntypedFormControl('', Validators.required),
      desc : new UntypedFormControl(''),
    }) 
    this.adsForm = new UntypedFormGroup({
      ad: new UntypedFormControl('',Validators.required),
    });
    this.fileUploadForm = new UntypedFormGroup({ 
      file: new UntypedFormControl('',[Validators.required]),
      firstName:new UntypedFormControl('',Validators.required),
      lastName:new UntypedFormControl('',Validators.required),
      email:new UntypedFormControl('',Validators.required),
      dateOfBirth:new UntypedFormControl('',Validators.required),
      dateFormat:new UntypedFormControl('M/d/yyyy',Validators.required),
      note:new UntypedFormControl('',Validators.required),
      
      phoneNumber:new UntypedFormControl('',Validators.required),

    })
    this.userService.getAll2().subscribe(res => {
      this.users = res;
    });
  }
  changeDob(){
    this.includedob = !this.includedob
    
     
     
     
    if(!this.includedob) {
      this.matchingValues[2]=''
      this.fileUploadForm.controls.dateOfBirth.setValue(null)
      this.fileUploadForm.controls.dateOfBirth.setValidators(null)
      this.fileUploadForm.controls.dateOfBirth.updateValueAndValidity() 
      this.fileUploadForm.controls.dateFormat.setValidators(null)
      this.fileUploadForm.controls.dateFormat.updateValueAndValidity()
    } else {
      this.fileUploadForm.controls.dateOfBirth.setValidators(Validators.required)
      this.fileUploadForm.controls.dateOfBirth.updateValueAndValidity()
      this.fileUploadForm.controls.dateFormat.setValidators(Validators.required)
      this.fileUploadForm.controls.dateFormat.updateValueAndValidity()

    }
  }
  changeFileName(event) {
    this.showError = false;
    this.fileToUpload = event.target.files.item(0);
    this.fileName = event.target.value.replace(/^.*\\/, '');
  }


  getHeaders(){
    if(!this.fileToUpload) {
      this.showError = true;
    } else {
        const data = {
          file: this.fileToUpload ,
        }
        this.workingListService.getCSVFileHeaders(data).subscribe(res => {
        this.headerValues = res;
        }, err => {
            this.workingListService.showError(err.error)
        })
    }
  }
  next(){
    if(this.workingForm.valid) {
      this.firstStep = false;
    } else {
      this.workingForm.markAllAsTouched()
    }
  }

  back() {
    this.firstStep = true;
  }
  back2() {
    this.errorStep = false;
  }
  importPatient() {
    if(this.adsForm.valid) {
      this.importing =true;
      const data = {
        adId: this.adsForm.controls.ad.value,
        file: this.fileToUpload ,
        request: {
          bindingValues: [
            {bindTo:'firstName', bindFrom:this.fileUploadForm.controls.firstName.value},
            {bindTo:'lastName', bindFrom:this.fileUploadForm.controls.lastName.value},
            {bindTo:'email', bindFrom:this.fileUploadForm.controls.email.value},
            {bindTo:'phoneNumber', bindFrom:this.fileUploadForm.controls.phoneNumber.value},
            {bindTo:'note', bindFrom:this.fileUploadForm.controls.note.value},
          ],
          expiresOn: this.workingForm.controls.expDate.value,
          assignTo: this.workingForm.controls.userId.value,
          description: this.workingForm.controls.desc.value,
          name:this.workingForm.controls.name.value,
          dateFormat: this.fileUploadForm.controls.dateFormat.value,

        }
      }
      if(this.includedob) {
        data.request.bindingValues.push({bindTo:'dateOfBirth', bindFrom:this.fileUploadForm.controls.dateOfBirth.value})
      }
      this.workingListService.importPatientFromCSV(data).subscribe(res => {
        this.importerrors = res;
        this.importing =false
        this.dialogRef.close(true);
      }, err=> {
        this.importerrors = err?.error
        this.importing =false;
        this.errorStep = true;

      })
    } else {
      this.importing =false
      this.imported = true;
      this.adsForm.markAllAsTouched();
    }
  }
}
