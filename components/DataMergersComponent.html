   makeid(length) {
    var result           = '';
    var characters       = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
    var charactersLength = characters.length;
    for ( var i = 0; i < length; i++ ) {
        result += characters.charAt(Math.floor(Math.random() * charactersLength));
    }
    return result;
}

searcher = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
  const data = {
    search,
    name:search,
    limit:pageSize,
    pageNumber: pageNumber-1,
    skip: (pageNumber-1)* pageSize,
    pageSize,
  }
  if(pageNumber - 1 != 0) {
    if(this.selectedData == 'SPONSOR') {
      this.sccipService.getSponsorsPaginated(data).subscribe(res => {
        this.froms = [...this.froms,...res]
      })
    } else if(this.selectedData == 'CRO') {
      this.sccipService.getCroPaginated(data).subscribe(res => {
        this.froms = [...this.froms,...res]
      })
    } else if(this.selectedData == 'INDICATION') {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.froms = [...this.froms,...res.content]
      })
    } else if(this.selectedData == 'COMPOUND') {
      this.sccipService.getCompundsPaginated(data).subscribe(res => {
        this.froms = [...this.froms,...res]
      })
    } else if(this.selectedData == 'POPULATION') {
      this.sccipService.getPopulationPaginated(data).subscribe(res => {
        this.froms = [...this.froms,...res]
      })
    } else if(this.selectedData == 'PROTOCOL') {
      this.sccipService.getProtocolsPaginated(data).subscribe(res => {
        this.froms = [...this.froms,...res]
      })
    } else if(this.selectedData == 'AD_CATEGORY') {
      this.sccipService.getAdCategories().subscribe(res => {
        this.froms = [...this.froms,...res]
      })
    }else if(this.selectedData == 'FINDING_CATEGORY') {
      this.sccipService.getFindingCategories(data).subscribe(res => {
        this.froms = [...this.froms,...res]
      })
    }
  
  } else {
    if(this.selectedData == 'SPONSOR') {
      this.sccipService.getSponsorsPaginated(data).subscribe(res => {
        this.froms = [...res];
      })
    } else if(this.selectedData == 'CRO') {
      this.sccipService.getCroPaginated(data).subscribe(res => {
        this.froms = [...res];
      })
    } else if(this.selectedData == 'INDICATION') {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.froms = [...res.content];
      })
    } else if(this.selectedData == 'COMPOUND') {
      this.sccipService.getCompundsPaginated(data).subscribe(res => {
        this.froms = [...res];
      })
    } else if(this.selectedData == 'POPULATION') {
      this.sccipService.getPopulationPaginated(data).subscribe(res => {
        this.froms = [...res];
      })
    } else if(this.selectedData == 'PROTOCOL') {
      this.sccipService.getProtocolsPaginated(data).subscribe(res => {
        this.froms = [...res];
      })
    }else if(this.selectedData == 'AD_CATEGORY') {
      this.sccipService.getAdCategories().subscribe(res => {
        this.froms = [...res]
      })
    }else if(this.selectedData == 'FINDING_CATEGORY') {
      this.sccipService.getFindingCategories(data).subscribe(res => {
        this.froms = [...res]
      })
    } 
    
  
  }
  return of(this.froms)
}

  loadData(dataType) {

      this.froms = this.sponsors;

  merge() {
      this.loading = true;
    if(this.mergeForm.valid) {
      if(this.mergeForm.controls.code.value != this.currentID) {
        this.sccipService.showError('Wrong confirmation code!');
        this.currentID = this.makeid(6);
        this.mergeForm.controls.code.reset('')
      } else {
        const data= {
          fromId: this.mergeForm.controls.from.value,
          toId: this.mergeForm.controls.to.value,

        }
        if(this.selectedData == 'SPONSOR') {
          this.sccipService.mergeSponsors(data).subscribe(res => {

            this.refreshLists();
            this.loading = false;

          }, err => {
            this.loading = false;
          })
        } else if(this.selectedData == 'CRO') {
          this.sccipService.mergeCros(data).subscribe(res => {
            this.refreshLists();
            this.loading = false;
          }, err => {
            this.loading = false;
          })
        } else if(this.selectedData == 'INDICATION') {
          this.sccipService.mergeIndications(data).subscribe(res => {
            this.refreshLists();
            this.loading = false;
          }, err => {
            this.loading = false;
          })
        } else if(this.selectedData == 'COMPOUND') {
          this.sccipService.mergeCompounds(data).subscribe(res => {

            this.refreshLists();
            this.loading = false;
          }, err => {
            this.loading = false;
          })
        } else if(this.selectedData == 'POPULATION') {
          this.sccipService.mergePopulation(data).subscribe(res => {

            this.refreshLists();
            this.loading = false;          
          }, err => {
            this.loading = false;
          })
        } else if(this.selectedData == 'PROTOCOL') {
          this.sccipService.mergeProtocols(data).subscribe(res => {

            this.refreshLists();

          }, err => {

        } else if(this.selectedData == 'GENERAL') {
