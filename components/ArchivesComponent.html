import { CdkDragDrop, transferArrayItem } from '@angular/cdk/drag-drop';
import { Component, OnInit } from '@angular/core';
import { UntypedFormGroup, UntypedFormControl, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { UserPreferencesService } from 'src/app/services/utils/user-preferences.service';
import { RegManagerService } from 'src/app/services/regulatory/reg-manager.service';
import { ScanBoxDialogComponent } from '../../common/dialogs/regulatory/scan-box-dialog/scan-box-dialog.component';
import { ArchiveBoxDetailsDialogComponent } from '../../common/dialogs/regulatory/archive-box-details-dialog/archive-box-details-dialog.component';
import { DatepickerHeaderComponent } from '../../common/dialogs/utils/datepicker-header/datepicker-header.component';
import { AuditService } from 'src/app/services/administration/audit.service';

@Component({
  selector: 'app-archives',
  templateUrl: './archives.component.html',
  styleUrls: ['./archives.component.scss']
})
export class ArchivesComponent implements OnInit {
  siteForm: UntypedFormGroup;
  noteForm: UntypedFormGroup;
  sites;
  loading = false;
  notesloading = false;
  archiveloading = false;
  sitearchives
  boxContents=[];
  audits=[];
  selectedStudy=null
  recordForm: UntypedFormGroup
  autorefresh = false;
  auditloading = false;
  dpheader
  notes
  
  constructor(private siteService: SiteService,
              private authService: AuthenticationService,
              private auditService: AuditService,
              private dialog:MatDialog,
              private userPreferences: UserPreferencesService,
              public utilService: UtilService,
              public archivesService: RegManagerService,
              private archiveService: RegManagerService,) { 
                this.dpheader = DatepickerHeaderComponent;

              }
  ngOnDestroy(): void {
  }

  ngOnInit(): void {
    this.siteForm = new UntypedFormGroup({
      siteID: new UntypedFormControl('',Validators.required),
    }); 
    this.noteForm = new UntypedFormGroup({
      note: new UntypedFormControl('', Validators.required),
    });
    this.recordForm = new UntypedFormGroup({
      desctructionDate: new UntypedFormControl('',Validators.required),
      irbdate: new UntypedFormControl('',),
      terms: new UntypedFormControl('',),
      contact: new UntypedFormControl(''),
    });
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.siteForm.controls.siteID.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.siteForm.controls.siteID.updateValueAndValidity();
      this.getLists()
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.siteForm.controls.siteID.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.siteForm.controls.siteID.updateValueAndValidity();
        this.getLists()
      });
    }
  

  }
  getAudits(studyID) {
    this.auditloading = true
    this.auditService.getAuditsForStudyArchiveView(studyID).subscribe(res => {
      this.audits = res;
    
      
      this.auditloading = false;
    }, err =>  {
      this.auditloading = false;
    });
  
  }
  getNotes(studyID) {
    this.notesloading = true
    this.archivesService.getStudyNotes(studyID).subscribe(res => {
    
      this.notes = res
      this.notes?.map(elem => {
        elem.createdDate = elem.createdDate.replace('T',' ') + ' UTC'
        elem.createdDate = elem.createdDate.toString()
      })
      this.notesloading = false;
    }, err =>  {
      this.notesloading = false;
    });
  }
  openScanBoxDialog() {
    const ref = this.dialog.open(ScanBoxDialogComponent, {
      width: 'auto',
      height: 'auto',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw',
      minWidth:'40vw',
      panelClass:'custom-dialog-container'
    });
    
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.getLists();
           
      }
    })
  }
  openBoxDetailsDialog(box){
    const ref = this.dialog.open(ArchiveBoxDetailsDialogComponent, {
      width: 'auto',
      height: 'auto',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw',
      minWidth:'40vw',
      data:{
        box
      },
      panelClass:'custom-dialog-container'
    });
    
    ref.afterClosed().subscribe(res => {
      if(res) {
           
      }
    })
  }

  viewBoxInventory(){
    const ref = this.dialog.open(ArchiveBoxDetailsDialogComponent, {
      width: 'auto',
      height: 'auto',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw',
      minWidth:'40vw',
      data:{

        type:1,
        manifest:false,

        types:['REGULATORY','PATIENT','CONTRACT']
      },
      panelClass:'custom-dialog-container'
    });
    
    ref.afterClosed().subscribe(res => {
      if(res) {
           
      }
    })
  }

  findPatients(){
    const ref = this.dialog.open(ArchiveBoxDetailsDialogComponent, {
      width: 'auto',
      height: 'auto',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw',
      minWidth:'40vw',
      data:{
        box:this.selectedStudy,
        types:['PATIENT'],
        manifest:false,
        type:2,
        
      },
      panelClass:'custom-dialog-container'
    });
    
    ref.afterClosed().subscribe(res => {
      if(res) {
           
      }
    })
  }
  findOtherDocuments(){
    const ref = this.dialog.open(ArchiveBoxDetailsDialogComponent, {
      width: 'auto',
      height: 'auto',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw',
      minWidth:'40vw',
      data:{
        box:this.selectedStudy,
        types:['REGULATORY','CONTRACT'],
        type:3
      },
      panelClass:'custom-dialog-container'
    });
    
    ref.afterClosed().subscribe(res => {
      if(res) {
           
      }
    })
  }


  loadStudy(box){
    this.selectedStudy = box;
    this.archiveloading = true;
    this.recordForm.controls.desctructionDate.setValue(new Date(this.selectedStudy?.estimatedDestructionDate))
    this.recordForm.controls.desctructionDate.setValidators(Validators.required)
    this.recordForm.controls.desctructionDate.updateValueAndValidity();
    this.recordForm.controls.terms.setValue(this.selectedStudy?.destructionTerms)
    this.recordForm.controls.terms.setValidators(null)
    this.recordForm.controls.terms.updateValueAndValidity();
    this.recordForm.controls.contact.setValue(this.selectedStudy?.destructionContact)
    this.recordForm.controls.contact.setValidators(null)
    this.recordForm.controls.contact.updateValueAndValidity();
  
    this.archiveloading = false;
    this.getNotes(box.id)
    this.getAudits(box.id)
    
  }
  
  getLists() {
    this.loading =true;
   
    if(this.siteForm.valid) {
      this.archiveService.getArchivesBySite(this.siteForm.controls.siteID.value.id).subscribe(res => {
        this.sitearchives = res;
        if(this.selectedStudy != null){
          this.loadStudy(this.sitearchives.filter(elem => elem.id==this.selectedStudy.id)[0])
          this.loading = false
        } else {

          this.loading = false
        }

      })
    } else {
      this.siteForm.markAllAsTouched()
      this.loading = false
    }

  }
  updateStudyArchiveDetails(){
    this.archiveloading=true;
    if(this.recordForm.valid){
        const data = {
          studyId:this.selectedStudy.id,
          req: {
            estimatedDestructionDate:this.recordForm.controls.desctructionDate.value,
            destructionTerms:this.recordForm.controls.terms.value,
            destructionContact:this.recordForm.controls.contact.value
          }
        }
        this.archiveService.updateStudyArchive(data).subscribe(res => {
         this.getLists()
         
         this.archiveloading=false
        })
    } else{
      this.recordForm.markAllAsTouched();
    }
  }


  sendNote(form) {
    this.notesloading = true;
    if(this.noteForm.valid ) {
      const data = {
        studyId:this.selectedStudy.id,
        req: {
          note:this.noteForm.controls.note.value
        }
      }
    
      this.archiveService.createStudyNote(data).subscribe(res => {
        this.getNotes(this.selectedStudy.id)
        this.getAudits(this.selectedStudy.id)
        this.noteForm.reset('')
      }, err => {
        this.notesloading = false;
      });
    } else {
      this.notesloading = false;
      this.noteForm.markAllAsTouched();
    }
  
  }

 
}
