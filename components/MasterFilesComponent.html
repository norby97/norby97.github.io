    ngOnInit() {

        this.sites = new UntypedFormGroup({
            site: new UntypedFormControl('')
        });
        this.companyID = JSON.parse(localStorage.getItem('siteCentric.loginUser')).company?.id;
        this.userService.getUserByID(JSON.parse(localStorage.getItem('siteCentric.loginUser')).id).subscribe(res => {
            this.user = res;
        });
        this.treeStructure.push({
            id: 1, parent: '#', text: 'Please select a research site.',
            icon: 'fa fa-folder', state: { opened: true }
        });
        this.numbers = Array(50).fill(0).map((x, i) => i);

        setTimeout(() => {
            $('#jstree1').jstree({
                core: {
                    check_callback: (operation, node, nodeParent) => {
                        if (operation === 'move_node') {
                            return node.type === 'file' && nodeParent.type === 'folder';
                        }
                    },
                    data: this.treeStructure
                },
                plugins: ['search', 'massload', 'types', 'dnd', 'sort'],
                types: {
                    folder: {
                        icon: 'fa fa-folder'
                    },
                    file: {
                        icon: 'fa fa-file'
                    },
                    placeholder: {
                        icon: 'fa fa-upload'
                    }
                },
                sort : function (a, b)   {
                    const a1 = this.get_node(a);
                    const b1 = this.get_node(b);
                    if(a1.parent != 1 && b1.parent != 1 ){
                        return (a1.uploadDate > b1.uploadDate) ? 1 : -1;
                        }
                    }
            }).bind('move_node.jstree', (e, data) => {
                const request = {
                    fileId: data.node.id,
                    siteId: this.siteID,
                    parentId: data.parent
                };
                this.masterService.moveFile(request).subscribe(res => {

                });
            }).bind('select_node.jstree', (e, data) => {
                this.validationIconClicked = false;
                this.actualNode = data;
                this.actualFileID = data.node.id;
                if (data.node.type === 'file') {
                    this.loading = true;
                    this.actualFilePreviousVersions = data.node.original.metadata.previousVersions;
                    this.ref.detectChanges();
                    this.internalValidation = new InternalValidation(data.node.original.metadata.intValidated,
                        data.node.original.metadata.currentHash, data.node.original.metadata.uploadedHash);
                    this.masterService.linkedStudies({
                        fileID: this.actualFileID,
                        siteID: this.siteID
                    }).subscribe(res => {
                        this.linkedStudies = res;
                    });
                    this.masterService.retrieveFileDetails({
                        fileID: this.actualFileID,
                        siteID: this.siteID,
                    }).subscribe(res => {
                        this.actualFileDetails = res;
                        this.actualFilePreviousVersions = res.previousVersions;
                    });
                    this.actualFileName = data.node.text;
                    this.folderSelected = false;
                    this.previousFileID = null;
                    this.previousFileName = null;
                    this.retrieveFile(data.node.id);
                } else {
                    this.folderSelected = true;
                }
            });
            this.initDone = true;
            this.studiesloading =true;
            this.siteService.getAll2(this.companyID).subscribe(res => {
                localStorage.setItem('siteCentric.sites', JSON.stringify(res));

                this.assignedSiteNumber = 0;
                this.siteSelected = true;
                this.companySites = res;
                this.assignedSite = res[0];
                this.siteID = this.assignedSite.id;
                this.sites.get('site').setValue(this.assignedSite.id + '_' + this.assignedSite.name);
                this.renderSiteFolders({ value: this.assignedSite.id + '_' + this.assignedSite.name } );
                this.studiesloading =false;

            });
        }, 200);
    }

    renderSiteFolders(event) {
        this.siteSelected = true;
        this.siteName = event.value.split('_')[1];
        this.siteID = event.value.split('_')[0];
        const data = {
            siteID: event.value.split('_')[0]
        };
        this.masterService.retrieveSiteFolders(data).subscribe(res => {
            this.treeStructure = [];
            this.justFolders = [];
            this.treeStructure.push({ id: 1, parent: '#', text: event.value.split('_')[1], type: 'folder', state: { opened: true } });
            Object.keys(res.filePathsByPath).map(key => {
                this.folders.push(res.filePathsByPath[key]);
                res.filePathsByPath[key].forEach(element => {
                    if (element.fileType === 'FOLDER') {
                        this.justFolders.push(element);
                    }
                    if (element.parent === null) {
                        if (element.fileType === 'FOLDER') {
                            this.treeStructure.push({
                                a_attr: {title: element.name},
                                id: element.id, parent: 1, text: element.name + ' (' +  this.getNumberOfChildrens(element, res.filePathsByPath)+ ')',
                                type: 'folder', metadata: element.path
                            });
                        } else if (element.fileType === 'FILE') {
                            this.treeStructure.push({
                                id: element.id, parent: 1,
                                a_attr: {title: element.name},
                                text: element.name, type: 'file', metadata: {
                                    previousVersions: element.previousVersions, path: element.path, intValidated: element.internallyValidated,
                                    currentHash: element.currentHash, uploadedHash: element.uploadedHash, uploadDate: element.createdOn
                                }
                            });
                        }
                    } else {
                        if (element.fileType === 'FOLDER') {
                            this.treeStructure.push({
                                a_attr: {title: element.name},
                                id: element.id, parent: element.parent.id,
                                text: element.name + ' (' +  this.getNumberOfChildrens(element, res.filePathsByPath)+ ')', type: 'folder', metadata: element.path
                            });
                        } else if (element.fileType === 'FILE') {
                            this.treeStructure.push({
                                a_attr: {title: element.name},
                                id: element.id, parent: element.parent.id,
                                text: element.name, type: 'file', metadata: {
                                    previousVersions: element.previousVersions, path: element.path, intValidated: element.internallyValidated,
                                    currentHash: element.currentHash, uploadedHash: element.uploadedHash, uploadDate: element.createdOn
                                }
                            });
                        }
                    }
                });
                $('#jstree1').jstree(true).settings.core.data = this.treeStructure;
                $('#jstree1').jstree(true).refresh();
                
            });
        });
    }

    getNumberOfChildrens(element, files:any ) {
        let count = 0;
        Object.keys(files).map(key => {
            files[key].forEach(elem => {
            if(elem?.parent?.id === element.id  && elem.fileType==='FILE') count++;
            })
        });
        return count;
    }

    retrieveFile(id) {
        const data = {
            fileID: id,
            siteID: this.siteID,

        this.initAuditTableService.next({companyId: this.authService.loginUserValue.company.id, studyId: '', siteId: this.siteID, filePathId: id, userId: '' });
        const file = this.masterService.retrieveFile(data);
        const fileDetails = this.masterService.retrieveFileDetails(data);
        forkJoin([file, fileDetails]).subscribe(res => {
            this.url = res[0];
            this.actualFile = res[0];
            this.actualFileDetails = res[1];
            this.previousSelect = this.actualFileDetails;
            this.actualFileName = this.actualFileDetails.name

                this.firstVersion = this.actualFileDetails;
            }

            this.loading = false
        }, err => {
            this.loading = false;
        });
    }

    printFile() {
        if (this.url === null) {
            this.studyService.showError('Please select a PDF first.');
        } else {
            const base64 = btoa(
                new Uint8Array(this.url)
                    .reduce((data, byte) => data + String.fromCharCode(byte), '')
            );

            printJS({ printable: base64, type: 'pdf', showModal: false, base64: true });
        }
    }

    downloadFile() {
        if (this.actualFile === null) {
            this.studyService.showError('Please select a PDF first.');
        } else {
            const blob = new Blob([this.actualFile], { type: 'application/pdf' });
            saveAs(blob, this.actualFileName);
        }
    }

                }

        }

      isFileAllowed(fileName: string) {
        let isFileAllowed = false;
        const allowedFiles = ['.pdf'];
        const regex = /(?:\.([^.]+))?$/;
        const extension = regex.exec(fileName);
        if (undefined !== extension && null !== extension) {
          for (const ext of allowedFiles) {
            if (ext === extension[0]) {
              isFileAllowed = true;
            }
          }
        }
        return isFileAllowed;
      }
