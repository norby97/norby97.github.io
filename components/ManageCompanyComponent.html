    calculateInterval( dateFrom: string, rowNumber) {
        if(rowNumber === 0) return false;
        else {  
            return new Date(dateFrom).getDate() -  new Date(this.calendarElements[rowNumber-1].dateTo).getDate() === 1 || new Date(dateFrom).getDate() -  new Date(this.calendarElements[rowNumber-1].dateTo).getDate() === -29 || new Date(dateFrom).getDate() -  new Date(this.calendarElements[rowNumber-1].dateTo).getDate() === -30 || new Date(dateFrom).getDate() -  new Date(this.calendarElements[rowNumber-1].dateTo).getDate() === -27 ? false : true;}
    }

        if(this.calendarElements.length === 1) {
            this.actualMonth = 1;
            this.actualWeek = 1;
            this.calendarElements.splice(this.calendarElements.length-1,1);

        if(this.calendarElements[this.calendarElements.length-1].mt === this.actualMonth && this.calendarElements[this.calendarElements.length-1].wk === 1 ) {
            this.calendarElements.splice(this.calendarElements.length-1,1);
            this.actualMonth--;
            this.actualWeek = this.calendarElements[this.calendarElements.length-1].wk +1;
        }
        if(this.calendarElements[this.calendarElements.length-1].mt === this.actualMonth && this.calendarElements[this.calendarElements.length-1].wk !== 1 ) {
            this.calendarElements.splice(this.calendarElements.length-1,1);

        }
        
    }
