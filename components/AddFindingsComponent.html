  ngOnInit(): void {
    this.findingForm = new UntypedFormGroup({
      siteId:new UntypedFormControl('',Validators.required),
      studyId:new UntypedFormControl(''),
      trackedVisitId:new UntypedFormControl(''),
      patientStudyLinkId:new UntypedFormControl(''),
      assignToId:new UntypedFormControl(''),

      findingNotes:new UntypedFormControl('',Validators.required)
    })
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.findingForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.findingForm.controls.siteId.updateValueAndValidity();
      this.studiesloading = true;
      this.searcher2('',1,30)
        this.getUsers({value:this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id)})

    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.findingForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.findingForm.controls.siteId.updateValueAndValidity();

    }
    this.findingcategoriesloading = true;
    this.sccipService.getFindingCategories({active:true}).subscribe(res => {
      this.findingcategories = res;
      this.findingcategoriescopy = res;

      this.findingcategoriesloading = false;

    });
    this.loadFindings();

  getUsers(event) {
    this.userPreferences.setuserState('selectedSite',this.findingForm.controls.siteId.value)
    this.usersloading = true;
    this.userService.getAll2({siteId:event.value.id}).subscribe(res => {
      res = res.sort((a, b) => a.investigator || b.investigator ? 1 : -1).reverse();
      this.users = res;
      this.usercopy = res;
      this.usersloading = false
    });
  }

  
  getPatients(){
    this.patientsloading = true;
    this.trackedVisitsService.getPatients(this.findingForm.controls.studyId.value.id).subscribe(res => {
      this.patients = res;
      this.patientscopy = res;
      this.patientsloading = false;
    } ,err => {
      this.patientsloading = false;

    })
  }

  getVisits() {
    this.visitsloading = true;
  
    this.trackingService.findAllVisitsForPatient(this.findingForm.controls.patientStudyLinkId.value.id).subscribe(res => {
      this.visits = [];
      this.visitscopy = [];
      res.forEach(elem => {
        if(elem.trackedVisitId != null) {
          this.visits.push(elem);
          this.visitscopy.push(elem)
        }
      })
      
      this.visitsloading = false;

    })
  }

  searcher3 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {

  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    this.userPreferences.setuserState('selectedSite',this.findingForm.controls.siteId.value)
    if(this.lastsearch2 != search || pageNumber ==1) {
      this.actualpage = 0
    }
    this.lastsearch2 = search
    const data = {
      siteId: this.findingForm.controls.siteId.value.id,
      trialType: '',
      dateFrom: '',
      dateTo:'',
      search,
      pageNumber: this.actualpage,
      pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.advertisingService.getStudiesLastActivityPaginated(data).subscribe(res => {
        res.forEach(element => {
            this.studies.push(element);
        });
      })
    } else {
      this.advertisingService.getStudiesLastActivityPaginated(data).subscribe(res => {
        this.studies = res;
      })
    }
    this.studiesloading = false;
    this.actualpage++
    return of(this.studies)
  }

  submit() {


    this.loading =true;
    if(this.findingForm.valid) {
      const data = { 
        siteId: this.findingForm.controls.siteId.value.id,
        studyId:this.findingForm.controls.studyId?.value =='' ?'' :this.findingForm.controls.studyId?.value.id,
        patientStudyLinkId:this.findingForm.controls.patientStudyLinkId.value =='' ?'' :this.findingForm.controls.patientStudyLinkId?.value.id,
        trackedVisitId:this.findingForm.controls.trackedVisitId.value =='' ?'' :this.findingForm.controls.trackedVisitId?.value.trackedVisitId,
        assignToId:this.findingForm.controls.assignToId?.value=='' ? '' :this.findingForm.controls.assignToId?.value.id,
        findingCategoryId:this.findingForm.controls.findingCategoryId.value =='' ?'' :this.findingForm.controls.findingCategoryId?.value.id,
        findingNotes: this.findingForm.controls.findingNotes.value,
      }
      this.findingService.addFinding(data).subscribe(res =>{
        this.findingForm.controls.findingNotes.reset('')
        this.loading = false;
        this.loadFindings()
      }, err => {
        this.loading = false;
      })
    } else {
      this.loading =false;
      this.findingForm.markAllAsTouched();
    }
  }

      findingCategoryId:this.findingForm.controls.assignToId?.value=='' &&  this.findingForm.controls.findingCategoryId.value =='' ?'' :this.findingForm.controls.findingCategoryId?.value == '' ? '':this.findingForm.controls.findingCategoryId?.value.id,

  calculateDiff(startDate, endDate){
    const oneDay = 24 * 60 * 60 * 1000; 
    return Math.round(Math.abs((new Date(startDate).getTime() - new Date(endDate).getTime()) / oneDay));
  }
}
