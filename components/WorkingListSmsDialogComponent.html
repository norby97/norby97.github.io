import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { MatDialog, MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { AppointmentService } from 'src/app/services/ctms/appointment.service';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';
import { SmsService } from 'src/app/services/administration/sms.service';
import { PatientProfileSmsDialogComponent } from '../patient-profile-sms-dialog/patient-profile-sms-dialog.component';
import { PatientsService } from 'src/app/services/ctms/patients.service';
import { BulkSmsDialogComponent } from '../../utils/bulk-sms-dialog/bulk-sms-dialog.component';

@Component({
  selector: 'app-working-list-sms-dialog',
  templateUrl: './working-list-sms-dialog.component.html',
  styleUrls: ['./working-list-sms-dialog.component.scss']
})
export class WorkingListSmsDialogComponent implements OnInit {
  patientSMS;
  loading=false;
  dialogloading=false;

  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  } 
  constructor(public dialogRef: MatDialogRef<WorkingListSmsDialogComponent>,
    private dialog:MatDialog,
    @Inject(MAT_DIALOG_DATA) public data: any, 
    private smsService: SmsService,
    private patientService: PatientsService) { }
  ngOnInit(): void {
  
  
  
    this.loading = true;
    this.smsService.getAppointmentSMSs({patientId: this.data?.patientID}).subscribe(res => {
      this.patientSMS =res
      this.patientSMS.forEach(elem => {
        elem.date = elem?.date?.replace('T',' ') + ' UTC'
        elem.date = elem?.date?.toString()
      })
      this.loading = false;

    })

  }
  openSMSDialog() {
    this.dialogloading = true;
    this.patientService.findPatientByID(this.data.patientID).subscribe(res=> {
      const ref = this.dialog.open(BulkSmsDialogComponent, {
        width: 'auto', 
      panelClass:'custom-dialog-container',
      maxHeight:'80vh',
      minWidth:'60vw',
        height:'auto',
        disableClose: true,
        data: {
          followup: false,
          patient:res,
          patientChecks:[true],
          workinglistId:this.data.workinglistId,
          selectedPatients:[this.data.patientID ],
          allSelected: this.data.allSelected,
          excludedPatients:this.data.excludedPatients,
          siteID: this.data?.siteID.id,
          patientIds: [this.data.patientID]
        }
      });
    ;
            this.dialogloading = false;

      ref.afterClosed().subscribe(res => {
        if(res) {
          this.dialogloading = false;
          this.dialogRef.close(true)
        }
      })
    })
    
  }
}
