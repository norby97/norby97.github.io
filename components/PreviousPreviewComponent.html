import { ChangeDetectorRef, Component, Inject, OnInit } from '@angular/core';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { ActivatedRoute } from '@angular/router';
import { forkJoin } from 'rxjs';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { FileService } from 'src/app/services/regulatory/file.service';
import { StudyService } from 'src/app/services/regulatory/study.service';
import { InitAuditTableService } from '../../administration/audit/audit-view-table.component/init-audit-table-service';
import { saveAs } from 'file-saver';
import * as  printJS from 'print-js';
import { MasterFilesService } from 'src/app/services/regulatory/master-files.service';

@Component({
  selector: 'app-previous-preview',
  templateUrl: './previous-preview.component.html',
  styleUrls: ['./previous-preview.component.scss']
})
export class PreviousPreviewComponent implements OnInit {
  actualFile = null;
  initDone = false;
  actualFileDetails = null;
  url = null;
  pdfDoc = null;
  validationIconClicked = false;
  studyName = '';
  folderSelected = false;
  openFile = false;
  openedFileID = null;
  treeStructure = [];
  actualFileTasks = null;
  actualFileOpenTasks;
  actualFileDoneTasks;
  actualFileNewTasks;
  actualFilePreviousVersions = null;
  reviewTreeStructure = [];
  reviewedFilesID = [];
  actualFileID;


  constructor(private dialog: MatDialog,
    public dialogRef: MatDialogRef<PreviousPreviewComponent>,
    private route: ActivatedRoute,
    private fileService: FileService,
    private ref: ChangeDetectorRef,
    private studyService: StudyService,
    public authService: AuthenticationService,
    public initAuditTableService: InitAuditTableService,
    private masterService: MasterFilesService,
    @Inject(MAT_DIALOG_DATA) public data: any) { 
    }

  ngOnInit(): void {
    this.retrieveFile(this.data.fileID);
  }

  backToTable() {
    this.dialogRef.close(false);
  }

  retrieveFile(id, withrender = true) {
    if (this.data.master) {
      const data = {
        fileID: id,
        siteID: this.data.siteID,

      };
      this.initAuditTableService.next({companyId: this.authService.loginUserValue.company.id, studyId: this.data.studyID, siteId: '', filePathId: id, userId: '' });
      const file = this.masterService.retrieveFile(data);
      const fileDetails = this.masterService.retrieveFileDetails(data);
      forkJoin([file, fileDetails]).subscribe(res => {
        const blob = new Blob([res[0]], { type: 'application/pdf' });

        this.url = res[0];
        this.actualFile = blob;
        this.actualFileDetails = res[1];
      });
    } else {
      const data = {
        fileID: id,
        studyID: this.data.studyID,
      };
      this.initAuditTableService.next({companyId: this.authService.loginUserValue.company.id, studyId: this.data.studyID, siteId: '', filePathId: id, userId: '' });
      const file = this.fileService.retrieveFile(data);
      const fileDetails = this.fileService.retrieveFileDetails(data);
      forkJoin([file, fileDetails]).subscribe(res => {
        const blob = new Blob([res[0]], { type: 'application/pdf' });

        this.url = res[0];
        this.actualFile = blob;
        this.actualFileDetails = res[1];
      });
    }

  }

  printFile() {
    if (this.url === null) {
      this.studyService.showError('Please select a PDF first.');
    } else {
      const base64 = btoa(new Uint8Array(this.url).reduce((data, byte) => data + String.fromCharCode(byte), ''));
      printJS({ printable: base64, type: 'pdf', showModal: false, base64: true });
    }
  }

  downloadFile() {
    if (this.actualFile === null) {
      this.studyService.showError('Please select a PDF first.');
    } else {
      const blob = new Blob([this.actualFile], { type: 'application/pdf' });
      saveAs(blob, this.data.actualFileName);
    }
  }
}
