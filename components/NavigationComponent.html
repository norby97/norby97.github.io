    clearLedger() {

   async startCounting() {
        const tasknr = this.taskService.getTaskNumber();
        const filednr = this.toBeFiledService.getEmailNumber();
        // const appnr = this.appointmentService.getDiscrepenciesNumber();
       setInterval( () => {
       
        if(this.authService.loginUserValue && this.router.url.toString().includes('regulatory')) {
            forkJoin([tasknr, filednr]).subscribe(res => {
                this.taskCount = res[0];
                this.emailCount = res[1];
                // this.dicrepanciesCount = res[2];
            });
        }
       }, 60000)
    }

    openChangePasswordDialog() {
       this.dialog.open(ProfileUpdatePasswordDialogComponent, {
        minWidth: 'auto',
        data:{
            hideclose:false,
        },
        maxHeight:'80vh',
        maxWidth:'80vw',
        width: 'auto',
        height:'auto',
        disableClose: true,
        panelClass:'custom-dialog-container'
        });
    }

    openInquiryCallDialog() {
        this.dialog.open(InquiryCallDialogComponent, {
            minWidth: '40vw',
            maxHeight:'80vh',
            maxWidth:'80vw',
            width: 'auto',
            height:'auto',
            disableClose: true,
            panelClass:'custom-dialog-container'
        });
    }

    openLogOutDialog() {
        const ref = this.dialog.open(FeedBackDialogComponent, {
            data: {
              message: 'Are you sure you want logout?'
            }
          })
          ref.afterClosed().subscribe(res => {
            if(res) {
              this.authService.logout()
            }
          })
    }

    getModulePermission(name) {
        this.authService.loginUserValue.modules[7].permissions.forEach(elem =>{
            if(elem.name == name) {
                if(elem.permission == 'NONE') return false;
                return true
            }
        })
        return false
    }
}
