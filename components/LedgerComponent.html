 }
  calcSums() {
    this.ledgerStats = {
      totalInv: 0,
      pastDue: 0,
      invoicedToSponsor:0,
      totalPayments: 0,
      currentBalance: 0,
      studyBalance: 0
    }
    this.invoices?.ledgerViews?.map(elem =>{
      if(elem.type == "Payment") {
        elem.payment = elem.balance
        this.ledgerStats.totalPayments += elem.payment;
        this.ledgerStats.currentBalance -= elem.balance;
        elem.currentBalance = this.ledgerStats.currentBalance;
      } else {
        elem.charge = elem.balance
        this.ledgerStats.totalInv += elem.charge;
        if(elem.type !='Tracked Visits' && this.utilService.checkIfTodayOrPast(elem.dueDate) && elem.dueDate) {
          this.ledgerStats.pastDue = this.add(this.ledgerStats.pastDue, elem.charge)
        }
        if(elem.type !='Tracked Visits') {
          this.ledgerStats.invoicedToSponsor = this.add(this.ledgerStats.invoicedToSponsor, elem.charge)
        }
        this.ledgerStats.currentBalance+= elem.balance;
        elem.currentBalance = this.ledgerStats.currentBalance;
        
      }
      
      
    })
    this.ledgerStats.studyBalance = this.ledgerStats.totalInv - this.ledgerStats.totalPayments;
  }
  add(y, z) {
    return  +y + +z;
  }
  getInvoices() {
    
      this.userPreferences.setuserState('selectedSite',this.ledgerForm.controls.site.value)

    if(this.ledgerForm.valid) {
      this.searched = false;
      this.loading = true;
      const data = {
        studyId: this.ledgerForm.controls.study.value,
        dateFrom:this.ledgerForm.controls.from.value ? formatDate(this.ledgerForm.controls.from.value, 'yyyy-MM-dd', 'en-Us') : '',
        dateTo: this.ledgerForm.controls.from.value ? formatDate(this.ledgerForm.controls.to.value, 'yyyy-MM-dd', 'en-Us') : '',
      }
      this.ledgerService.getInvoices(data).subscribe(res => {
        this.invoices = res;
        this.calcSums();
        this.loading = false;
        this.searched = true;

      }, err => {
        this.loading = false;
      })
    } else {
      this.ledgerForm.markAllAsTouched()
    }
  
  }

  setStudyName(event) {
    this.studyName = this.studies.filter(elem => elem?.id == event?.value)[0]?.name
  }



  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    this.userPreferences.setuserState('selectedSite',this.ledgerForm.controls.site.value)

    const data = {
      dateFrom:this.ledgerForm.controls.from.value ? formatDate(this.ledgerForm.controls.from.value, 'yyyy-MM-dd', 'en-Us'): '',
      dateTo:this.ledgerForm.controls.to.value ? formatDate(this.ledgerForm.controls.to.value, 'yyyy-MM-dd', 'en-Us') : '',
      siteId: this.ledgerForm.controls.site.value.id,
      search,
      pageNumber: pageNumber-1,
      pageSize,
      statuses: ['MOVING_FORWARD','ADD_ON','ACTIVE']
    }
    if(pageNumber - 1 != 0) {
      this.siteService.getStudies(data).subscribe(res => {
        this.studies = [...this.studies, ...res.content];
      })
    } else {
      this.studiesloading =true;
      this.siteService.getStudies(data).subscribe(res => {
        this.studies = [...res.content];
        this.studiesloading =false;
      })
    }
    return of(this.studies)
  }
  

}
