export class MyQaFindingsComponent implements OnInit {

  constructor(private siteService: SiteService,
    private utilService: UtilService,
    private router:Router,
    private sccipService:SccipService,

  ngOnInit(): void {
    this.findingForm = new UntypedFormGroup({
      siteId:new UntypedFormControl('',Validators.required),
      studyId:new UntypedFormControl('',),
      trackedVisitId:new UntypedFormControl('',),
      patientStudyLinkId:new UntypedFormControl('',),
      assignToId:new UntypedFormControl(''),
      findingCategoryId:new UntypedFormControl(''),
      findingNotes:new UntypedFormControl('')
    })
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.findingForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.findingForm.controls.siteId.updateValueAndValidity();
      this.searcher2('',1,30);

    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.findingForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.findingForm.controls.siteId.updateValueAndValidity();
        this.searcher2('',1,30);

      });
    }
    this.findingcategoriesloading = true;
    this.sccipService.getFindingCategories({active:true}).subscribe(res => {
      this.findingcategories = res;
      this.findingcategoriescopy = res;

      this.findingcategoriesloading = false;

    })
    this.loadFindings();
  }

  searcher7 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    this.userPreferences.setuserState('selectedSite',this.findingForm.controls.siteId.value)
    let temp = []
    if(search == '') {
        this.findingcategories = this.findingcategoriescopy;
    } else {
        this.findingcategoriescopy.map(elem => {
            if(elem?.name?.toLowerCase().indexOf(search.toLowerCase()) > -1) {
                temp.push(elem);
            }
        });
        this.findingcategories = temp;
    }
    
    return of(this.findingcategories)
  }


  searcher6 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    this.userPreferences.setuserState('selectedSite',this.findingForm.controls.siteId.value)
    let temp = []
    if(search == '') {
        this.categories = this.categoriescopy;
    } else {
        this.categoriescopy.map(elem => {
            if(elem?.name?.toLowerCase().indexOf(search.toLowerCase()) > -1) {
                temp.push(elem);
            }
        });
        this.categories = temp;
    }
    
    return of(this.categories)
  }
  
  getPatients(){
    this.patientsloading = true;
    this.findingService.getQAPatients(this.findingForm.controls.studyId.value.id).subscribe(res => {
      this.patients = res;
      this.patientscopy = res;
      this.patientsloading = false;
    } ,err => {
      this.patientsloading = false;

    })
  }

  getVisits() {
    this.visitsloading = true;
    this.findingService.getQAVisits(this.findingForm.controls.patientStudyLinkId.value.id).subscribe(res => {
      this.visits = res;
      this.visitscopy = res;
      this.visitsloading = false;

    })
  }

  searcher4 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    this.userPreferences.setuserState('selectedSite',this.findingForm.controls.siteId.value)
    let temp = []
    if(search == '') {
        this.visits = this.visitscopy;
    } else {
        this.visitscopy.map(elem => {
            if(elem?.title?.toLowerCase().indexOf(search.toLowerCase()) > -1) {
                temp.push(elem);
            }
        });
        this.visits = temp;
    }
    
    return of(this.visits)
  }


  searcher3 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    this.userPreferences.setuserState('selectedSite',this.findingForm.controls.siteId.value)
    let temp = []
    if(search == '') {
        this.patients = this.patientscopy;
    } else {
        this.patientscopy.map(elem => {
            if(elem?.alias?.toLowerCase().indexOf(search.toLowerCase()) > -1  || elem?.subjectNumber?.toLowerCase().indexOf(search.toLowerCase()) > -1) {
                temp.push(elem);
            }
        });
        this.patients = temp;
    }
    
    return of(this.patients)
  }

  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    this.userPreferences.setuserState('selectedSite',this.findingForm.controls.siteId.value)
    if(this.lastsearch != search || pageNumber == 1) {
      this.studypageNumber = 0
    }
    if(search =='') {
      this.findingForm.controls.studyId.reset('')
      this.findingForm.controls.studyId.updateValueAndValidity();
    }
    this.lastsearch = search
 
   
    this.studiesloading = true;

     if(this.studypageNumber != 0) {
      this.findingService.getQAStudies(this.findingForm.controls.siteId.value.id).subscribe(res => {
        res.forEach(element => {
          this.studies.push(element);
      });
      this.studiesloading = false;
      })
    } else {
      this.findingService.getQAStudies(this.findingForm.controls.siteId.value.id).subscribe(res => {
        this.studies =[...res]
        this.studiesloading = false;
      })
    }
   
    this.studypageNumber++;
    return of(this.studies)
  }
  getFindingNotes(finding,i){
    this.findingService.getFindingResponses(finding.id).subscribe(res=> {
      this.existingresponses[i]=res
      this.existingresponses[i].notes?.map(elem => {
        elem.createdDate = elem.createdDate.replace('T',' ') + ' UTC'
        elem.createdDate = elem.createdDate.toString()
      })
    })
  }
  loadFindings(){
  
    if(this.findingForm.valid) {
      this.responses = [];
      this.existingresponses = [];
      this.findingsloading = true;
      this.patientFindings = [];
      const data = {
        siteId: this.findingForm.controls.siteId.value.id,
        studyId:this.findingForm.controls.studyId.value == '' ? '' : this.findingForm.controls.studyId.value.id,
        patientStudyLinkId:this.findingForm.controls.patientStudyLinkId.value == '' ? '' : this.findingForm.controls.patientStudyLinkId.value.id,
        trackedVisitId:this.findingForm.controls.trackedVisitId.value == '' ? '' : this.findingForm.controls.trackedVisitId.value.id,
        assignToId:this.authService.loginUserValue.id,
        findingCategoryId:this.findingForm.controls.findingCategoryId.value == '' ? '' : this.findingForm.controls.findingCategoryId.value.id,
        dateFrom: '',
        dateTo: '',
        statuses:['OPEN','RESPONDED']
      }
      this.showOpen ? data.statuses.push('OPEN') : null;
      this.showResponded ? data.statuses.push('RESPONDED') : null;
        this.findingService.getFindings(data).subscribe(res => {
          this.patientFindings = res.sort((a, b) => {
           return this.compare(a.dueDate?.toLowerCase() , b.dueDate?.toLowerCase(), true);
            })

          this.patientFindings.forEach((elem,i) =>{
            this.existingresponses.push('');

              this.findingService.getFindingResponses(elem.id).subscribe(res => {
                this.existingresponses[i] = res;
                this.existingresponses[i].notes?.map(elem => {
                  elem.createdDate = elem.createdDate.replace('T',' ') + ' UTC'
                  elem.createdDate = elem.createdDate.toString()
                })
              })
              
              this.responses.push('');
          })
          this.findingsloading = false;
        },err => {
          this.findingsloading = false;

        })

    } else {
      this.findingForm.markAllAsTouched();
    }
    
    
  }

  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
  calculateDiff(startDate, endDate){
    const oneDay = 24 * 60 * 60 * 1000; 
    return Math.round(Math.abs((new Date(startDate).getTime() - new Date(endDate).getTime()) / oneDay));
  }

  moveToOpen(finding) {
    const ref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: `Are you sure you want to move to open this finding?`
      }
    })
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.findingService.moveToOpen({qaFindingId:finding.id}).subscribe(res=> {
          this.loadFindings();
        })
      }
    })
    
  }

  submit(finding,index){
  
  
    const data = {
      qaFindingId:finding.id,
      req:{
        note:this.responses[index]
      }
    }
    this.findingService.submitMyResponse(data).subscribe(res => {
      this.loadFindings();
    })
  }
}
