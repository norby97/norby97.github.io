import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormGroup, UntypedFormBuilder, Validators, UntypedFormControl } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialog, MatDialogRef } from '@angular/material/dialog';
import { startWith, map } from 'rxjs/operators';
import { ContactService } from 'src/app/services/crm_bd/contact.service';
import { SccipService } from 'src/app/services/crm_bd/sccip.service';
import { StudyStartsService } from 'src/app/services/crm_bd/study-starts.service';
import { ContactAddForStudyComponent } from '../contact-add-for-study/contact-add-for-study.component';

@Component({
  selector: 'app-contact-search-results-for-studies',
  templateUrl: './contact-search-results-for-studies.component.html',
  styleUrls: ['./contact-search-results-for-studies.component.scss']
})
export class ContactSearchResultsForStudiesComponent implements OnInit {
  resultsStep = false;
  searchStep = true;
  roles = ['','','','','','','','','','',''].fill('');
  studyRoleForm: UntypedFormGroup;
  findContactForm: UntypedFormGroup;
  searchedFName = '';
  searchedLName = '';
  contacts;
  filteredRoles;
  loading = false;
  contactID = null;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(@Inject(MAT_DIALOG_DATA) public data: any,
              private dialog: MatDialog,
              private contactService: ContactService,
              private sccip: SccipService,
              private studyStartsService: StudyStartsService,
              public dialogRef: MatDialogRef<ContactSearchResultsForStudiesComponent>,
              private formBuilder: UntypedFormBuilder,
              ) { }

  ngOnInit(): void {
    this.sccip.getStudyRoles().subscribe(res => {
      this.roles = res;
    });
    this.findContactForm = new UntypedFormGroup({
      firstName: new UntypedFormControl('', [Validators.required]),
      lastName: new UntypedFormControl('', [Validators.required]),
    });
    this.studyRoleForm =this.formBuilder.group({
      role: this.formBuilder.group({ name: ['', Validators.required]}),
    });
    this.filteredRoles = this.studyRoleForm.get('role.name').valueChanges.pipe(
      startWith(''),
      map(value => this._filter(value, this.roles))
    );
  }
  
  private _filter(value: string, items: Array<any>) {
    const filterValue = value?.toLowerCase();
    return items?.filter(item => {
        return item.name?.toLowerCase().indexOf(filterValue) === 0;
    });
  } 
  
  openAddNewForLead() {
    var ref = this.dialog.open(ContactAddForStudyComponent, {
      data: {
        siteID: this.data.siteID,
        contactID: this.contactID,
        firstName:this.searchedFName,
        lastName: this.searchedLName,
        studyStartID: this.data.studyStartID,
      },
      width: 'auto',
      height: 'auto',
      disableClose: true,
      maxHeight:'80vh',
      maxWidth:'80vw',
      panelClass:'custom-dialog-container',
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.dialogRef.close(res);
      }
    })
  }

  nextStep(contact: any, role: any) {
    this.loading= true;

    this.contactID = contact.id;
    if(this.studyRoleForm.valid) {
      const data = {
        siteID: this.data.siteID,
        contactID: this.contactID,
        studyStartID: this.data.studyStartID,
        request: {
          name: role
        }
      }
      if(this.data?.scorecard) {
        this.studyStartsService.addScoreCardContact(data).subscribe(res => {
          this.loading= false;
          this.dialogRef.close(true);
        })
      } else {
        this.studyStartsService.addContact(data).subscribe(res => {

          this.loading= false;
          this.dialogRef.close(true);
        })
      }
    } else {
      this.loading= false;
      this.studyRoleForm.markAllAsTouched();
    }
  }

  assignUser(form:any) {
    if(this.studyRoleForm.valid) {
      const data = {
        siteID: this.data.siteID,
        contactID: this.contactID,
        studyStartID: this.data.studyStartID,
        request: {
          name: this.studyRoleForm.get('role.name').value
        }
      }
      if(this.data?.scorecard) {
        this.studyStartsService.addScoreCardContact(data).subscribe(res => {
          this.dialogRef.close(true);
        })
      } else {
        this.studyStartsService.addContact(data).subscribe(res => {
          this.dialogRef.close(true);
        })
      }
      
    } else {
      this.studyRoleForm.markAllAsTouched();
    }
    
  }
  
  findContact(form) {
    if(this.findContactForm.valid) {
      this.searchedFName = this.findContactForm.controls.firstName.value;
      this.searchedLName = this.findContactForm.controls.lastName.value;
      this.contactService.findContact(form.value).subscribe(res => {
        this.searchStep = false;
        this.contacts = res;
        this.resultsStep = true;
      });
    } else {
      this.findContactForm.markAllAsTouched();
    }
  }

}
