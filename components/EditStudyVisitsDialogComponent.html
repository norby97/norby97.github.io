import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { ScoreCardService } from 'src/app/services/ctms/score-card.service';
import { StudyBuildService } from 'src/app/services/finance/study-build.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { AddVersionDialogComponent } from '../add-version-dialog/add-version-dialog.component';

@Component({
  selector: 'app-edit-study-visits-dialog',
  templateUrl: './edit-study-visits-dialog.component.html',
  styleUrls: ['./edit-study-visits-dialog.component.scss']
})
export class EditStudyVisitsDialogComponent implements OnInit {
  visitsError = false;
  visitsFieldsError = false;
  editvisitRows=[];
  visitPayments=[]
  path = [];
  scores= [];
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(true);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor( @Inject(MAT_DIALOG_DATA) public data: any,
  private studyBuildService: StudyBuildService,
  private scoreCardService: ScoreCardService,
  public utilService:UtilService,
  public dialogRef: MatDialogRef<EditStudyVisitsDialogComponent>) { }

  ngOnInit(): void {

    this.scoreCardService.getStudyScores(this.data?.study).subscribe(res => {
      this.scores = res;
    })
   this.editvisitRows = this.data.visitsRow.map(obj => Object.assign({}, obj));
   this.editvisitRows.map(elem => elem.length = parseInt(elem.length));
   this.path = this.data.paths.map(obj => Object.assign({}, obj));
  }

  deleteVisitRowEdit(index) {
    this.editvisitRows.splice(index, 1);
  }


  addNewVisitRowEdit() {
    this.editvisitRows.push({
      name: '',
      type: '',
      path: {id:null},
      randomization :false,
      enrollmentBaselineVisit: false,
      sendSms: false,
      unscheduledVisit: false,
      countVisit: false,
      projectedVisit: false,
      canHappenScreenFail: false,
      method: 'NONE',
      length: 15,
      stipendCost: 0,
      phase: {id:null},
      daysFromPrevious: 0,
      windowAfter:0,
      windowBefore:0  
    });
  }

  changeMethodEdit(index, status) {
    if(this.editvisitRows[index].method == status) {
      this.editvisitRows[index].method = 'NONE'
    } else{
        this.editvisitRows[index].method = status;
    }
  }
  
  
  edit() {
    this.visitsError = false;
    this.visitsFieldsError = false;
    if(this.editvisitRows.length == 0) {
      this.visitsError = true;
    } else {
      let sources = [];
      let baselineCheckedVisits = [];
      this.editvisitRows.forEach(elem => {
        if(elem?.enrollmentBaselineVisit) {
          baselineCheckedVisits.push(elem?.path);
        } 
        if(elem?.name == '' || elem?.path.id == null || elem?.type == '' || elem?.phase=='') {
          this.visitsFieldsError = true;
        }
      });
        this.path.forEach(elem => {
          if(elem?.pathDepend.split(',')[1] == '') {
            sources.push({id: elem?.pathId, basecount: 0})
          }
        });

        baselineCheckedVisits.forEach(elem => {
          sources.forEach(elem2 => {
            if(elem2.id == this.getSource(elem)) {
              elem2.basecount++;
            }
          }) ;
        });
        sources.forEach(elem => {
          if(elem?.basecount < 1) {
            this.visitsError = true;
          }
        });

        if(!this.visitsError && !this.visitsFieldsError) {
          const data ={
            budgetVersionId: this.data.budgetVersionId,
            req: {
              visits:[]
            }
          }
          this.editvisitRows.forEach(elem => {
            data.req.visits.push({
              countVisit:elem?.countVisit,
              daysFromPrevious:elem?.daysFromPrevious,
              enrollmentBaselineVisit:elem?.enrollmentBaselineVisit,
              id:elem?.id ? elem?.id : null,
              length:elem?.length,
              masterStudyVisitId:elem?.masterStudyVisitId,
              method:elem?.method,
              name:elem?.name,
              order:elem?.order,
              invoiceable:elem?.invoiceable,
              pathId:elem?.path.id,
              canHappenScreenFail: elem?.canHappenScreenFail,
              phaseId:elem?.phase?.id ,
              projectedVisit:elem?.projectedVisit,
              randomization:elem?.randomization,
              sendSms:elem?.sendSms,
              stipendCost:elem?.stipendCost,
              type:elem?.type,
              unscheduledVisit:elem?.unscheduledVisit,
              windowAfter:elem?.windowAfter,
              windowBefore:elem?.windowBefore,
            })
          })
          this.studyBuildService.editStudyVisits(data).subscribe(res => {
            this.dialogRef.close(true);
          })
        }
    }
  }

  currentVisit(visitID) {
    let found =false;
    this.data.visitsRow.forEach(elem => {
      if(elem?.id == visitID) found =true;
    });

    return found;
  }

  getSource(path) {
    if(path.pathParent == 0) {
        return path.pathId;
    } else {
      let parent = null;
      this.editvisitRows.forEach(elem => {
        if(elem?.path?.pathId == path.pathParent) {
          parent = elem.path;
        }
      });
      return this.getSource(parent)
    }
  }
}
