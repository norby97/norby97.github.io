import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { CalendarEvent } from 'src/app/models/advertising/calendar-event.model';
import { MediaEventFileType } from 'src/app/models/advertising/media-event-file-type.model';
import { CalendarService } from 'src/app/services/advertising/calendar.service';
import { InvoicesService } from 'src/app/services/advertising/invoices.service';
import { MediaEventService } from 'src/app/services/advertising/media-event.service';
import { saveAs } from 'file-saver';
import { EditEventDialogComponent } from '../edit-event-dialog/edit-event-dialog.component';
import { PdfPreviewDialogComponent } from '../../../pdf-preview-dialog/pdf-preview-dialog.component';
import { FeedBackDialogComponent } from '../../utils/feed-back-dialog/feed-back-dialog.component';
import { ImagePreviewDialogComponent } from '../../utils/image-preview-dialog/image-preview-dialog.component';
import { SelectDialogComponent } from '../../utils/select-dialog/select-dialog.component';

@Component({
  selector: 'app-view-event-dialog',
  templateUrl: './view-event-dialog.component.html',
  styleUrls: ['./view-event-dialog.component.scss']
})
export class ViewEventDialogComponent implements OnInit {
  eventDetails!:  CalendarEvent
  eventFilesTypes=  Object.keys(MediaEventFileType);
  fileUploadForm: UntypedFormGroup;
  fileName = ''
  loading = false;
  addNoteForm: UntypedFormGroup
  eventFiles;
  eventNotes;
  eventInvoices;
  invoicedToSponsor;
  fileToUpload =null;
  actualDetails;
  detailsloading = true;
  reportView = false
  landingPage = false;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(@Inject(MAT_DIALOG_DATA) public data: any,
              private caledarService: CalendarService,
              private invoiceService: InvoicesService,
              private dialog: MatDialog,
              private mediaEventsService: MediaEventService,
              public dialogRef: MatDialogRef<ViewEventDialogComponent>) { }
  ngOnInit(): void {
    
    
    this.refreshFiles();
    this.refreshNotes();
    this.mediaEventsService.getMediaEventInvoices({adCalendarId:this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {
      this.eventInvoices = res;
    })
    this.mediaEventsService.getInvoicedToSponsor({adCalendarId:this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {
      this.invoicedToSponsor = res;
    })
    this.caledarService.getMediaEventEditDetails({id:this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {
      this.actualDetails = res;
      this.detailsloading=false;

    });
    this.addNoteForm = new UntypedFormGroup({
        note: new UntypedFormControl('', Validators.required)
    })
    this.fileUploadForm = new UntypedFormGroup({
      file: new UntypedFormControl('', Validators.required),
      type: new UntypedFormControl('', Validators.required),
    })
    this.caledarService.getCalendarEventByID({eventID: this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {
      this.eventDetails = res;
    })
  }

  changeFileName(event) {
    this.fileToUpload = event.target.files.item(0);
    this.fileName = event.target.value.replace(/^.*\\/, '');
  }

  toogleActions( type) {
    if(type == 0) {
      this.caledarService.toggleInCalendar({resourceID: this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {

      })      
    } else if(type == 1) {
      this.caledarService.toggleZeroWeeks({resourceID: this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {

      })
    } else {
      this.caledarService.toggleTriage({resourceID: this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {

      })
    }
  }


  copied() {
    this.caledarService.showCopied();
  }
  openPreviewDialog(file) {

    const data = {
      adCalendarId: this.data.arg.event.extendedProps.adCalendarId,
      fileId:file.id
    }
    this.caledarService.getMediaEventFile(data).subscribe(res => {
      if(file.fileName.split('.').pop()=='png' || file.fileName.split('.').pop()=='PNG' || file.fileName.split('.').pop()=='jpg' || file.fileName.split('.').pop()=='jpeg' || file.fileName.split('.').pop()=='JPG' || file.fileName.split('.').pop()=='JPEG') {
        const blob = new Blob([res], { type: 'image/png' });
        this.dialog.open(ImagePreviewDialogComponent, {
            data: {
                  actualFile: blob
            },
            height: 'auto',
            maxHeight:'95vh',
            minWidth:'85vw',
            width: 'auto',
            disableClose: true,
            panelClass:"custom-dialog-container"
          });
          this.loading = false;
      } else if(file.fileName.split('.').pop()=='mp3' || file.fileName.split('.').pop()=='MP3') {
        const blob = new Blob([res], { type: "audio/mp3" });
        
        saveAs(blob, file.fileName);
      } else {
        const blob = new Blob([res], { type: 'application/pdf' });
        this.dialog.open(PdfPreviewDialogComponent, {
            data: {
                  showprint:true,
                  actualFile: blob,
                  deletefile:false,

            },
            height: 'auto',
            maxHeight:'95vh',
            minWidth:'85vw',
            width: 'auto',
            disableClose: true,
            panelClass:"custom-dialog-container"
          });
          this.loading = false;
      }
   
    }, err => {
      this.loading = false;
  })
  }
  refreshNotes() {
    this.mediaEventsService.getMediaEventNotes({adCalendarId:this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {
      this.eventNotes = res;
    })
  }

  refreshFiles() {
    this.mediaEventsService.getMediaEventFiles({adCalendarId: this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {
      this.eventFiles = res;
    })
  }

  uploadFile(form) {
    this.loading = true;
    if(this.fileUploadForm.valid && this.fileToUpload != null) {
      const data = {
        file: this.fileToUpload,
        type: this.fileUploadForm.controls.type.value,
        adCalendarId: this.data.arg.event.extendedProps.adCalendarId
      }
      this.mediaEventsService.addFile(data).subscribe(res => {
        this.refreshFiles();
        this.fileUploadForm.reset('');
        this.fileName = '';
        this.loading = false;
      }, err => {
        this.loading = false;
      });
    } else {
      this.loading = false;
      this.fileUploadForm.markAllAsTouched();
    }
  }

  sendNote(form) {
    this.loading = true
    if(this.addNoteForm.valid) {
      this.mediaEventsService.sendNote({request : {note: this.addNoteForm.controls.note.value}, adCalendarId: this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {

        this.loading = false; 
        this.addNoteForm.reset();
        this.refreshNotes();
      }, err => {
        this.loading = false;
      })
    } else {
      this.loading = false;
      this.addNoteForm.markAllAsTouched();
    }

  }

  downloadSponsorFile(invoiceNr, id) {
    const data = {
      invoiceId: id
    }
    this.invoiceService.downloadSponsorInvoice(data).subscribe(res => {
      const blob = new Blob([res], { type: 'application/pdf' });
      this.dialog.open(PdfPreviewDialogComponent, {
        data: {
              showprint:true,
              deletefile:false,

              actualFile: blob
        },
        height: 'auto',
        maxHeight:'95vh',
        minWidth:'85vw',
        width: 'auto',
        disableClose: true,
        panelClass:"custom-dialog-container"
      })

    }) 
  }

  changeStatusDialog(eventdetails){
    const ref = this.dialog.open(SelectDialogComponent, {
      data: {
        currentValue:this.eventDetails.status,
        type:'ad_event_status',
        details:eventdetails
      },
      height: 'auto',
      maxHeight:'45vh',
      minWidth:'40vw',
      maxWidth:'60vw',

      width: 'auto',
      disableClose: true,
      panelClass:"custom-dialog-container"
    })
    ref.afterClosed().subscribe(res => {
      this.caledarService.getCalendarEventByID({eventID: this.data.arg.event.extendedProps.adCalendarId}).subscribe(res => {
        this.eventDetails = res;
      })
    })
  }

  downloadFile(invoiceNr, id) {
    const data = {
      invoiceId: id
    }
    this.invoiceService.downloadInvoice(data).subscribe(res => {
      const blob = new Blob([res], { type: 'application/pdf' });
      this.dialog.open(PdfPreviewDialogComponent, {
        data: {
              showprint:true,
              actualFile: blob,
              deletefile:false,

        },
        height: 'auto',
        maxHeight:'95vh',
        minWidth:'85vw',
        width: 'auto',
        disableClose: true,
        panelClass:"custom-dialog-container"
      })

    }) 
  }

removeMediaEvent() {
  const ref = this.dialog.open(FeedBackDialogComponent, {
    data: {
      message: 'Are you sure you want to remove this event from the calendar?'
    }
  });
  ref.afterClosed().subscribe(res => {
    if(res) {
    this.caledarService.removeMediaEvent(this.data.arg.event.extendedProps.adCalendarId).subscribe(res => {

      this.dialogRef.close(true);   
      })
    }
  });
}

openEditMediaEventDialog() {
  const dialogref = this.dialog.open(EditEventDialogComponent, {
    width: 'auto',
    height: 'auto',
    disableClose: true,
    panelClass:'custom-dialog-container',
    maxHeight:'80vh',
     maxWidth:'80vw',
    data: {
      data: this.data,
      actualDetails: this.actualDetails,
      actualDetails2: this.eventDetails
    }
    
  })
  dialogref.afterClosed().subscribe(res => {
    if(res) {
      this.dialogRef.close(true)
    }
  })
}

}
