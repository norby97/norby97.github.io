import { AfterViewInit, Component, HostListener, Inject, OnInit } from '@angular/core';

@Component({

  templateUrl: './lead-details-dialog.component.html',

})
export class LeadDetailsDialogComponent implements OnInit, AfterViewInit {
  editLeadForm: UntypedFormGroup;
  noteForm: UntypedFormGroup;
  notes;
  loading = false;
  notesloading = false;
  eventFilesTypes=  Object.keys(LeadsStudyStartsFileTypeModel);

  sites = [];
  selected = [];
  articleForm: UntypedFormGroup;
  articles = [];
  contacts = [];
  indications =[] ;
  sponsors =[];
  cro =[];
  users;
  compound =[];
  leadDetails; 
  eventFiles; 
  fileToUpload; 
  fileName =''; 
  dpheader;
  fileUploadForm
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(@Inject(MAT_DIALOG_DATA) public data: any,
    private authService: AuthenticationService,
    private dialog: MatDialog,
    public dialogRef: MatDialogRef<LeadDetailsDialogComponent>,
    private siteService: SiteService,
    private contactService: ContactService,
    private sccipService:SccipService,
    private userService: UserService,
    private leadService: LeadsService,
    private formBuilder: UntypedFormBuilder,) {
      this.dpheader = DatepickerHeaderComponent;

     }
  ngAfterViewInit(): void {
     }
     uploadFile(form) {
      this.loading = true;
      if(this.fileUploadForm.valid && this.fileToUpload != null) {
        const data = {
          file: this.fileToUpload,
          type: this.fileUploadForm.controls.type.value,
          leadId: this.leadDetails.id
          }
          this.leadService.addFileToLead(data).subscribe(res => {
            this.loading = false;
            this.fileUploadForm.reset('')
            this.fileToUpload = null;
            this.fileName =''
            this.refreshFiles()
          })
       
      } else {
        this.loading = false;
        this.fileUploadForm.markAllAsTouched();
      }
    }
    deleteFile(file) {
      const ref = this.dialog.open(FeedBackDialogComponent, {
        data: {
          message: 'Are you sure you want to delete ' +file.fileName + '?'
        }
      })
      ref.afterClosed().subscribe(res => {
        if(res) {
        const data = {
            leadId: this.data?.lead2?.id,
            fileId:file.id
          }
          this.leadService.deleteLeadFile(data).subscribe(res => {

            this.refreshFiles();   
          })
        }
      })
    }
  refreshFiles() {
    this.leadService.getLeadFiles(this.data?.lead2.id).subscribe(res=> {
      this.eventFiles = res;
    })
  }
  openPreviewDialog(file) {

    const data = {
      leadId: this.data?.lead2?.id,
      fileId:file.id
    }
    this.leadService.getLeadFile(data).subscribe(res => {
      if(file.fileName.split('.').pop()=='png' || file.fileName.split('.').pop()=='PNG' || file.fileName.split('.').pop()=='jpg' || file.fileName.split('.').pop()=='jpeg' || file.fileName.split('.').pop()=='JPG' || file.fileName.split('.').pop()=='JPEG') {
        const blob = new Blob([res], { type: 'image/png' });
        this.dialog.open(ImagePreviewDialogComponent, {
            data: {
                  actualFile: blob
            },
            height: 'auto',
            maxHeight:'95vh',
            minWidth:'85vw',
            width: 'auto',
            disableClose: true,
            panelClass:"custom-dialog-container"
          });
          this.loading = false;
      } else if(file.fileName.split('.').pop()=='mp3' || file.fileName.split('.').pop()=='MP3') {
        const blob = new Blob([res], { type: "audio/mp3" });
        
        saveAs(blob, file.fileName);
      } else {
        const blob = new Blob([res], { type: 'application/pdf' });
        this.dialog.open(PdfPreviewDialogComponent, {
            data: {
                  showprint:true,
                  actualFile: blob,
                  deletefile:false,

            },
            height: 'auto',
            maxHeight:'95vh',
            minWidth:'85vw',
            width: 'auto',
            disableClose: true,
            panelClass:"custom-dialog-container"
          });
          this.loading = false;
      }
   
    }, err => {
      this.loading = false;
  })
  }
  changeFileName(event) {
    this.fileToUpload = event.target.files.item(0);
    this.fileName = event.target.value.replace(/^.*\\/, '');
  }

  ngOnInit(): void {
    this.fileUploadForm = new UntypedFormGroup({
      file: new UntypedFormControl('', Validators.required),
      type: new UntypedFormControl('', Validators.required),
    })
    this.leadDetails = this.data?.lead2
    
    this.selected=this.leadDetails?.sites?.map(obj => obj.id)
      this.articles = this.leadDetails?.inTheNews
    this.userService.getAll2().subscribe(res => {
      this.users = res;
    })
    this.leadService.getLeadFiles(this.data?.lead2.id).subscribe(res=> {
      this.eventFiles = res;
    })
   
    
    
    this.editLeadForm =this.formBuilder.group({
      sponsor: new UntypedFormControl(this.leadDetails?.sponsor?.id, Validators.required),
      compound:new UntypedFormControl(this.leadDetails?.compound?.id, Validators.required),
      cro: new UntypedFormControl(this.leadDetails?.cro?.id),
      ntcID: new UntypedFormControl(this.leadDetails?.ntcId,),
      phase: new UntypedFormControl(this.leadDetails?.phase, Validators.required),
      indication: new UntypedFormControl(this.leadDetails?.indication?.id, Validators.required),
      owner: new UntypedFormControl(this.leadDetails?.owner?.id, Validators.required),
      status: new UntypedFormControl(this.leadDetails?.status, Validators.required),
      sites: new UntypedFormControl(this.selected),
      followUp: new UntypedFormControl(this.leadDetails?.followUpDate),
    });
  
    this.getNotes(this.data?.lead2?.id)


      this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
      this.sites =res;
      this.sites.forEach(elem => {

        if(this.data.lead.sites.includes(elem.name)) {
          this.selected.push(elem.id)
        }
      })
    });

    this.noteForm = new UntypedFormGroup({
      note: new UntypedFormControl('', Validators.required)
    });
    this.articleForm = new UntypedFormGroup({
      url: new UntypedFormControl('', Validators.required)
    })

   
  }

 
  searcher = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize:pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.indications = [this.indications,...res.content];
      
      })
    } else {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.indications = [...res.content];
        if(this.leadDetails?.indication != null) {
          this.indications.push({id:this.leadDetails?.indication?.id, name:this.leadDetails?.indication?.name, ngxSelected:1,active:true})
        }
      })
    }
    return of(this.indications)
  }
  private _filter(value: string, items: Array<any>) {
    const filterValue = value?.toLowerCase();
    return items?.filter(item => {
        return item.name?.toLowerCase().indexOf(filterValue) === 0;
    });
  }

  edit(form) {
    this.loading =true;
    if(this.editLeadForm.valid) {
         
          const data = {
            leadID: this.data.lead.id,
            request : {
              compoundId: this.editLeadForm.controls.compound.value,
              croId: this.editLeadForm.controls.cro.value,
              followUpNeededDate: this.editLeadForm.controls.followUp.value,
              indicationId: this.editLeadForm.controls.indication.value,
              ntcId: this.editLeadForm.controls.ntcID.value,
              ownerId: this.editLeadForm.controls.owner.value,
              phase: this.editLeadForm.controls.phase.value,
              sites: this.editLeadForm.controls.sites.value,
              sponsorId: this.editLeadForm.controls.sponsor.value,
              status: this.editLeadForm.controls.status.value
            }
          }
          this.leadService.editLead(data).subscribe(res => {
            this.loading =false;
            this.dialogRef.close(true);

          })
        } else {
          this.loading =false;

          this.editLeadForm.markAllAsTouched();
        }
  }

  getNotes(leadID) {
    this.leadService.getLeadbyID(leadID).subscribe(res => {
      this.data.lead.notes = res.notes;
      this.data.lead?.notes.map(elem => {
        elem.createdDate = elem.createdDate.replace('T',' ') + ' UTC'
        elem.createdDate = elem.createdDate.toString()
      })
      this.notesloading = false;

    }, err => {
      this.notesloading = false;

    });
  }

  openAddContact() {
   const ref = this.dialog.open(ContactSearchResultsForLeadsComponent, {
      data: {
        leadID: this.data.lead.id,
      },
      width: 'auto',
      height: 'auto',
      disableClose: true,
      maxHeight:'80vh',
      maxWidth:'80vw',
      panelClass:'custom-dialog-container',
      });
    ref.afterClosed().subscribe(res => {
      if(res) {

        this.leadService.getLeadbyID(this.data.lead.id).subscribe(res => {
          this.contacts = res.contacts;
        });
      }
    })  
  }

  sendNote(form) {
    this.notesloading = true;
    if(this.noteForm.valid) {
      const data = {
        leadID: this.data.lead.id,
        request: {
          name: this.noteForm.controls.note.value
        }
      }
      this.leadService.sendLeadNote(data).subscribe(res => {
        this.noteForm.reset();

        this.getNotes(this.data.lead.id)
      }, err => {
        this.notesloading = false;
      });
    } else {
      this.notesloading = false;
      this.noteForm.markAllAsTouched();
    }
  
  }

  addArticle(form) {
    this.notesloading = true;
    if(this.articleForm.valid) {
      const data = {
        leadID: this.data.lead.id,
        url: this.articleForm.controls.url.value
      }
      this.articleForm.reset('');
      this.leadService.addArticleToLead(data).subscribe(res => {
      
        this.leadService.getLeadbyID(this.data.lead.id).subscribe(res => {
          this.articles = res.inTheNews;
          this.notesloading = false;
        })

      }, err => {
        this.notesloading = false;
      })
    } else {
      this.notesloading = false;
      this.articleForm.markAllAsTouched();
    }
  }
  openAddDialog(type) {
    if(type =='indication') {
      const dialogref = this.dialog.open(AddIndicationDialogComponent, {
        disableClose: true,
        minWidth: "30rem",width: 'auto',
        height: 'auto',
        maxHeight:'80vh',
        maxWidth:'80vw',    
        panelClass:'custom-dialog-container',
        data:{indicationObject: null, returndata:true}
      });

      dialogref.afterClosed().subscribe(res => {
        if(res) {
          this.sccipService.addIndication(res).subscribe((result:any) => {
            this.indications.push({...result,ngxSelected:1,active:true})
            this.editLeadForm.controls.indication.setValue(result?.id)
            this.editLeadForm.controls.indication.updateValueAndValidity(result?.id)
      })
    }
      })
    } else {

    const ref = this.dialog.open(SimpleNameImputDialogComponent, {
      data: {
        type,
      },
      width: 'auto',
      minWidth:'30vw',
      height: 'auto',
      panelClass:'custom-dialog-container',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw',
    });      
    ref.afterClosed().subscribe(res => {
      if(res) {
        if(type =='sponsor') {
        this.sccipService.addSponsor(res).subscribe((result:any) => {
