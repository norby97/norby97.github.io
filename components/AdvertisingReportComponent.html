  sortData(sort: Sort) {
    const data = this.studyAdvertisingReportDetails;
    if (!sort.active || sort.direction === '') {
      this.sortedData = data;
      return;
    }
    this.sortedData = data.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'totalReimbursable':
          return this.compare(a.totalReimbursable, b.totalReimbursable, isAsc);
        case 'available':
          return this.compare(a.available, b.available, isAsc);
        case 'totalNonReimbursable':
          return this.compare(a.totalNonReimbursable, b.totalNonReimbursable, isAsc);
        case 'invoiced':
          return this.compare(a.invoiced, b.invoiced, isAsc);
        case 'totalBudget':
          return this.compare(a.totalBudget, b.totalBudget, isAsc);
        case 'studyName':
            return this.compare(a.name.toLowerCase(), b.name.toLowerCase(), isAsc);
      }
    });
  }

  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
