import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { SpecialAccessReportComponent } from 'src/app/components/reports/special-access-report/special-access-report.component';
import { CalendarService } from 'src/app/services/advertising/calendar.service';
import { SpecialAccessReportsService } from 'src/app/services/reports/special-access-reports.service';

@Component({
  selector: 'app-view-daily-ops-report-dialog',
  templateUrl: './view-daily-ops-report-dialog.component.html',
  styleUrls: ['./view-daily-ops-report-dialog.component.scss']
})
export class ViewDailyOpsReportDialogComponent implements OnInit {
  loading = false;
  currentDate;
  dailyOpsReport;
  dailyOpsTotals = {
    inquirycalls: 0,
    pscomp: 0,
    psns: 0,
    psrsch:0,
    psdb:0,
    scdb:0,
    wtpt:0,
    scrcomp:0,
    scschps:0,    
    visitschscr:0,
    scrdnc:0,
    scrfail:0,
    scrns:0,
    scrcanc:0,
    scrresch:0,
    visitns:0,
    visitresch:0,
    drops:0,
    visittracked:0,
    scrvisittracked:0
  }
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(private specialReportService:SpecialAccessReportsService,
    private calendarService:CalendarService,
    public dialogRef: MatDialogRef<ViewDailyOpsReportDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any, ) { }

  ngOnInit(): void {
    this.loading= true
    this.calendarService.getCurrentDate().subscribe(res => {
      this.currentDate = res;

      this.dailyOpsReport = []
      const data = {
        siteIds: this.data.siteId == null ? [...this.data?.sites?.map(elem => elem.id)]: [this.data.siteId],
        year: this.currentDate.yr,
        month: this.currentDate.mt,
        week: this.currentDate.wk
      }
      this.specialReportService.getDailyOpsReport(data).subscribe(res => {
        this.dailyOpsReport = res;
        this.getDailyOpsTotals();
        this.loading = false;

      }, err => {
        this.loading = false;
      })
    }, err => {
      this.loading = false;
    })
    
  }
  
  getDailyOpsTotals() {
    this.dailyOpsTotals = {
      inquirycalls: 0,
      pscomp: 0,
      psns: 0,
      psrsch:0,
      psdb:0,
      scdb:0,
      wtpt:0,
      scrcomp:0,
      scschps:0,
      scrdnc:0,
      scrfail:0,
      scrns:0,
      scrcanc:0,
      scrresch:0,
      visitns:0,
      visitschscr:0,
      visitresch:0,
      drops:0,
      visittracked:0,
      scrvisittracked:0
    }
    this.dailyOpsReport.actualProductivityList.map(elem => {
      this.dailyOpsTotals.inquirycalls += parseInt(elem.inquiryCalls);
      this.dailyOpsTotals.pscomp += parseInt(elem.prescreenCompleted);
      this.dailyOpsTotals.psdb += parseInt(elem.prescreenScheduledFromDB);
      this.dailyOpsTotals.psns += parseInt(elem.prescreenThatNoShowed);
      this.dailyOpsTotals.psrsch += parseInt(elem.prescreenThatRescheduled);
      this.dailyOpsTotals.scrfail += parseInt(elem.screenFails);
      this.dailyOpsTotals.scdb += parseInt(elem.screenScheduledFromDB);
      this.dailyOpsTotals.scrcomp += parseInt(elem.screensCompleted);
      this.dailyOpsTotals.scrdnc += parseInt(elem.screensDNC);
      this.dailyOpsTotals.scschps += parseInt(elem.screensScheduledFromPrescreen);
      this.dailyOpsTotals.scrcanc += parseInt(elem.screensThatCanceled);
      this.dailyOpsTotals.scrns += parseInt(elem.screensThatNoShowed);
      this.dailyOpsTotals.scrresch += parseInt(elem.screensThatRescheduled);
      this.dailyOpsTotals.scrvisittracked += parseInt(elem.visitsAndScreensTracked);
      this.dailyOpsTotals.visitschscr += parseInt(elem.visitsScheduledIncludingScreens);
      this.dailyOpsTotals.visitns += parseInt(elem.visitsThatNoShowed);
      this.dailyOpsTotals.visitresch += parseInt(elem.visitsThatRescheduled);
      this.dailyOpsTotals.visittracked += parseInt(elem.visitsTracked);
      this.dailyOpsTotals.wtpt += parseInt(elem.waitlistedPatients);
      this.dailyOpsTotals.drops += parseInt(elem.drops);

    })
  }
  
}
