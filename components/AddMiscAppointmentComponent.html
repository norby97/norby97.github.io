import { formatDate } from '@angular/common';
import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { MiscellaneousService } from 'src/app/services/ctms/miscellaneous.service';
import { PatientsService } from 'src/app/services/ctms/patients.service';

import { ShowCalendarDialogComponent } from '../show-calendar-dialog/show-calendar-dialog.component';
import moment from 'moment';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';

@Component({
  selector: 'app-add-misc-appointment',

  styleUrls: ['./add-misc-appointment.component.scss']
})
export class AddMiscAppointmentComponent implements OnInit {
  addMiscAppForm: UntypedFormGroup;
  miscUsers = []
  miscResources = [];
  showDays = false;
  todayapp = false;
  loading = false;
  rescheks = [];
  usercheks = [];
  allDayBLocks = false;
  selectedTimeto= '01:00 PM'
  days= [false,true,true,true,true,true,true,true]
  daystoshow= [false,false,false,false,false,false,false,false]
  selectedTimefrom= '08:00 AM' 
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }

  constructor(public dialogRef: MatDialogRef<AddMiscAppointmentComponent>,
              private dialog:MatDialog,
              private miscService: MiscellaneousService,
              @Inject(MAT_DIALOG_DATA) public data: any,) { 
                this.dpheader = DatepickerHeaderComponent;
              }

  ngOnInit(): void {
  if(this.data?.reschedule) {
    this.days.fill(false);
    this.data?.patientDetails?.days?.forEach(element => {
      this.days[element] = true;
    });
    this.selectedTimefrom= this.convert24To12(this.data?.patientDetails2?.dateFrom.split('T')[1])
    this.selectedTimeto= this.convert24To12(this.data?.patientDetails2?.dateTo.split('T')[1])
    this.addMiscAppForm = new UntypedFormGroup({
      dateTo:  new UntypedFormControl(this.data?.patientDetails2?.dateTo,),
      dateFrom: new UntypedFormControl(this.data?.patientDetails2?.dateFrom,Validators.required),
      timeBlockTo: new UntypedFormControl(this.selectedTimeto),
      timeBlockFrom: new UntypedFormControl(this.selectedTimefrom),
      details: new UntypedFormControl(this.data?.patientDetails2?.description,Validators.required),
      notes :  new UntypedFormControl(this.data?.patientDetails2?.appointmentNotes,),
    });
  this.miscService.getMiscUsers(this.data?.site).subscribe(res => {
    this.miscUsers = res;
      this.miscUsers.forEach(elem => {
        if(this.data?.patientDetails2?.providerIds.includes(elem.id)) {
          this.usercheks.push(true);
        } else {
          this.usercheks.push(false)
        }
        
      })
    });
    this.miscService.getMiscResources(this.data?.site).subscribe(res => {
      this.miscResources = res;
      this.miscResources.forEach(elem => {
        if(this.data?.patientDetails2?.providerIds.includes(elem.id)) {
          this.rescheks.push(true);
        } else {
          this.rescheks.push(false)
        }
        
      })
    });
    this.checkdays();
  } else {
    this.addMiscAppForm = new UntypedFormGroup({
      dateTo:  new UntypedFormControl(null,),
      dateFrom: new UntypedFormControl(null,Validators.required),
      timeBlockTo: new UntypedFormControl(this.selectedTimeto,),
      timeBlockFrom: new UntypedFormControl(this.selectedTimefrom),
      details: new UntypedFormControl(null,Validators.required),
      notes :  new UntypedFormControl(null,),
    });
    this.miscService.getMiscUsers(this.data?.site).subscribe(res => {
      this.miscUsers = res;
        this.miscUsers.forEach(elem => {
          this.miscUsers.forEach(elem => this.usercheks.push(false) )
        })
      });
      this.miscService.getMiscResources(this.data?.site).subscribe(res => {
        this.miscResources = res;
        this.miscResources.forEach(elem => {
          this.miscResources.forEach(elem => this.rescheks.push(false) )
        })
      });
  }
 
    this.addMiscAppForm.controls.dateFrom.valueChanges.subscribe(res => {
      if(new Date(res).getDate() == new Date().getDate()) {
        this.todayapp = true;
        this.addMiscAppForm.controls.dateTo.setValidators(Validators.required);
        this.addMiscAppForm.controls.dateTo.setValue(this.addMiscAppForm.controls.dateFrom.value);
        this.addMiscAppForm.controls.dateTo.updateValueAndValidity();
      } else {
        this.todayapp = false;
        this.addMiscAppForm.controls.dateTo.setValidators(Validators.required);
        this.addMiscAppForm.controls.dateTo.setValue(this.addMiscAppForm.controls.dateFrom.value);
        this.addMiscAppForm.controls.dateTo.updateValueAndValidity();
      }
    })
  }
  convert24To12(time12h)  {
      const [hourStr, minuteStr, secondStr] = time12h.split(':');
      const hour = parseInt(hourStr, 10);
      const minute = parseInt(minuteStr, 10);
      const period = hour < 12 ? 'AM' : 'PM';
      const hour12 = hour % 12 || 12;
      return `${hour12 < 10 ? '0' + hour12 : hour12}:${minute < 10 ? '0' + minute : minute} ${period}`;
     
  }


  checkdays() {

    let diff = new Date(this.addMiscAppForm.controls.dateTo.value).getTime() - new Date(this.addMiscAppForm.controls.dateFrom.value).getTime();
    if(diff >= 0) {
      if(Math.ceil(diff / (1000 * 60 * 60 * 24))+1 > 2 && Math.ceil(diff / (1000 * 60 * 60 * 24))+1 < 7) {
        this.daystoshow.fill(false);
        if(new Date(this.addMiscAppForm.controls.dateFrom.value).getDay() < new Date(this.addMiscAppForm.controls.dateTo.value).getDay()) {
          for(let i = new Date(this.addMiscAppForm.controls.dateFrom.value).getDay(); i<= new Date(this.addMiscAppForm.controls.dateTo.value).getDay();i++){
            this.daystoshow[i] = true;
          }
        } else {
          for(let i = new Date(this.addMiscAppForm.controls.dateFrom.value).getDay(); i<=7;i++){
            this.daystoshow[i] = true;
          }
          for(let i = new Date(this.addMiscAppForm.controls.dateTo.value).getDay(); i>=0;i--){
            this.daystoshow[i] = true;
          }
        }
      
      }
      if(Math.ceil(diff / (1000 * 60 * 60 * 24))+1 >= 7){
        this.daystoshow.fill(true);
      }
    
      this.showDays =  Math.ceil(diff / (1000 * 60 * 60 * 24))+1 > 2;
    } else {
      this.miscService.showError('date-from-to-error')
    }
   
  }
  convert12To24(time: string): string {
    const [timeStr, period] = time.split(' ');
    const [hourStr, minuteStr, secondStr] = timeStr.split(':');
    let hour = parseInt(hourStr, 10);
    const minute = parseInt(minuteStr, 10);
    const second = parseInt(secondStr || '00', 10);
    if (period === 'PM' && hour < 12) {
      hour += 12;
    } else if (period === 'AM' && hour === 12) {
      hour = 0;
    }
    const hourStr24 = hour < 10 ? '0' + hour : hour;
    return `${hourStr24}:${minute < 10 ? '0' + minute : minute}`;
  }
  addMiscAppointment(form) {
    this.loading = true;
    if(this.addMiscAppForm.valid) {
    let diff = new Date(this.addMiscAppForm.controls.dateTo.value).getTime() - new Date(this.addMiscAppForm.controls.dateFrom.value).getTime();
    if(diff >= 0) {
      let days = [];
      if(this.days[0] == false) {
        this.days.map((elem,i) => {if(elem == true){days.push(i)}})
      }
      const data = {
        dateFrom: formatDate(this.addMiscAppForm.controls.dateFrom.value, 'yyyy-MM-dd','en-Us'),
        dateTo:this.addMiscAppForm.controls.dateTo.value == null ?  formatDate(this.addMiscAppForm.controls.dateFrom.value, 'yyyy-MM-dd','en-Us') : formatDate(this.addMiscAppForm.controls.dateTo.value, 'yyyy-MM-dd','en-Us'), 
        details: this.addMiscAppForm.controls.details.value,
        internalNotes: this.addMiscAppForm.controls.notes.value,
        resourcesIds: [],
        days:this.days[0] == true ? [1,2,3,4,5,6,7] : days,
        setTimeBlockForAllDay: this.allDayBLocks,
        siteId: this.data?.site,
        timeblockFrom: this.convert12To24(this.selectedTimefrom),
        timeblockTo: this.convert12To24(this.selectedTimeto),
        userIds: []
      }
      if(this.todayapp && data.dateTo == null) {
        data.dateTo = null;
      } 
      this.miscUsers.forEach((elem,i) => {
        if(this.usercheks[i]) {
          data.userIds.push(elem.id);
        }
      });
      this.miscResources.forEach((elem,i) => {
        if(this.rescheks[i]) {
          data.resourcesIds.push(elem.id);
        }
      });   

      const data2 = {
        userIds: data?.userIds,
        resourceIds: data?.resourcesIds,
         dateFrom: data?.dateFrom,
         dateTo: data?.dateTo,
         timeblockFrom: data?.timeblockFrom,
         days:this.days[0] == true ? [1,2,3,4,5,6,7] : days,
         timeblockTo: data?.timeblockTo,
        setTimeBlockForAllDay: data?.setTimeBlockForAllDay,
      }

      
      this.miscService.checkOverlap(data2).subscribe(res => {
        if(res.length == 0) {
          if(this.data?.reschedule) {
            const data3 = {
              appointmentId: this.data?.arg?.event?.extendedProps?.appointmentId,
              req: data
            }
            this.miscService.reschedulePatientlessMiscApp(data3).subscribe(res => {
                  this.loading = false;
                  this.dialogRef.close(true);
            }, err => {
              this.loading = false;
            })
          } else {
            this.miscService.createMiscApp(data).subscribe(res => {
                  this.loading = false;
                  this.dialogRef.close(true);
            }, err => {
              this.loading = false;
            })
          }
        } else {
          const ref = this.dialog.open(OverlapDialogComponent,{
            width: 'auto',
            height: 'auto',
            panelClass:'custom-dialog-container',
            disableClose: true,
            maxHeight:'80vh',
            minWidth:'40vw',
            data:{res,
            misc: true,
            type:'MISC',
                  siteId:this.data?.site,
                  date:data?.dateFrom
            },
          })
          ref.afterClosed().subscribe(res => {  
            if(res) {
              if(this.data?.reschedule) {
                const data3 = {
                  appointmentId: this.data?.arg?.event?.extendedProps?.appointmentId,
                  req: data,

                }
                this.miscService.reschedulePatientlessMiscApp(data3).subscribe(res => {
                  this.dialogRef.close(true);
                  this.loading = false;

                }, err => {
                  this.loading = false;
                })
              } else {
                this.miscService.createMiscApp(data).subscribe(res => {
                  this.dialogRef.close(true);
                  this.loading = false;

                }, err => {
                  this.loading = false;
                })
              }
            }else {
              this.loading = false;
            }
          })
        }
      },err => {
        this.loading = false;
      })
    }else {
    this.loading = false;
    this.miscService.showError('date-from-to-error')
    }
    

      
    } else {
    this.loading = false;

      this.addMiscAppForm.markAllAsTouched();
    }

  }

  showCalendarDialog() {
    this.dialog.open(ShowCalendarDialogComponent, {
      data: {

        type: 'RESOURCES_VISITS_SCREENS',
        date:  formatDate(this.addMiscAppForm.controls.dateFrom.value, 'yyyy-MM-ddTHH:mm','en-Us')

      width: 'auto',
      height: 'auto',
      panelClass:'custom-dialog-container',
      disableClose: true,
      maxHeight:'80vh',
      maxWidth:'90vw',
      minWidth:'70vw',
    })
  }
}
