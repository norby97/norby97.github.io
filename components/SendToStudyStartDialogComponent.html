export class SendToStudyStartDialogComponent implements OnInit {
  sendStudyForm: UntypedFormGroup;

  constructor(@Inject(MAT_DIALOG_DATA) public data: any,
              private authService: AuthenticationService,
              private dialog: MatDialog,
              private sccipService: SccipService,

                this.dpheader = DatepickerHeaderComponent;

  ngOnInit(): void {
    var cats = this.giccsppp.getGeneralCategories();
    var pops = this.giccsppp.getPopulations();
    var prots = this.giccsppp.getProtocols();
    forkJoin([cats, pops, prots]).subscribe(res => {
      this.generalCategories = res[0];
      this.populations = res[1];
      this.protocols = res[2];
    })
    this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
    });
   
    
    this.sendStudyForm =new UntypedFormGroup({
      sponsor: new UntypedFormControl(this.data?.lead2?.sponsor?.id, Validators.required),
      compound: new UntypedFormControl(this.data?.lead2?.compound?.id, Validators.required),
      protocol: new UntypedFormControl(),
      cro: new UntypedFormControl(this.data?.lead2?.cro?.id),
      ntcID: new UntypedFormControl(this.data?.lead2?.ntcId),
      phase:  new UntypedFormControl(this.data?.lead2?.phase, Validators.required),
      indication: new UntypedFormControl(this.data?.lead2?.indication?.id, Validators.required),
      bdindication: new UntypedFormControl(this.data?.lead2?.bdIndicationName),
      population:new UntypedFormControl('', Validators.required),
      status: new UntypedFormControl('', Validators.required),
      protocolTitle: new UntypedFormControl('', ),
      site: new UntypedFormControl(''),
      ext: new UntypedFormControl(this.extendedStudy),
      ind: new UntypedFormControl(''),
      nickname: new UntypedFormControl(''),
      user: new UntypedFormControl(null),
      trialType: new UntypedFormControl('', Validators.required),
      startDate: new UntypedFormControl('', Validators.required),
      
    });
   

  }

  toggle() {
    this.removeFromLeads = !this.removeFromLeads;
  }
  
  private _filter(value: string, items: Array<any>) {
    const filterValue = value?.toLowerCase();
    return items?.filter(item => {
        return item.name?.toLowerCase().indexOf(filterValue) === 0;
    });
  }
  
  addSite() {
    if(!this.sendStudyForm.get('site').value) {
      this.error = true;
    } else {
      this.sites.forEach(element => {
        if(element.id === this.sendStudyForm.get('site').value ) {
          this.addedSites = [...this.addedSites, element];
          this.sites = this.sites.filter(item => !this.addedSites.includes(item));
        }
      })
      this.studyService.getInvestigators( this.sendStudyForm.get('site').value).subscribe(res => {
        this.users = res;
        this.addedUsers = [...this.addedUsers, this.users]

       })
    
    }
    this.sendStudyForm.get('site').reset('');
    
  }

  pairUser(event,item) {
    var exists = false;
    
    this.siteReq.forEach(element => {
      if (element.siteId === item.id) {
        element.principalInvestigatorId = event == 'null' ? null : event;
        exists = true;
      }
    })

    if(!exists) {
      this.siteReq.push( {
        principalInvestigatorId: event == 'null' ? null : event,
        siteId: item.id
      });
    }
    
  }
  toggleStatus(status) {
    this.extendedStudy = status;
  }
  sendLead(sendStudyForm) {
    this.loading = true;
    if(this.sendStudyForm.valid) {
      const data = {
        compoundId: this.sendStudyForm.get('compound').value,
        croId: this.sendStudyForm.get('cro').value,
        ext: this.extendedStudy,
        indicationId: this.sendStudyForm.get('indication').value,
        bdIndicationName: this.sendStudyForm.get('bdindication').value,
        ntcId: this.sendStudyForm.get('ntcID').value,
        leadId: this.data.lead.id,
        phase: this.sendStudyForm.get('phase').value,
        populationId: this.sendStudyForm.get('population').value,
        protocolName: this.sendStudyForm.get('protocol').value,
        protocolTitle: this.sendStudyForm.get('protocolTitle').value,
        sites: this.siteReq,
        nickname: this.sendStudyForm.get('nickname').value,
        ind: this.sendStudyForm.get('ind').value,
        sponsorId: this.sendStudyForm.get('sponsor').value,
        removeFromLead: this.removeFromLeads,
        startDate: this.sendStudyForm.get('startDate').value,
        status: this.sendStudyForm.get('status').value,
        trialType: this.sendStudyForm.get('trialType').value
      }
      this.leadService.sendLeadToStudyStart(data).subscribe(res => {

        this.dialogRef.close(true);
      }, err => {
        this.loading = false;
      })
    } else {
      this.loading = false;
      this.sendStudyForm.markAllAsTouched();
    }
   }

   searcher = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    const data = {
      search,
      active:true,
      limit:pageSize-1,
      skip: 0 + ((pageNumber-1)*pageSize),
    }
    if(pageNumber-1 != 0) {
      this.sccipService.getProtocolsPaginated(data).subscribe(res => {
        this.protocol = [...this.protocol, ...res];
      })
    } else {
     this.sccipService.getProtocolsPaginated(data).subscribe(res => { 
      this.protocol = [ ...res];
    
      })
    }
    return of(this.protocol)
  }

  openAddDialog(type) {
    if(type =='indication') {
      const dialogref = this.dialog.open(AddIndicationDialogComponent, {
        disableClose: true,
        minWidth: "30rem",width: 'auto',
        height: 'auto',
        maxHeight:'80vh',
        maxWidth:'80vw',    
        panelClass:'custom-dialog-container',
        data:{indicationObject: null, returndata:true}
      });

      dialogref.afterClosed().subscribe(res => {
        if(res) {
          this.sccipService.addIndication(res).subscribe((result:any) => {
            this.diagnosis.push({...result,ngxSelected:1,active:true})
            this.sendStudyForm.controls.indication.setValue(result?.id)
            this.sendStudyForm.controls.indication.updateValueAndValidity(result?.id)
      })
    }
      })
    } else {

    const ref = this.dialog.open(SimpleNameImputDialogComponent, {
      data: {
        type,
      },
      width: 'auto',
      minWidth:'30vw',
      height: 'auto',
      panelClass:'custom-dialog-container',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw',
    });      
    ref.afterClosed().subscribe(res => {
      if(res) {
        if(type =='sponsor') {
        this.sccipService.addSponsor(res).subscribe((result:any) => {
        this.sponsors=  [{...result,ngxSelected:1,active:true,ngxHide:false}, ...this.sponsors]
        this.sendStudyForm.controls.sponsor.setValue(result?.id)
            this.sendStudyForm.controls.sponsor.updateValueAndValidity(result?.id)
        })

        }
        if(type =='cro') {
          this.sccipService.addCro(res).subscribe((result:any) => {
      this.cro.push({...result,ngxSelected:1,active:true})
            this.sendStudyForm.controls.cro.setValue(result?.id)
            this.sendStudyForm.controls.cro.updateValueAndValidity(result?.id)
          })
        }
     
     
        if(type =='compound') {
          this.sccipService.addCompunds(res).subscribe((result:any) => {
           this.compound.push({...result,ngxSelected:1,active:true})
           this.sendStudyForm.controls.compound.setValue(result?.id)
           this.sendStudyForm.controls.compound.updateValueAndValidity(result?.id)

        }

      }
    })
