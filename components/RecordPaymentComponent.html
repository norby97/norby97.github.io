  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
      siteId:   this.recordForm.controls.siteID.value.id,
      statuses: ['MOVING_FORWARD','ADD_ON','ACTIVE']
    }
    if(pageNumber - 1 != 0) {
      this.siteService.getStudies(data).subscribe(res => {
        this.studies =[...this.studies,...res.content]
      })
    } else {
      this.siteService.getStudies(data).subscribe(res => {
        this.studies =[...res.content]
      })
    }
    return of(this.studies)
  }

  changeStudy(event) {
    if(this.paymentDetailsSubmitted) {
      const ref =this.dialog.open(FeedBackDialogComponent, {
        data:{
          message:"You will lose all your process on the previous study? \n Are you sure you want to continue?"
        }
      });
      ref.afterClosed().subscribe(res => {
        if(res) {
          this.actualStudy = event.value;
          this.addedInvoices = [];
          this.addedInvoicesCopy = [];
          this.paymentForm.reset();
          this.totalAmount = 0;
          this.studyStartService.getPaymentTypesForStudyStarts(this.recordForm.controls.study.value.id).subscribe(res => {
            this.pTypes = res;
          })
          this.dataSource = new MatTableDataSource(null);
        } else {
          this.recordForm.controls.study.setValue(this.actualStudy);
          this.recordForm.controls.study.updateValueAndValidity();

        }
      })
    }  else {
      this.actualStudy = event.value;
    }
  }
  selectType(event) {
    if(event.value =='Unassigned Payment'){
      const ref = this.dialog.open(UnassignedPaymentDialogComponent, {
        data: {
          paymentDetails: this.recordForm.value,
          siteId:this.paymentForm.controls.type.value
        },
        panelClass: 'custom-dialog-container',
        width: 'auto',
        height: 'auto',
        disableClose: true,
        maxHeight:'80vh',
        maxWidth:'30vw',
      });
      ref.afterClosed().subscribe(res =>{

        if(res) {
          this.recordForm.reset();
          this.paymentForm.reset();
        } else {
          this.paymentForm.controls.stype.reset();
          this.paymentForm.controls.stype.updateValueAndValidity();
        }
      })
    }
  }

  submit(form) {
    if(this.recordForm.valid) {

  addFullPaid(invoice) {
    invoice.patientStudyLinkId = this.paymentForm.controls.patient.value.id;
    invoice.type  = this.paymentForm.controls.stype.value;
    if(invoice.type == 'Visits') {
    invoice.patient = this.paymentForm.controls.patient.value

      invoice.trackedProcedures.map(elem=>  {
        elem.visitId = invoice.visitId;
       
        elem.patientStudyLinkId = invoice.patientStudyLinkId;
        elem.visitScheduleId = invoice.visitScheduleId;
      });
    }
    invoice.paymentAmount = invoice.trackedAmount;
    invoice.trackedProcedures.forEach(elem => {
      elem.paymentAmount = elem.amount;
      elem.withHoldingAmount = 0;
    });

    this.addedInvoices.push(invoice);
    this.addedInvoicesCopy.push(invoice);
    this.totalAmount = 0;
    this.addedInvoices.map(elem => this.totalAmount+= elem.paymentAmount)
  }

  clearPayment(invoice) {
    invoice.paymentAmount = null
    invoice.trackedProcedures.forEach(elem => {
      elem.paymentAmount = null;
      elem.withHoldingAmount = null;
    });
    this.addedInvoices.splice(this.addedInvoices.indexOf(invoice),1)
    this.addedInvoicesCopy.splice(this.addedInvoices.indexOf(invoice),1)
    this.totalAmount = 0;
    this.addedInvoices.map(elem => this.totalAmount+= elem.paymentAmount)
  }

    invoice.patient = this.paymentForm.controls.patient.value
      invoice.trackedProcedures.map(elem=>  {
        elem.visitId = invoice.visitId;
       
        elem.patientStudyLinkId = invoice.patientStudyLinkId;
        elem.visitScheduleId = invoice.visitScheduleId;
      });
    }
    invoice.paymentAmount = invoice.trackedAmount;
   invoice.trackedProcedures.forEach((elem,i) => {
      elem.paymentAmount = parseFloat(elem.amount) - parseFloat(this.percentage(invoice.withHoldingAmount,elem.amount))
      elem.withHoldingAmount = this.percentage(invoice.withHoldingAmount,elem.amount) || 0;
    });

    this.addedInvoices.push(invoice);
    this.addedInvoicesCopy.push(invoice);
    this.totalAmount = 0;
    this.addedInvoices.map(elem => this.totalAmount+= elem.paymentAmount)
  }
  percentage(percent, total) {
    return ((percent/ 100) * total).toFixed(2)
  }
  savePayment() {
    if(this.totalAmount != this.recordForm.controls.amount.value) {
      this.siteService.showError('Please match total amount values!')
    } else {
      const data = {
        paymentDate: formatDate(this.recordForm.controls.pdate.value, 'yyyy-MM-dd', 'en-Us'),
        paymentNr: this.recordForm.controls.pref.value,
        studyStartId: this.recordForm.controls.study.value.id,
        siteId:this.paymentForm.controls.type.value,
        paymentType:this.recordForm.controls.ptype.value == 'WIRE_TRANSFER' ? 'WIRE' :this.recordForm.controls.ptype.value,
        paymentTotalAmount:this.recordForm.controls.amount.value,
        unassignedPayment:false,
        paymentNote: '',
        procedurePayments:  [],
        miscellaneousPayments: [],
        advertisingPayments: []
      }
      this.addedInvoices.map(elem => {
        if(elem.type == 'Visits') {
          elem?.trackedProcedures?.map(procedure => {
              data.procedurePayments.push(procedure)
          })
         } else if(elem.type == 'Miscellaneous') {
          elem?.miscellaneousDetails?.map(procedure => {
            data.miscellaneousPayments.push(procedure)
        })
        }  else {
          elem?.invoiceDetails?.map(procedure => {
            data.advertisingPayments.push(procedure)
        });
        }
      });
      this.addedInvoicesCopy.map(elem => {
        if(elem.type == 'Visits') {
          elem?.trackedProcedures?.map(procedure => {
              data.procedurePayments.push(procedure)
          })
         } else if(elem.type == 'Miscellaneous') {
          elem?.miscellaneousDetails?.map(procedure => {
            data.miscellaneousPayments.push(procedure)
        })
        }  else {
          elem?.invoiceDetails?.map(procedure => {
            data.advertisingPayments.push(procedure)
        });
        }
      });
      this.recordPaymentService.savePayment(data).subscribe(res => {

        this.recordForm.reset();
        this.paymentForm.reset();
        this.addedInvoices = [];
        this.addedInvoicesCopy = [];
        this.paymentDetailsSubmitted = false;
        this.dataSource = new MatTableDataSource();
        this.totalAmount = 0;
      })
    }
    }
    sortData(sort: Sort) {
      const data = this.addedInvoices;
      if (!sort.active || sort.direction === '') {
        this.addedInvoicesCopy = data;
        this.addedInvoices = this.addedInvoicesCopy;
        return;
      }
       this.addedInvoicesCopy = data?.sort((a, b) => {
        const isAsc = sort.direction === 'asc';
        switch (sort.active) {
          case 'details': return this.compare(a.subjectNumber?.toLowerCase(), b?.subjectNumber?.toLowerCase(), isAsc);
          case 'amount': return this.compare(a.paymentAmount, b?.paymentAmount, isAsc);
          case 'decription': return this.compare(a.description?.toLowerCase(), b?.description?.toLowerCase(), isAsc);
          default: return 0;
        }
      });
      this.addedInvoices = this.addedInvoicesCopy;
  
    }
    
    compare(a: number | string | Date, b: number | string | Date, isAsc: boolean) {
      return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
    }
  
}
