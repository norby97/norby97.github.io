import { Component, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { Observable, of } from 'rxjs';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { SccipService } from 'src/app/services/crm_bd/sccip.service';
import { StudyStartsService } from 'src/app/services/crm_bd/study-starts.service';
import { RegManagerService } from 'src/app/services/regulatory/reg-manager.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { TrainingService } from 'src/app/services/regulatory/training.service';
import { EditTrainingDialogComponent } from '../../common/dialogs/regulatory/edit-training-dialog/edit-training-dialog.component';
import { AddTraineeDialogComponent } from '../../common/dialogs/utils/add-trainee-dialog/add-trainee-dialog.component';
import { NameInputDialogComponent } from '../../common/dialogs/utils/name-input-dialog/name-input-dialog.component';
import { NameUpdateDialogComponent } from '../../common/dialogs/utils/name-update-dialog/name-update-dialog.component';
import { UserPreferencesService } from 'src/app/services/utils/user-preferences.service';
import { FeedBackDialogComponent } from '../../common/dialogs/utils/feed-back-dialog/feed-back-dialog.component';

@Component({
  selector: 'app-training',
  templateUrl: './training.component.html',
  styleUrls: ['./training.component.scss']
})
export class TrainingComponent implements OnInit {
  trainingForm: UntypedFormGroup;
  noteForm: UntypedFormGroup;
  studySelected = false;
  projectMNote = false;
  includepm = true;
  sites = [];
  studies=[];
  studyID;
  notes;
  currentTraining;
  scales = []
  trainees = []
  trainings =[]
  lastsearch=''
  studypageNumber= 1;
  constructor(private siteService: SiteService,
              private dialog: MatDialog,
              private trainingService: TrainingService,
              private userPreferences: UserPreferencesService,
              private sccipService: SccipService,
              private authService: AuthenticationService) { }

  ngOnInit(): void {
    this.noteForm = new UntypedFormGroup({
      note: new UntypedFormControl('', Validators.required),
      sites: new UntypedFormControl([])
    });
    this.trainingForm = new UntypedFormGroup({
      site: new UntypedFormControl('', Validators.required),
      study: new UntypedFormControl('', Validators.required),
    }); 
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.trainingForm.controls.site.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.trainingForm.controls.site.updateValueAndValidity();
    } else {
      
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.trainingForm.controls.site.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.trainingForm.controls.site.updateValueAndValidity();
      });
    }
   
  }

  refreshNotes() {
    const data = {
      siteID: this.trainingForm.controls.site.value.id,
      id:this.studyID,
      includepm: this.includepm
    }
    this.trainingService.getNotes(data).subscribe(res => {
      this.notes = res;
    })
  }

  toggle() {
    this.projectMNote = !this.projectMNote;
    if(this.projectMNote) {
      this.noteForm.controls.sites.setValidators(Validators.required);
      this.noteForm.controls.sites.updateValueAndValidity();
    } else {
      this.noteForm.controls.sites.setValidators(null);
      this.noteForm.controls.sites.updateValueAndValidity();
    }
 
  }

  sendNote(form) {
    if (this.noteForm.valid) {
      const data = {
        siteID: this.trainingForm.controls.site.value.id,
        studyStartID: this.studyID,
        request: {
          projectManagementNote:this.projectMNote ,
          note: this.noteForm.controls.note.value,
          sites:  this.noteForm.controls.sites.value
        }
      }
   
      this.trainingService.sendNote(data).subscribe(res => {

        this.refreshNotes()
        this.noteForm.reset();
        this.noteForm.controls.sites.setValue([]);
        this.noteForm.controls.sites.updateValueAndValidity();
        this.projectMNote = false;
      })
    } else {
      this.noteForm.markAllAsTouched();
    }
  }

  getTrainingIfExists(userId, scaleId) {
    this.currentTraining = null;
    if(this.trainings.length == 0){ this.currentTraining = null;}
    else {
      this.trainings.forEach(elem => {
        if(elem.userId == userId && elem.scaleId == scaleId) this.currentTraining = elem;     
      })
    }
  }

  refreshTable(event) {
    this.studySelected = true;
    this.studyID = event.value.studySiteId || event.value.id;
    this.sccipService.getScales(this.studyID).subscribe(res => {
      this.scales = res;
    });
    this.trainingService.getTraineesForStudy(this.studyID).subscribe(res =>{
      this.trainees = res;
    })
    this.trainingService.getTrainings(this.studyID).subscribe(res => {
      this.trainings = res;
    })
    this.refreshNotes();
  }

  openActionDialog(i,j, trainee, scale,) {
    this.getTrainingIfExists(trainee.userId, scale.scaleId)
    const ref = this.dialog.open(EditTrainingDialogComponent, {
      data: {
        i,
        j,
        trainee,
        scale,
        training: this.currentTraining,
        study: this.trainingForm.controls.study.value
      },
      width: 'auto',
      minWidth:'30vw',
      height: 'auto',
      panelClass:'custom-dialog-container',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw'
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.refreshTable({value: {studySiteId: this.studyID}});
      }
    })
  }

  addScale() {
   const ref = this.dialog.open(NameInputDialogComponent, {
      data: {
        type: 'Scale',
        study: this.trainingForm.controls.study.value
      },
      width: 'auto',
      minWidth:'30vw',
      height: 'auto',
      panelClass:'custom-dialog-container',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw',
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.refreshTable({value: {studySiteId: this.studyID}});
      }
    })
  }

  openUpdateScaleDialog(scaleId) {
    const ref = this.dialog.open(NameUpdateDialogComponent, {
      data: {
        type: 'Scale',
        study: this.trainingForm.controls.study.value,
        scaleId
      },
      width: 'auto',
      minWidth:'30vw',
      height: 'auto',
      panelClass:'custom-dialog-container',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw'
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.refreshTable({value: {studySiteId: this.studyID}});

      }
    })
  }

  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    this.userPreferences.setuserState('selectedSite',this.trainingForm.controls.site.value,true)
    if(this.lastsearch != search || pageNumber == 1) {
      this.studypageNumber = 0
    }
    if(search =='') {
      this.trainingForm.controls.study.reset('')
      this.trainingForm.controls.study.updateValueAndValidity();
    }
    this.lastsearch = search
 
    if(pageNumber ==1 ) {
      this.studypageNumber = 0;
      this.trainingForm.controls.study.reset('')
      this.trainingForm.controls.study.updateValueAndValidity();
    }
    const data = {
      search,
      pageNumber: this.studypageNumber,
      pageSize,
      siteId:  this.trainingForm.controls.site.value.id || this.sites[0].id,
      statuses: ['MOVING_FORWARD','ADD_ON','ACTIVE']
    }
    if(this.studypageNumber!= 0) {
      this.siteService.getStudies(data).subscribe(res => {
        this.studies = [...this.studies,...res.content];
      })
    } else {
      this.siteService.getStudies(data).subscribe(res => {
        this.studies =res.content
      })
    }
    this.studypageNumber++;
    return of(this.studies)
  }


  addTrainee() {
    const ref = this.dialog.open(AddTraineeDialogComponent, {
      data: {
        type: 'Trainee',
        site: this.trainingForm.controls.site.value.id,
        study: this.trainingForm.controls.study.value
      },
      width: 'auto',
      height: 'auto',
      minWidth:'30vw',
      panelClass:'custom-dialog-container',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw'
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.refreshTable({value: {studySiteId: this.studyID}});
      }
    })
  }

  deleteScale(scale) {
  
    const ref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to delete ' + scale?.name + ' from the scales?'
      }
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.trainingService.deleteScale(scale?.id).subscribe(res => {
          this.refreshTable({value: {studySiteId: this.studyID}});
        })
      }
    });
  }

  deleteTrainee(trainee) {
  
    const ref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to remove ' + trainee?.userFullName + ' from trainees?'
      }
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.trainingService.deleteTrainee(trainee?.id).subscribe(res => {
          this.refreshTable({value: {studySiteId: this.studyID}});
        })
      }
    });
  }

}
