  refreshRange() {
    if(this.rangeForm.valid) {
        if(this.countweekends) {
            if(this.includeholidays){
                this.rangevalue = `${this.calculateDiff(new Date(this.rangeForm.controls.dateFrom.value), new Date(this.rangeForm.controls.dateTo.value))} days`
            } else {
                this.rangevalue = ``
            }
        } else {
            if(this.includeholidays) {
                this.rangevalue = ` ${this.getBusinessDatesCount(new Date(this.rangeForm.controls.dateFrom.value), new Date(this.rangeForm.controls.dateTo.value))} days`
            } else {
                this.rangevalue = ``
                
            }
        }
    } else {
        this.rangeForm.markAllAsTouched();
    }
  }

  calculateDiff(startDate, endDate){
    const oneDay = 24 * 60 * 60 * 1000; 
    return Math.round(Math.abs((startDate - endDate) / oneDay));
  }

  getBusinessDatesCount(startDate, endDate) {
    let count = 0;
    const curDate = new Date(startDate.getTime());
    while (curDate <= endDate) {
        const dayOfWeek = curDate.getDay();
        if(dayOfWeek !== 0 && dayOfWeek !== 6) count++;
        curDate.setDate(curDate.getDate() + 1);
    }
    
    return count;
}

  calculateBmi(h,w){
    const bmi = (parseFloat(w) / Math.pow(parseFloat(h), 2)) * 10000;
    if(isNaN(bmi)) {
        this.bmivalue = null;
    } else {
        this.bmivalue= bmi.toFixed(2)
    }
  }

  calculateBmi2(h,w){
    const bmi2 = 703 *(parseFloat(w) / Math.pow(parseFloat(h), 2))
    if(isNaN(bmi2)) {
        this.bmi2value = null;
    } else {
        this.bmi2value= bmi2.toFixed(2)
    }
  }

    if(this.drugForm.valid) {

    } else {
