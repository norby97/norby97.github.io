import { Component, OnInit, Inject, HostListener } from '@angular/core';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { UntypedFormGroup, UntypedFormBuilder, UntypedFormControl, Validators } from '@angular/forms';
import { TaskService } from 'src/app/services/regulatory/task.service';
import { StudyService } from 'src/app/services/regulatory/study.service';

@Component({
  selector: 'app-internal-dialog',
  templateUrl: './internal-dialog.component.html',
  styleUrls: ['./internal-dialog.component.scss']
})
export class InternalDialogComponent implements OnInit {
  internalForm: UntypedFormGroup;
  messages;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.matDialog.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(private formBuilder: UntypedFormBuilder,
    public matDialog: MatDialogRef<InternalDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any,
    private taskService: TaskService,
    private studyService: StudyService
  ) { }

  ngOnInit(): void {
    this.internalForm = this.formBuilder.group({
      message: new UntypedFormControl('', [Validators.required]),
      taskID: this.data.taskID
    });
    this.taskService.retrieveTaskMessages({ taskID: this.data.taskID }).subscribe(res => {
      this.messages = res.messages;
    });
  }

  sendMessage(form) {
    if (this.internalForm.valid) {
      this.taskService.sendTaskMessages(form.value).subscribe(res => {

        this.matDialog.close(true);
      });
    } else {
      this.internalForm.markAllAsTouched();
    }

  }

  closeTask() {
    const data = {
      taskID: this.data.taskID
    };
    this.taskService.updateTask(data).subscribe(res => {
      this.matDialog.close(true);

    });
  }
}
