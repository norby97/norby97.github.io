import { NgxMatDateFormats, NgxMatDateAdapter, NGX_MAT_DATE_FORMATS } from '@angular-material-components/datetime-picker';
import { formatDate } from '@angular/common';
import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormGroup, UntypedFormControl, Validators } from '@angular/forms';
import { MAT_MOMENT_DATE_ADAPTER_OPTIONS } from '@angular/material-moment-adapter';
import { MAT_DATE_LOCALE } from '@angular/material/core';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Observable, of } from 'rxjs';

import { MY_FORMATS } from 'src/app/app.component';
import { CustomNgxDatetimeAdapter } from 'src/app/directives/ngx-mat-select/ngx-date-format';

@Component({
  selector: 'app-call-scheduling-dialog',
  templateUrl: './call-scheduling-dialog.component.html',
  styleUrls: ['./call-scheduling-dialog.component.scss'],

    { provide: NGX_MAT_DATE_FORMATS , useValue: MY_FORMATS}

export class CallSchedulingDialogComponent implements OnInit {
  indications;
  indications2;
  indications3;
  users;
  callForm:UntypedFormGroup
  prescreensYesForm: UntypedFormGroup;
  prescreensNoForm: UntypedFormGroup;
  allowsms = {
    one: false,
    two: false
  }
  sites;
  guardian = false;
  formsubmitted = false;
  schedulePreScreen = false;
  confirms = {
    sms: false,
    email: false
  }
  communityPartners;
  reasonCats;
  defaultTime = [new Date().getHours(), 0 , 0]
  selectedRefSource = '';
  loading = false; 
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<CallSchedulingDialogComponent>,
    private appointmentService: AppointmentService,
    private patientService: PatientsService,    
    public userService: UserService,

    private sccipService: SccipService ,
       @Inject(MAT_DIALOG_DATA) public data: any,) {
        this.dpheader = DatepickerHeaderComponent;
       }

  ngOnInit(): void {
    this.sccipService.getindIcationsPaginated({search:'',page:0,size:30 }).subscribe(res => {
      this.indications = res.content;
      this.indications2 = res.content;
      this.indications3 = res.content;

    });
    this.userService.getAll2({siteId:this.data?.siteID}).subscribe(res => {
      this.users = res;
    });
    this.patientService.getActiveReasonCategoriesByApptType('INFO_CALL').subscribe(res => {
      this.reasonCats = res.sort((a,b)=> this.compare(a?.name?.toLowerCase() , b?.name?.toLowerCase(),true));
      
    })
    this.prescreensYesForm = new UntypedFormGroup({
      prescreens: new UntypedFormControl('', Validators.required),
      prescreenType: new UntypedFormControl('',Validators.required),
      notes: new UntypedFormControl(''),
    });
    this.prescreensNoForm = new UntypedFormGroup({
      reasonCat: new UntypedFormControl('', Validators.required),
      followUpDate: new UntypedFormControl(''),
      waitlist1: new UntypedFormControl(''),
      waitlist2: new UntypedFormControl(''),
      notes: new UntypedFormControl(''),
    });
    this.callForm = new UntypedFormGroup({
      callDate: new UntypedFormControl(new Date(), Validators.required),
      userId: new UntypedFormControl('',Validators.required),
      type: new UntypedFormControl('info_call',Validators.required),
      interest: new UntypedFormControl('', Validators.required),
      
    });
  }
  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
}
  searcher = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        res.content.forEach(element => {
            this.indications.push(element);
        });
      })
    } else {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.indications = res.content;
      })
    }
    return of(this.indications)
  }

  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        res.content.forEach(element => {
            this.indications2.push(element);
        });
      })
    } else {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.indications2 = res.content;
      })
    }
    return of(this.indications2)
  }
  
  searcher3 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        res.content.forEach(element => {
            this.indications3.push(element);
        });
      })
    } else {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.indications3 = res.content;
      })
    }
    return of(this.indications3)
  }

  submitInquiry(form, phase) {
    this.loading = true;
      if(phase == 1) {
        if(this.prescreensNoForm.valid) {
          const data = {
          patientId: this.data.patientId,
          patientTrackId: this.data.trackId,    
          req: {
            callDateTime: formatDate(this.callForm.controls.callDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),
            userId: this.callForm.controls.userId.value,
            callType:this.callForm.controls.type.value,
            interestId:this.callForm.controls.interest.value,
            appointmentSchedule: this.schedulePreScreen,
            noSchedulePrescreen: {
              reasonCategory:this.prescreensNoForm.controls.reasonCat.value,
              followUpNeededOn: this.prescreensNoForm.controls.followUpDate.value,
              waitlist: this.prescreensNoForm.controls.waitlist1.value,
              waitlist2: this.prescreensNoForm.controls.waitlist2.value,
              notes: this.prescreensNoForm.controls.notes.value,
            }
          }
            
          }
          this.patientService.addCall(data).subscribe(res =>{
            this.loading = false;

            this.dialogRef.close(true);
          }, err => {
            this.loading = false;
          });
        } else {
            this.loading = false;
            this.prescreensNoForm.markAllAsTouched();
        }
      } else {
        if(this.prescreensYesForm.valid) {
          const data = {
          patientId: this.data.patientId,
          patientTrackId: this.data.trackId,    
          req: {
            callDateTime: formatDate(this.callForm.controls.callDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),
            userId: this.callForm.controls.userId.value,
            callType:this.callForm.controls.type.value,
            interestId:this.callForm.controls.interest.value,
            appointmentSchedule: this.schedulePreScreen,
            schedulePrescreen: {
              appointmentDateTime: formatDate(this.prescreensYesForm.controls.prescreens.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),
              prescreenType: this.prescreensYesForm.controls.prescreenType.value,
              smsConfirmation: this.confirms.sms,
              emailConfirmation: this.confirms.email,
              notes: this.prescreensYesForm.controls.notes.value
            }
          }
            
          }
          this.patientService.addCall(data).subscribe(res =>{
            this.loading = false;

            this.dialogRef.close(true);
          }, err => {
            this.loading = false;
          });
  
        } else {
            this.loading = false;
            this.prescreensYesForm.markAllAsTouched();
        }
      }
    
  }

  next(form) {
    if(this.callForm.valid) {
      this.formsubmitted = true;
    
    } else {
      this.callForm.markAllAsTouched();
    }
    
  }
  toggleStatus(status) {
    this.schedulePreScreen = status;
  }
}
