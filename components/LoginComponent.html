export class LoginComponent implements OnInit {
  actualYear = new Date().getFullYear()
  loginForm: UntypedFormGroup;
  loginResponse: AuthenticationResponse = {
    showSecurityCodeField: false,
    loginUser: null,
    message: ''
  };
  lengthRegex = new RegExp('[a-zA-Z0-9@!%;$#?&]{8,}');
  upperRegex = new RegExp('[A-Z]');
  lowerRegex = new RegExp('[a-z]');
  numberRegex = new RegExp('[0-9]');
  specialRegex = new RegExp('[@!%;$#?&]');
  constructor(
    public authService:AuthenticationService,
    private formBuilder: UntypedFormBuilder,
    private router:Router
  ) {
  }

  ngOnInit(): void {

    this.loginForm =new UntypedFormGroup({
      email: new UntypedFormControl('', Validators.required),
      password: new UntypedFormControl('', Validators.required),
      securityCode: new UntypedFormControl('')
    });
  }

  login(loginForm) {
      if(this.loginForm.valid) {
      this.loginResponse = this.authService.login(loginForm);
      if(this.loginResponse.showSecurityCodeField) {
        this.loginForm.controls.securityCode.setValidators([Validators.minLength(7),Validators.required]);
        this.loginForm.controls.securityCode.updateValueAndValidity()
      }
    } else {
      this.loginForm.markAllAsTouched();
    }
      
    
  }

  testpws(p1,seccode,email) { 
    if(this.loginResponse.showSecurityCodeField) {
      if(email == '') return false;
      return seccode?.length == 7 ;
    } else {
      return true;
    }
  }
}
