import { trigger, state, style, transition, animate } from '@angular/animations';
import { Component, OnInit, ViewChild } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';
import { MatTableDataSource } from '@angular/material/table';
import { Observable, of } from 'rxjs';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { CalendarService } from 'src/app/services/advertising/calendar.service';
import { InvoicesService } from 'src/app/services/advertising/invoices.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { UserPreferencesService } from 'src/app/services/utils/user-preferences.service';
import { DatepickerHeaderComponent } from '../../common/dialogs/utils/datepicker-header/datepicker-header.component';

@Component({
  selector: 'app-invoices',
  templateUrl: './invoices.component.html',
  animations: [trigger('detailExpand', [
    state('collapsed', style({ height: '0px', minHeight: '0' })),
    state('expanded', style({ height: '*' })),
    transition('expanded <=> collapsed', animate('225ms cubic-bezier(0.4, 0.0, 0.2, 1)')),
])],
  styleUrls: ['./invoices.component.scss']
})
export class InvoicesComponent implements OnInit {
  columnsToDisplay: string[] = ['Week/Name'];
  dataSource: MatTableDataSource<any[]> = new MatTableDataSource(null);
  expandedElement: any[] | null;
  showZeroWeeks = JSON.parse(localStorage.getItem('scUserSettings'))?.recordAdShow0Weeks;
  sites;
  loading = false;
  invoiceForm: UntypedFormGroup;
  vendors;
  sum = 0;
  reimbsum = 0
  studypageNumber = 1;
  nreimbsum = 0;
  costs: any[] = [];
  sortedWeeks: any[];
  invoiceWeeks: any[]=[]; 
  dpheader;
  isTableExpanded = JSON.parse(localStorage.getItem('scUserSettings'))?.invoicesExpanded;
  @ViewChild(MatPaginator, { static: true }) paginator: MatPaginator;
  @ViewChild(MatSort, { static: true }) sort: MatSort;
  
  constructor(private invoiceService: InvoicesService,
              private calendarService: CalendarService,
              private authService: AuthenticationService,
              private userPreferences: UserPreferencesService,
              private siteService: SiteService) { 
                this.dpheader = DatepickerHeaderComponent;
              }

  ngOnInit(): void {
    this.invoiceForm = new UntypedFormGroup({
      siteId: new UntypedFormControl('',Validators.required),
      vendorId: new UntypedFormControl('', Validators.required),
      invoiceNr: new UntypedFormControl(JSON.parse(localStorage.getItem('scUserSettings'))?.recordAdInvoiceNr,Validators.required),
      invDate: new UntypedFormControl(new Date(JSON.parse(localStorage.getItem('scUserSettings'))?.recordAdInvoiceDate == undefined ? '' :JSON.parse(localStorage.getItem('scUserSettings'))?.recordAdInvoiceDate ),Validators.required),     
      totalInvoice: new UntypedFormControl('',Validators.required)
    });


    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.invoiceForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.invoiceForm.controls.siteId.updateValueAndValidity();

    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.invoiceForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.invoiceForm.controls.siteId.updateValueAndValidity();
      });
    }
  }
  validateFloatKeyPress(el, evt) {

    var charCode = (evt.which) ? evt.which : evt.keyCode;
    if (charCode != 46 && charCode > 31
        && (charCode < 48 || charCode > 57)) {
        return false;
    }

    if (charCode == 46 && el.value.indexOf(".") !== -1) {
        return false;
    }

    return true;
}

  searcher3 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    if(pageNumber == 1) {  

      this.studypageNumber = 1;
      this.invoiceForm.controls.vendorId.reset('')
      this.invoiceForm.controls.vendorId.updateValueAndValidity();
    }
    const data = {
      search,
      active:true,
      limit:pageSize,
      skip: 0 + (( this.studypageNumber-1)*pageSize),
      siteId: this.invoiceForm.controls.siteId.value.id,
    }

   
    if(pageNumber-1 != 0) {
      this.calendarService.getVendors(data).subscribe(res => {
        this.vendors = [...this.vendors,...res];

      })
    } else {
     this.calendarService.getVendors(data).subscribe(res => { 
        this.vendors = res
      
      })
    }
    this.studypageNumber++
    return of(this.vendors)
  }


  getInvoiceWeeks() {
    this.loading = true;
    if(this.invoiceForm.valid) {
      this.userPreferences.setuserState('selectedSite',this.invoiceForm.controls.siteId.value)
      this.userPreferences.setuserState('recordAdInvoiceDate',this.invoiceForm.controls.invDate.value)
      this.userPreferences.setuserState('recordAdInvoiceNr',this.invoiceForm.controls.invoiceNr.value)
      this.userPreferences.setuserState('recordAdShow0Weeks',this.showZeroWeeks,true)
      this.invoiceService.getInvoiceWeeks({vendorId: this.invoiceForm.controls.vendorId.value}).subscribe(res => {
        this.sortedWeeks = []
        this.invoiceWeeks = res;
        this.invoiceWeeks.forEach(elem => this.costs.push({eventId: elem.eventId, budgetWeeklyId: elem.id,  wkReimb: null, wkNonreimb: null, weekTotal: 0  }));
        this.sortedWeeks = res.filter((value, index, self) => index === self.findIndex((t) => (t.weekId === value.weekId)))
        this.dataSource = new MatTableDataSource(this.sortedWeeks);
        this.dataSource.paginator = this.paginator;
        this.dataSource.sort = this.sort;
        this.loading = false;


      }, err => {
        this.loading = false;
      });
    } else {
      this.loading = false;
      this.invoiceForm.markAllAsTouched();
    }
  }

  updateWeekTotal(index,type) {
    if(type == 1 && this.costs[index].wkNonreimb == null) {
      this.costs[index].wkNonreimb = 0;  
    } else  if(type == 2 && this.costs[index].wkReimb == null){
      this.costs[index].wkReimb = 0;  
    }
    this.costs[index].weekTotal = parseFloat(this.costs[index].wkReimb) +  parseFloat(this.costs[index].wkNonreimb)
    this.sum = 0
    this.reimbsum = 0;
    this.nreimbsum = 0;
    this.costs.forEach(elem => {
      if(elem.wkReimb !== null && elem.wkNonreimb !== null) {
        this.reimbsum += parseFloat(elem.wkReimb);
        this.nreimbsum += parseFloat(elem.wkNonreimb);
        this.sum += parseFloat(elem.weekTotal);
      }
      
    });
  }
  
  toogleZeroWeeks(status) {
    this.showZeroWeeks = status;
  }

  toggleTableRows() {
    this.isTableExpanded = !this.isTableExpanded;
    this.userPreferences.setuserState('invoicesExpanded',this.isTableExpanded)

    this.dataSource.data.forEach((row: any) => {
      row.isExpanded = this.isTableExpanded;
    })
  }
  

  saveChanges(form) {
      if(this.invoiceForm.valid) {
        this.userPreferences.setuserState('selectedSite',this.invoiceForm.controls.siteId.value)
        this.userPreferences.setuserState('recordAdInvoiceDate',this.invoiceForm.controls.invDate.value)
        this.userPreferences.setuserState('recordAdInvoiceNr',this.invoiceForm.controls.invoiceNr.value)
        this.userPreferences.setuserState('recordAdShow0Weeks',this.showZeroWeeks)
        this.loading = true;
          this.sum = 0;
          this.reimbsum = 0;
          this.nreimbsum = 0;
          this.costs.forEach(elem => {
            if(elem.wkReimb !== null && elem.wkNonreimb !== null) {
              this.reimbsum += parseFloat(elem.wkReimb);
              this.nreimbsum += parseFloat(elem.wkNonreimb);
              this.sum += parseFloat(elem.weekTotal);
            }
          });
          if(this.sum.toPrecision(2) ==parseFloat(this.invoiceForm.controls.totalInvoice.value).toPrecision(2)) {
            
            const data = {  
              vendorId: this.invoiceForm.controls.vendorId.value,
              invoiceNumber: this.invoiceForm.controls.invoiceNr.value,
              invoiceDate: this.invoiceForm.controls.invDate.value,
              invoiceTotal: this.invoiceForm.controls.totalInvoice.value,
              invoiceDetails: this.costs
            }
            this.invoiceService.createInvoice(data).subscribe(res => {
              this.sortedWeeks = []
              this.invoiceWeeks = []
              this.costs =[]
              this.invoiceForm.reset('');
              this.invoiceForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
              this.invoiceForm.controls.siteId.updateValueAndValidity();
              this.invoiceWeeks.forEach(elem => this.costs.push({eventId: elem.eventId, budgetWeeklyId: elem.id, wkReimb: 0, wkNonreimb: 0, weekTotal: 0  }));
              
              
              this.sortedWeeks = []
              this.dataSource = new MatTableDataSource([]);
              this.dataSource.paginator = this.paginator;
              this.dataSource.sort = this.sort;
              this.loading = false;

            }, err => {
              this.loading = false;
          });
          } else {
            this.loading = false;
            this.calendarService.showError("Weekly amounts entered don't equal total invoice cost")
          }

      } else {
        this.loading = false;

        this.invoiceForm.markAllAsTouched();
      }
  }
}
