import { Component, OnInit } from '@angular/core';
import { UntypedFormGroup, UntypedFormControl, Validators } from '@angular/forms';
import { MatDialog, MatDialogRef } from '@angular/material/dialog';
import { AdsService } from 'src/app/services/advertising/ads.service';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { ApprovalEditDialogComponent } from '../../common/dialogs/advertising/approval-edit-dialog/approval-edit-dialog.component';
import { CreateAdDialogComponent } from '../../common/dialogs/advertising/create-ad-dialog/create-ad-dialog.component';
import { FinalAdsApprovalDialogComponent } from '../../common/dialogs/advertising/final-ads-approval-dialog/final-ads-approval-dialog.component';
import { Observable, of } from 'rxjs';
import { UserPreferencesService } from 'src/app/services/utils/user-preferences.service';

@Component({
  selector: 'app-ad-approvals',
  templateUrl: './ad-approvals.component.html',
  styleUrls: ['./ad-approvals.component.scss']
})
export class AdApprovalsComponent implements OnInit {
  sites;
  siteForm: UntypedFormGroup;
  selectedStudy;
  studySelected = false;
  selectedSite;
  studies
  printchecks = []
  internetchecks = []
  actualPage = 0
  tvchecks = []
  radiochecks = []
  checkedAds = [];
  otherchecks = []
  ads: {
    radios:any[],
    internets:any[] ,
    others:any[],
    prints:any[],
    tvs: any[]
  }
  loading = false;

  constructor( private authService: AuthenticationService,
               private dialog: MatDialog,
    private siteService: SiteService,
    private userPreferences: UserPreferencesService,
    private adsService: AdsService) { }

  ngOnInit(): void {
    this.siteForm = new UntypedFormGroup({
      siteID: new UntypedFormControl('', Validators.required),
      studyStartID: new UntypedFormControl('', Validators.required)
    });
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )  {   
     this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.siteForm.controls.siteID.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.siteForm.controls.siteID.updateValueAndValidity();

    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.siteForm.controls.siteID.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.siteForm.controls.siteID.updateValueAndValidity();
      });
    }
  }

  getStudies(event) {
    this.selectedSite = event.value;
    this.userPreferences.setuserState('selectedSite',this.siteForm.controls.siteID.value)
    this.studySelected = false;
    this.actualPage = 0;
    const data = {
      search:'',
      pageNumber: 0,
     pageSize: 30,
      siteId: this.siteForm.controls.siteID.value.id,
      statuses: ['MOVING_FORWARD','ADD_ON','ACTIVE']
    }
    this.siteService.getStudies(data).subscribe(res => {
      this.studies = [...this.studies,...res.content];
    })
  }
  openCreateAdDialog() {
    const ref = this.dialog.open(CreateAdDialogComponent, {
      data: {
        sites: this.sites,
        study: this.selectedStudy
      },
      height: 'auto',
        maxHeight:'80vh',
        maxWidth:'80vw',
        width: 'auto',
        disableClose: true,
        panelClass:"custom-dialog-container"
    })

    ref.afterClosed().subscribe(res => {
      if(res) {
        this.getAds();

      }
    })
  }

  showBulkBtn() {
    this.otherchecks.map(elem => {
      if(elem.includes(true) ){
        return true;
      }
    })
    this.radiochecks.map(elem => {
      if(elem.includes(true) ){
        return true;
      }
    })
    
    this.internetchecks.map(elem => {
      if(elem.includes(true) ){
        return true;
      }
    })
    
    this.tvchecks.map(elem => {
      if(elem.includes(true) ){
        return true;
      }
    })
    
    this.printchecks.map(elem => {
      if(elem.includes(true) ){
        return true;
      }
    })
    
    return false;
  }

  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    this.userPreferences.setuserState('selectedSite',this.siteForm.controls.siteID.value)
    if(pageNumber ==1 )  {
      this.actualPage = 0;
      this.siteForm.controls.studyStartID.reset('')
      this.siteForm.controls.studyStartID.updateValueAndValidity();
    }
    const data = {
      search,
      pageNumber: this.actualPage,
      pageSize,
      siteId: this.siteForm.controls.siteID.value.id,
      statuses: ['MOVING_FORWARD','ADD_ON','ACTIVE']
    }
    if(pageNumber - 1 != 0) {
      this.siteService.getStudies(data).subscribe(res => {
        res.content.forEach(element => {
          this.studies.push(element);
      });
      })
    } else {
      this.siteService.getStudies(data).subscribe(res => {
        this.studies = []
        this.studies =[...this.studies,...res.content]
      })
    }
    this.actualPage++;
    return of(this.studies)
  }

  openBulkApproveDialog() {
    this.checkedAds = [];
    this.ads.others.map((elem,i) => {
      for(let j = 0; j++; j<4) {
        if(this.otherchecks[i][j]) {
          this.checkedAds.push({adId:elem.id, request:elem.types[j]})
        }
      }
    })
    this.ads.internets.map((elem,i) => {
      for(let j = 0; j++; j<4) {
          if(this.internetchecks[i][j]) {
          this.checkedAds.push({adId:elem.id, request:elem.types[j]})
        }
      }
    })
    this.ads.tvs.map((elem,i) => {
      for(let j = 0; j++; j<4) {
        if(this.tvchecks[i][j]) {
          this.checkedAds.push({adId:elem.id, request:elem.types[j]})
        }
      }
    })
    this.ads.radios.map((elem,i) => {
      for(let j = 0; j++; j<4) {
        if(this.radiochecks[i][j]) {
          this.checkedAds.push({adId:elem.id, request:elem.types[j]})
        }
      }
    })
    this.ads.prints.map((elem,i) => {
      for(let j = 0; j++; j<4) {
        if(this.printchecks[i][j]) {
          this.checkedAds.push({adId:elem.id, request:elem.types[j]})
        }
      }
    })
    
  }

  toggle(i,row) {
      const ref =this.dialog.open(ApprovalEditDialogComponent, {
        data: {
          allrow: true,
          row,
          study: this.selectedStudy
        },
        height: 'auto',
        maxHeight:'80vh',
        maxWidth:'80vw',
        width: 'auto',
        disableClose: true,
        panelClass:"custom-dialog-container"

      })
      ref.afterClosed().subscribe(res => {
        if(res) {
          this.getAds();
        }
      })
  }

  getByStudy(form) {
    if(this.siteForm.valid) {
      this.selectedStudy = this.siteForm.controls.studyStartID.value;
      this.studySelected = true;

      this.getAds();

    } else {
      this.siteForm.markAllAsTouched();
    }
    
  }

  getAds() {
    this.loading = true;

    this.adsService.getAds(this.selectedStudy.id).subscribe(res => {
      this.ads = res;
      this.ads.others.map(elem => {
        this.otherchecks.push([false,false,false,false])
      })
      this.ads.radios.map(elem => {
        this.radiochecks.push([false,false,false,false])
      })
      this.ads.internets.map(elem => {
        this.internetchecks.push([false,false,false,false])
      })
      this.ads.tvs.map(elem => {
        this.tvchecks.push([false,false,false,false])
      })
      this.ads.prints.map(elem => {
        this.printchecks.push([false,false,false,false])
      })
      this.loading = false;

    }, err => {
      this.loading = false;
    })
  }

  openItemEditDialog(item, row) {
      const ref =this.dialog.open(ApprovalEditDialogComponent, {
        data: {
          item,
          allrow: false,
          row,
          study: this.selectedStudy
        },
        height: 'auto',
        maxHeight:'80vh',
        maxWidth:'80vw',
        width: 'auto',
        disableClose: true,
        panelClass:"custom-dialog-container"

      })
      ref.afterClosed().subscribe(res => {
        if(res) {
          this.getAds();
        }
      })
    }

    openApprovalDialog(item) {
      const ref =this.dialog.open(FinalAdsApprovalDialogComponent, {
        data: {
          item,
          study: this.selectedStudy
        },
        height: 'auto',
        maxHeight:'80vh',
        maxWidth:'80vw',
        width: 'auto',
        disableClose: true,
        panelClass:"custom-dialog-container"

      })
      ref.afterClosed().subscribe(res => {
        if(res) {
          this.getAds();
        }
      })
    }
}
