  getTotals(index){
      this.visitmatrix[index]?.forEach((elem,i) =>{
        elem.forEach(visit => {
          if(visit != null) {
            visit.forEach(p => {
              this.hsums[index][i] += p.amount;
              this.hcounts[index][i]++
            })
          }
        })
      })
      for(let i = 0; i<this.visits[index].length; i++) {
        this.sums[index][i] = this.getColumnSum(i, index)
        this.counts[index][i]= this.getColumnCount(i,index);
      }
      let costtemp =this.sums[index].reduce((partialSum, a) => partialSum + a, 0);
      let visittemp = this.counts[index].reduce((partialSum, a) => partialSum + a, 0);
      this.ttotal[index] = {cost:costtemp, visits:visittemp}

  }

  getColumnSum(j,index) {
    let sum = 0
    for(let i = 0; i<this.visitmatrix[index].length; i++) {
      if(this.visitmatrix[index][i][j]!=null) {
        this.visitmatrix[index][i][j].forEach(element => {
          sum+=element.amount
        });
      }
    }
    return sum;
  }

  getColumnCount(j,index) {
    let count = 0
    for(let i = 0; i<this.visitmatrix[index].length; i++) {
      if(this.visitmatrix[index][i][j]!=null) {
        this.visitmatrix[index][i][j].forEach(element => {
          count++
        });
      }
    }
    return count;

  }

  sortData(sort: Sort) {
    const data = this.studyActivityreportDetails;
    if (!sort.active || sort.direction === '') {
      this.sortedData = data;
      return;
    }
    this.sortedData = data.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'initial-dollar-per-visit':
          return this.compare(a.studyDetailReportDto.initialDollarsPerVisit, b.studyDetailReportDto.initialDollarsPerVisit, isAsc);
        case 'visits':
          return this.compare(a.visitsNr, b.visitsNr, isAsc);
        case 'actual-dollar-per-visit':
          return this.compare( a.averageVisit, b.averageVisit, isAsc);
        case 'total-revenue':
          return this.compare(a.totalAmountOfStudies, b.totalAmountOfStudies, isAsc);
       case 'studyName':
          return this.compare(a.studyName.toLowerCase(), b.studyName.toLowerCase(), isAsc);
      }
    });
  }

  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
