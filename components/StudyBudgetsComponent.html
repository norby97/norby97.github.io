import { trigger, state, style, transition, animate } from '@angular/animations';
import { AfterViewInit, Component, OnInit, ViewChild } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { MatPaginator } from '@angular/material/paginator';
import { MatSelect } from '@angular/material/select';
import { MatSort, Sort } from '@angular/material/sort';
import { MatTableDataSource } from '@angular/material/table';
import { Router } from '@angular/router';
import { Observable, of } from 'rxjs';
import { Budgets } from 'src/app/models/advertising/budgets.model';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { BudgetService } from 'src/app/services/advertising/budget.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { CreateBudgetDialogComponent } from '../../common/dialogs/advertising/create-budget-dialog/create-budget-dialog.component';
import { PdfPreviewDialogComponent } from '../../common/pdf-preview-dialog/pdf-preview-dialog.component';
import { UserPreferencesService } from 'src/app/services/utils/user-preferences.service';
@Component({
  selector: 'app-study-budgets',
  animations: [trigger('detailExpand', [
    state('collapsed', style({ height: '0px', minHeight: '0' })),
    state('expanded', style({ height: '*' })),
    transition('expanded <=> collapsed', animate('225ms cubic-bezier(0.4, 0.0, 0.2, 1)')),
])],
  templateUrl: './study-budgets.component.html',
  styleUrls: ['./study-budgets.component.scss']
})
export class StudyBudgetsComponent implements OnInit, AfterViewInit {
  data = {
    pageNumber: 0,
   pageSize: 30,
  }
  viewForm: UntypedFormGroup;
  sites;
  
  loading = false;
  notesloading = false;
  actualPage= 0;
  studies=[];
  blist=[];
  blistcopy= [];
  @ViewChild('studyselect') select;
  @ViewChild(MatSort, { static: true }) sort: MatSort;
  projectMNote = false;
  selectedStudy;
  studySelected = false;
  selectedSite;
  studyBudget: Budgets
  studyBudgetCopy:any =[]
  noteForm: UntypedFormGroup;
  notes;
  includepm = true;
  constructor(private router: Router,
              private dialog: MatDialog,
              public preferencesService:UserPreferencesService,
              private budgetService: BudgetService,
              private authService: AuthenticationService,
              private userPreferences: UserPreferencesService,
              private siteService: SiteService) { }
  ngAfterViewInit(): void {
    
  }

  ngOnInit(): void {
  
    this.viewForm = new UntypedFormGroup({
      siteID: new UntypedFormControl('', Validators.required),
      studyStartID: new UntypedFormControl('', Validators.required)
    });
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.viewForm.controls.siteID.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.viewForm.controls.siteID.updateValueAndValidity();
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.viewForm.controls.siteID.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.viewForm.controls.siteID.updateValueAndValidity();
      });
    }
    this.noteForm = new UntypedFormGroup({
      note: new UntypedFormControl('', Validators.required),
      sites: new UntypedFormControl([],),
    });

  }

  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    if(pageNumber ==1  ) {
      this.actualPage =0
    }
    const data = {
      search,
      pageNumber: this.actualPage,
      pageSize,
      siteId:   this.viewForm.controls.siteID.value.id,
      statuses: ['MOVING_FORWARD','ADD_ON','ACTIVE']
    }
    if(pageNumber - 1 != 0) {
      this.siteService.getStudies(data).subscribe(res => {
        this.studies =[...this.studies,...res.content]

      })
    } else {
      this.siteService.getStudies(data).subscribe(res => {
        this.studies =[...res.content]
      })
    }
    this.actualPage++;
    return of(this.studies)
  }

  refreshNotes() {
    this.notesloading = true;
    this.budgetService.getBudgetNotes({studyID: this.selectedStudy, includepm: this.includepm}).subscribe(res => {
      this.notes = res;
      this.notes?.map(elem => {
        elem.createdDate = elem.createdDate.replace('T',' ') + ' UTC'
        elem.createdDate = elem.createdDate.toString()
      })
      this.notesloading = false;
    })
  }
  
  toggle() {
    this.projectMNote = !this.projectMNote;
  }
  sendNote(form) {
    if(this.noteForm.valid) {
      const data = {
        studyID: this.selectedStudy,
        request: {
          note: this.noteForm.controls.note.value,
          projectManagementNote: this.projectMNote,
          sites: this.noteForm.controls.sites.value,
        }
        
      }
     
      this.budgetService.addBudgetNotes(data).subscribe(res => {

        this.refreshNotes()
        this.noteForm.reset('');
        this.noteForm.controls.sites.setValue([]);
        this.noteForm.controls.sites.updateValueAndValidity();
        this.projectMNote = false;
      }, err => {
        this.loading = false;
      })      
    
    } else {
      this.noteForm.markAllAsTouched();
    }
  }

  getStudy(form) {
    this.loading = true;
    this.userPreferences.setuserState('selectedSite',this.viewForm.controls.siteID.value,true)
    if(this.viewForm.valid) {
      this.selectedStudy = this.viewForm.controls.studyStartID.value
      this.refreshNotes();
      this.budgetService.getStudyBudgets(this.selectedStudy).subscribe(res => {
        this.studySelected = true;
        this.studyBudget = res;
        this.blist = res.budgetList;
        this.blistcopy = res.budgetList;
        this.blist.map(elem => {
          elem.budgeted
        })
        this.loading = false;
      }, err => {
        this.loading = false;
      })

    } else {
      this.loading = false;
      this.viewForm.markAllAsTouched();
    }
  }

  getStudies(event) {
    this.selectedSite = event.value;
    this.studySelected = false;
    this.actualPage = 0;
    const data = {
      search:'',
      pageNumber: 0,
     pageSize: 30,
      siteId:   this.viewForm.controls.siteID.value.id,
      statuses: ['MOVING_FORWARD','ADD_ON','ACTIVE']
    }
    this.siteService.getStudies(data).subscribe(res => {
      this.studies =[...res.content]
    })
  }

  openMaintainBudgetDialog() {
  const dialogref = this.dialog.open(CreateBudgetDialogComponent, {
      data: {
        study: this.selectedStudy,
        site: this.selectedSite
      },
      height: 'auto',
      maxHeight:'80vh',
      maxWidth:'80vw',
      width: 'auto',
      disableClose: true,
      panelClass:"custom-dialog-container"
    });
    dialogref.afterClosed().subscribe(res => {
      if(res) {
        this.refreshSelectedStudy();
      }
    })
  }

  refreshSelectedStudy() {
    this.budgetService.getStudyBudgets(this.selectedStudy).subscribe(res => {
      this.studySelected = true;
      this.studyBudget = res;
      this.blist = res.budgetList;
        this.blistcopy = res.budgetList;
    })
    
  }

  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
  
  sortData(sort: Sort) {
    const data = this.blistcopy
    if (!sort.active || sort.direction === '') {
        this.studyBudget.budgetList = data;
        return;
    }
    this.blistcopy = data.sort((a, b) => {
        const isAsc = sort.direction === 'asc';
        switch (sort.active) {
            case 'date': return this.compare(new Date(a?.date).getTime(), new Date(b?.date).getTime(), isAsc);
            case 'description': return this.compare(a?.title?.toLowerCase(), b?.title?.toLowerCase(), isAsc);
            case 'vendor': return this.compare(a?.vendorName?.toLowerCase(), b?.vendorName?.toLowerCase(), isAsc);
            case 'budget': return this.compare(a?.budget, b?.budget, isAsc);
            case 'total': return this.compare(a?.totalCost, b?.totalCost, isAsc);
            case 'reimbursable': return this.compare(a?.reimbursable, b?.reimbursable, isAsc);
            case 'nonreimbursable': return this.compare(a?.nonReimbursable, b?.nonReimbursable, isAsc);
          
            default: return 0;
        }
    });
    this.studyBudget.budgetList =  this.blistcopy;
  }

  downloadDocumentation(id) {
    this.budgetService.downloadDocumentation(id).subscribe(res => {
      const blob = new Blob([res], {  type: "application/pdf"} );

      this.dialog.open(PdfPreviewDialogComponent, {
        data:{
            showprint:true,
            deletefile:false,

            actualFile: blob
        }, 
        height: 'auto',
        maxHeight:'95vh',
        minWidth:'85vw',
        width: 'auto',
        disableClose: true,
        panelClass:"custom-dialog-container"
        })
    })
  }
}
