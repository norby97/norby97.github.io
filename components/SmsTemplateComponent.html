import { Component, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { SmsService } from 'src/app/services/administration/sms.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { FeedBackDialogComponent } from '../feed-back-dialog/feed-back-dialog.component';

@Component({
  selector: 'app-sms-template',
  templateUrl: './sms-template.component.html',
  styleUrls: ['./sms-template.component.scss']
})
export class SmsTemplateComponent implements OnInit {
  templateForm: UntypedFormGroup;
  sites;
  tags = ['APPOINTMENT_DATE', 'APPOINTMENT_TIME', 'PATIENT_FIRST_NAME', 'PATIENT_LAST_NAME', 'LOGGED_USER_FIRST_NAME', 'GUARDIAN_FIRST_NAME','GUARDIAN_LAST_NAME']
  templates = [];
  selectedTemplate;
  viewCase = false;
  editCase = false;
  constructor(private smsService: SmsService,
              private dialog:MatDialog,
              private siteService: SiteService,
              public utilService: UtilService,
              private authService: AuthenticationService) { }

  ngOnInit(): void {
    this.refreshTemplates();
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
      });
    }
    this.templateForm = new UntypedFormGroup({
      name: new UntypedFormControl('New Template Name', Validators.required),
      site: new UntypedFormControl('',Validators.required),
      option: new UntypedFormControl(false,Validators.required),
      text: new UntypedFormControl('',Validators.required),
    })
  }

  addCase(){
    this.viewCase = true;
    this.editCase = false;
    this.selectedTemplate = null;
    this.templateForm.reset()
    this.templateForm.controls.name.setValue('New Template Name');
    this.templateForm.controls.option.setValue(false);
    this.templateForm.controls.option.updateValueAndValidity();
    this.templateForm.controls.name.updateValueAndValidity();
    
  }

  addToSMS(item) {
    this.templateForm.controls.text.setValue(this.templateForm.controls.text.value + ' #' + item + '# ');
    this.templateForm.controls.text.updateValueAndValidity();
  }

  createSms(form){
    if(this.templateForm.valid) {
      const data = {
        appointmentOption: this.templateForm.controls.option.value,
        name: this.templateForm.controls.name.value,
        siteId: this.templateForm.controls.site.value,
        textMessage: this.templateForm.controls.text.value
      }
      this.smsService.create(data).subscribe(res => {
        this.viewCase = false;

        this.refreshTemplates()
      })
    } else {
      this.templateForm.markAllAsTouched();
    }
  }

  refreshTemplates() {
    this.smsService.getTemplatesByCompany().subscribe(res => {
      this.templates = res;
    })
  }



  loadTemplate(template) {
    this.viewCase = true;
    this.editCase = true;
    this.selectedTemplate = template;
    this.templateForm.controls.name.setValue(template.name);
    this.templateForm.controls.option.setValue(template.appointmentOption);
    this.templateForm.controls.site.setValue(template.siteId);
    this.templateForm.controls.text.setValue(template.textMessage);
    this.templateForm.controls.text.updateValueAndValidity();
    this.templateForm.controls.site.updateValueAndValidity();
    this.templateForm.controls.option.updateValueAndValidity();
    this.templateForm.controls.name.updateValueAndValidity();
    

  }

  editSms(form) {
    if(this.templateForm.valid) {
      const data = {
        templateId: this.selectedTemplate.id,
        req : {
          appointmentOption: this.templateForm.controls.option.value,
          name: this.templateForm.controls.name.value,
          siteId: this.templateForm.controls.site.value,
          textMessage: this.templateForm.controls.text.value
        }
        
      }
      this.smsService.update(data).subscribe(res => {
        this.viewCase = false;

        this.refreshTemplates()
      })
    } else {
      this.templateForm.markAllAsTouched();
    }
  }

  deleteTemplate(template) {
    const ref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to delete ' + template.name + '?'
      }
    })
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.smsService.delete(template.id).subscribe(res => {
          this.viewCase = false;
    
          this.refreshTemplates()    
        })
      }
    })
    
  }

}
