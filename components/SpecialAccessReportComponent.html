  calculateWeeksElapsed(date: Date): number {
    date = new Date(date)
    const oneWeekInMilliseconds = 1000 * 60 * 60 * 24 * 7;

    const currentDate = new Date();
    const dateInMilliseconds = date.getTime();
    const currentInMilliseconds = currentDate.getTime();

    const timeDiff = currentInMilliseconds - dateInMilliseconds;
    const weeksElapsed = Math.floor(timeDiff / oneWeekInMilliseconds);

    return weeksElapsed;
  }
