import { ChangeDetectorRef, Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import * as  printJS from 'print-js';
import { forkJoin } from 'rxjs';
import { FileService } from 'src/app/services/regulatory/file.service';
import { StudyService } from 'src/app/services/regulatory/study.service';
import { saveAs } from 'file-saver';
import { InitAuditTableService } from '../../../administration/audit/audit-view-table.component/init-audit-table-service';
import { TaskService } from 'src/app/services/regulatory/task.service';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { MatDialog } from '@angular/material/dialog';
import { AddTaskRoleDialogComponent } from 'src/app/components/common/dialogs/regulatory/add-task-role-dialog/add-task-role-dialog.component';
import { CreateTaskDialogComponent } from 'src/app/components/common/dialogs/regulatory/create-task-dialog/create-task-dialog.component';
import { TaskMessagesDialogComponent } from 'src/app/components/common/dialogs/regulatory/task-messages-dialog/task-messages-dialog.component';
import { UpdateFileDialogComponent } from 'src/app/components/common/dialogs/regulatory/update-file-dialog/update-file-dialog.component';

@Component({
    selector: 'app-my-tasks',
    templateUrl: './my-tasks.component.html',
    styleUrls: ['./my-tasks.component.scss']
})
export class MyTasksComponent implements OnInit {
    url = null;
    studyName = '';
    actualFileID;
    actualFile = null;
    actualFileName = null;
    actualFileDetails = null;
    actualFileOpenTasks;
    actualFileDoneTasks;
    actualFileNewTasks;
    studyID;
    myTasks = [];
    myOverdueTasks = [];
    myFileOverdueTasks = [];
    constructor(private route: ActivatedRoute,
        private fileService: FileService,
        private ref: ChangeDetectorRef,
        private dialog: MatDialog,
        private taskService: TaskService,
        private authService: AuthenticationService,
        private studyService: StudyService,
        public initAuditTableService: InitAuditTableService) { }

    ngOnInit(): void {
        this.myTasks = [];
        this.taskService.retrieveTasks({
            fileID: '', status: ['NEW', 'OPEN'],
            type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? 'EXTERNAL' : 'INTERNAL', userID: this.authService.loginUserValue.id
        }).subscribe(res => {
            let pushed = false;
            res.tasks.forEach(task => {
                pushed = false;
                task.assignedUsers.forEach(user => {
                    if (user.id === this.authService.loginUserValue.id) {
                        this.myTasks.push(task);
                        pushed = true;
                        this.ref.detectChanges();
                    }
                });
                if (task.creator.id === this.authService.loginUserValue.id && pushed === false) {
                    this.myTasks.push(task);
                    this.ref.detectChanges();
                }
            });
        });
        this.taskService.retrieveTasks({
            fileID: '', status: ['OVERDUE'],
            type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? ['EXTERNAL'] : ['INTERNAL'], userID: this.authService.loginUserValue.id
        }).subscribe(res => {
            this.myOverdueTasks = res.tasks;
        });
    }

   
   
    retrieveFile(id, studyID, withrender = true) {
        this.actualFileID = id;
        this.studyID = studyID;
        this.retrieveFileTasks(id);
        const data = {
            fileID: id,
            studyID: this.studyID,
        };
        this.initAuditTableService.next({companyId: this.authService.loginUserValue.company.id, studyId: this.studyID, siteId: '', filePathId: id, userId: '' });
        const file = this.fileService.retrieveFile(data);
        const fileDetails = this.fileService.retrieveFileDetails(data);
        forkJoin([file, fileDetails]).subscribe(res => {
            this.url = res[0];
            this.actualFile = res[0];
            this.actualFileDetails = res[1];
        });
    }

    printFile() {
        if (this.url === null) {
            this.studyService.showError('Please select a PDF first.');
        } else {
            const base64 = btoa(new Uint8Array(this.url).reduce((data, byte) => data + String.fromCharCode(byte), ''));
            printJS({ printable: base64, type: 'pdf', showModal: false, base64: true });
        }
    }

    downloadFile() {
        if (this.actualFile === null) {
            this.studyService.showError('Please select a PDF first.');
        } else {
            const blob = new Blob([this.actualFile], { type: 'application/pdf' });
            saveAs(blob, this.actualFileName);
        }
    }

    openAddRoleDialog(taskID, taskName, taskDescription) {
        const dialogRef = this.dialog.open(AddTaskRoleDialogComponent, {
            width: 'auto', 
            panelClass:'custom-dialog-container',
            minWidth:'30vw',
            maxHeight:'80vh',
            height:'auto',
            disableClose: true,
            data: {
                taskID,
                taskName,
                taskDescription
            }
        });
    }

    retrieveFileTasks(fileID) {
        const newTasks = this.taskService.retrieveTasks({
            fileID, status: ['NEW'],
            type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? ['EXTERNAL'] : ['INTERNAL'], userID: this.authService.loginUserValue.id
        });
        const openTasks = this.taskService.retrieveTasks({
            fileID, status: ['OPEN'],
            type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? ['EXTERNAL'] : ['INTERNAL'], userID: this.authService.loginUserValue.id
        });
        const doneTasks = this.taskService.retrieveTasks({
            fileID, status: ['DONE'],
            type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? 'EXTERNAL' : ['INTERNAL'], userID: this.authService.loginUserValue.id
        });
        const overdueTasks = this.taskService.retrieveTasks({
            fileID, status: ['OVERDUE'],
            type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? ['EXTERNAL'] : ['INTERNAL'], userID: this.authService.loginUserValue.id
        });
        forkJoin([openTasks, doneTasks, newTasks, overdueTasks]).subscribe(result => {
            this.actualFileOpenTasks = result[0].tasks;
            this.actualFileDoneTasks = result[1].tasks;
            this.actualFileNewTasks = result[2].tasks;
            this.myFileOverdueTasks = result[3].tasks;
        });
    }
    retriveTaskMessages(taskID, taskName, taskDescription) {
        this.taskService.retrieveTaskMessages({ taskID }).subscribe(res => {
            const dialogRef = this.dialog.open(TaskMessagesDialogComponent, {
                width: 'auto', 
            panelClass:'custom-dialog-container',
            minWidth:'30vw',
            maxHeight:'80vh',
            height:'auto',
            disableClose: true,
                data: {
                    taskID,
                    taskName,
                    taskDescription,
                    messages: res.messages,
                }
            });
        });
    }

    openCreateTaskDialog() {
        const dialogRef = this.dialog.open(CreateTaskDialogComponent, {
            width: 'auto', 
            panelClass:'custom-dialog-container',
            minWidth:'30vw',
            maxHeight:'80vh',
            height:'auto',
            disableClose: true,
            data: {
                studyID: this.studyID,
                fileID: this.actualFileID,
                details: this.actualFileDetails
            }
        });
        dialogRef.afterClosed().subscribe(res => {
            if(res === true) {
                const newTasks = this.taskService.retrieveTasks({
                    fileID: '', status: ['NEW'],
                    type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? ['EXTERNAL'] : ['INTERNAL'], userID: this.authService.loginUserValue.id
                });
                const openTasks = this.taskService.retrieveTasks({
                    fileID: '', status: ['OPEN'],
                    type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? ['EXTERNAL'] : ['INTERNAL'], userID: this.authService.loginUserValue.id
                });
                const doneTasks = this.taskService.retrieveTasks({
                    fileID: '', status: ['DONE'],
                    type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? ['EXTERNAL'] : ['INTERNAL'], userID: this.authService.loginUserValue.id
                });
                forkJoin([openTasks, doneTasks, newTasks]).subscribe(resarray => {
                    resarray.forEach(result => {
                        result.tasks.forEach(task => {
                            task.assignedUsers.forEach(user => {
                                if (user.id === this.authService.loginUserValue.id) {
                                    this.myTasks.push(task);
                                }
                            });
                        });
                    });
                });
            }
        });
    }

    openUpdateFileDialog() {
        const dialogRef = this.dialog.open(UpdateFileDialogComponent, {
            width: 'auto', 
            panelClass:'custom-dialog-container',
            minWidth:'30vw',
            maxHeight:'80vh',
            height:'auto',
            disableClose: true,
            data: {
                studyID: this.studyID,
                fileID: this.actualFileID,
                details: this.actualFileDetails
            }
        });
    }
}
