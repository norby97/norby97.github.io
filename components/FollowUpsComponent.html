import { Component, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { FollowUpService } from 'src/app/services/crm_bd/follow-up.service';
import { UserService } from 'src/app/services/administration/user.service';
import { ContactDetailsDialogComponent } from '../../common/dialogs/crmbd/contact-details-dialog/contact-details-dialog.component';
import { FeedBackDialogComponent } from '../../common/dialogs/utils/feed-back-dialog/feed-back-dialog.component';
import { Sort } from '@angular/material/sort';
import { DatepickerHeaderComponent } from '../../common/dialogs/utils/datepicker-header/datepicker-header.component';
import { Observable, of } from 'rxjs';

@Component({
  selector: 'app-follow-ups',
  templateUrl: './follow-ups.component.html',
  styleUrls: ['./follow-ups.component.scss']
})
export class FollowUpsComponent implements OnInit {
  date  = new Date();
  users: any = [];
  userscopy: any = [];
  followUps: any = [];
  followUpsCopy: any = [];
  loadedTaskNotes = null;
  loadedTaskName = null;
  loadedTaskDue = null;
  loadedTaskContact = null;
  userClicked = false;
  noteForm: UntypedFormGroup;
  currentUserID='';
  loadedTaskID: null;
  historyNotes;
  loading = false; 
  usersloading = false;
  dpheader;
  constructor(private userService: UserService,
              public authService: AuthenticationService,
              private followService: FollowUpService,
              private dialog: MatDialog) { 
                this.dpheader = DatepickerHeaderComponent;
              }

  ngOnInit(): void {
    this.noteForm = new UntypedFormGroup({
      note: new UntypedFormControl('', Validators.required),
      assignTo: new UntypedFormControl('',Validators.required),
      dueDate: new UntypedFormControl('',Validators.required)
    })
 
    this.usersloading = true;
    this.userService.getAll2().subscribe(res => {
      this.users = res;
      this.userscopy = res;
      this.usersloading = false;
      this.getFollowUpsByUser({value:this.authService.loginUserValue.id })
    })
  }
  searcher5 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    let temp = []
    if(search == '') {
        this.users = this.userscopy;
    } else {
        this.userscopy.map(elem => {
            if(elem?.firstName?.toLowerCase().indexOf(search.toLowerCase()) > -1 || elem?.lastName?.toLowerCase().indexOf(search.toLowerCase()) > -1) {
                temp.push(elem);
            }
        });
        this.users = temp;
    }
    
    return of(this.users)
  }
  getFollowUpsByUser(event) {
    this.loading = true;
    this.userClicked = true;
    this.currentUserID = event.value;
    this.followService.getAllFollowUpsByUserID(event.value).subscribe(res => {
      this.followUps = res;
      this.followUpsCopy = res;
      this.loading = false
    });
  }

  loadTask(task) {
    this.loadedTaskName = task.name;
    this.loadedTaskDue = task.dueDate;
    this.loadedTaskID = task.followTaskId;
    this.loadedTaskContact = {    
      id: task.contactId,
      firstName: task.contactFullName.split(' ')[0],
      lastName: task.contactFullName.split(' ')[1],
    }
    this.followService.getAllNotesByFollowTaskID(task.followTaskId).subscribe(res => {
      this.loadedTaskNotes = res[0];
      this.historyNotes = res;
    });
  }

  openContactDetailsDialog(contactID){
    this.dialog.open(ContactDetailsDialogComponent, {
      data: {
        contact: this.loadedTaskContact
      },
      height: 'auto',
        maxHeight:'80vh',
    maxWidth:'80vw',
        width: 'auto',
        disableClose: true,
        panelClass:'custom-dialog-container',    
    });
  }

  closeTask() {
    const dialogref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to delete this task from Follow-Ups?',
      },
      height: 'auto',
        maxHeight:'80vh',
    maxWidth:'80vw',
        width: 'auto',
        disableClose: true,      
    } ) 
    dialogref.afterClosed().subscribe(res => {
      if(res) {
        this.followService.closeTask({followTaskID: this.loadedTaskID}).subscribe(res => {

         
        })
        this.loadedTaskNotes = null;;
        this.loadedTaskName = null;
        this.loadedTaskDue = null;
        this.loadedTaskContact = null;
        this.followService.getAllFollowUpsByUserID(this.currentUserID).subscribe(res => {
          this.followUps = res;
          this.followUpsCopy = res;
        });
      }
    })
  }
    
  addNote(form) {
    this.loading = true;
    if(this.noteForm.valid) {
      const data = {
        followTaskID: this.loadedTaskID,
        request: {
          dueDate: this.noteForm.controls.dueDate.value,
          note: this.noteForm.controls.note.value,
          userAssignedToId: this.noteForm.controls.assignTo.value,
        }
      }
      this.followService.addNoteToFollowUpTask(data).subscribe(res => {

        this.noteForm.reset('');
        this.followService.getAllNotesByFollowTaskID(this.loadedTaskID).subscribe(res => {
          this.loadedTaskNotes = res[0];
          this.historyNotes = res;
        });
        this.loading = false;
      }, err => {
        this.loading = false;
      })
    } else {
      this.loading = false;
      this.noteForm.markAllAsTouched();
    }
  }

  sortData(sort: Sort) {
    const data = this.followUps;
    if (!sort.active || sort.direction === '') {
      this.followUpsCopy = data;
      this.followUps = this.followUpsCopy;
      return;
    }
     this.followUpsCopy = data?.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'date': return this.compare(new Date(a?.dueDate), new Date(b?.dueDate), isAsc);
        case 'name': return this.compare(a.name?.toLowerCase(), b?.name?.toLowerCase(), isAsc);
        default: return 0;
      }
    });
    this.followUps = this.followUpsCopy;

  }
  
  compare(a: number | string | Date, b: number | string | Date, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
}
