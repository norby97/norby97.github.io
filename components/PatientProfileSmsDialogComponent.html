import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormGroup, UntypedFormControl, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { SmsService } from 'src/app/services/administration/sms.service';
import { UtilService } from 'src/app/services/utils/UtilService';

@Component({
  selector: 'app-patient-profile-sms-dialog',
  templateUrl: './patient-profile-sms-dialog.component.html',
  styleUrls: ['./patient-profile-sms-dialog.component.scss']
})
export class PatientProfileSmsDialogComponent implements OnInit {
  smsForm: UntypedFormGroup;
  smshistory = [];
  phones = [];
  smstemplates = [];
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<PatientProfileSmsDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any,
    public utilService: UtilService,
    private smsService: SmsService
    ) { }

  ngOnInit(): void {

    this.smsForm = new UntypedFormGroup({
      phone: new UntypedFormControl('', Validators.required),
      template: new UntypedFormControl(''),
      message: new UntypedFormControl('', Validators.required),
    })
    if((!this.data?.patientDetails?.allowSmsNumberOne && !this.data?.patientDetails?.allowSmsNumberTwo) || (!this.data?.patientDetails?.phoneNumberOne && !this.data?.patientDetails?.phoneNumberTwo)) {
      this.smsService.showError('no-phone-allowed');
      this.dialogRef.close();
    }
    if(this.data?.patientDetails?.allowSmsNumberOne && this.data?.patientDetails?.phoneNumberOne) {
      this.phones.push(this.data?.patientDetails?.phoneNumberOne);
      this.smsForm.controls.phone.setValue(this.data?.patientDetails?.phoneNumberOne);
      this.smsForm.controls.phone.updateValueAndValidity();
    }
    if(this.data?.patientDetails?.allowSmsNumberTwo && this.data?.patientDetails?.phoneNumberTwo) {
      this.phones.push(this.data?.patientDetails?.phoneNumberTwo)
    }    
    this.smsService.getTemplatesBySite({siteId: this.data?.patientDetails?.siteId, option:false}).subscribe(res => {
      this.smstemplates = res;
    });
   this.smsService.getAppointmentSMSs({patientId: this.data?.patient?.patientId}).subscribe(res => {
    this.smshistory = res;
   })
  
   

  }

  loadMessage(event) {
    this.smsService.getsmsTemplateMessagePatient({smsTemplateId: event.value, patientId:this.data?.patientDetails?.id}).subscribe(res => {
      this.smsForm.controls.message.setValue(res);
      this.smsForm.controls.message.updateValueAndValidity();
    })
  }


  sendSms(form){
    if(this.smsForm.valid) {
      const data = {
        patientId: this.data?.patientDetails?.id,
        req: {
          message: this.smsForm.controls.message.value,
          phoneNumber:this.smsForm.controls.phone.value,
        }
      }
      this.smsService.sendSMSPatient(data).subscribe(res => {

        this.dialogRef.close(true);
      })
    } else {
      this.smsForm.markAllAsTouched()
    }
  }

}
