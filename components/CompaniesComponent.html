import {Component, OnInit, ViewChild} from '@angular/core';
import {MatDialog} from '@angular/material/dialog';
import {CompanyDetailsDto} from '../../../models/administration/company/CompanyDetailsDto';
import {UserService} from '../../../services/administration/user.service';
import {CompanyService} from '../../../services/administration/company.service';
import {User} from '../../../models/administration/user/User';
import {Company} from '../../../models/administration/company/Company';
import {CompanyDialogComponent} from './company-dialog.component';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';
import { MatTableDataSource } from '@angular/material/table';
import { trigger, state, style, transition, animate } from '@angular/animations';
import { CompanySitesDialogComponent } from '../../common/dialogs/utils/company-sites-dialog/company-sites-dialog.component';

@Component({
    selector: 'app-companies', 
    animations: [trigger('detailExpand', [
        state('collapsed', style({ height: '0px', minHeight: '0' })),
        state('expanded', style({ height: '*' })),
        transition('expanded <=> collapsed', animate('225ms cubic-bezier(0.4, 0.0, 0.2, 1)')),
      ])],
    templateUrl: './companies.component.html',
    styleUrls: ['./companies.component.scss'],
})
export class CompaniesComponent implements OnInit {
    companiesDetails: CompanyDetailsDto[] = [];
    users: User[] = [];
    columnsToDisplay: string[] = ['company-name', 'manage-company.research-sites', 'studies', 'status', 'action'];
    dataSource: MatTableDataSource<any>;
    expandedElement: [] | null;
    loading = true;
    constructor(
        private companyService: CompanyService,
        private userService: UserService,
        private dialog: MatDialog,
        public auth: AuthenticationService
    ) {
    }

    openSitesDialog(company){
        const ref = this.dialog.open(CompanySitesDialogComponent,{
            height: 'auto',
            maxWidth:'85vw',
            minWidth:'85vw',
            width: 'auto',
            disableClose: true,
            panelClass:'custom-dialog-container',
            data:{
                company:company
            }
           });
        ref.afterClosed().subscribe(res => {
            if(res) {
                //TODO:
            }
        })
    }

    ngOnInit(): void {

        this.companyService.getAll().subscribe(res => {
            this.companiesDetails = res.companyDetailsDtos;
            this.dataSource = new MatTableDataSource(res.companyDetailsDtos);
            this.loading = false;

        },err => {
            this.loading = false;
        });
   
    }

    openCreateCompanyDialog(company: Company) {
        const dialogRef = this.dialog.open(CompanyDialogComponent, {
            width: 'auto',
            data: company,
            disableClose: true
        });
        const sub = dialogRef.componentInstance.formSubmitted.subscribe(() => {
            this.ngOnInit();
        });
    }

    makeInactive(companyID) {
        this.companyService.getByID(companyID).subscribe(res => {
            res.company.active = !res.company.active;
            const newCompany = res.company;
            this.companyService.save(newCompany).subscribe(result => {

            });
        });
    }

}
