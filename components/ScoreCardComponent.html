import { Component, OnInit, ViewChild } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { MatSort, Sort } from '@angular/material/sort';
import { Router } from '@angular/router';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { ScoreCardService } from 'src/app/services/ctms/score-card.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { UserPreferencesService } from 'src/app/services/utils/user-preferences.service';
import { WindowscoreService } from 'src/app/services/utils/window-score.service';
import { WindowService } from 'src/app/services/utils/window.service';
import { WindowrefService } from 'src/app/services/utils/windowref.service';
import { ContactSearchResultsForStudiesComponent } from '../../common/dialogs/crmbd/contact-search-results-for-studies/contact-search-results-for-studies.component';
import { EditScoreboardDialogComponent } from '../../common/dialogs/ctms/edit-scoreboard-dialog/edit-scoreboard-dialog.component';
import { ChangeStudyStatusDialogComponent } from '../../common/dialogs/ctms/change-study-status-dialog/change-study-status-dialog.component';
import { StudyKnowledgebaseDialogComponent } from '../../common/dialogs/utils/study-knowledgebase-dialog/study-knowledgebase-dialog.component';

@Component({
  selector: 'app-score-card',
  templateUrl: './score-card.component.html',
  styleUrls: ['./score-card.component.scss']
})
export class ScoreCardComponent implements OnInit {
  siteForm: UntypedFormGroup;
  sites;
  scores=[];
  scorescopy=[];
  actualSearch = ''
  scoreloading = [].fill(0,2500,0);
  loading = false;  
  studyDetails=[]
  scoreCommitment ;  
  @ViewChild(MatSort, { static: true }) sort: MatSort;
  @ViewChild('searchinput', { static: true }) input ;

  constructor(private scoreCardService: ScoreCardService,
    private authService: AuthenticationService,
    public router: Router,
    public dialog: MatDialog,
    private utilService:UtilService,
    public windowService:WindowscoreService,
    public windowRef: WindowrefService,
    private userPreferences:UserPreferencesService,
    private siteService: SiteService,) { }

  ngOnInit(): void {
    
    this.siteForm = new UntypedFormGroup({
      sites: new UntypedFormControl('',Validators.required)
    });
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.siteForm.controls.sites.setValue(JSON.parse(localStorage.getItem('scUserSettings'))?.scoreCardSelectedSite == undefined ? [this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id).id] :JSON.parse(localStorage.getItem('scUserSettings'))?.scoreCardSelectedSite);
      this.siteForm.controls.sites.updateValueAndValidity();
      this.getScores()
     
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.siteForm.controls.sites.setValue(JSON.parse(localStorage.getItem('scUserSettings'))?.scoreCardSelectedSite == undefined ? [this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id).id] :JSON.parse(localStorage.getItem('scUserSettings'))?.scoreCardSelectedSite);
        this.siteForm.controls.sites.updateValueAndValidity();
        this.sites = res;
      this.getScores()
     
      });
    }

  }
  openStudyKnowledgeBaseDialog(study){
    
      
      const dialogRef = this.dialog.open(StudyKnowledgebaseDialogComponent, {
        height: 'auto',
        maxHeight:'80vh',
        minWidth:'80vw',
        minHeight:'80vh',
        width: 'auto',
        disableClose: true,
        data:{
          studyBuild:false,
          study
        },
        panelClass:'custom-dialog-container',
       
    });
    }
  getScores() {
      this.loading = true;
      if(this.siteForm.valid) {
        this.userPreferences.setuserState('scoreCardSelectedSite',this.siteForm.controls.sites.value,true)
        this.scoreCardService.getScoresBySites({sites:this.siteForm.controls.sites.value}).subscribe(res => {
          this.scores = res;
          this.scorescopy = res;
          this.scores.map(elem => this.studyDetails.push(null))
          if(JSON.parse(localStorage.getItem('scUserSettings'))?.saveScoreCardSearch ==true) {
            this.actualSearch = JSON.parse(localStorage.getItem('scUserSettings'))?.scoreCardSearchText;
            
            this.sortData(this.sort)
            this.applyFilter({target:{value:JSON.parse(localStorage.getItem('scUserSettings'))?.scoreCardSearchText}})
          } 
          this.loading = false;
        }, err => {
          this.loading = false;
        });
      } else {
          this.loading = false;
          this.siteForm.markAllAsTouched();
      }
      
  }

  
  applyFilter(event) {
    
    let temp = []
    this.actualSearch = event?.target?.value;
    this.userPreferences.setuserState('scoreCardSearchText',this.actualSearch)
      if(event.target.value == '') {
        this.scores = this.scorescopy;
      } else {
        this.scorescopy.map(elem => {
          if(elem.studyName?.replace(/\s+/g, ' ').toLowerCase().indexOf(event.target.value.toLowerCase()) > -1 ) {
            temp.push(elem);
            }
        });
        this.scores = temp;
    }
  }

  sortData(sort: Sort) {
    const data = this.scorescopy;
    if (!sort.active || sort.direction === '') {
      this.scores = data;
      return;
    }
    this.scores = data.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'study': return this.compare(a.studyName?.toLowerCase(), b.studyName?.toLowerCase(), isAsc);
        case 'leadcrc': return this.compare(a.leadCrcFullName?.toLowerCase(), b.leadCrcFullName?.toLowerCase(), isAsc);
        case 'backupcrc': return this.compare(a.backupCrcFullName?.toLowerCase(), b.backupCrcFullName?.toLowerCase(), isAsc);
        case 'indication': return this.compare(a.indicationName?.toLowerCase(), b.indicationName?.toLowerCase(), isAsc);
        case 'pi': return this.compare(a.principalInvestigatorFullName?.toLowerCase(), b.principalInvestigatorFullName?.toLowerCase(), isAsc);
        case 'status': return this.compare(this.utilService.getMasterStatus(a.studyStatus),this.utilService.getMasterStatus(b.studyStatus) , isAsc);
        case 'cro': return this.compare(a.croName?.toLowerCase(), b.croName?.toLowerCase(), isAsc);

      }
    });
  }
  openAddContact(scoreDetails,i) {
    this.scoreloading[i]=1;
    if(this.studyDetails[i] != null) {
      const ref = this.dialog.open(ContactSearchResultsForStudiesComponent, {
        data: {
         studyStartID:  this.studyDetails[i].studyStartSiteId,
         siteID:this.studyDetails[i].siteId,

         maxWidth:'80vw',

      ref.afterClosed().subscribe(res => {
        if(res) {
        }
        this.scoreloading[i]=0;

      })  
    } else {

        this.studyDetails[i]=res;
        const ref = this.dialog.open(ContactSearchResultsForStudiesComponent, {
          data: {
           studyStartID:  res.studyStartSiteId,
           siteID:res.siteId,
           scorecard: true,
          },
           width: 'auto',

           disableClose: true,
           maxHeight:'80vh',
           maxWidth:'80vw',
           panelClass:'custom-dialog-container',
        });
        ref.afterClosed().subscribe(res => {
          if(res) {
          }
          this.scoreloading[i]=0;

        })  
      })
    }
   
   }
   openChangeStatus(scoreDetails,i) {
    this.scoreloading[i]=1;
    if(this.studyDetails[i] != null) {
      const ref = this.dialog.open(ChangeStudyStatusDialogComponent, {
        data: {
         studyStartID:  this.studyDetails[i].studyStartSiteId,
         siteID:this.studyDetails[i].siteId,
         status: scoreDetails.actionStatus,
        },
         width: 'auto',
         height: 'auto',
         disableClose: true,
         maxHeight:'80vh',
         maxWidth:'80vw',
         panelClass:'custom-dialog-container',
      });
      ref.afterClosed().subscribe(res2 => {
        if(res2) {
         this.getScores()
         
        }
        this.scoreloading[i]=0;

      }) 
    } else {
      this.scoreCardService.getStudyScoreDetails(scoreDetails.id).subscribe((res:any) => {
        this.studyDetails[i]=res;
        const ref = this.dialog.open(ChangeStudyStatusDialogComponent, {
          data: {
           studyStartID:  res.studyStartSiteId,
           siteID:res.siteId,
           status: scoreDetails.actionStatus,
          },
           width: 'auto',
           height: 'auto',
           disableClose: true,
           maxHeight:'80vh',
           maxWidth:'80vw',
           panelClass:'custom-dialog-container',
        });
        ref.afterClosed().subscribe(res2 => {
          if(res2) {
           this.getScores()
          }
          this.scoreloading[i]=0;

        }) 
      })
    }
  
    
   }
   openEditscoreboardDialog(scoreDetails,i) {
    this.scoreloading[i]=1;
    if(this.studyDetails[i] != null) {
      this.scoreCardService.getStudyScoreCommitment(scoreDetails.id).subscribe(res =>{
        this.scoreCommitment =res

  
      const ref =this.dialog.open(EditScoreboardDialogComponent, {
        data: {
          scoreId:scoreDetails.id,
          commitment:this.scoreCommitment,
          scoreDetails:this.studyDetails[i]
        },
        width: 'auto',
        panelClass:'custom-dialog-container',
        disableClose: true,
        height: 'auto',
        maxHeight:'80vh'
      });
        ref.afterClosed().subscribe(res => {
          if(res) {
            this.getScores()
          }
          this.scoreloading[i]=0

        })    })  
    } else {
      this.scoreCardService.getStudyScoreDetails(scoreDetails.id).subscribe((res:any) => {
        this.studyDetails[i]=res;
        this.scoreCardService.getStudyScoreCommitment(scoreDetails.id).subscribe(res1 =>{
          this.scoreCommitment =res1

  
        const ref =this.dialog.open(EditScoreboardDialogComponent, {
          data: {
            scoreId:scoreDetails.id,
            commitment:this.scoreCommitment,
            scoreDetails:res
          },
          width: 'auto',
          panelClass:'custom-dialog-container',
          disableClose: true,
          height: 'auto',
          maxHeight:'80vh'
        });
          ref.afterClosed().subscribe(res => {
            if(res) {
              this.getScores()
            }
            this.scoreloading[i]=0

          })    })
      })
    }


  }

  loadScoreCard(score,i) {
    this.scoreloading[i] = 1;
    this.userPreferences.setuserState('scoreCardSearchText',this.actualSearch)
    this.userPreferences.setuserState('saveScoreCardSearch',true)
    this.router.navigate(['/ctms/score-cards', score.id, score.studyName]);
    this.scoreloading[i] = 0;

  }


  openInNewWindow(score,i) {
    if(score?.regManager ==1) {
      this.scoreloading[i] = 1;
      this.windowService.open( score.id,  score.studyName.replace('(','').replace(')','').replace('/',''))
      this.scoreloading[i] = 0;

    }
    }

  compare(a: number | string | Date, b: number | string | Date, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
}
