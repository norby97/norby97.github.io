import { Component, OnInit } from '@angular/core';
import { HttpClient } from "@angular/common/http";
import { MatDialog } from "@angular/material/dialog";
import { ErrorDialogService } from "../generic-warning-dialog/error-dialog.service";
import { UntypedFormControl, UntypedFormGroup } from "@angular/forms";
import { Sort } from '@angular/material/sort';
import { CategoriesService } from 'src/app/services/regulatory/categories.service';
import { AddCategoryDialogComponent } from '../add-category-dialog/add-category-dialog.component';


@Component({
  selector: 'app-reason-category',
  templateUrl: './reason-category.component.html',
  styleUrls: ['./reason-category.component.scss']
})
export class ReasonCategoryComponent implements OnInit {

  loadedCategories;
  isFetching = false;
  categoryForm: UntypedFormGroup;
  sortedData: any;
  term;

  constructor(
    private categoriesService: CategoriesService,
    private dialog: MatDialog,) { }

  ngOnInit(){
    this.getCategory();

    this.categoryForm = new UntypedFormGroup({
      active: new UntypedFormControl(true),
      name: new UntypedFormControl(""),
    });

    this.refreshCategories();

  }


  getCategory() {
    this.isFetching = true;

    this.categoriesService.getAll().subscribe((data) => {
      this.isFetching = false;
      this.loadedCategories = data;
    });
  }

 addNewReasonCategoryDialog() {
  const dialogref = this.dialog.open(AddCategoryDialogComponent, {
    disableClose: true,
    minWidth: "30rem",width: 'auto',
    height: 'auto',
    maxHeight:'80vh',
    maxWidth:'80vw',
    panelClass:'custom-dialog-container',
    data:{categoryObject: null}
  });

  dialogref.afterClosed().subscribe(res => {
    if(res) {
      this.refreshCategories()
    }
  })
}

editCategoryDialog(data) {
  const dialogref = this.dialog.open(AddCategoryDialogComponent, {
    width: 'auto',
    height: 'auto',minWidth: "30rem",
    disableClose: true,
    maxHeight:'80vh',
    maxWidth:'80vw',
    panelClass:'custom-dialog-container',
    data:{categoryObject: data}
  });

  dialogref.afterClosed().subscribe(res => {
    if(res) {
      this.refreshCategories()
    }
  })
}

toggleStatus(event, reasonCategory) {
  this.categoriesService.tooglePermissionStatus(reasonCategory).subscribe(
    (res) => {
      this.refreshCategories()
    }
  );
}
toggleStatus2(event, reasonCategory,type) {



  if(type ==0 ) {
    reasonCategory.activeInInfoCall = event;
  }
  if(type ==1 ) {
    reasonCategory.activeInPrescreen = event;
  }

    reasonCategory.activeInScreen = event;
  }

    reasonCategory.activeInVisit = event;

    activeInScreen: reasonCategory.activeInScreen,
    activeInVisit: reasonCategory.activeInVisit,
  })
  .subscribe((res) => {

  });
}


  sortData(sort: Sort) {
    const data = this.loadedCategories;
    if (!sort.active || sort.direction === '') {
      this.sortedData = data;
      return;
    }
    this.sortedData = data.sort((a, b) => {
      const isAsc = sort.direction === 'asc';
      switch (sort.active) {
        case 'categoryName': return this.compare(a.name?.toLowerCase(), b.name?.toLowerCase(), isAsc);
        case 'active': return this.compare(a.active, b.active, isAsc);
        default:
          return 0;
      }
    });
  }

  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }

  refreshCategories(){
    this.categoriesService.getAll().subscribe(res => {
      this.loadedCategories = res;
    })
  }
}
