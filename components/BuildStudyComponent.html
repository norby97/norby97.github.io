  toggleStatus(status) {
    this.failHandling = status;
    if(this.failHandling) {
      this.firstFormGroup.controls.sfnr.setValidators(Validators.required);
      this.firstFormGroup.controls.sfnr.updateValueAndValidity();
      this.firstFormGroup.controls.ernr.setValidators(Validators.required);
      this.firstFormGroup.controls.ernr.updateValueAndValidity();
      this.firstFormGroup.controls.ratioperer.setValue('ENROLLED');
      this.firstFormGroup.controls.ratioperer.setValidators(Validators.required);
      this.firstFormGroup.controls.ratioperer.updateValueAndValidity();
      this.firstFormGroup.controls.sfpayment.setValidators(Validators.required);
      this.firstFormGroup.controls.sfpayment.updateValueAndValidity();
    } else {
      this.firstFormGroup.controls.sfnr.setValidators(null);
      this.firstFormGroup.controls.sfnr.updateValueAndValidity();
      this.firstFormGroup.controls.ernr.setValidators(null);
      this.firstFormGroup.controls.ernr.updateValueAndValidity();
      this.firstFormGroup.controls.ratioperer.setValue('');
      this.firstFormGroup.controls.ratioperer.setValidators(null);
      this.firstFormGroup.controls.ratioperer.updateValueAndValidity();
      this.firstFormGroup.controls.sfpayment.setValidators(null);

    }

    var result = new Date(date);

  checkIfExists(event) {
    this.warningmessage = false;
    this.data2.forEach(elem => {
      if(event.target.value  == elem.pName || event.target.value  == elem.pCaption ) {
        this.warningmessage = true;
      }
    });
  }
  
  drop(event: CdkDragDrop<string[]>, status) {
    if(status ==1) {
      moveItemInArray(this.procedures, event.previousIndex, event.currentIndex);
    } else {
      moveItemInArray(this.visitRows, event.previousIndex, event.currentIndex);

    }
    
  }

  deleteLast() {
    if(this.pathForm.controls.name.value == 'EOS' || this.pathForm.controls.name.value == 'eos') {
      this.pathForm.enable();
      this.pathForm.reset('');
      this.data2 = this.data2.slice(0, -1);
      if(this.data2.length == 0) {
        this.previousToEOS = [];
      }
     }  else {
      this.data2 = this.data2.slice(0, -1);
     }
  }

  deleteVisitRow(index) {
    this.visitRows.splice(index, 1);
    this.procedureTotals.splice(index, 1);
    this.visitPayments.splice(index, 1);
  }

    this.procedures.splice(index, 1);
    if(this.procedures.length == 0) {
      this.procedureTotals.forEach((procedure,index) => {
          procedure.subtotal = 0;
          procedure.adjusted = 0;
          procedure.overheadtotal = 0;
      });
    } else {
      this.refreshTotals()
    }
  }

  changeData(form) {

          this.data2 = [... this.data2,{

              pPlanStart: this.addDays(new Date('2017-03-05'), 2),

              pParent: 0,
              pDepend: '0,',
              }];
          this.pathForm.reset('');

          this.data2 = [... this.data2,{

          }
    }

      this.pathForm.markAllAsTouched();
    }
  }

  visitsCheck(stepper: MatStepper) {
    this.visitsError = false;
    this.visitsFieldsError = false;
    if(this.visitRows.length == 0) {
      this.visitsError = true;
    } else {
      let sources = [];
      let baselineCheckedVisits = [];
      this.visitRows.forEach(elem => {
        if(elem.details.enrollment) {
          baselineCheckedVisits.push(elem.path);
        } 
        if(elem.name == '' || elem.path == '' || elem.type == '') {
          this.visitsFieldsError = true;
        }
      });
        this.data2.forEach(elem => {
          if(elem.pDepend.split(',')[1] == '') {
            sources.push({id: elem.pID, basecount: 0})
          }
        });

        baselineCheckedVisits.forEach(elem => {
          sources.forEach(elem2 => {
            if(elem2.id == this.getSource(elem)) {
              elem2.basecount++;
            }
          });
        });
        sources.forEach(elem => {
          if(elem.basecount < 1) {
            this.visitsError = true;
          }
        });
        if(!this.visitsError && !this.visitsFieldsError) {
          this.visitRows.forEach((elem,i) => {
            this.visitPayments[i].visitName = elem.name;
            this.visitPayments[i].pathName = elem.path.pCaption
          })
          stepper.next();
          
        }
    }
  }

  sortData(sort: Sort) {
    const data = this.visitRows.slice();
    if (!sort.active || sort.direction === '') {
        this.visitRows = data;
        return;
    }
    this.visitRows = data.sort((a, b) => {
        const isAsc = sort.direction === 'asc';
        switch (sort.active) {
            case 'name': return this.compare(a.name?.toLowerCase(), b.name?.toLowerCase(), isAsc);
          
            default: return 0;
        }
    });
}


compare(a: number | string, b: number | string, isAsc: boolean) {
  return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
}
