  ngOnInit(): void {
   
    this.queryForm = new UntypedFormGroup({
      condition: new UntypedFormControl('', Validators.required),
      siteID: new UntypedFormControl('', Validators.required),
      operator: new UntypedFormControl('', Validators.required),
      value: new UntypedFormControl('', Validators.required),
    })
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sitevalues = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.queryForm.controls.siteID.setValue(this.sitevalues.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.queryForm.controls.siteID.updateValueAndValidity();
      this.getStudies({value:this.sitevalues.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id)})
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sitevalues = res;
        this.queryForm.controls.siteID.setValue(this.sitevalues.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.queryForm.controls.siteID.updateValueAndValidity();
        this.getStudies({value:this.sitevalues.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id)})
      });
    }
    
  }

  groupNumbers() {

  exporCSV() {
    this.csvloading = true;
    const data = {
      allPatients: true,
      patients: [],
      conditionRequest: {conditions:this.queryGroups}
    }
    this.recruitingService.queryCSV(data).subscribe(res => {
      const blob = new Blob([res], {  type: "text/csv;charset=utf-8"} );
      this.csvloading = false;

      saveAs(blob,'Query Results.csv'); 
    }, err =>{
      this.csvloading = false;
    })
  }

  getOperators(event) {
    if(event.value == 'Has Address' || event.value == 'Has Email' || event.value == 'SMS Allowed') {
      this.resetConditions();
      this.yesnoValues = true;
      this.operators = ['=']
      this.queryForm.controls.operator.setValue('=');
      this.queryForm.controls.operator.updateValueAndValidity()
    } else if(event.value == 'Gender'){
      this.operators = ['=']
      this.resetConditions()
      this.queryForm.controls.operator.setValue('=');
      this.queryForm.controls.operator.updateValueAndValidity()
      this.genderValues = true;
    } else if(event.value == 'Waitlist'){
      
      this.resetConditions()
      this.operators = ['=']
      this.queryForm.controls.operator.setValue('=');
      this.queryForm.controls.operator.updateValueAndValidity()
      this.waitlistValues = true;
      this.sccipService.getindIcations().subscribe(res => {
        this.indications = res;
      });
    } else if(event.value == 'Protocol'){
      this.resetConditions()
      this.protocolValues = true;
      this.operators = ['LIKE']
      this.queryForm.controls.operator.setValue('LIKE');
      this.queryForm.controls.operator.updateValueAndValidity()
    }  else if(event.value == 'Age'){
      this.resetConditions()
      this.ageValues = true;
      this.operators = ['<', '>', '=']
    } else if(event.value == 'Sponsor'){
      this.resetConditions()
      this.sponsorValues = true;
      this.operators = ['LIKE']
      this.queryForm.controls.operator.setValue('LIKE');
      this.queryForm.controls.operator.updateValueAndValidity()
    } else if(event.value == 'Ethnicity'){
      this.operators = ['=']
       this.resetConditions();
        this.raceValues = true;
        this.queryForm.controls.operator.setValue('=');
        this.queryForm.controls.operator.updateValueAndValidity()
    } else if (event.value == 'Notes Contains'){ 
      this.operators = ['LIKE']
       this.resetConditions();
        this.notesValues = true;
        this.queryForm.controls.operator.setValue('LIKE');
        this.queryForm.controls.operator.updateValueAndValidity()
    }else if (event.value == 'Participated in a Study'){ 
      this.operators = ['=']
       this.resetConditions();
        this.studyParticipationValues = true;
        this.queryForm.controls.operator.setValue('=');
        this.queryForm.controls.operator.updateValueAndValidity()
    }else if (event.value == 'Active in a Study'){ 
      this.operators = ['=']
       this.resetConditions();
        this.activeStudyValues = true;
        this.queryForm.controls.operator.setValue('=');
        this.queryForm.controls.operator.updateValueAndValidity()
    }else if (event.value == 'Diagnosis'){ 
      this.operators = ['=']
       this.resetConditions();
        this.diagnosisValues = true;
        this.queryForm.controls.operator.setValue('=');
        this.queryForm.controls.operator.updateValueAndValidity()
        this.sccipService.getindIcations().subscribe(res => {
          this.diagnosises = res;
        });
    } else if (event.value == 'Activity Date') { 
      this.operators = ['=','<','>']
       this.resetConditions();
        this.activityDateValues = true;
    } else if (event.value == 'Status in a Study') { 
      this.operators = ['=']
       this.resetConditions();
        this.statusInStudyValues = true;
        this.queryForm.controls.operator.setValue('=');
        this.queryForm.controls.operator.updateValueAndValidity()
    } else if (event.value == 'Mailing List') { 
      this.operators = ['=']
       this.resetConditions();
        this.mailingValues = true;
        this.queryForm.controls.operator.setValue('=');
        this.queryForm.controls.operator.updateValueAndValidity()
    } else if (event.value == 'Patient Interest') { 
      this.operators = ['=']
       this.resetConditions();
        this.interestValues = true;
        this.queryForm.controls.operator.setValue('=');
        this.queryForm.controls.operator.updateValueAndValidity()
        this.sccipService.getindIcations().subscribe(res => {
          this.indications = res;
        });
    } else if (event.value == 'Tagged for Study') { 
      this.operators = ['=']

        this.tagValues = true;
        this.queryForm.controls.operator.setValue('=');
        this.queryForm.controls.operator.updateValueAndValidity()
        this.appointmentService.getDenovoStudies({pageSize:200,pageNumber:0,siteId:this.queryForm.controls?.siteID.value.id,statuses:['MOVING_FORWARD','ACTIVE'],search:''}).subscribe(res=>{
        
          this.studies = res;
          this.studies = this.studies.sort((a, b) => {
            if (a.name< b.name) {
              return -1;
            } else if (a.name > b.name) {
              return 1;
            } else {
              return 0;
            }
          });
          this.studiescopy = res;
        })
    } else if (event.value == 'Study' ){ 
      this.operators = ['=']
       this.resetConditions();
      this.querystudiesvalue = true;
        this.queryForm.controls.operator.setValue('=');
        this.queryForm.controls.operator.updateValueAndValidity()
       
    }else if (event.value == 'Advertising Event ID' ){ 
      this.resetConditions()
      this.operators = ['=', 'LIKE']
      this.queryForm.controls.operator.setValue('=');
      this.queryForm.controls.operator.updateValueAndValidity()
      this.protocolValues = true;
   
    }
  }

  addQuery(form) {
    if(this.queryForm.valid) {
      this.queryGroups[this.groupValue-1].queries.push({condition: this.queryForm.controls.condition.value, operator: this.queryForm.controls.operator.value, value: this.genderValues ? this.queryForm.controls.value.value.name : this.ageValues || this.notesValues || this.protocolValues || this.sponsorValues  ? this.queryForm.controls.value.value : this.activityDateValues ? formatDate(this.queryForm.controls.value.value, 'yyyy-MM-dd', 'en-Us') : this.tagValues ? this.queryForm.controls.value.value.id : this.queryForm.controls.value.value.id})
      this.queryGroupsToShow[this.groupValue-1].queries.push({condition: this.queryForm.controls.condition.value, operator: this.queryForm.controls.operator.value, value: this.genderValues ? this.queryForm.controls.value.value.name : this.ageValues || this.notesValues || this.protocolValues || this.sponsorValues ? this.queryForm.controls.value.value : this.activityDateValues ? formatDate(this.queryForm.controls.value.value, 'MM/dd/yyyy', 'en-Us')  : this.queryForm.controls.value.value.name});
      let temp = this.queryForm.controls.siteID.value;
      this.queryForm.reset('')
      this.queryForm.controls.siteID.setValue(temp);
      this.queryForm.controls.siteID.updateValueAndValidity();
      
      this.resetConditions()
    } else {
      this.queryForm.markAllAsTouched()
    }
  }

  addQueryGroup(form) {
    if(this.queryForm.valid) {
      this.queryGroups.push({queries: [{condition:'Site', operator:'=',value:this.queryForm.controls.siteID.value.id}]});
      this.queryGroupsToShow.push({queries: [{condition:'Site', operator:'=',value:this.queryForm.controls.siteID.value.alias}]});
      this.queryGroups[this.queryGroups.length-1].queries.push({condition: this.queryForm.controls.condition.value, operator: this.queryForm.controls.operator.value, value: this.genderValues ? this.queryForm.controls.value.value.name : this.ageValues || this.notesValues || this.protocolValues || this.sponsorValues  ? this.queryForm.controls.value.value : this.activityDateValues ? formatDate(this.queryForm.controls.value.value, 'yyyy-MM-dd', 'en-Us') : this.tagValues ? this.queryForm.controls.value.value.id : this.queryForm.controls.value.value.id});
      this.queryGroupsToShow[this.queryGroups.length-1].queries.push({condition: this.queryForm.controls.condition.value, operator: this.queryForm.controls.operator.value, value: this.genderValues ? this.queryForm.controls.value.value.name : this.ageValues || this.notesValues || this.protocolValues || this.sponsorValues ? this.queryForm.controls.value.value : this.activityDateValues ? formatDate(this.queryForm.controls.value.value, 'MM/dd/yyyy', 'en-Us') :  this.queryForm.controls.value.value.name});
      let temp = this.queryForm.controls.siteID.value;
      this.queryForm.reset('')
      this.queryForm.controls.siteID.setValue(temp);
      this.queryForm.controls.siteID.updateValueAndValidity();
      this.resetConditions()
    } else {
      this.queryForm.markAllAsTouched()
    }
  }

  deleteQuery(groupIndex, index){
    this.queryGroups[groupIndex].queries.splice(index, 1);
    this.queryGroupsToShow[groupIndex].queries.splice(index, 1);
    if(this.queryGroups[groupIndex].queries.length == 1) {
      this.queryGroups.splice(groupIndex,1).slice()
      this.queryGroupsToShow.splice(groupIndex,1).slice()
    }
   
  }
  
  deleteVisitGroup(index){

    const ref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to delete Group ' +(index+1) + '?'
      }
    })
    ref.afterClosed().subscribe(res => {
      if(res) {
        if(this.queryGroupsToShow.length ==1 && this.queryGroups.length == 1) {
        this.queryGroups = []
        this.queryGroupsToShow = []
        } else {
        this.queryGroups.splice(index,1).slice()
        this.queryGroupsToShow.splice(index,1).slice()
        }
        
      }
    })
    
  }
  back() {
    this.queryResults =[]
  }
  onScroll() {
    this.scrollloading= true;
    this.skip = this.skip + this.limit
    this.recruitingService.query({skip:this.skip, limit:this.limit,req:this.queryGroups}).subscribe(res => {
      if(res?.length == 0) {
        this.noMore = true; 
        this.scrollloading = false;
      } else {
        this.queryResults = [...this.queryResults,...res];
        res.map(elem => {this.patientChecks.push(false);})
        this.dataSource = new MatTableDataSource(this.queryResults);
        this.scrollloading = false;
      }
    },err=> {
      this.scrollloading = false;
    })
    
    
  }
  generateList() {
    this.loading = true;
    this.noMore = false;
    this.skip = 0;
    this.limit = 30;
    
    this.userPreferences.setuserState('selectedSite',this.queryForm.controls.siteID.value,true)
    this.recruitingService.query({skip:0, limit:30,req:this.queryGroups}).subscribe(res => {

      this.queryResults = res;
      this.queryResults.map(elem => {this.patientChecks.push(false);})
      this.dataSource = new MatTableDataSource(res);
      this.loading = false;
    },err=> {
      this.loading = false;
    })
  }
