    constructor(private dialog: MatDialog,
        private route: ActivatedRoute,

    ngOnInit() {
        this.route.params.subscribe(params => { this.studyID = params.id; });
        this.studyService.getStudy(this.studyID).subscribe(res => {
            this.parsedStudy = res;
            this.studyName = this.parsedStudy?.sponsor?.name + ' ' + this.parsedStudy?.protocol + ' ' + (this.parsedStudy?.nickname ? '"' + this.parsedStudy?.nickname + '"' : '');
        })
        this.fileService.getReviewedFiles({ studyID: this.studyID }).subscribe(res => {
            this.reviewedFilesID = res;
        });
        this.studyService.getStudy(this.studyID).subscribe(res => {
            this.siteID = res.site.id;
        });
        this.route.queryParams.subscribe(res => {
            if (res.fileID) {
                this.openFile = true;
                this.openedFileID = res.fileID;
            } else {
                this.openFile = false;
                this.openedFileID = null;
            }
        });
       

        if (history.state.data !== undefined) {
            localStorage.setItem('study', JSON.stringify(history.state.data));
        }
       
        this.numbers = Array(100).fill(0).map((x, i) => i);
        $('#jstree1').jstree({
            core: {
                check_callback: (operation, node, nodeParent) => {
                    if (operation === 'move_node') {
                        return node.type === 'file' && nodeParent.type === 'folder';
                    }
                },
                data: this.treeStructure,
                multiple: false,

            },
            types: {
                folder: {
                    icon: 'fa fa-folder'
                },
                file: {
                    icon: 'fa fa-file'
                },
                placeholder: {
                    icon: 'fa fa-upload'
                },
                link: {
                    icon: 'fa fa-link',
                }
            },
            dnd: {
                check_while_dragging: true
            },
            plugins: ['massload', 'search', 'types', 'dnd', 'sort'],
            sort : function (a, b)   {
                const a1 = this.get_node(a);
                const b1 = this.get_node(b);
                if(a1.parent != 1 && b1.parent != 1 ){
                    if (a1.icon == b1.icon){
                        return (a1.text > b1.text) ? -1 : 1;
                    } else {
                        return (a1.icon < b1.icon) ?  1 : -1;
                    }
                }
            }
        }).bind('move_node.jstree', (e, data) => {
            const request = {
                fileId: data.node.id,
                studyId: JSON.parse(localStorage.getItem('study')).id,
                parentId: data.parent
            };
            this.fileService.moveFile(request).subscribe(res => {

            });
        }).bind('select_node.jstree', (e, data) => {
            this.validationIconClicked = false;
            this.actualNode = data;
            this.actualFileType = data.node.type;
            if(this.ontasktab) {
                this.getTasks(this.actualNode);
            }
            if(this.oneesigntab) {
                this.getEsignTasks(this.actualNode);
            }
            this.internalValidation = new InternalValidation(data.node.original.metadata.intValidated,
                data.node.original.metadata.currentHash, data.node.original.metadata.uploadedHash);
            this.actualFileID = data.node.id;
            if (data.node.type === 'file' || data.node.type === 'link') {
                    this.loading = true

                data.node.type === 'link' ?  this.isLink = true : this.isLink = false;
                this.actualFilePreviousVersions = data.node.original.metadata.previousVersions;
                if (this.actualFilePreviousVersions !== null) {
                    this.actualFilePreviousVersions.forEach(element => {
                        if (element.validityTo === '1970-01-01') {
                            element.validityTo = 'Not Set'
                        }
                        if (element.validityFrom === '1970-01-01') {
                            element.validityFrom = 'Not Set'
                        }
                    });
                }
                this.isPlaceholder = false
                this.actualFileName = data.node.text;
                this.folderSelected = false;
                this.previousFileID = null;
                this.previousFileName = null;
                this.retrieveFile(data.node.id);

            } else if (data.node.type === 'placeholder') {
                 this.loading = true
                this.actualFileName = data.node.text;
                this.folderSelected = false;
                this.isPlaceholder = true;
                this.previousFileID = null;
                this.previousFileName = null;
                this.retrieveFile(data.node.id, false);
            } else {
                this.folderSelected = true;
                
            }
        }).bind('hover_node.jstree', (e, data) => {
        });
        this.getTreeStucture();
        this.initDone = true;
        const searchFileID = localStorage.getItem('searchFile') ? localStorage.getItem('searchFile') : null;
        if (searchFileID !== null) {
            this.retrieveFile(searchFileID);
            localStorage.removeItem('searchFile');
        }
    }

    getEsignTasks(data) {
        this.ontasktab = false;
        this.oneesigntab = true;
        const newEsignTasks = this.taskService.retrieveTasks({ fileID: data.node.id, status: ['NEW'], type: ['ESIGN'], userID: this.authService.loginUserValue.id });
        const fullySignedEsignTasks = this.taskService.retrieveTasks({ fileID: data.node.id, status: ['FULLY_SIGNED'], type: ['ESIGN'], userID: this.authService.loginUserValue.id });
        const openEsignTasks = this.taskService.retrieveTasks({ fileID: data.node.id, status: ['OPEN'], type: ['ESIGN'], userID: this.authService.loginUserValue.id });
        const doneEsignTasks = this.taskService.retrieveTasks({ fileID: data.node.id, status: ['DONE'], type: ['ESIGN'], userID: this.authService.loginUserValue.id });
        forkJoin([newEsignTasks, openEsignTasks, doneEsignTasks, fullySignedEsignTasks]).subscribe(res => {
            this.actualFileNewEsignTasks = res[0].tasks;
            this.actualFileOpenEsignTasks = res[1].tasks;
            this.actualFileDoneEsignTasks = res[2].tasks;
            this.actualFileFullySignedEsignTasks = res[3].tasks;
        });
    }

    getTasks(data) {
        this.ontasktab = true;
        this.oneesigntab = false;
        this.taskloading = true;
        const newTasks = this.taskService.retrieveTasks({
            fileID: data.node.id, status: ['NEW'],
            type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? ['EXTERNAL'] : ['INTERNAL'], userID: this.authService.loginUserValue.id
        });
        const openTasks = this.taskService.retrieveTasks({
            fileID: data.node.id, status: ['OPEN'],
            type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? ['EXTERNAL'] : ['INTERNAL'], userID: this.authService.loginUserValue.id
        });
        const doneTasks = this.taskService.retrieveTasks({
            fileID: data.node.id, status: ['DONE'],
            type: this.authService.loginUserValue.accountType === 'EXTERNAL' ? ['EXTERNAL'] : ['INTERNAL'], userID: this.authService.loginUserValue.id
        });
        forkJoin([openTasks, doneTasks, newTasks]).subscribe(res => {
            this.actualFileOpenTasks = res[0].tasks;
            this.actualFileDoneTasks = res[1].tasks;
            this.actualFileNewTasks = res[2].tasks;
            this.taskloading = false;

        }, err => {
            this.taskloading = false;
          });
    }

    back(event) {

    getTreeStucture() {

    isFileAllowed(fileName: string) {
    let isFileAllowed = false;
    const allowedFiles = ['.pdf'];
    const regex = /(?:\.([^.]+))?$/;
    const extension = regex.exec(fileName);
    if (undefined !== extension && null !== extension) {
        for (const ext of allowedFiles) {
        if (ext === extension[0]) {
            isFileAllowed = true;
        }
        }
    }
    return isFileAllowed;
    }
