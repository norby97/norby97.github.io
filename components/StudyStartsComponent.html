  searcher = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    if(search == ''){
      this.selectedIndications = [];
    }
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        res.content.forEach(element => {
            this.indications.push(element);
        });
      })
    } else {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.indications = res.content;
      })
    }
    return of(this.indications)
  }

  sortData(sort: Sort) {
    const data = this.studies;
    if (!sort?.active || sort?.direction === '') {
      this.studiesCopy = data;
      this.studies = this.studiesCopy;
      return;
    }
     this.studiesCopy = data?.sort((a, b) => {
      const isAsc = sort?.direction === 'asc';
      switch (sort.active) {
        case 'idnumber': return this.compare(a.idNumber, b?.idNumber, isAsc);
        case 'cro': return this.compare(a.croName?.toLowerCase(), b?.croName?.toLowerCase(), isAsc);
        case 'indication': return this.compare(a.indicationName?.toLowerCase(), b?.indicationName?.toLowerCase(), isAsc);
        case 'status': return this.compare(this.utilService.getMasterStatus(a.studyStatus),this.utilService.getMasterStatus(b.studyStatus), isAsc);
        case 'fpv': return this.compare(new Date(a.startDate),new Date(b.startDate), isAsc);
        case 'sponsor': return this.compare(a.sponsorName?.toLowerCase(), b?.sponsorName?.toLowerCase(), isAsc);
        case 'protocol': return this.compare(a.protocolName?.toLowerCase(), b?.protocolName?.toLowerCase(), isAsc);
        case 'compound': return this.compare(a.compoundName?.toLowerCase(), b?.compoundName?.toLowerCase(), isAsc);
        case 'investigator': return this.compare(a.principalInvestigatorFullName?.toLowerCase(), b?.principalInvestigatorFullName?.toLowerCase(), isAsc);
        case 'population': return this.compare(a.populationName?.toLowerCase(), b?.populationName?.toLowerCase(), isAsc);
        case 'phase': return this.compare(a.phase?.toLowerCase(), b?.phase?.toLowerCase(), isAsc);
        case 'startfpv': return this.compare(new Date(a?.startDate).getTime(), new Date(b?.startDate).getTime(), isAsc);
        case 'type': return this.compare(a.trialType?.toLowerCase(), b?.trialType?.toLowerCase(), isAsc);
        default: return 0;
      }
    });
    this.studies = this.studiesCopy;
    this.dataSource = new MatTableDataSource(this.studies);

  }
  
  compare(a: number | string | Date, b: number | string | Date, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
