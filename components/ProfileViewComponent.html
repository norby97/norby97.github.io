export class ProfileViewComponent implements OnInit {

    loginUser: User;
    serverUrl: string;
    disableSMSNotification = false;
    phoneNumbers;
    notifications=[];
    constructor(
        private authService: AuthenticationService,
        public initAuditTableService: InitAuditTableService,
        public utilService: UtilService,
        private userService: UserService,
        private companyService: CompanyService,
        private dialog: MatDialog,
        private changeRef: ChangeDetectorRef
    ) {
    }

    ngOnInit(): void {
        this.serverUrl = environment.serverUrl;
        
        this.userService.getUserByID(this.authService.loginUserValue.id).subscribe(res => {
            this.loginUser = res;
            this.companyService.getCompanyPhoneNumbers({companyId:this.authService.loginUserValue.company.id}).subscribe((res:any) => {
                this.phoneNumbers = res;
                if(this.phoneNumbers?.phoneNumberCtms == this.loginUser.phone || this.phoneNumbers?.phoneNumberRecruiting == this.loginUser.phone){
                    this.disableSMSNotification = true;
                } else {
                    this.disableSMSNotification = false;
                    this.utilService.checkIfBadPhoneNumberSites(this.loginUser.phone.replace(/\D/g,'').slice(-10)).subscribe(res => {
                        this.disableSMSNotification = res;
                    })
                }
            
            })
        });
        
        this.userService.getUserNotifications(this.authService.loginUserValue.id).subscribe(res => {
            this.notifications = res;
        })
       
        
        this.initAuditTableService.next({companyId: this.authService.loginUserValue.company.id,
            studyId: '',
            siteId: '',
            filePathId: '',
            userId: this.authService.loginUserValue.id
        });
    }

    openProfileChangeDetailsDialog() {
        const dialogRef = this.dialog.open(ProfileChangeDetailsDialogComponent, {
            minWidth: 'auto',
            maxHeight:'80vh',
            maxWidth:'80vw',
            width: 'auto',
            height:'auto',
            data:{
                loginUser:this.loginUser
            },
            disableClose: true,
            panelClass:'custom-dialog-container'
        });
        dialogRef.afterClosed().subscribe(() => {
            this.userService.getUserByID(this.authService.loginUserValue.id).subscribe(res => {
                this.loginUser = res;
            this.userService.getUserNotifications(this.authService.loginUserValue.id).subscribe(res => {
                this.notifications = res;
            })
            this.companyService.getCompanyPhoneNumbers({companyId:this.authService.loginUserValue.company.id}).subscribe((res:any) => {
                this.phoneNumbers = res;
                if(this.phoneNumbers?.phoneNumberCtms == this.loginUser.phone || this.phoneNumbers?.phoneNumberRecruiting == this.loginUser.phone){
                    this.disableSMSNotification = true;
                } else {
                    this.disableSMSNotification = false;
                    this.utilService.checkIfBadPhoneNumberSites(this.loginUser.phone.replace(/\D/g,'').slice(-10)).subscribe(res => {
                        this.disableSMSNotification = res;
                    })
                }
            });

            })
        });
    }

    openUpdateProfileDialog() {
        const dialogRef = this.dialog.open(ProfileUpdatePasswordDialogComponent, {
            minWidth: 'auto',
            maxHeight:'80vh',
            maxWidth:'80vw',
            width: 'auto',
            height:'auto',
            disableClose: true,
            panelClass:'custom-dialog-container'
        });
        dialogRef.afterClosed().subscribe(() => {
            
        });
    }

    removeProfilePic() {
        const dialogRef = this.dialog.open(FeedBackDialogComponent, {
            minWidth: 'auto',
            maxHeight:'80vh',
            maxWidth:'80vw',
            width: 'auto',
            height:'auto',
            disableClose: true,
            data: {
                message: 'Are you sure you want to remove your profile picture ?',
            }
        });
        dialogRef.afterClosed().subscribe((res) => {
            if (res === true) {
            
              
            } else {
                this.dialog.closeAll();
            }
        });

    }
    updateNotificationType(notification,type){
        const data ={
            userNotificationId: notification.id,
            req:{
                receiveEmail: type == 1 ? !notification.receiveEmail : notification.receiveEmail,
                receiveSms: type == 2 ? !notification.receiveSms : notification.receiveSms,
                receiveOnPlatform: type == 3 ? !notification.receiveOnPlatform : notification.receiveOnPlatform,
            }
        }
        this.userService.updateUserNotification(data).subscribe(res => {
            this.userService.getUserNotifications(this.authService.loginUserValue.id).subscribe(res => {
                this.notifications = res;
            })
        })
    }
}
