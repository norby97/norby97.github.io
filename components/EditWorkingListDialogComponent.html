import { formatDate } from '@angular/common';
import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormGroup, UntypedFormControl, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { UserService } from 'src/app/services/administration/user.service';
import { RecruitingService } from 'src/app/services/ctms/recruiting.service';
import { WorkingListService } from 'src/app/services/ctms/working-list.service';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';

@Component({
  selector: 'app-edit-working-list-dialog',
  templateUrl: './edit-working-list-dialog.component.html',
  styleUrls: ['./edit-working-list-dialog.component.scss']
})
export class EditWorkingListDialogComponent implements OnInit {
  workingForm: UntypedFormGroup;
  users;
  listDetails;
  loading = false; 
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<EditWorkingListDialogComponent>,
    private userService: UserService,
    private workingListService: WorkingListService,
    @Inject(MAT_DIALOG_DATA) public data: any) {
      this.dpheader = DatepickerHeaderComponent;
     }

  ngOnInit(): void {
    this.workingForm = new UntypedFormGroup({
      name : new UntypedFormControl('', Validators.required),
      expDate : new UntypedFormControl('', Validators.required),
      userId : new UntypedFormControl('', Validators.required),
      desc : new UntypedFormControl(''),
    })  
    this.workingListService.getWorkingListDetails({workingListId: this.data?.workinglist?.id}).subscribe(res =>{
      this.listDetails = res;
      this.workingForm = new UntypedFormGroup({
        name : new UntypedFormControl(this.listDetails?.name, Validators.required),
        expDate : new UntypedFormControl(this.listDetails?.expiresOn, Validators.required),
        userId : new UntypedFormControl(this.listDetails?.assignTo?.id, Validators.required),
        desc : new UntypedFormControl(this.listDetails?.description),
      })  
    })
    this.userService.getAll2().subscribe(res => {
      this.users = res;
    });
    
  }

  submit(form) {
  this.loading = true;
  if(this.workingForm.valid) {
      const data = {
        workingListId: this.data?.workinglist?.id,
        req: {
          name: this.workingForm.controls.name.value,
          expiresOn: formatDate(this.workingForm.controls.expDate.value, 'yyyy-MM-dd', 'en-Us'),
          assignTo: this.workingForm.controls.userId.value,
          description: this.workingForm.controls.desc.value,
        }
      }
      this.workingListService.editWorkingListDetails(data).subscribe(res => {
        this.loading = false;
        this.dialogRef.close(true);

      }, err => {
        this.loading = false;
      })
    } else {
      this.loading = false;
      this.workingForm.markAllAsTouched();
    }
  }
}
