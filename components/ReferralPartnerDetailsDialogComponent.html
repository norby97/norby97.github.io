import { Component, EventEmitter, HostListener, Inject, Input, OnInit, Output } from '@angular/core';
import { UntypedFormGroup, UntypedFormBuilder, UntypedFormControl, Validators } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { startWith, map } from 'rxjs/operators';
import { ContactFileModel } from 'src/app/models/crmbd/contact-file-model.model';
import { ContactModel } from 'src/app/models/crmbd/contact.model';
import { countries } from 'src/app/models/crmbd/countries.model';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { UserService } from 'src/app/services/administration/user.service';
import { ReferralPartnersService } from 'src/app/services/advertising/referral-partners.service';
import { ContactCompanyService } from 'src/app/services/crm_bd/contact-company.service';
import { ContactService } from 'src/app/services/crm_bd/contact.service';
import { FollowUpService } from 'src/app/services/crm_bd/follow-up.service';
import { SccipService } from 'src/app/services/crm_bd/sccip.service';
import { StudyStartsService } from 'src/app/services/crm_bd/study-starts.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { ContactDetailsDialogComponent } from '../../crmbd/contact-details-dialog/contact-details-dialog.component';
import { saveAs } from 'file-saver';
import { forkJoin } from 'rxjs';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';

@Component({
  selector: 'app-referral-partner-details-dialog',
  templateUrl: './referral-partner-details-dialog.component.html',
  styleUrls: ['./referral-partner-details-dialog.component.scss']
})
export class ReferralPartnerDetailsDialogComponent implements OnInit {

  contactDetailsForm: UntypedFormGroup;
  otherDetailsForm: UntypedFormGroup;
  linkedInForm: UntypedFormGroup;
  followUpsForm: UntypedFormGroup;
  linkForm: UntypedFormGroup;
  fileUploadForm: UntypedFormGroup;
  details: ContactModel = {
    contactDetails: null,
    otherDetails: null,
    linkedinUrl: null,
    responseStatus: '',
    responseMessage: ''
  };
  loading = false;
  buttonClicked = false;
  emailstring = '';
  contactFileTypes=  Object.values(ContactFileModel);;
  fileToUpload = null;
  fileName = '';
  webstring = '';
  roles = []
  filteredRoles: any;
  linkedinstring = '';
  countries: any = countries
  contactStudies: any = [];
  filteredCompanies: any;
  users:any =[];
  contactFiles: any = [];
  followTasks: any = [];
  existingCompanies: any;
  @Input() back: boolean;
  @Input() contact: any;
  @Output() backToTable = new EventEmitter();
  studies = []
  selectedSite;
  sites; 
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(@Inject(MAT_DIALOG_DATA) public data: any,
              private partnerService: ReferralPartnersService,
              private contactCompanyService: ContactCompanyService,
              private userService: UserService,
              private siteService: SiteService,
              private studyStartService: StudyStartsService,
              private sccip: SccipService,
              public dialogRef: MatDialogRef<ContactDetailsDialogComponent>,
              private authService: AuthenticationService,
              private followService: FollowUpService,
              private formBuilder: UntypedFormBuilder) { 
                this.dpheader = DatepickerHeaderComponent;
              }
              

  ngOnInit(): void {
    
    const existingCompanies = this.contactCompanyService.getAll();
    const roles =  this.sccip.getStudyRoles();
    const contactStudies = this.partnerService.getLinkedStudyRoles({contactID: this.back ? this.contact.id : this.data.contact.id});
    const users = this.userService.getAll2();
    const sites = this.siteService.getAll2(this.authService.loginUserValue.company.id);

    forkJoin([existingCompanies, roles, contactStudies, users, sites]).subscribe(res => {
      localStorage.setItem('siteCentric.sites', JSON.stringify(res[4]));

      this.existingCompanies = res[0];
      this.roles = res[1];
      this.contactStudies = res[2];
      this.users = res[3];
      this.sites = res[4];
    });


    this.resetForm();

    this.linkForm =this.formBuilder.group({
      siteId: new UntypedFormControl(''),
      studyId: new UntypedFormControl(''),
      role: this.formBuilder.group({ name: ['', Validators.required]}),
    });


    this.getFollowUps();

    this.retrieveFiles(this.back ? this.contact.id : this.data.contact.id);

    this.partnerService.getPartnerDetails({contactID: this.back ? this.contact.id : this.data.contact.id}).subscribe(res => {
      this.details = res;
      this.emailstring = 'mailto:' + this.details.contactDetails.email;
      this.webstring = this.details.contactDetails.webSite;
      this.linkedinstring = this.details.linkedinUrl;
      this.resetForm();
    });

    this.filteredRoles = this.linkForm.get('role.name').valueChanges.pipe(
      startWith(''),
      map(value => this._filter(value, this.roles))
    );
  }



  linkStudy(form) {
    if(this.linkForm.valid) {
      const data = {
        contactId: this.back ? this.contact.id : this.data.contact.id,
        studyStartId: this.linkForm.controls.studyId.value,
        siteId: this.linkForm.controls.siteId.value,
        role : this.linkForm.get('role.name').value
      }
      this.partnerService.createContactStudy(data).subscribe(res =>{
        

    });
    } else {
      this.linkForm.markAllAsTouched()
    }
      
  }

  getStudies(event) {
    this.selectedSite = event.value;
    this.studyStartService.getSiteStudies({siteID:event.value}).subscribe(res => {
      this.studies = res
    });
  }

  deleteContactStudy(contactStudyID) {
    this.partnerService.deleteContactStudy({contactStudyID}).subscribe(res => {

    })
  }

  saveContactOtherDetails(form) {
    if(this.otherDetailsForm.valid) {
      const data = {
        contactID: this.back ? this.contact.id : this.data?.contact.id,
        request : {
          busAddressLineOne: form.value.businessaddress1,
          busAddressLineThree: form.value.businessaddress3,
          busAddressLineTwo: form.value.businessaddress2,
          busCity: form.value.businesscity,
          busCountry: form.value.businesscountry,
          busState: form.value.businessstate,
          busZip: form.value.businesszip,
          homeAddressLineOne: form.value.homeaddress1,
          homeAddressLineThree: form.value.homeaddress3,
          homeAddressLineTwo: form.value.homeaddress2,
          homeCity: form.value.homecity,
          homeCountry: form.value.homecountry,
          homeState: form.value.homestate,
          homeZip: form.value.homezip,
          otherAddressLineOne: form.value.otheraddress1,
          otherAddressLineThree: form.value.otheraddress3,
          otherAddressLineTwo: form.value.otheraddress2,
          otherCity: form.value.othercity,
          otherCountry: form.value.othercountry,
          otherState: form.value.otherstate,
          otherZip: form.value.otherzip
        }
      };
      this.partnerService.updateContactOtherDetails(data).subscribe(res => {

      })
    } else {
      this.otherDetailsForm.markAllAsTouched();
    }
  }
  
  getFollowUps() {
    this.followService.getFollowUpTasksForContact(this.back ? this.contact.id : this.data.contact.id).subscribe(res => {
      this.followTasks = res;
    })
  }

  addFollowUp(form) {
    if(this.followUpsForm.valid) {
      const data = {
        contactID: this.back ? this.contact.id : this.data.contact.id,
        followTaskDto: {
          dueDate: this.followUpsForm.controls.dateTo.value,
          name: this.followUpsForm.controls.taskName.value,
          note: this.followUpsForm.controls.note.value,
          userAssignedToId: this.followUpsForm.controls.assignTo.value
        }
      }
    this.followService.createFollowUpTask(data).subscribe(res => {

      this.getFollowUps();
      this.followUpsForm.reset('');
    })

    } else {
      this.followUpsForm.markAllAsTouched();
    }
  }

  changeFileName(event) {
    this.fileToUpload = event.target.files.item(0);
    this.fileName = event.target.value.replace(/^.*\\/, '');
  }

  resetForm() {
    this.linkedInForm = new UntypedFormGroup({
      url: new UntypedFormControl(this.details !== null ? this.details?.linkedinUrl : '', [Validators.required])
    });
    this.fileUploadForm = new UntypedFormGroup({ 
      type: new UntypedFormControl('',[Validators.required]),
      file: new UntypedFormControl('',[Validators.required])
    })
    this.followUpsForm = new UntypedFormGroup({
      taskName: new UntypedFormControl('', [Validators.required]),
      assignTo: new UntypedFormControl('', [Validators.required]),
      dateTo: new UntypedFormControl('', [Validators.required]),
      note: new UntypedFormControl(''),
    });
    this.contactDetailsForm = this.formBuilder.group({
      category: new UntypedFormControl( this.details?.contactDetails?.category, [Validators.required]),
      subCategory: new UntypedFormControl( null),
      company: this.formBuilder.group({ name: [this.details !== null ? this.details?.contactDetails?.companyName : '', Validators.required]}),
      firstName: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.firstName : '', [Validators.required]),
      mi: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.middleName: ''),
      lastName: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.lastName : '', [Validators.required]),
      suffix: new UntypedFormControl( this.details !== null ?this.details?.contactDetails?.sufix : ''),
      department: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.department : ''),
      jobTitle: new UntypedFormControl( this.details !== null ?this.details?.contactDetails?.jobTitle : ''),
      busPhone1: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.busPhoneOne : ''),
      busPhone1Ext: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.busPhoneOneExt : ''),
      busPhone2: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.busPhoneTwo : ''),
      busPhone2Ext: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.busPhoneTwoExt : ''),
      phone: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.cellPhone : ''),
      email: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.email : ''),
      website: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.webSite : ''),
      fax: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.fax : ''),
      status: new UntypedFormControl(this.details !== null ? this.details?.contactDetails?.active : '', [Validators.required]),
    });  
    this.filteredCompanies = this.contactDetailsForm.get('company.name').valueChanges.pipe(
      startWith(''),
      map(value => this._filter(value, this.existingCompanies  ))
  );
    this.otherDetailsForm = new UntypedFormGroup({
      businessaddress1: new UntypedFormControl(this.details !== null ? this.details?.otherDetails?.busAddressLineOne : ''),
      businessaddress2: new UntypedFormControl(this.details !== null ? this.details?.otherDetails?.busAddressLineTwo : ''),
      businessaddress3: new UntypedFormControl(this.details !== null ? this.details?.otherDetails?.busAddressLineThree : ''),
      businesscity: new UntypedFormControl(this.details !== null ? this.details?.otherDetails?.busCity : ''),
