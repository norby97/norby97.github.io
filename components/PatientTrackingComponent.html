  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    this.userPreferences.setuserState('selectedSite',this.siteForm.controls.siteID.value)
    if(this.lastsearch != search || pageNumber == 1) {
      this.studypageNumber = 0
      this.studySelected = false;
    }
   
    this.lastsearch = search
 
    const data = {
      search,
      pageNumber: this.studypageNumber,
      pageSize,
      siteId: this.siteForm.controls.siteID.value.id,
      statuses: ['ACTIVE']
    }
    this.studiesloading = true;
     if(this.studypageNumber != 0) {
      this.siteService.getStudies(data).subscribe(res => {
        res.content.forEach(element => {
          element.value = element.id
          element.ngxHide = false;
          element.selected = 0
          this.studies.push(element);
      });
      this.studiesloading = false;
      }, err => {
        this.studiesloading = false;
      })
    } else {
      this.siteService.getStudies(data).subscribe(res => {
        this.studies =[...res.content]
        this.studies .forEach(element => {
          element.value = element.id
          element.ngxHide = false;
          element.selected = 0
      });
        this.studiesloading = false;
      }, err => {
        this.studiesloading = false;
      })
    }
    this.studypageNumber++;

    return of(this.studies)
  }

    }
  }
  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }
  
  sortData(sort: Sort) {
    const data = this.trackedPatients.slice();
    if (!sort.active || sort.direction === '') {
        this.trackedPatients = data;
        return;
    }
    this.trackedPatients = data.sort((a, b) => {
        const isAsc = sort.direction === 'desc';
        switch (sort.active) {
            case 'name': return this.compare(a?.patientFullName?.toLowerCase(), b?.patientFullName?.toLowerCase(), isAsc);
            default: return 0;
        }
    });
}

  trackVisit() {
    this.loading =true;

    const data = {
      patientStudyLinkId: this.selectedPatient.id,
      trackedVisitId: null,
      req: {
        questionChangePath: [],
        procedures: {
          additionalProcedures: [],
          optionalProcedures: [],
          requiredProcedures: [],
          groupsProcedure: []
        },
        screenFail: this.showSfDateAndReason,
        visitId: this.selectedVisit.visitId,
        visitScheduleId: this.selectedVisit.id
      }
      
    }

    this.visitDetails?.procedureGroups?.additionalProcedures?.forEach((elem,i) => {
      if(elem.ckeched) {
        data.req.procedures.additionalProcedures.push(elem.id);
      }
    });
    this.visitDetails?.questions?.forEach((elem,i) => {
        data.req.questionChangePath.push({fieldName: elem.inputName, value: elem.value});
    });
    
    this.visitDetails?.procedureGroups?.requiredProcedures?.forEach((elem,i) => {
      if(elem.ckeched) {
        data.req.procedures.requiredProcedures.push(elem.id);
      }
    });
    let duplicateCheck = false;
    // this.visitDetails?.procedureGroups?.groupsProcedure?.forEach((elem,i) => {
    //   let count = 0

    //   });

    //     this.hasError[i].error=true;
    //     duplicateCheck = true
    //   } 
    // })
    
    this.visitDetails?.procedureGroups?.groupsProcedure?.forEach((elem,i) => {
      elem.forEach((elem2,i) =>  {
        if(elem2.ckeched)
          data.req.procedures.requiredProcedures.push(elem2.id);
      })
    })


    this.visitDetails?.procedureGroups?.optionalProcedures?.forEach((elem,i) => {
      if(elem.ckeched) {
        data.req.procedures.optionalProcedures.push(elem.id);
      }
    });

    // if(data.req.procedures.requiredProcedures.length == 0 && data.req.procedures.optionalProcedures.length == 0 && data.req.procedures.additionalProcedures.length == 0) {
    //   this.siteService.showError('required-procedure');
    //   this.loading =false;
    // } else { 
      if(this.showSfDateAndReason){
        if(this.sfFailFrom.valid) {
          data.req.questionChangePath.push({fieldName:'screenFailDate', value: this.sfFailFrom.controls.sfDate.value == null ? null :formatDate(this.sfFailFrom.controls.sfDate.value, 'yyyy-MM-dd', 'en-Us')})
          data.req.questionChangePath.push({fieldName:'reason', value: this.sfFailFrom.controls.reason.value})
          data.req.questionChangePath.push({fieldName:'subStatus', value: this.sfFailFrom.controls.substatus.value})
            if(!duplicateCheck) {
              if(this.visitDetails.trackedVisitId != null || this.visitDetails.trackedVisitId != undefined) {
                data.trackedVisitId = this.visitDetails.trackedVisitId;
                this.trackingService.retrackVisit(data).subscribe((res:any) => {

                  this.visitCase = false;
                  this.firstTab = res?.status == 'SCREEN_FAIL' || res?.status == 'DROP' || res?.status == 'COMPLETED'  ? false : true;

                  this.loading = false;

              } else {

                  this.trackingService.trackVsit(data).subscribe((res:any) => {

                    this.visitCase = false;
                    this.firstTab = res?.status == 'SCREEN_FAIL' || res?.status == 'DROP' || res?.status == 'COMPLETED'  ? false : true;
                    this.showSfDateAndReason = false;
                    this.getByStudy(this.siteForm,this.firstTab)
                  }, err => {
                    this.loading = false; 
                  })
    
              }
              
            }
          
        } else {
          this.loading =false;
          this.sfFailFrom.markAllAsTouched();
        }
          
      } else {
        if(this.lastVisit) {
          if(this.lastVisitForm.valid) {
            if(this.lastVisitStatus == 'DROP') {
              data.req.questionChangePath.push({fieldName:'dropped', value: 'true'})
              data.req.questionChangePath.push({fieldName:'dropDate', value: this.lastVisitForm.controls.date.value == null ? null : formatDate(this.lastVisitForm.controls.date.value, 'yyyy-MM-dd', 'en-Us')})
              data.req.questionChangePath.push({fieldName:'reason', value: this.lastVisitForm.controls.reason.value})

            } else {

              data.req.questionChangePath.push({fieldName:'completedDate', value: this.lastVisitForm.controls.date.value == null ? null : formatDate(this.lastVisitForm.controls.date.value, 'yyyy-MM-dd', 'en-Us')})
            }
            if(this.visitDetails.trackedVisitId != null || this.visitDetails.trackedVisitId != undefined) {
              data.trackedVisitId = this.visitDetails.trackedVisitId;

              this.trackingService.retrackVisit(data).subscribe((res:any) => {

                this.visitCase = false;
                this.firstTab = res?.status == 'SCREEN_FAIL' || res?.status == 'DROP' || res?.status == 'COMPLETED'  ? false : true;
                this.getByStudy(this.selectedPatient,this.firstTab)
              }, err => {
                this.loading = false;

            } else {

              this.trackingService.trackVsit(data).subscribe((res:any) => {

                this.visitCase = false;
                this.firstTab = res?.status == 'SCREEN_FAIL' || res?.status == 'DROP' || res?.status == 'COMPLETED'  ? false : true;

                this.getByStudy(this.selectedPatient,this.firstTab)
              }, err => {
                this.loading = false; 
              })
            }
          } else {
            this.loading =false;
            this.lastVisitForm.markAllAsTouched();
          }
        } else {
          if(this.visitDetails.trackedVisitId != null || this.visitDetails.trackedVisitId != undefined) {
            data.trackedVisitId = this.visitDetails.trackedVisitId;
            this.trackingService.retrackVisit(data).subscribe((res:any) => {

              this.visitCase = false;
