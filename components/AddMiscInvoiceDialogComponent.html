import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { LedgerService } from 'src/app/services/finance/ledger.service';
import { StudyBuildService } from 'src/app/services/finance/study-build.service';
import { UtilService } from 'src/app/services/utils/UtilService';

@Component({
  selector: 'app-add-misc-invoice-dialog',
  templateUrl: './add-misc-invoice-dialog.component.html',
  styleUrls: ['./add-misc-invoice-dialog.component.scss']
})
export class AddMiscInvoiceDialogComponent implements OnInit {
  totalInvoice = 0;
  invoiceForm: UntypedFormGroup;
  lines = [{name: '', cost: 0}];
  loading = false;
  billingForm;
  locations
  showbilling = false;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<AddMiscInvoiceDialogComponent>,
              private ledgerService:LedgerService,
              public utilService:UtilService,
              private studyBuildService:StudyBuildService,
    @Inject(MAT_DIALOG_DATA) public data: any, ) { } 

  ngOnInit(): void {
  
    this.invoiceForm = new UntypedFormGroup({
      description: new UntypedFormControl('',Validators.required)
    })

      this.studyBuildService.getStudyBillingSetup(this.data?.study?.studyDetailId).subscribe((res :any)=>{
        this.billingForm = new UntypedFormGroup({
          company: new UntypedFormControl(res.company ? res.company : ''),
          attn: new UntypedFormControl(res.billingAttnTo ? res.billingAttnTo : '',),
          email: new UntypedFormControl(res.payEmail ? res.payEmail : ''),
          billingemail: new UntypedFormControl(res.billingEmail ? res.billingEmail : ''),
          phone: new UntypedFormControl(res.phone ? res.phone : ''),
          address1: new UntypedFormControl(res.address1 ? res.address1 : ''),
          address2: new UntypedFormControl(res.address2 ? res.address2 : ''),
          city: new UntypedFormControl(res.city ? res.city : ''),
          state: new UntypedFormControl(res.state ? res.state : ''),
          zip: new UntypedFormControl(res.zip ? res.zip : ''),
          ponr: new UntypedFormControl(res.poNumber ? res.poNumber : ''),
          remitloc:new UntypedFormControl(res.remitLocationId ? res.remitLocationId : '', Validators.required),
          receivableContact: new UntypedFormControl(res.receivableContact ? res.receivableContact : '',Validators.required), 
          term: new UntypedFormControl(res.netTerms ? res.netTerms : '', Validators.required),
          notes: new UntypedFormControl(res.invoiceNotes ? res.invoiceNotes : ''),  
        })
        
      })
      this.studyBuildService.findAllRemitLocations().subscribe(res => {
        this.locations = res;
      })
  } 

  costSum() {
    this.totalInvoice = 0;
    this.lines.forEach(elem => {
      this.totalInvoice = this.add(elem.cost,this.totalInvoice) 

    })
  }
  add(y, z) {
    return  +y + +z;
  }
  removeLine(index) {
  
   this.lines.splice(index,1)
    this.costSum();
  }
  addLine() {
    this.lines.push({name: '', cost: 0})
  }

  createInvoice() {
    this.loading = true;
    let nameError = false;
    if(this.invoiceForm.valid) {
      this.lines.forEach((elem,i) => {
        if(elem.name == '') {
          nameError = true;
        }
       
      })
      if(nameError) {
        this.ledgerService.showError('Please fill all the names!')
        this.loading = false;

      } else {
        const data = {
          studyId: this.data?.study?.studySiteId ||this.data?.study ,
          req: {
              description: this.invoiceForm.controls.description.value,
              miscellaneousInvoiceDetails: this.lines
          }
        }
        this.ledgerService.createMiscInvoice(data).subscribe(res => {

          this.dialogRef.close(true);
        }, err => {
          this.loading = false;
        })
      }
    } else {
      this.loading = false;
      this.invoiceForm.markAllAsTouched();
    }
  }
}
