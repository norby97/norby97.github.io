import { Component, EventEmitter, OnInit, Output, ViewChild } from '@angular/core';

@Component({

    styleUrls: ['./dashboard.component.scss']
})
export class DashboardComponent implements OnInit {
    @Output() refreshStats = new EventEmitter();
    clicked = [].fill(false);
    initDashboardDto: InitDashboardDto;
    investigators = [];
    selectedSite = null;
    overviewDetails: OverViewDetailsModel;
    studyFilterForm: UntypedFormGroup;
    public UserRolesEnum = UserRoles;
    public loginUserIsCompanyAdmin: boolean;
    latestTasks = null;
    studies;
    loading = true;
    taskloading = true;
    sites;
    researchSites: any;
    companyId: any;
    sortedData;
    @ViewChild(MatSort) sort:MatSort;
    constructor(
        public dialog: MatDialog,
        private formBuilder: UntypedFormBuilder,
        public utilService:UtilService,
        public initAuditTableService: InitAuditTableService,
        private studyService: StudyService,
        private fileService: FileService,
        public authService: AuthenticationService,
        private taskService: TaskService,
        private dialogService: ErrorDialogService,
        private siteService: SiteService
    ) {
    }

    messageHour(createdOn) {
        const p = new Date(createdOn);
        const n = new Date();
        if (p.getDay() === n.getDay()) {
            return Math.floor(Math.abs(p.getTime() - n.getTime()) / 36e5) + ' hours ago';
        } else {
            return '';
        }

    }
    
    sortData(sort: Sort) {
        const data = this.studies;
        if (!sort.active || sort.direction === '') {
          this.sortedData = data;
          return;
        }
    
        this.sortedData = data.sort((a, b) => {
          const isAsc = sort.direction === 'asc';
          switch (sort.active) {
            case 'name':
              return this.compare(a.studyName?.toLowerCase(), b.studyName?.toLowerCase(), isAsc);
              case 'files':
                return this.compare(a.totalFiles, b.totalFiles, isAsc);
                 case 'modified':
              return this.compare(new Date(a.modified).getTime(), new Date(b.modified).getTime(), isAsc);
            default:
              return 0;
          }
        });
      }

     compare(a: number | string, b: number | string, isAsc: boolean) {
        return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
      }

    openStudyDialog(study: Study) {
        const dialogRef = this.dialog.open(StudyDialogComponent, {
            height: 'auto',
            maxHeight:'80vh',
            maxWidth:'80vw',
            width: 'auto',
            disableClose: true,
            panelClass:'custom-dialog-container',
            data: {
                study,
            }
        });

         this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
            this.researchSites = res;
            if (!this.researchSites.length) {
                this.dialogService.openDialog({
                    title: 'No Research Site created.',
                    message: 'No site has been created on your company, create a site to add a new study.',
                    confirmText: 'Ok, got it!'
                })
            }
        });

        dialogRef.componentInstance.studySuccessfullySaved.subscribe(() => {
            this.refreshStudies();
        });
    }

    refreshStudies() {
        this.loading = true;
        this.studyService.getStudies(null).subscribe(res => {
            this.studies = res;
            this.loading = false;
            this.sortData(this.sort)
        })
    }


    ngOnInit() {
            this.taskloading = true;
            this.loading = true;
        if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
            this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
          } else {
             this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
              this.sites = res;
            });
          }
        const tasks =this.taskService.retrieveTasks({ userID: this.authService.loginUserValue.id, fileID: '', type: ['ESIGN', 'INTERNAL'], status: ['NEW', 'OPEN'] })
             this.studyService.getStudies(null).subscribe(res => {
            this.studies = res;
            this.loading = false;

        }, err => {
            this.loading = false;
          });
            const overview =  this.taskService.getOverviewDetails();
        forkJoin([ tasks, overview]).subscribe(res => {
            this.latestTasks = res[0].tasks.reverse();
            this.overviewDetails = res[1];
            this.taskloading = false;
        }, err => {
            this.taskloading = false;
        });
        this.studyFilterForm = this.formBuilder.group({
            companyId: '',
            siteId: '',
            investigatorId: '',
            keywords: ''
        });
        this.loginUserIsCompanyAdmin = JSON.parse(localStorage.getItem('siteCentric.loginUser')).accountType === AccountType.COMPANY_ADMINISTRATOR;
    }

    showPopup(i) {
        this.clicked[i] = !this.clicked[i];
    }

    siteSelected(event: any) {
        this.selectedSite = event.value;
        this.studyService.getInvestigators(this.selectedSite).subscribe(res => {
            this.investigators = res;
        })
    }

    private showError(errMessage: string) {
        this.dialog.open(InfoPopupComponent, {
            data: errMessage
        });
    }

    filter(studyFilterForm: UntypedFormGroup) {
        this.loading = true;
        this.studyService.getStudies({
            siteId: studyFilterForm.value.siteId,
            investigatorId: studyFilterForm.value.investigatorId,
            keywords: studyFilterForm.value.keywords
        }).subscribe(res => {
            this.studies = res;
            this.sortData(this.sort)
            this.loading = false;
        }, err => {
            this.loading = false;
          })
    }

    closeStudy(study) {
        const dialogRef = this.dialog.open(FeedBackDialogComponent, {
            minWidth: 'auto',
            maxHeight:'80vh',
            maxWidth:'80vw',
            width: 'auto',
            height:'auto',
            disableClose: true,
            data: {
                message: `Are you sure you want to close ${study.studyName} ?  `,
            }
        });
        dialogRef.afterClosed().subscribe((res) => {
            if (res === true) {
                this.studyService.close(study.id).subscribe(res => {

                    this.refreshStudies();
                });
            } 
        });
    }

    downloadStudy(id, name) {
        this.fileService.downloadStudy(id).subscribe(res => {

        });
    }

    openTaskMessage(task) {
        const dialogRef = this.dialog.open(InternalDialogComponent, {
            height: 'auto',
            maxHeight:'80vh',
            maxWidth:'80vw',
            width: 'auto',
            disableClose: true,
            panelClass:'custom-dialog-container',
            data: {
                taskID: task.id,
                taskName: task.name,
                taskDescription: `${task.study.sponsor.name}  ${task.study.protocol}  ${task.study.nickname} '${task.study.nickname}'`
            }
        });
    }
}
