import { NgxMatDateFormats, NgxMatDateAdapter, NGX_MAT_DATE_FORMATS } from '@angular-material-components/datetime-picker';
import { formatDate } from '@angular/common';
import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MAT_MOMENT_DATE_ADAPTER_OPTIONS } from '@angular/material-moment-adapter';
import { MAT_DATE_LOCALE } from '@angular/material/core';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { forkJoin, Observable, of } from 'rxjs';

import { MY_FORMATS } from 'src/app/app.component';
import { CustomNgxDatetimeAdapter } from 'src/app/directives/ngx-mat-select/ngx-date-format';
import { SccipService } from 'src/app/services/crm_bd/sccip.service';
import { AppointmentService } from 'src/app/services/ctms/appointment.service';
import { MiscellaneousService } from 'src/app/services/ctms/miscellaneous.service';
import { PatientsService } from 'src/app/services/ctms/patients.service';
import { ScreenAppointmentService } from 'src/app/services/ctms/screen-appointment.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { OverlapDialogComponent } from '../overlap-dialog/overlap-dialog.component';
import { FeedBackDialogComponent } from '../../utils/feed-back-dialog/feed-back-dialog.component';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';
import moment from 'moment';
import { BusinessHoursHolidaysDialogComponent } from '../../utils/business-hours-holidays-dialog/business-hours-holidays-dialog.component';


@Component({
  selector: 'app-complete-screening-form-dialog',
  templateUrl: './complete-screening-form-dialog.component.html',
  styleUrls: ['./complete-screening-form-dialog.component.scss'],
  providers:[
    { provide: NGX_MAT_DATE_FORMATS , useValue: MY_FORMATS}  
  ]
})
export class CompleteScreeningFormDialogComponent implements OnInit {
  screenForm: UntypedFormGroup;
  studies;
  reasonCats;
  miscResources;
  loading = false;
  miscUsers;
  resChecks = [];
  userChecks = [];
  consentSigned = true;
  scheduleAnother = false;
  defaultTime = new Date().setMinutes(0) 
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<CompleteScreeningFormDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any, 
    public utilService: UtilService,
    public dialog: MatDialog,
    private siteService:SiteService,
    private miscService: MiscellaneousService,
    private screenService: ScreenAppointmentService,
    private appointmentService: AppointmentService,
    private patientService: PatientsService) { 
      this.dpheader = DatepickerHeaderComponent;
    }

  ngOnInit(): void {
    const data = {
      search:'',
      pageNumber: 0,
     pageSize: 30,
      siteId:   this.data?.site,
      statuses: ['MOVING_FORWARD','ADD_ON','ACTIVE']
    }
    const r1 = this.siteService.getStudies(data)
    const r2 = this.patientService.getActiveReasonCategoriesByApptType('SCREEN')
    const r3 = this.miscService.getMiscResources(this.data?.site)
    const r4 = this.miscService.getMiscUsers(this.data?.site)
    forkJoin([r1,r2,r3,r4]).subscribe(res => {
      this.studies = res[0].content;
      this.reasonCats = res[1].sort((a,b)=> this.compare(a?.name?.toLowerCase() , b?.name?.toLowerCase(),true));
      this.miscResources = res[2];
      this.miscUsers = res[3];
    });
    this.screenForm = new UntypedFormGroup({
      screenDate :new UntypedFormControl('',),
      study :new UntypedFormControl('',),
      duration :new UntypedFormControl(''),
      notes: new UntypedFormControl(this.data?.patientDetails?.internalNotes),
      refnotes: new UntypedFormControl(),
      reasonCat: new UntypedFormControl(this.data?.patientDetails?.reasonId,),
      followUpDate: new UntypedFormControl(''),
      alias: new UntypedFormControl(   this.data?.patientDetails?.firstName[0] +( this.data?.patientDetails?.middleName == null || this.data?.patientDetails?.middleName == '' ? '-' : this.data?.patientDetails?.middleName ) + this.data?.patientDetails?.lastName[0]
      , Validators.required),
      subjectnr: new UntypedFormControl(this.data?.patientDetails?.subjectNumber),
      address1:new UntypedFormControl(this.data?.patientDetails?.addressOne, Validators.required),
      address2:new UntypedFormControl(this.data?.patientDetails?.addressTwo),  
      city:new UntypedFormControl(this.data?.patientDetails?.city, Validators.required),
      state:new UntypedFormControl(this.data?.patientDetails?.state, Validators.required),
      zip:new UntypedFormControl(this.data?.patientDetails?.zip, Validators.required),
    });
    this.toggleStatus(!this.data?.patientDetails?.doNotConsent)
  }

  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
}

  undoConsent(data) {
    this.loading = true;
    const dialogRef = this.dialog.open(FeedBackDialogComponent, {
      minWidth: 'auto',
      maxHeight:'80vh',
      maxWidth:'80vw',
      width: 'auto',
      height:'auto',
      disableClose: true,
      data: {
          message: 'Are you sure you want to undo the consent form outcome for ' +this.data?.patientDetails?.lastName +', '+ this.data?.patientDetails?.firstName +'?',
      }
  });
  dialogRef.afterClosed().subscribe((res) => {
      if (res === true) {
        this.screenService.markUndoCompleted(this.data.arg.event.extendedProps.appointmentId).subscribe(res => {
          this.loading = false;

          this.dialogRef.close(true);

        }, err => {
          this.loading = false;
        });  
      }  else {
        this.loading = false;

      }
  });
     
  }

  screen(form) {
    this.loading = true;

    if(this.screenForm.valid) {
        if(this.consentSigned) {
          const data = {
            screenAppointmentId: this.data.arg.event.extendedProps.appointmentId, 
            req : {
              alias: this.screenForm.controls.alias.value,
              notes: this.screenForm.controls.notes.value,
              subjectNumber: this.screenForm.controls.subjectnr.value,
              addressOne: this.screenForm.controls.address1.value,
              addressTwo: this.screenForm.controls.address2.value,
              city: this.screenForm.controls.city.value,
              state: this.screenForm.controls.state.value,
              zip: this.screenForm.controls.zip.value,
              internalNotes: this.screenForm.controls.notes.value,
            }
              
          }
          this.screenService.consentSigned(data).subscribe(res => {
            this.loading = false;

            this.dialogRef.close(true);

          }, err => {
            this.loading = false;
          });

        } else {
          const data = {
            screenAppointmentId: this.data.arg.event.extendedProps.appointmentId, 
            req : {
              anotherScreen: {
                duration: this.screenForm.controls.duration.value != '' ? this.screenForm.controls.duration.value  : 180,
                reasonId: this.screenForm.controls.reasonCat.value,
                resourcesIds: [],
                screenDateTime: null,
                studyId: this.screenForm.controls.study.value,
                userIds: []
              },
              followUpOn: this.screenForm.controls.followUpDate.value != '' ?  formatDate(this.screenForm.controls.followUpDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us') : null,
              notes: this.screenForm.controls.notes.value,
              reasonId: this.screenForm.controls.reasonCat.value,
              scheduleAnotherScreen: this.scheduleAnother
            }
              
          }
          if(this.scheduleAnother) {
            data.req.anotherScreen.screenDateTime = this.screenForm.controls.screenDate.value ? formatDate(this.screenForm.controls.screenDate.value, 'yyyy-MM-ddTHH:mm', 'en-Us'): ''
            this.miscResources.forEach((elem,i) => {
              if(this.resChecks[i]) {
                data.req.anotherScreen.resourcesIds.push(elem.id);
              }
            });
            this.miscUsers.forEach((elem,i) => {
              if(this.userChecks[i]) {
                data.req.anotherScreen.userIds.push(elem.id);
              }
            });
            const data2 = {
              userIds: data?.req?.anotherScreen?.userIds,
              resourceIds: data?.req?.anotherScreen?.resourcesIds,
              appointmentDate:data?.req?.anotherScreen?.screenDateTime,
              duration:data?.req?.anotherScreen?.duration,
              removeAppointmentId: this.data.arg.event.extendedProps.appointmentId,

            }


                this.utilService.isWithinBusinessHours(this.data?.site, new Date(data?.req?.anotherScreen?.screenDateTime),formatDate(data?.req?.anotherScreen?.screenDateTime,'hh:mm a','en-US'),data?.req?.anotherScreen?.duration).subscribe(result => {
                  if(result) {
                    this.appointmentService.checkOverlap(data2).subscribe(res => {
                      if(res.length == 0) {
                        this.screenService.consentNotSigned(data).subscribe(res => {
                          this.loading = false;
                          this.dialogRef.close(true);

                        }, err => {
                          this.loading = false;
                        });
                      } else {
                        const ref = this.dialog.open(OverlapDialogComponent,{
               
                          data: {
                            siteId:this.data?.site,
                            misc: false,              
                            date:data?.req?.anotherScreen?.screenDateTime,
                            type:'SCREEN',
                            res,
                          },
                          width: 'auto',
                          height: 'auto',
                          minWidth:'40vw',
                          panelClass:'custom-dialog-container',
                          disableClose: true,
                          maxHeight:'80vh'
                        })
                        ref.afterClosed().subscribe(res => {
                          if(res) {
                            this.screenService.consentNotSigned(data).subscribe(res => {
                              this.loading = false;

                              this.dialogRef.close(true);

                            }, err => {
                              this.loading = false;
                            });
                          }else {
                            this.loading = false;
                          }
                        })
                      }
                    }, err => {
                      this.loading = false;
                    })
                  } else {
                    const ref12 = this.dialog.open(BusinessHoursHolidaysDialogComponent,{
                      data: {
                        siteId: this.data?.site,
                        isholiday:false,
                        appointmentDate:data?.req?.anotherScreen?.screenDateTime,
                      },
                      width: 'auto',
                      height: 'auto',
                      panelClass:'custom-dialog-container',
                      disableClose: true,
                      maxHeight:'80vh',
                      minWidth:'40vw'
                    })

                    ref12.afterClosed().subscribe(result2 => {
                      if(result2) {
                        this.appointmentService.checkOverlap(data2).subscribe(res => {
                          if(res.length == 0) {
                            this.screenService.consentNotSigned(data).subscribe(res => {
                              this.loading = false;
                              this.dialogRef.close(true);

                            }, err => {
                              this.loading = false;
                            });
                          } else {
                            const ref = this.dialog.open(OverlapDialogComponent,{
               
                              data: {
                                siteId:this.data?.site,
                                misc: false,              
                                date:data?.req?.anotherScreen?.screenDateTime,
                                type:'SCREEN',
                                res,
                              },
                              width: 'auto',
                              height: 'auto',
                              minWidth:'40vw',
                              panelClass:'custom-dialog-container',
                              disableClose: true,
                              maxHeight:'80vh'
                            })
                            ref.afterClosed().subscribe(res => {
                              if(res) {
                                this.screenService.consentNotSigned(data).subscribe(res => {
                                  this.loading = false;

                                  this.dialogRef.close(true);

                                }, err => {
                                  this.loading = false;
                                });
                              }else {
                                this.loading = false;
                              }
                            })
                          }
                        }, err => {
                          this.loading = false;
                        })
                      } else {
                        this.loading = false;
                      }
                    })
                  }
                })
                 
             
          } else {
            data.req.anotherScreen.screenDateTime = null
            this.screenService.consentNotSigned(data).subscribe(res => {
              this.loading = false;
              this.dialogRef.close(true);

            }, err => {
              this.loading = false;
            });
            
          }
          
         
        }
    } else {
      this.screenForm.markAllAsTouched();
      this.loading = false;

    }
  }
  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
      siteId:  this.data?.site,
      statuses: ['MOVING_FORWARD','ADD_ON','ACTIVE']
    }
    if(pageNumber - 1 != 0) {
      this.siteService.getStudies(data).subscribe(res => {
        res.content.forEach(element => {
          this.studies.push(element);
      });
      })
    } else {
      this.siteService.getStudies(data).subscribe(res => {
        this.studies =[...this.studies,...res.content]
      })
    }
    return of(this.studies)
  }



  toggleAnotherStatus(status) {
    this.scheduleAnother = status;
    if(this.scheduleAnother) {
      this.screenForm.controls.alias.setValidators(null);
      this.screenForm.controls.alias.updateValueAndValidity();
      this.screenForm.controls.study.setValidators(Validators.required);
      this.screenForm.controls.study.updateValueAndValidity();
      this.screenForm.controls.screenDate.setValidators(Validators.required);
      this.screenForm.controls.screenDate.updateValueAndValidity();
      this.screenForm.controls.duration.setValidators(Validators.required);
      this.screenForm.controls.duration.updateValueAndValidity();
    }
  }


  toggleStatus(status) {
    this.consentSigned = status;
    if(this.consentSigned) {
      this.screenForm.controls.address1.setValidators(Validators.required);
      this.screenForm.controls.address1.updateValueAndValidity();
      this.screenForm.controls.address2.setValidators(null);
      this.screenForm.controls.address2.updateValueAndValidity();
      this.screenForm.controls.city.setValidators(Validators.required);
      this.screenForm.controls.city.updateValueAndValidity();
      this.screenForm.controls.state.setValidators(Validators.required);
      this.screenForm.controls.state.updateValueAndValidity();
      this.screenForm.controls.zip.setValidators(Validators.required);
      this.screenForm.controls.zip.updateValueAndValidity();
      this.screenForm.controls.alias.setValidators(Validators.required);
      this.screenForm.controls.alias.updateValueAndValidity();
      this.screenForm.controls.reasonCat.setValidators(null);
      this.screenForm.controls.reasonCat.updateValueAndValidity();
      this.screenForm.controls.followUpDate.setValidators(null);
      this.screenForm.controls.followUpDate.updateValueAndValidity();
      this.screenForm.controls.study.setValidators(null);
      this.screenForm.controls.study.updateValueAndValidity();

      this.screenForm.controls.duration.updateValueAndValidity();

      this.screenForm.controls.reasonCat.setValidators(Validators.required);
      this.screenForm.controls.reasonCat.updateValueAndValidity();
      this.screenForm.controls.address1.setValidators(null);
      this.screenForm.controls.address1.updateValueAndValidity();
      this.screenForm.controls.address2.setValidators(null);
      this.screenForm.controls.address2.updateValueAndValidity();
      this.screenForm.controls.alias.setValidators(null);    
      this.screenForm.controls.followUpDate.setValidators(null);
      this.screenForm.controls.followUpDate.updateValueAndValidity();
      this.screenForm.controls.alias.updateValueAndValidity();
      this.screenForm.controls.city.setValidators(null);
      this.screenForm.controls.city.updateValueAndValidity();

    }
  }
}
