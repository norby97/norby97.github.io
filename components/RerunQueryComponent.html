import { trigger, state, style, transition, animate } from '@angular/animations';
import { Component, OnInit, ViewChild } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';
import { MatTableDataSource } from '@angular/material/table';
import { ActivatedRoute, Router } from '@angular/router';
import { WorkingListService } from 'src/app/services/ctms/working-list.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { CreateWorkingListDialogComponent } from '../../common/dialogs/ctms/create-working-list-dialog/create-working-list-dialog.component';
import { PatientMiniViewDialogComponent } from '../../common/dialogs/ctms/patient-mini-view-dialog/patient-mini-view-dialog.component';

@Component({
  selector: 'app-rerun-query',
  animations: [trigger('detailExpand', [
    state('collapsed', style({ height: '0px', minHeight: '0' })),
    state('expanded', style({ height: '*' })),
    transition('expanded <=> collapsed', animate('225ms cubic-bezier(0.4, 0.0, 0.2, 1)')),
])],
  templateUrl: './rerun-query.component.html',
  styleUrls: ['./rerun-query.component.scss']
})
export class RerunQueryComponent implements OnInit {
  listId;
  groupValue = 1;
  allCheck = false;
  queryResults =[];
  listDetails; 
  loading = false;
  noMore = false;
  skip =0;
  limit = 30;
  patientChecks = [];
  columnsToDisplay: string[] = ['Name (Gender/Age)', 'Email','Phone Number One/Phone Number Two','Last Activity', 'Interest'];
  dataSource: MatTableDataSource<any[]>;
  expandedElement: any | null;
  @ViewChild(MatPaginator, { static: true }) paginator: MatPaginator;
  @ViewChild(MatSort, { static: true }) sort: MatSort;
  constructor(private route: ActivatedRoute,
              public utilService: UtilService,
              private router: Router,
              private dialog: MatDialog,
              private workingListService: WorkingListService) { }

  ngOnInit(): void {
    this.loading = true;
    this.route.params.subscribe(params => { this.listId = params.id; });
    this.workingListService.getWorkingListDetails({workingListId: this.listId}).subscribe(res =>{
      this.listDetails = res;
    })
    this.workingListService.rerunQuery({skip:this.skip, limit:this.limit,workingListId: this.listId}).subscribe(res => {

        this.queryResults = res;
        this.queryResults.map(elem => {this.patientChecks.push(false);})
        this.dataSource = new MatTableDataSource(res);
        this.loading =false;
    })
  }

  toogleAll(status) {
    this.allCheck = status;
    for(let i =0; i< this.patientChecks.length; i++) {
      this.patientChecks[i] =status;
    }
  }


  
  toggleStatus(status, index) {               
    this.patientChecks[index] = status;
  }

  openNewWorkingListDialog() {
    const selected = []
    this.patientChecks.map((elem,i) => {
      if(elem) {
          selected.push(this.queryResults[i].id);
      }
    });
    if(selected.length) {
      const ref = this.dialog.open(CreateWorkingListDialogComponent, {
        width: 'auto',
        height: 'auto',
        disableClose: true,
    panelClass:'custom-dialog-container',
    maxHeight:'80vh',
    maxWidth:'80vw',
        data: {
          addPatient: true,
          selectedPatients: selected,
          workinglist: this.listDetails,
          queryGroups:this.listDetails?.conditionRequest
        }
  
      });
      ref.afterClosed().subscribe(res => {
        if(res) {
          this.router.navigate(['/ctms/working-list'])
        }
      })
    } else {
      this.workingListService.showError('Please select patient(s) first.')
    }
   
  }
  onScroll() {
    this.loading= true;
    this.skip = this.skip + this.limit
    this.workingListService.rerunQuery({skip:this.skip, limit:this.limit,workingListId: this.listId}).subscribe(res => {
      if(res?.length == 0) {
        this.noMore = true; 
        this.loading = false;
      } else {
        this.queryResults = [...this.queryResults,...res];
        res.map(elem => {this.patientChecks.push(false);})
        this.dataSource = new MatTableDataSource(this.queryResults);
        this.loading = false;
      }
    },err=> {
      this.loading = false;
    })
    
    
  }
  openPatientMiniViewDialog(patient) {
    patient.patientId = patient.id
    this.dialog.open(PatientMiniViewDialogComponent, {
       width: 'auto',
       maxHeight:'80vh',
    maxWidth:'80vw',
    panelClass:'custom-dialog-container',
    height:'auto',
       disableClose: true,
       data: {
        fromRecruiting:true,
         fromTracking:false,
         allSelected: false,
        excludedPatients:[],
        site: {id:patient?.siteId},
        patient
       }
     });
   }
}
