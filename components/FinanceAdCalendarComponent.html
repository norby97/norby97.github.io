  buildMonths(data?: any) {
    if(data) {
      const year = parseInt(data.yr);
      let months =[]
      for (let i = parseInt(data.mth) - 2; i <= parseInt(data.mth); ++i) { 
        if (i <= 0) {
          const temp = 12+i;
           months.push({mth: temp, yr: year - 1});
        } else {
          months.push({mth: i, yr: year});
        }
    }
      
    for (let i = parseInt(data.mth) + 1; i <= parseInt(data.mth) + 4; ++i) {
        if (i > 12) {
          const temp = Math.abs(12 - i) 
          months.push({mth: temp, yr: year + 1});
        } else {
          months.push({mth: i, yr: year});
        }
    }
    this.buildedMonths = months;
    } else {
      const year = parseInt(this.current.yr);
      let months =[]
      for (let i = parseInt(this.current.mth) - 2; i <= parseInt(this.current.mth); ++i) { 
        if (i <= 0) {
          const temp = 12+i;
           months.push({mth: temp, yr: year - 1});
        } else {
          months.push({mth: i, yr: year});
        }
    }
      
    for (let i = parseInt(this.current.mth) + 1; i <= parseInt(this.current.mth) + 4; ++i) {
        if (i > 12) {
          const temp = Math.abs(12 - i) 
          months.push({mth: temp, yr: year + 1});
        } else {
          months.push({mth: i, yr: year});
        }
    }
    this.buildedMonths = months;
    }

  }
