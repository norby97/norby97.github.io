import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormBuilder, UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog, MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';
import { Observable, of } from 'rxjs';
import { startWith, map } from 'rxjs/operators';

import { Cro } from 'src/app/models/common/Cro';
import { Indication } from 'src/app/models/common/indication.model';

import { SccipService } from 'src/app/services/crm_bd/sccip.service';
import { SimpleNameImputDialogComponent } from '../../utils/simple-name-imput-dialog/simple-name-imput-dialog.component';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';
import { AddIndicationDialogComponent } from '../../utils/add-indication-dialog/add-indication-dialog.component';

export class AddNewLeadDialogComponent implements OnInit {
  addLeadForm: UntypedFormGroup;
  filteredSponsors: Observable<Sponsor[]>;
  filteredCros: Observable<Cro[]>;
  filteredCompounds: Observable<Compound[]>;
  filteredIndications: Observable<Indication[]>;
  filteredPhases: Observable<Phase[]>;
  sites = [];
  loading = false;
  sponsors=[];
  compound=[];
  users
  cro =[]
  diagnosis=[] 
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogref.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(private leadService: LeadsService,
    public dialog: MatDialog,
    private authService: AuthenticationService,
    private siteService: SiteService,
    private userService: UserService,
    private sccipService: SccipService,
    public dialogref: MatDialogRef<AddNewLeadDialogComponent>,
    private formBuilder: UntypedFormBuilder,
    @Inject(MAT_DIALOG_DATA) public data: any) { 
      this.dpheader = DatepickerHeaderComponent;
    }

  ngOnInit(): void {
    this.userService.getAll2().subscribe(res => {
      this.users = res;
    })

    this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
     this.sites = res;
    });
    this.addLeadForm =new UntypedFormGroup({
      sponsor: new UntypedFormControl(this.data?.sponsor ? this.data?.sponsorName : '', Validators.required),
      compound: new UntypedFormControl(this.data?.compoundName ? this.data?.compoundName : '', Validators.required),
      cro: new UntypedFormControl(),
      ntcID: new UntypedFormControl(''),
      phase: new UntypedFormControl(''),
      indication:new UntypedFormControl('',Validators.required),
      owner: new UntypedFormControl('', Validators.required),
      status: new UntypedFormControl('', Validators.required),
      sites: new UntypedFormControl('',  Validators.required),
      followUp: new UntypedFormControl(''),
    });

  }

  private _filter(value: string, items: Array<any>) {
    const filterValue = value?.toLowerCase();
    return items?.filter(item => {
        return item.name?.toLowerCase().indexOf(filterValue) === 0;
    });
  }


  add(form) {
    this.loading = true;
    if(this.addLeadForm.valid ) {
     
      const data = {
        compoundId: this.addLeadForm.get('compound').value,
        croId: this.addLeadForm.get('cro').value,
        followUpNeededDate: this.addLeadForm.controls.followUp.value,
        indicationId: this.addLeadForm.get('indication').value,
        ntcId: this.addLeadForm.controls.ntcID.value,
        ownerId: this.addLeadForm.controls.owner.value,
        phase: this.addLeadForm.get('phase').value,
        sites: this.addLeadForm.controls.sites.value,
        sponsorId: this.addLeadForm.get('sponsor').value,
        status: this.addLeadForm.controls.status.value
      }
     
      this.leadService.addNewLead(data).subscribe(res => {

        this.dialogref.close(true)
      }, err => {
        this.loading = false;
      })
    } else {
      this.loading = false;
      this.addLeadForm.markAllAsTouched();
    }
  
  }
    
  searcher2 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    const data = {
      search,
      active:true,
      limit:pageSize,
      skip: 0 + ((pageNumber-1)*pageSize),
    }
    if(pageNumber-1 != 0) {
      this.sccipService.getSponsorsPaginated(data).subscribe(res => {
        this.sponsors = [...this.sponsors, ...res];

      })
    } else {
     this.sccipService.getSponsorsPaginated(data).subscribe(res => { 
      this.sponsors = [...res];

      
      })
    }
    return of(this.sponsors)
  }

  searcher6 = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => { 
    const data = {
      search,
      active:true,
      pageSize,
      pageNumber: pageNumber-1,
    }
    if(pageNumber-1 != 0) {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.diagnosis = [...this.diagnosis, ...res.content];

      })
    } else {
     this.sccipService.getindIcationsPaginated(data).subscribe(res => { 
      this.diagnosis = [ ...res.content];
      
      })
    }
    return of(this.diagnosis)
  }
  
  openAddDialog(type) {
    if(type =='indication') {
      const dialogref = this.dialog.open(AddIndicationDialogComponent, {
        disableClose: true,
        minWidth: "30rem",width: 'auto',
        height: 'auto',
        maxHeight:'80vh',
        maxWidth:'80vw',    
        panelClass:'custom-dialog-container',
        data:{indicationObject: null, returndata:true}
      });

      dialogref.afterClosed().subscribe(res => {
        if(res) {
          this.sccipService.addIndication(res).subscribe((result:any) => {
            this.diagnosis.push({...result,ngxSelected:1,active:true})
            this.addLeadForm.controls.indication.setValue(result?.id)
            this.addLeadForm.controls.indication.updateValueAndValidity(result?.id)
      })
    }
      })
    } else {

    const ref = this.dialog.open(SimpleNameImputDialogComponent, {
      data: {
        type,
      },
      width: 'auto',
      minWidth:'30vw',
      height: 'auto',
      panelClass:'custom-dialog-container',
      disableClose: true,
      maxHeight:'80vh',
    maxWidth:'80vw',
    });      
    ref.afterClosed().subscribe(res => {
      if(res) {
        if(type =='sponsor') {
        this.sccipService.addSponsor(res).subscribe((result:any) => {
        this.sponsors=  [{...result,ngxSelected:1,active:true,ngxHide:false}, ...this.sponsors]
        this.addLeadForm.controls.sponsor.setValue(result?.id)
            this.addLeadForm.controls.sponsor.updateValueAndValidity(result?.id)
        })

        }
        if(type =='cro') {
          this.sccipService.addCro(res).subscribe((result:any) => {
      this.cro.push({...result,ngxSelected:1,active:true})
            this.addLeadForm.controls.cro.setValue(result?.id)
            this.addLeadForm.controls.cro.updateValueAndValidity(result?.id)
          })
        }
     
     
        if(type =='compound') {
          this.sccipService.addCompunds(res).subscribe((result:any) => {
           this.compound.push({...result,ngxSelected:1,active:true})
           this.addLeadForm.controls.compound.setValue(result?.id)
           this.addLeadForm.controls.compound.updateValueAndValidity(result?.id)
          })
        }

      }
    }) 
   }    }

}
