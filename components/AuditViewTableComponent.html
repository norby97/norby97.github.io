        'GMT+5', 'GMT+6', 'GMT+7', 'GMT+8', 'GMT+9', 'GMT+10', 'GMT+11', 'GMT+12','GMT+13','GMT+14'];
    dataSource: MatTableDataSource<AuditTrail>;
    expandedElement: AuditTrail | null;
    filterForm: UntypedFormGroup;
    auditTrails;
    initAuditFilterDto: InitAuditFilterDto;
    @ViewChild(MatSort, { static: true }) sort: MatSort;
 
    dpheader;
    constructor(
        private auditService: AuditService,
        private formBuilder: UntypedFormBuilder,
        public authService: AuthenticationService,
        private initAuditTableService: InitAuditTableService
    ) {
        this.dpheader = DatepickerHeaderComponent;

    }

    ngOnInit(): void {

        this.initFormFields();
        if (this.showAllAuditFilters) {
            this.filter(this.filterForm);
        } else {
            this.initAuditObservable();
        }

    }

    private initAuditObservable() {
        this.initAuditTableService.observable.subscribe(auditFilterData => {
            this.filterForm.value.siteId = auditFilterData.siteId;
            this.filterForm.value.studyId = auditFilterData.studyId;
            this.filterForm.value.filePathId = auditFilterData.filePathId;
            this.filterForm.value.userId =  this.studyaudit ? null :auditFilterData.userId;
            this.filterForm.value.companyId = auditFilterData.companyId;
            this.actualFilePathId = auditFilterData.filePathId;
            this.filter(this.filterForm);
        });
    }

    toggleTableRows() {
        this.isTableExpanded = !this.isTableExpanded;
    
        this.dataSource.data.forEach((row: any) => {
          row.isExpanded = this.isTableExpanded;
        })
      }

    private initFormFields() {
        this.filterForm = this.formBuilder.group({
            companyId: this.authService.loginUserValue.company.id,
            siteId: '',
            studyId: '',
            userId: this.showAllAuditFilters ? '' :this.authService.loginUserValue.id,
            filePathId: '',
            ip: '',
            actions: [],
            status: '',
            from: new Date(new Date().getTime() - 1 * 24 * 60 * 60 * 1000),
            to: new Date(),
            timezone: new Date().toString().match(/([A-Z]+[\+-][0-9]+)/)[1].replace(/[0]/g, ""),
        });
        if(this.contactAudits) {
            this.auditService.initFilterContacts(this.contactID).subscribe(
                res => {
                    if (res.responseStatus === 'FAILURE') {
                        this.auditService.showError(res.responseMessage);
                    } else {
                        this.initAuditFilterDto = res;
                    }
                }
            );
        } else {
            this.auditService.initFilter().subscribe(
                res => {
                    if (res.responseStatus === 'FAILURE') {
                        this.auditService.showError(res.responseMessage);
                    } else {
                        this.initAuditFilterDto = res;
                    }
                }
            );
        }
        
    }







    export() {
        var newData = [];
        var finalData = []
        let tempData = this.dataSource

        tempData.data.forEach((t: any, index) => {
            newData.push(t);
            let data = {
               company : newData[index]!== null ?  newData[index].company?.name : '',
               study : newData[index].study !== null ?  newData[index].study?.sponsor?.name + ' ' +  newData[index].study?.protocol : '',
                site : newData[index] !== null ?  newData[index].site?.name : '',
                userName :  newData[index] !== null ?  newData[index].user?.firstName + '' + newData[index].user?.lastName  : '',
                ip :  newData[index] !== null ?  newData[index].ip  : '',
                action : newData[index]!== null ?  newData[index].action : '',
                status :  newData[index] !== null ?  newData[index].status : '',
                message :  newData[index] !== null ?  newData[index].message: '',
                createdOn :  newData[index] !== null ?  newData[index].createdOn : '',
              createdOnGMT: newData[index] !== null ? new Date( newData[index].createdOn ).toString().match(/([A-Z]+[\+-][0-9]+)/).input : new Date( newData[index].createdOn ).toString().match(/([A-Z]+[\+-][0-9]+)/).input
            }
            finalData.push(data);
            
        });
        const workSheet = XLSX.utils.json_to_sheet(finalData);
        const workBook: XLSX.WorkBook = XLSX.utils.book_new();
        XLSX.utils.book_append_sheet(workBook, workSheet, 'SheetName');
        XLSX.writeFile(workBook, 'Audit.xlsx');
    }

    filter(filterForm: UntypedFormGroup) {
        this.loading = true;
        if(this.contactAudits) {
            this.auditService.getAllForContacts(filterForm.value.companyId,
                filterForm.value.siteId,
                filterForm.value.studyId,
                this.studyaudit ? null : filterForm.value.userId,
                filterForm.value.filePathId,
                filterForm.value.ip,
                filterForm.value.actions,
                formatDate(filterForm.value.from, 'yyyy-MM-dd', 'en-US'),
                formatDate(filterForm.value.to, 'yyyy-MM-dd', 'en-US'),
                filterForm.value.status,
                this.contactID)
                .subscribe(res => {
                    this.auditTrails = res;
                    this.dataSource = new MatTableDataSource(res.auditTrails);
                    this.dataSource.sort = this.sort;                    
                    this.loading = false;
                    this.dataSource.sortingDataAccessor = (item, property) => {
                        switch (property) {
                            case 'performed':
                                return new Date(item.createdOn);
                            default:
                                return item[property];
                        }
                    };
                }, err => {
                    this.loading = false;
                  });
        } else {
            this.auditService.getAll(filterForm.value.companyId,
                filterForm.value.siteId,
                filterForm.value.studyId,
                this.studyaudit ? null : filterForm.value.userId,
                this.actualFilePathId,
                filterForm.value.ip,
                filterForm.value.actions,
                formatDate(filterForm.value.from, 'yyyy-MM-dd', 'en-US'),
                formatDate(filterForm.value.to, 'yyyy-MM-dd', 'en-US'),
                filterForm.value.status)
                .subscribe(res => {
                    this.auditTrails = res;
                    this.dataSource = new MatTableDataSource(res.auditTrails);
                    this.dataSource.sort = this.sort;
                    this.loading = false;
                    this.dataSource.sortingDataAccessor = (item, property) => {
                        switch (property) {
                            case 'performed':
                                return new Date(item.createdOn);
                            default:
                                return item[property];
                        }
                    };
                }, err => {
                    this.loading = false;
                  });
        }
        
    }

}
