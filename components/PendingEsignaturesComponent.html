import { Component, OnInit } from '@angular/core';
import { MatDialog } from '@angular/material/dialog';
import { ActivatedRoute } from '@angular/router';
import { saveAs } from 'file-saver';
import * as  printJS from 'print-js';
import { forkJoin } from 'rxjs';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { FileService } from 'src/app/services/regulatory/file.service';
import { StudyService } from 'src/app/services/regulatory/study.service';
import { TaskService } from 'src/app/services/regulatory/task.service';
import { InitAuditTableService } from '../../../administration/audit/audit-view-table.component/init-audit-table-service';
import { EsignatureDialogComponent } from 'src/app/components/common/dialogs/regulatory/esignature-dialog/esignature-dialog.component';
import { RejectDocumentDialogComponent } from 'src/app/components/common/dialogs/regulatory/reject-document-dialog/reject-document-dialog.component';

@Component({
    selector: 'app-pending-esignatures',
    templateUrl: './pending-esignatures.component.html',
    styleUrls: ['./pending-esignatures.component.scss']
})
export class PendingEsignaturesComponent implements OnInit {
    url = null;
    pdfDoc = null;
    pageNum = 1;
    pageRendering = false;
    pageNumPending = null;
    totalPageNumber = null;
    scale = 1;
    studyName = '';
    zoomRange = 0.25;
    canvas = null;
    ctx = null;
    actualFile = null;
    actualFileName = null;
    actualFileDetails = null;
    actualFileID = null;
    studyID;
    taskId;
    esigns = null;
    overdueEsigns = null;

    actualTask = null;
    pageCounter = 1;

    constructor(private route: ActivatedRoute,
        private fileService: FileService,
        private authService: AuthenticationService,
        public dialog: MatDialog,
        private taskService: TaskService,
        private studyService: StudyService,
        public initAuditTableService: InitAuditTableService) {

         }

    ngOnInit(): void {
        this.taskService.getFiles({ esign: true, overdue: true }).subscribe(res => {
            this.overdueEsigns = res;
        });
        this.taskService.getFiles({ esign: true, overdue: false }).subscribe(res => {
            this.esigns = res;
        });
    }


    retrieveFile(id, studyID, task, withrender = true) {
        this.pageNum = 1;
        this.actualFileID = id;
        this.actualTask = task;
        this.studyID = studyID;
        this.taskId = task.id;
        const data = {
            fileID: id,
            studyID,
        };
        this.initAuditTableService.next({companyId: this.authService.loginUserValue.company.id, studyId: studyID, siteId: '', filePathId: id, userId: '' });
        const file = this.fileService.retrieveFile(data);
        const fileDetails = this.fileService.retrieveFileDetails(data);
        forkJoin([file, fileDetails]).subscribe(res => {
            this.url = res[0];
            this.actualFile = res[0];
            this.actualFileDetails = res[1];
        });
    }

    printFile() {
        if (this.url === null) {
            this.studyService.showError('Please select a PDF first.');
        } else {
            const base64 = btoa(new Uint8Array(this.url).reduce((data, byte) => data + String.fromCharCode(byte), ''));
            printJS({ printable: base64, type: 'pdf', showModal: false, base64: true });
        }
    }

    downloadFile() {
        if (this.actualFile === null) {
            this.studyService.showError('Please select a PDF first.');
        } else {
            const blob = new Blob([this.actualFile], { type: 'application/pdf' });
            saveAs(blob, this.actualFileName);
        }
    }

    openEsignatureDialog() {
        const dialogRef = this.dialog.open(EsignatureDialogComponent, {
            height: 'auto',
            width: 'auto',
            disableClose: true,
            autoFocus: false,
            data: {
                fileID: this.actualFileID,
                studyId: this.studyID,
                taskId: this.taskId
            }
        });
        dialogRef.afterClosed().subscribe(res => {
            if (res === true) {
                this.taskService.getFiles({ esign: true, overdue: false }).subscribe(result => {
                    this.esigns = result;
                });
                this.actualFileName = null;
                if (this.ctx) {
                    this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
                    this.ctx.beginPath();
                }
            }
        });
    }

    openRejectDocumentDialog() {
        const dialogRef = this.dialog.open(RejectDocumentDialogComponent, {
            height: 'auto',
            width: 'auto',
            disableClose: true,
            autoFocus: false,
        });
    }

    checkIfUserIsAssigned(task) {
        return task.assignedUsers.filter(e => e.id === this.authService.loginUserValue.id).length > 0 ||
            task.finalSigner?.id === this.authService.loginUserValue.id || task.usersWhoSigned.filter(e => e.id === this.authService.loginUserValue.id).length === 0 || task.creator?.id === this.authService.loginUserValue.id;
    }

    checkIfUserIsAssignedArray(task) {
        if (task) {
            task.forEach(element => {
                if (element.task.assignedUsers.filter(e => e.id === this.authService.loginUserValue.id).length > 0 ||
                    element.task.finalSigner?.id === this.authService.loginUserValue.id || element.task.usersWhoSigned.filter(e => e.id === this.authService.loginUserValue.id).length === 0) {
                    return false;
                }
            });
        }
        return true;
    }
}
