import { Component, HostListener, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { ActivatedRoute, Router } from '@angular/router';
import { forkJoin } from 'rxjs';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { ScoreCardService } from 'src/app/services/ctms/score-card.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { ContactDetailsDialogComponent } from '../../common/dialogs/crmbd/contact-details-dialog/contact-details-dialog.component';
import { ContactSearchResultsForStudiesComponent } from '../../common/dialogs/crmbd/contact-search-results-for-studies/contact-search-results-for-studies.component';

import { DatePipe } from '@angular/common';

@Component({
  selector: 'app-pm-score-card',
  templateUrl: './pm-score-card.component.html',
  styleUrls: ['./pm-score-card.component.scss']
})
export class PmScoreCardComponent implements OnInit {
  scoreID;
  hoverTimeout
  noteForm: UntypedFormGroup;
  projectMNote = false;
  includepm = true;
  mouseX;
  mouseY
  showNoteForm= false;
  notes = [];
  initialMousePosition
  scoreCommitment;
  scoreContacts = []
  scoreDetails;
  screenFails;

  contactremoveloading = false;
  imports
  actualStatus;
  sites;
  @HostListener('document:mousemove', ['$event'])
  onMouseMove(event: MouseEvent): void {
    this.mouseX = event.clientX;
    this.mouseY = event.clientY;
  }
  constructor(private route: ActivatedRoute,
              private siteService:SiteService,
              private contactService:ContactService,
              private router:Router,
              private toastr: ToastrService,
              private studyStarService: StudyStartsService,
              private dialog: MatDialog,
              private authService: AuthenticationService,
    private scoreCardService: ScoreCardService,) { }

  ngOnInit(): void {
    this.scoreID = this.route.snapshot.params.id
    this.noteForm = new UntypedFormGroup({
      sites:new UntypedFormControl([]),
      note: new UntypedFormControl('', Validators.required)
    })
    this.scoreCardService.getScoreContacts(this.scoreID).subscribe(res => {
      this.scoreContacts = res;
    })
    this.refreshDetails();
    this.refreshNotes();
    
  }
  arePointsWithinDistance(x1: number, y1: number, x2: number, y2: number): boolean {
    const distance = Math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2);
    return distance <= 200 && distance >= 0;
  }
  showPopup(event,contact,info) {
    clearTimeout(this.hoverTimeout);
    event.target.style.cursor = 'wait'
    this.initialMousePosition = { x: event.clientX, y: event.clientY };
    const rect = event.target.getBoundingClientRect();
    this.hoverTimeout = setTimeout(() => {
      if( this.arePointsWithinDistance(this.initialMousePosition.x, this.initialMousePosition.y,rect.x,rect.y)){
        this.contactService.getContactDetails({contactID:contact.contactId}).subscribe(res => {
  
          tippy(info as HTMLElement, {
            content: `<h5>
            Company: ${res?.contactDetails?.companyName} <br>
            Job title: ${res?.contactDetails?.jobTitle} <br>
            Bus.Phone: ${res?.contactDetails?.busPhoneOne} <br> 
            Cell Phone: ${res?.contactDetails?.cellPhone} <br>
            Email: ${res?.contactDetails?.email == null ? '':res?.contactDetails?.email} <br>
            </h5>`,
            maxWidth: 400,
            interactive: true,
            interactiveDebounce:500,
        placement: 'top-start',
        allowHTML: true,
            appendTo: 'parent',  
          }).show()
          event.target.style.cursor = 'default'
          info.onmouseout = (arg) => {
            clearTimeout(this.hoverTimeout);
            hideAll()
          }
        })
      }  
      
    }, 1000);
   
      
    
   
   
    
  }

  hidepops() {
    clearTimeout(this.hoverTimeout);
    hideAll({duration:0})
  }
  openContactDetailsDialog(contact){
    contact.id = contact.contactId;
    contact.firstName= contact.contactFullName.split(' ')[0],
      contact.lastName= contact.contactFullName.split(' ')[1],
    this.dialog.open(ContactDetailsDialogComponent, {
      data: {
        contact,
        
      },
      width: 'auto',
      disableClose: true,
      height: 'auto',
      maxHeight:'80vh'

    });
  }

  openEditscoreboardDialog() {
  
    const ref =this.dialog.open(EditScoreboardDialogComponent, {
      data: {
        scoreId:this.scoreID,
        commitment:this.scoreCommitment,
        scoreDetails: this.scoreDetails
      },
      width: 'auto',
      panelClass:'custom-dialog-container',
      disableClose: true,
      height: 'auto',
      maxHeight:'80vh'
    });
      ref.afterClosed().subscribe(res => {
        if(res) {
          this.refreshDetails(true);

        }
      })
  }
   getNoteStudies(){
    if(this.showNoteForm){
      this.notessiteloading = true;
       this.studyStarService.getSitesInStudy(this.scoreID).subscribe(res => {
        this.sites = res
        this.notessiteloading = false
       },err => {
        this.notessiteloading = false

       });

    }
   }

  refreshDetails(withoutCommitment=false) {
    this.loading = true;
    const r2 = this.scoreCardService.getStudyScoreFlow(this.scoreID)
    const r4 = this.scoreCardService.getStudyScoreDetails(this.scoreID)
    const r5 =  this.scoreCardService.getStudyScoreCommitment(this.scoreID)
   
    const r7 = this.scoreCardService.getStudyScoreImport(this.scoreID)
    
    forkJoin([r2,r4,r5,r7]).subscribe(res => {
      this.screenFails = res[0];
      this.scoreDetails = res[1];
      this.scoreCommitment = withoutCommitment ? this.scoreCommitment :res[2];
      this.imports = res[3];
      
      this.actualStatus = this.scoreDetails.actionStatus
      if(withoutCommitment) {
        this.updateScoreCard();
        
      } else {
        this.loading = false;

      }
    }, err => {
      this.loading =false;
    })
    
  }

  openAddContact() {
    const ref = this.dialog.open(ContactSearchResultsForStudiesComponent, {
       data: {
        studyStartID:  this.scoreDetails.studyStartSiteId,
        siteID:this.scoreDetails.siteId,
        scorecard: true,
       },
        width: 'auto',
        height: 'auto',
        disableClose: true,
        maxHeight:'80vh',
        maxWidth:'80vw',
        panelClass:'custom-dialog-container',
     });
     ref.afterClosed().subscribe(res => {
       if(res) {
        this.refreshDetails();
       }
     })  
   }

  getPercetage(partialValue, totalValue) {
    if(isNaN((100 * partialValue) / totalValue) || !isFinite((100 * partialValue) / totalValue)) {
       return 0;
    }  else {
      return ((100 * partialValue) / totalValue).toFixed(2);
    }
  }

  toggle() {
    this.projectMNote = !this.projectMNote;
  }

  sendNote(form) {
  
    if(this.noteForm.valid) {
      this.notesloading = true;
      const data = {
        studyScoreID: this.scoreID,
        req: {
          note: this.noteForm.controls.note.value,
          projectManagementNote: this.projectMNote,
          sites: this.noteForm.controls.sites.value,
        }
        
      }
     
      this.scoreCardService.sendStudyScoreNote(data).subscribe(res => {

        this.noteForm.reset('');
        this.noteForm.controls.sites.setValue([]);
        this.noteForm.controls.sites.updateValueAndValidity();
        this.projectMNote = false;
        this.notesloading = false;

        this.refreshNotes()
      })      
    
    } else {
      this.noteForm.markAllAsTouched();
      this.notesloading = false;

    }
  }

  refreshNotes() {
    this.notesloading = true;
    this.scoreCardService.getStudyScoreNotes({studyScoreID:this.scoreID, includepm:this.includepm}).subscribe(res => {
        this.notes = res;
        
          this.notes?.map(elem => {
            elem.createdDate = elem.createdDate.replace('T',' ') + ' UTC'
            elem.createdDate = elem.createdDate.toString()
          })
        this.notesloading = false;
      })
  }
   

  openPatientMiniViewDialog(patient) {
    if(!patient?.doNotContact) {
      patient.patientId = patient.patientId
      const ref = this.dialog.open(PatientMiniViewDialogComponent, {
        width: 'auto',
        maxHeight:'80vh',
      maxWidth:'80vw',
        height:'auto',
        panelClass:'custom-dialog-container',
        disableClose: true,
        data: {
          fromTracking:false,
          patient,
          site: {id:this.scoreDetails.siteId},
        }
      });
      
    } else {
      this.siteService.showError('Patient marked as DO NOT CONTACT.')
    }
  }
  calculateWeeks(date) {
    date = new Date(date)
    const oneWeekInMilliseconds = 1000 * 60 * 60 * 24 * 7;

    const currentDate = new Date();
    const dateInMilliseconds = date.getTime();
    const currentInMilliseconds = currentDate.getTime();

    const timeDiff = currentInMilliseconds - dateInMilliseconds;
    const weeksElapsed = Math.floor(timeDiff / oneWeekInMilliseconds);

    return weeksElapsed;
  }

  calculateWeeksFuture(startDate) {
    let today =new Date();
    let temp = new Date(startDate)
      return Math.round((temp.getTime() - today.getTime())/1000/60/60/24/7)
  }


  openRemoveContactDialog(contact){
    const dialogref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to remove this contact?',
      },
      disableClose: true,
    } ) 
    dialogref.afterClosed().subscribe(res => {
      if(res) {
this.contactremoveloading = true;
        const data = {
          siteID: this.scoreDetails.siteId,
          studyStartID: this.scoreDetails.studyStartSiteId,
          contactID: contact.id
        }
        this.studyStarService.removeContactFromStudy(data).subscribe(res => {
          this.scoreCardService.getScoreContacts(this.scoreID).subscribe(res => {
            this.scoreContacts =res;
            this.contactremoveloading = false;
          })
        })
      }
    })
  
  }

  changeStatus(status) {
    this.actualStatus = status;
    const data = {
      siteID: this.scoreDetails.siteId,
      studyStartID: this.scoreDetails.studyStartSiteId,
      status
    }
    this.loading = true;
    this.studyStarService.changeStatus(data).subscribe(res => {
      this.refreshDetails()
     })
  }

  updateScoreCard() {
    this.scoreCardService.updateActualDetails(this.scoreID).subscribe(res => {
      this.toastr.success('Scorecard actuals updated!', 'Success!',{
        toastClass:'bg-info ngx-toastr',
        timeOut:3000,
        newestOnTop:false,
        titleClass:' font-weight-bold',
        messageClass:'dotsdots'
      });
      this.scoreCardService.getStudyScoreCommitment(this.scoreID).subscribe(res => {
        this.scoreCommitment = res;
        this.loading = false;
      })
    })
  }
}
