import { NgxMatDateFormats, NgxMatDateAdapter, NGX_MAT_DATE_FORMATS } from '@angular-material-components/datetime-picker';
import { formatDate } from '@angular/common';
import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MAT_MOMENT_DATE_ADAPTER_OPTIONS } from '@angular/material-moment-adapter';
import { MAT_DATE_LOCALE } from '@angular/material/core';
import { MatDialogRef, MAT_DIALOG_DATA, MatDialog } from '@angular/material/dialog';
import { Observable, forkJoin, of } from 'rxjs';

import { MY_FORMATS } from 'src/app/app.component';
import { CustomNgxDatetimeAdapter } from 'src/app/directives/ngx-mat-select/ngx-date-format';
import { AppointmentService } from 'src/app/services/ctms/appointment.service';
import { MiscellaneousService } from 'src/app/services/ctms/miscellaneous.service';
import { PatientsService } from 'src/app/services/ctms/patients.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { FeedBackDialogComponent } from '../../utils/feed-back-dialog/feed-back-dialog.component';
import { ManagePatientInterestDialogComponent } from '../manage-patient-interest-dialog/manage-patient-interest-dialog.component';
import { PatientMiniViewDialogComponent } from '../patient-mini-view-dialog/patient-mini-view-dialog.component';
import { UserService } from 'src/app/services/administration/user.service';
import { DatepickerHeaderComponent } from '../../utils/datepicker-header/datepicker-header.component';
import { SccipService } from 'src/app/services/crm_bd/sccip.service';
import { WindowService } from 'src/app/services/utils/window.service';
import { FollowUpsPatientProfileService } from 'src/app/services/utils/follow-ups-patient-profile.service';
import { BusinessHoursHolidaysDialogComponent } from '../../utils/business-hours-holidays-dialog/business-hours-holidays-dialog.component';


@Component({
  selector: 'app-follow-up-patient-dialog',
  templateUrl: './follow-up-patient-dialog.component.html',
  styleUrls: ['./follow-up-patient-dialog.component.scss'],
  providers:[
    { provide: NGX_MAT_DATE_FORMATS , useValue: MY_FORMATS} 
  ]
})
export class FollowUpPatientDialogComponent implements OnInit {
  patientDetails;
  notes;
  indications= [];
  detailStep= true;
  screenStep = false;
  rescheduleStep = false;
  prescreenStep = false
  confirms = {
    sms: false,
    email: false
  }
  miscResources;
  miscUsers;
  resChecks = [];
  userChecks = [];
  studies;
  rescheduleForm: UntypedFormGroup;
  prescreensForm: UntypedFormGroup;
  screenForm: UntypedFormGroup;
  tryAgain = false;
  defaultTime = [new Date().getHours(), 0 , 0] 
  users = [] 
  loading =false;
  dpheader;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<FollowUpPatientDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any,
    public utilService: UtilService,
    private miscService: MiscellaneousService,
    private sccipService: SccipService,
    private userService: UserService,
    private appointmentService: AppointmentService,
    private patientService: PatientsService,
    private windowService: FollowUpsPatientProfileService,
    private dialog: MatDialog,) { 
      this.dpheader = DatepickerHeaderComponent;
    }

    openInNewWindow() {
      this.windowService.open( this.data.patientDetails.patientId, this.data.patientDetails.patientFullName)
      }
  ngOnInit(): void {
  
    this.rescheduleForm = new UntypedFormGroup({
      outcome: new UntypedFormControl('', Validators.required),
      userId:new UntypedFormControl(this.data?.patientDetails?.createdById, Validators.required),
      followUpDate: new UntypedFormControl('', Validators.required),
      notes: new UntypedFormControl(),
    })
    if(this.data?.patientDetails.attempts == 3){
      this.tryAgain = true;
      this.rescheduleForm.controls.followUpDate.setValidators(null);
      this.rescheduleForm.controls.followUpDate.updateValueAndValidity();
 
    }
    this.loading = true;
    this.prescreensForm = new UntypedFormGroup({
      prescreens: new UntypedFormControl('', Validators.required),
      prescreenType: new UntypedFormControl('',Validators.required),
      notes: new UntypedFormControl(''),
      indication:new UntypedFormControl('', Validators.required),

    });
    this.userService.getAll2({siteId:this.data?.site}).subscribe(res=> {
      this.users = res;

    })
    const r1 = this.appointmentService.getAppointmentStudies(this.data?.site)
    const r3 = this.miscService.getMiscResources(this.data?.site)
    const r4 = this.miscService.getMiscUsers(this.data?.site)
    forkJoin([r1,r3,r4]).subscribe(res => {
      this.studies = res[0];
      this.miscResources = res[1];
      this.miscUsers = res[2];
      this.loading = false;

    });
    this.screenForm = new UntypedFormGroup({
      screenDate :new UntypedFormControl('',Validators.required),
      study :new UntypedFormControl('',Validators.required),
      duration :new UntypedFormControl('',Validators.required),
      notes: new UntypedFormControl(),
      refnotes: new UntypedFormControl(),
    });
    this.patientService.findPatientByID(this.data.patientDetails.patientId).subscribe(res => {
      this.patientDetails = res; 
    });
  }
  receiveDataFromChild(event){

    
    this.patientDetails = event;
        }
  toggleStatus(status) {
    this.tryAgain = !this.tryAgain;
    if(this.tryAgain) {
      this.rescheduleForm.controls.followUpDate.setValidators(Validators.required);
      this.rescheduleForm.controls.followUpDate.updateValueAndValidity();
      this.rescheduleForm.controls.userId.setValidators(Validators.required);
      this.rescheduleForm.controls.userId.updateValueAndValidity();
    } else {
      this.rescheduleForm.controls.followUpDate.setValidators(null);
      this.rescheduleForm.controls.followUpDate.updateValueAndValidity();
      this.rescheduleForm.controls.userId.setValidators(null);
      this.rescheduleForm.controls.userId.updateValueAndValidity();
    }
  }

  openPatientMiniViewDialog(patient) {
    const ref = this.dialog.open(PatientMiniViewDialogComponent, {
      width: 'auto',
      maxHeight:'80vh',
    maxWidth:'80vw',
      height:'auto',
      panelClass:'custom-dialog-container',
      disableClose: true,
      data: {
        fromTracking:false,
        patient,
        site: this.data?.site,
      }
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
        this.dialogRef.close(true);
      }
    })
  }

  activateStep(step) {
    if(step == 1) {
      this.detailStep = true;
      this.screenStep = false;
      this.rescheduleStep = false 
      this.prescreenStep = false 
      
    } else if (step == 2) {
      this.detailStep = false;
      this.screenStep = false;
      this.rescheduleStep = true 
      this.prescreenStep = false 

    } else if(step ==3) {
      this.detailStep = false;
      this.screenStep = true;
      this.rescheduleStep = false 
      this.prescreenStep = false 

    } else {
      this.prescreenStep = true;
      this.detailStep = false;
      this.screenStep = false;
      this.rescheduleStep = false
    }
  }

  screen(form) {
    if(this.screenForm.valid) {
      const data ={
        dailyTaskId: this.data.patientDetails.id,
        req: {
          duration: this.screenForm.controls.duration.value,
          internalNotes: this.screenForm.controls.notes.value,
          resourcesIds: [],
          screenDateTime: this.screenForm.controls.screenDate.value,
          studyId: this.screenForm.controls.study.value,
          userIds: []

        }
          
      }
      this.miscResources.forEach((elem,i) => {
        if(this.resChecks[i]) {
          data.req.resourcesIds.push(elem.id);
        }
      });
      this.miscUsers.forEach((elem,i) => {
        if(this.userChecks[i]) {
          data.req.userIds.push(elem.id);
        }
      });


      this.utilService.isWithinBusinessHours(this.data?.site, new Date(this.screenForm.controls.screenDate.value),formatDate(this.screenForm.controls.screenDate.value,'hh:mm a','en-US'),this.screenForm.controls.duration.value).subscribe(result => {
        if(result) {
  

          this.patientService.screenFollowUp(data).subscribe(res => {

            this.dialogRef.close(true);
          })
        } else {
          const ref12 = this.dialog.open(BusinessHoursHolidaysDialogComponent,{
            data: {
              siteId: this.data?.site,
              isholiday:false,
              appointmentDate:this.screenForm.controls.screenDate.value
            },
            width: 'auto',
            height: 'auto',
            panelClass:'custom-dialog-container',
            disableClose: true,
            maxHeight:'80vh',
            minWidth:'40vw'
          })
  
          ref12.afterClosed().subscribe(result2 => {
            if(result2) {

              this.patientService.screenFollowUp(data).subscribe(res => {

                this.dialogRef.close(true);
              })
            } else {
              this.loading = false;
            }
          })
        }
      })


    } else {
      this.screenForm.markAllAsTouched();
    }
  }
  openManageInterestDialog(patient) {
    patient.patientId = patient.id
    const ref = this.dialog.open(ManagePatientInterestDialogComponent,  {
      width: 'auto',
      height: 'auto',
      disableClose: true,
    panelClass:'custom-dialog-container',
    maxHeight:'80vh',
    maxWidth:'80vw',
      data: {
        patientDetails: patient
      },
    });
    ref.afterClosed().subscribe(res => {
      
       
    }) 
  }

  reschedule(form) {
    if(this.rescheduleForm.valid) {
       const data = {
        dailyTaskId: this.data.patientDetails.followupid,
         req: {
          followUpDate: this.data?.patientDetails.attempts == 3 ?  this.tryAgain ? this.rescheduleForm.controls.followUpDate.value : null : this.rescheduleForm.controls.followUpDate.value ,
          userId: this.rescheduleForm.controls.userId.value,
          notes: this.rescheduleForm.controls.notes.value,
          taskOutcome: this.rescheduleForm.controls.outcome.value,
          
         }
        
       }
       this.patientService.rescheduleFollowUp(data).subscribe(res => {

         this.dialogRef.close(true)
       })
    } else {
      this.rescheduleForm.markAllAsTouched()
    }
  }
  searcher = (search: string, pageNumber: number, pageSize: number):Observable<any[]> => {
    const data = {
      search,
      pageNumber: pageNumber-1,
      pageSize,
    }
    if(pageNumber - 1 != 0) {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        res.content.forEach(element => {
          element.value = element.id;
            this.indications.push(element);
        });
      })
    } else {
      this.sccipService.getindIcationsPaginated(data).subscribe(res => {
        this.indications = res.content;
      })
    }
    return of(this.indications)
  }
  prescreen(form) {
    if(this.prescreensForm.valid) {
     const data = {
       dailyTaskId: this.data.patientDetails.followupid,
       req: {
        appointmentDateTime: formatDate(this.prescreensForm.controls.prescreens.value, 'yyyy-MM-ddTHH:mm', 'en-Us'),
        prescreenType: this.prescreensForm.controls.prescreenType.value,
        smsConfirmation: this.confirms.sms,
        emailConfirmation: this.confirms.email,
        interestId:this.prescreensForm.controls.indication.value,
        notes: this.prescreensForm.controls.notes.value
       }
        
      }


      this.utilService.isWithinBusinessHours(this.data?.site, new Date(this.prescreensForm.controls.prescreens.value),formatDate(this.prescreensForm.controls.prescreens.value,'hh:mm a','en-US'),30).subscribe(result => {
        if(result) {
  
          this.patientService.prescreenFollowUp(data).subscribe(res => {

            this.dialogRef.close(true)
          })
        } else {
          const ref12 = this.dialog.open(BusinessHoursHolidaysDialogComponent,{
            data: {
              siteId: this.data?.site,
              isholiday:false,
              appointmentDate:this.prescreensForm.controls.prescreens.value
            },
            width: 'auto',
            height: 'auto',
            panelClass:'custom-dialog-container',
            disableClose: true,
            maxHeight:'80vh',
            minWidth:'40vw'
          })
  
          ref12.afterClosed().subscribe(result2 => {
            if(result2) {

              this.patientService.prescreenFollowUp(data).subscribe(res => {

                this.dialogRef.close(true)
              })
            } else {
              this.loading = false;
            }
          })
        }
      })




     
    } else {
      this.prescreensForm.markAllAsTouched();
    }
  }

  openCancelDialog() {
    const dialogref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to cancel this follow-up?'
      },
      disableClose: true,
    } ) 
    dialogref.afterClosed().subscribe(res => {
      if(res) {
        const  data =  {
          dailyTaskId: this.data.patientDetails.followupid,
        }
        this.patientService.cancelFollowUp(data).subscribe(res => {

          this.dialogRef.close(true);
        })
      }
   })
  }


}
