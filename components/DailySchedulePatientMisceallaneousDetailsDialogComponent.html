import { Component, HostListener, Inject, OnInit } from '@angular/core';
import { MatDialogRef, MAT_DIALOG_DATA, MatDialog } from '@angular/material/dialog';
import { AppointmentService } from 'src/app/services/ctms/appointment.service';
import { MiscellaneousService } from 'src/app/services/ctms/miscellaneous.service';
import { PatientsService } from 'src/app/services/ctms/patients.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { FeedBackDialogComponent } from '../../utils/feed-back-dialog/feed-back-dialog.component';
import { AddMiscAppointmentComponent } from '../add-misc-appointment/add-misc-appointment.component';
import { CancelAppointmentDialogComponent } from '../cancel-appointment-dialog/cancel-appointment-dialog.component';
import { CompleteScreeningFormDialogComponent } from '../complete-screening-form-dialog/complete-screening-form-dialog.component';
import { PatientMiniViewDialogComponent } from '../patient-mini-view-dialog/patient-mini-view-dialog.component';
import { RescheduleAppointmentDialogComponent } from '../reschedule-appointment-dialog/reschedule-appointment-dialog.component';
import { ScheduleMiscellaneousDialogComponent } from '../schedule-miscellaneous-dialog/schedule-miscellaneous-dialog.component';
import { SmsMessageDialogComponent } from '../sms-message-dialog/sms-message-dialog.component';
import { WindowService } from 'src/app/services/utils/window.service';
import { MiscellaneousPatientProfileService } from 'src/app/services/utils/miscellaneous-patient-profile.service';

@Component({
  selector: 'app-daily-schedule-patient-misceallaneous-details-dialog',
  templateUrl: './daily-schedule-patient-misceallaneous-details-dialog.component.html',
  styleUrls: ['./daily-schedule-patient-misceallaneous-details-dialog.component.scss']
})
export class DailySchedulePatientMisceallaneousDetailsDialogComponent implements OnInit {
  patientDetails;
  patientDetails2;
  patientDetails3;
  notes=[];
  completedStudies=[];
  loading = true;
  studiesloading = false;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<DailySchedulePatientMisceallaneousDetailsDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any,
    public utilService: UtilService,
    private dialog: MatDialog,
    private patientService: PatientsService,
    private windowService: MiscellaneousPatientProfileService,

    private appointmentService: AppointmentService,
    private miscService: MiscellaneousService) { }
    getCompletedStudies(){
this.studiesloading= true;
      this.patientService.getCompletedStudies(this.data.arg.event.extendedProps.patientId).subscribe(res => {
        
  this.completedStudies=res;
  this.studiesloading= false
      })
    }


    receiveDataFromChild(event){
      
    
      this.patientDetails2 = event;
    }

  ngOnInit(): void {





    this.miscService.getMiscDetails(this.data?.arg.event.extendedProps.appointmentId).subscribe(res => {
      this.patientDetails = res;
     
      this.loading = false;

    }, err => {
      this.loading = false;
    })
    
    if(this.data?.patientId != null) {
      this.patientService.findPatientByID(this.data?.patientId).subscribe(res => {
        this.patientDetails2 = res;
        if(this.data?.site == null) {
          this.data.site = res?.siteId ;
        }
        this.loading = false;
        }, err => {
          this.loading = false;
        })
    }
  }
  openInNewWindow() {
    this.windowService.open( this.data.arg.event.extendedProps.patientId, this.data.arg.event.title)
    }
markArrived() {
  this.miscService.markArrived(this.data?.arg.event.extendedProps.appointmentId).subscribe(res => {

    this.dialogRef.close(true);
  });  
}
openPatientMiniViewDialog(patient) {
  const ref = this.dialog.open(PatientMiniViewDialogComponent, {
    width: 'auto',
    maxHeight:'80vh',
    maxWidth:'80vw',
    panelClass:'custom-dialog-container',
    height:'auto',
    disableClose: true,
    data: {
      fromTracking:false,
      patient,
      site: this.data?.site,
    }
  });
  ref.afterClosed().subscribe(res => {
    if(res) {
      this.dialogRef.close(true);
    }
  })
}
markNotConfirmed() {
  this.miscService.changeStatus({appointmentId: this.data?.arg.event.extendedProps.appointmentId, status: 'NOT_CONFIRMED'}).subscribe(res => {

    this.dialogRef.close(true);
  });  
}
markConfirmed() {
  this.miscService.changeStatus({appointmentId: this.data?.arg.event.extendedProps.appointmentId, status: 'CONFIRMED'}).subscribe(res => {

    this.dialogRef.close(true);
  });  
}

markLeftMSG() {
  this.miscService.changeStatus({appointmentId: this.data?.arg.event.extendedProps.appointmentId, status: 'LEFT_MESSAGE'}).subscribe(res => {

    this.dialogRef.close(true);
  });  
}

markAsNoShow() {
  this.miscService.markNoShow(this.data?.arg.event.extendedProps.appointmentId).subscribe(res => {

    this.dialogRef.close(true);
  });
}

undoNoShow() {
  this.miscService.markAsShow(this.data.arg.event.extendedProps.appointmentId).subscribe(res => {

    this.dialogRef.close(true);
  });
}

undoArrived() {
  this.miscService.markNotArrived(this.data.arg.event.extendedProps.appointmentId).subscribe(res => {

    this.dialogRef.close(true);
  });
}


  openCancelAppointmentDialog() {
    const dialogref = this.dialog.open(FeedBackDialogComponent, {
      data: {
        message: 'Are you sure you want to cancel this miscellaneous appointment?'
      },
      disableClose: true,
    } ) 
    dialogref.afterClosed().subscribe(res => {
      if(res) {
       
        this.miscService.cancelMiscAppointement(this.data.arg.event.extendedProps.appointmentId).subscribe(res => {

          this.dialogRef.close(true);
        })
      }
   })
  }

  openRescheduleAppointmentDialog() {
    if(this.data?.arg?.event?.extendedProps?.patientId != null) {
      if(this.utilService.checkPermission('RESCHEDULE_MISCELLANEOUS_PATIENT')){
        const ref = this.dialog.open(ScheduleMiscellaneousDialogComponent,  {
          width: 'auto',
          height: 'auto',
          disableClose: true,           
          panelClass:'custom-dialog-container',
          maxHeight:'80vh',
          maxWidth:'60vw',
          data: {
            reschedule: true,
            arg: this.data.arg,
            site: this.data?.site,
            patientDetails: this.patientDetails2,
            patientDetails2: this.patientDetails
  
          },
        });
        ref.afterClosed().subscribe(res => {
          if(res) {
            this.dialogRef.close(true);
          }
        })
      }
      
    } else {
      if(this.utilService.checkPermission('RESCHEDULE_MISCELLANEOUS')) {
        const ref = this.dialog.open(AddMiscAppointmentComponent,  {
          width: 'auto',
          height: 'auto',
          disableClose: true,
          maxHeight:'80vh',            
          panelClass:'custom-dialog-container',
          maxWidth:'60vw',
          data: {
            reschedule: true,
            arg: this.data.arg,        
            dailyschedule:true,
            site: this.data?.site,
            patientDetails: this.patientDetails,
            patientDetails2: this.patientDetails

          },
        });
        ref.afterClosed().subscribe(res => {
          if(res) {
            this.dialogRef.close(true);
          }
        })
      }
    
    }
   
  }

  openSMSMessageDialog(patient) {
    const ref = this.dialog.open(SmsMessageDialogComponent, {
      width: 'auto',
       maxHeight:'80vh',
    maxWidth:'80vw',
      panelClass:'custom-dialog-container',
      height:'auto',
      disableClose: true,
      data: {
        arg:this.data.arg,
        patientDetails: patient,
        site: this.data?.site,
        appointmentType:'MISCELLANEOUS'
      }
    });
    ref.afterClosed().subscribe(res => {
      if(res) {
      }
    })
  }
}
