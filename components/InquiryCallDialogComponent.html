import { Component, EventEmitter, HostListener, OnInit, Output } from '@angular/core';
import { UntypedFormBuilder, UntypedFormControl, UntypedFormGroup, Validators } from '@angular/forms';
import { MatDialog, MatDialogRef } from '@angular/material/dialog';
import { AuthenticationService } from 'src/app/services/administration/authentication.service';
import { PatientsService } from 'src/app/services/ctms/patients.service';
import { SiteService } from 'src/app/services/regulatory/site.service';
import { UtilService } from 'src/app/services/utils/UtilService';
import { AddPatientDialogComponent } from '../add-patient-dialog/add-patient-dialog.component';
import { AddSelectedPatientDialogComponent } from '../add-selected-patient-dialog/add-selected-patient-dialog.component';
import { AnonymousCallDialogComponent } from '../anonymous-call-dialog/anonymous-call-dialog.component';
import { PatientMiniViewDialogComponent } from '../patient-mini-view-dialog/patient-mini-view-dialog.component';
import { UserPreferencesService } from 'src/app/services/utils/user-preferences.service';
import { Sort } from '@angular/material/sort';

@Component({
  selector: 'app-inquiry-call-dialog',
  templateUrl: './inquiry-call-dialog.component.html',
  styleUrls: ['./inquiry-call-dialog.component.scss']
})
export class InquiryCallDialogComponent implements OnInit {
  resultsStep = false;
  searchStep = true;
  findPatientForm: UntypedFormGroup;
  findedPartners;
  searchedFName = null;
  searchedLName = null;
  sites;
  textLength;
  loading = false;
  @HostListener('window:keyup.esc') onKeyUp() {
    this.dialogRef.close(false);
  }
  
  @HostListener("window:beforeunload", ["$event"]) unloadHandler(event: Event) {
      event.returnValue = false;
  }
  constructor(public dialogRef: MatDialogRef<InquiryCallDialogComponent>,
    private formBuilder: UntypedFormBuilder,
    private dialog: MatDialog,
    public utilService: UtilService,
    private authService: AuthenticationService,
    private userPreferences: UserPreferencesService,
    private siteService: SiteService,
    private patientService: PatientsService) { }


  ngOnInit(): void {
    this.findPatientForm =  this.formBuilder.group({
      firstName: new UntypedFormControl('', [Validators.required]),
      lastName: new UntypedFormControl('', []),
      siteId: new UntypedFormControl('',Validators.required),
    });
    if(JSON.parse(localStorage.getItem('siteCentric.sites'))?.length != 0 && JSON.parse(localStorage.getItem('siteCentric.sites')) != null )   {
      this.sites = JSON.parse(localStorage.getItem('siteCentric.sites'));
      this.findPatientForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
      this.findPatientForm.controls.siteId.updateValueAndValidity();
    } else {
       this.siteService.getAll2(this.authService.loginUserValue.company.id).subscribe(res => {
        localStorage.setItem('siteCentric.sites', JSON.stringify(res));
        this.sites = res;
        this.findPatientForm.controls.siteId.setValue(this.sites.find(elem => elem?.id == JSON.parse(localStorage.getItem('scUserSettings'))?.selectedSite?.id));
        this.findPatientForm.controls.siteId.updateValueAndValidity();
      });
    }
    
  }

  length(ev,e: string) {
  
    this.textLength = e.length < 3 ? false : true;
    this.process(e);
    if(this.textLength && ev?.keyCode ==13) {
      this.findPatient(this.findPatientForm);
    }
}

process(value) {
    value = value.replace(/^a-zA-Z0-9 ]/g, '');
}

  openPatientMiniViewDialog(patient) {
    if(!patient?.doNotContact) {
      patient.patientId = patient.id
      const ref = this.dialog.open(PatientMiniViewDialogComponent, {
        width: 'auto',
        maxHeight:'80vh',
      maxWidth:'80vw',
        height:'auto',
        panelClass:'custom-dialog-container',
        disableClose: true,
        data: {
          dontshowfullprofilebutton:true,
          fromTracking:false,
          patient:patient,
          site: this.findPatientForm.controls.siteId.value
        }
      });
  
    } else {
      this.siteService.showError('Patient marked as DO NOT CONTACT.')
    }
 
  }
  findPatient(form) {
    this.loading = true;
    this.userPreferences.setuserState('selectedSite',this.findPatientForm.controls.siteId.value,true)

    if(this.findPatientForm.valid) {
      const data = form.value;
      this.patientService.searchPatient({text:this.findPatientForm.controls.firstName.value}).subscribe(res => {
        this.resultsStep = true;
        this.searchStep = false;
        this.findedPartners = res;
        this.loading = false;
      }, err => {
        this.loading = false;
      });
    
    } else {
      this.loading = false;
      this.findPatientForm.markAllAsTouched();
    }

  }
findPatient2(event,form) {

  if(event.keyCode == 13 ){
    this.loading = true;
    this.userPreferences.setuserState('selectedSite',this.findPatientForm.controls.siteId.value,true)

    if(this.findPatientForm.valid) {
      const data = form.value;
      this.patientService.searchPatient({text:this.findPatientForm.controls.firstName.value}).subscribe(res => {
        this.resultsStep = true;
        this.searchStep = false;
        this.findedPartners = res;
        this.loading = false;
      }, err => {
        this.loading = false;
      });
    
    } else {
      this.loading = false;
      this.findPatientForm.markAllAsTouched();
    }

  }
    
  }

  schedule(patient) {
    let patientDetails;
     this.patientService.findPatientByID(patient.id).subscribe(res => {
      patientDetails = res;
      const ref =  this.dialog.open(AddSelectedPatientDialogComponent, {
        maxHeight:'80vh',
    maxWidth:'80vw',
        width: 'auto',
        minWidth:'40vw',
        height:'auto',
        disableClose: true,
        panelClass:"custom-dialog-container",
        data: {
          inquirycall:true,
          trackwithcall:false,
          newtrack:false,
          schedule: true,
          patient,
          patientDetails,
        siteID: this.findPatientForm.controls.siteId.value.id,
        siteId: this.findPatientForm.controls.siteId.value.id
        }
      });
      ref.afterClosed().subscribe(res =>{
        if(res) {
         this.utilService.inquiryAdded();
          this.dialogRef.close(true);
        }
      })
    })
    
  }

  openAddPatientDialog() {
   const ref =  this.dialog.open(AddPatientDialogComponent, {
    maxHeight:'80vh',
maxWidth:'80vw',
    width: 'auto',
    minWidth:'40vw',
    height:'auto',
    disableClose: true,
    panelClass:"custom-dialog-container",
      data: {
        schedule: false,
        patientNames: {
          firstName: null,
          lastName: null
        },
        patientDetails : {
          firstName: this.searchedFName,
          lastName: this.searchedLName
        },
        siteId: this.findPatientForm.controls.siteId.value.id,
        siteID: this.findPatientForm.controls.siteId.value.id
      }
    });
    ref.afterClosed().subscribe(res =>{
      if(res) {
        this.utilService.inquiryAdded();
        this.dialogRef.close(true);
      }
    })
  }

  openAnonymousDialog() {
    const ref =  this.dialog.open(AnonymousCallDialogComponent, {
     minWidth: '50vw',
     maxHeight:'80vh',
     maxWidth:'80vw',
     width: 'auto',
     height:'auto',
     disableClose: true,
     panelClass:"custom-dialog-container",
       data: {
         patientNames: {
           firstName: '',
           lastName: ''
         },
        siteID: this.findPatientForm.controls.siteId.value.id,
        siteId: this.findPatientForm.controls.siteId.value.id
       }
     });
     ref.afterClosed().subscribe(res =>{
       if(res) {
         this.utilService.inquiryAdded();
         this.dialogRef.close(true);
       }
     })
   }
   sortData(sort: Sort) {
    const data = this.findedPartners;
    if (!sort.active || sort.direction === '') {
    this.findedPartners = data;
    return;
    }
    this.findedPartners = data.sort((a, b) => {
    const isAsc = sort.direction === 'asc';
    switch (sort.active) {
        case 'name':
          return this.compare(a.fullName?.toLowerCase(), b.fullName?.toLowerCase(), isAsc);
     case 'dob':
          return this.compare(new Date(a.birthDate).getTime(), new Date(b.birthDate).getTime(), isAsc);
     case 'phone':
          return this.compare(a.phone?.toLowerCase(), b.phone?.toLowerCase(), isAsc);
     case 'email':
          return this.compare(a.email?.toLowerCase(), b.email?.toLowerCase(), isAsc);
     case 'city':
          return this.compare(a.city?.toLowerCase(), b.city?.toLowerCase(), isAsc);
    
    }
    });
}

compare(a: number | string, b: number | string, isAsc: boolean) {
  return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
}
  
}
