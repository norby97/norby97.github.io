import { Injectable } from '@angular/core';
import { ActivatedRouteSnapshot, CanActivate, Router, RouterStateSnapshot, UrlTree } from '@angular/router';
import { ToastrService } from 'ngx-toastr';
import { Observable } from 'rxjs';
import { AuthenticationService } from '../services/administration/authentication.service';
import { UserService } from '../services/administration/user.service';

@Injectable({
  providedIn: 'root'
})
export class RegulatoryGuard implements CanActivate {
  constructor(public auth: AuthenticationService, public router: Router, private toastr:ToastrService, private userService:UserService) {}

  canActivate(
    route: ActivatedRouteSnapshot,
    state: RouterStateSnapshot)
    : Observable<boolean | UrlTree> | Promise<boolean | UrlTree> | boolean | UrlTree {

      if(state.url =='/regulatory/approvals') {
        this.auth?.loginUserValue?.modules[1].permissions.map(item => {
          if(item.name =='REGULATORY_ADVERTISING_APPROVALS') {
            if(item.permission == 'NONE') {
              this.cantActivate()
            } else return true;    
          }
        })
      }
      if(state.url =='/regulatory/reg-manager') {
        this.auth?.loginUserValue?.modules[1].permissions.map(item => {
          if(item.name =='REG_MANAGER') {
            if(item.permission == 'NONE') {
              this.cantActivate()
            } else return true;    
          }
        })
      }
      if(state.url =='/regulatory/training') {
        this.auth?.loginUserValue?.modules[1].permissions.map(item => {
          if(item.name =='REGULATORY_TRAINING') {
            if(item.permission == 'NONE') {
              this.cantActivate()
            } else return true;    
          }
        })
      }
    return true;
  }
  cantActivate() {
    if(this.auth?.loginUserValue?.accountType == 'EXTERNAL') {
      this.router.navigate(['/regulatory/dashboard']) 
    } else {
      this.router.navigate(['/my-dashboard', this.auth.loginUserValue?.firstName + ' ' + this.auth.loginUserValue?.lastName]) 
    }
    this.toastr.error(`Contact company administrator!`,'Access Denied!', {
      timeOut: 5000,
      extendedTimeOut:0,
      tapToDismiss:true,
      toastClass:'bg-danger ngx-toastr',
      titleClass:'dotsdotstitle font-weight-bold',
      messageClass:'dotsdots'
    }).onTap.subscribe(ress => {
      this.userService.showError('Action not permitted.Please contact company administrator for access.',1)
    });
    return false;
  }
}
